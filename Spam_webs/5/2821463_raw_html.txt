His Jade label absorbed those God-fearing artists: family bands with wailing kids and barely amateur groups sourced from local parishes, infused with reverberations of country and western and deep soul.         		if(cart_total == 0)

                    $("#billing_info_container").html("No credit card needed, this order is free, man.");
        		}    
        		    		
    			$("#order_button").show();
    			$("#cart_total").html(cart_total.toFixed(2));
    			$("#charge_amount").html(cart_total.toFixed(2))
                //$("#charge_amt").val(cart_total.toFixed(2));
                $("#charge_amt").attr("value",cart_total.toFixed(2));
    			$("#charge_amt_cash").html(cart_total.toFixed(2));
                $("#customer_credit_after_transaction").val(credit_balance_after_order.toFixed(2));
                $("#credit_used").val(credit_to_use.toFixed(2));
	
			}
		});   
    return false;
});

   <script type="text/javascript">
    jQuery(function($){

    // Create variables (in this scope) to hold the API and image size
    var jcrop_api,
        boundx,
        boundy,

        // Grab some information about the preview pane
        $preview = $('#preview-pane'),
        $pcnt = $('#preview-pane .preview-container'),
        $pimg = $('#preview-pane .preview-container img'),

        xsize = $pcnt.width(),
        ysize = $pcnt.height();
    
    $('#target').Jcrop({
      onChange: updatePreview,
      onSelect: updatePreview,
      aspectRatio: xsize / ysize
    },function(){
      // Use the API to get the real image size
      var bounds = this.getBounds();
      boundx = bounds[0];
      boundy = bounds[1];
      // Store the API in the jcrop_api variable
      jcrop_api = this;

      // Move the preview into the jcrop container for css positioning
      $preview.appendTo(jcrop_api.ui.holder);
    });

    function updatePreview(c)
    {
      if (parseInt(c.w) &gt; 0)
      {
        var rx = xsize / c.w;
        var ry = ysize / c.h;

		$('#x1').val(c.x);
	    $('#y1').val(c.y);
	    $('#x2').val(c.x2);
	    $('#y2').val(c.y2);
	    $('#w').val(c.w);
	    $('#h').val(c.h);

        $pimg.css({
          width: Math.round(rx * boundx) + 'px',
          height: Math.round(ry * boundy) + 'px',
          marginLeft: '-' + Math.round(rx * c.x) + 'px',
          marginTop: '-' + Math.round(ry * c.y) + 'px'
        });
      }
    };
    
  });
   </script>
