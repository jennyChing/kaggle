                    embedVideo = [embedVideo.slice(0, qstart), 'autoplay=1&amp;', embedVideo.slice(qstart)].join('');
                }
                else {
                    var srcstart = embedVideo.indexOf('src="');
                    var srcend = embedVideo.indexOf('"', srcstart + 5);
                    embedVideo = [embedVideo.slice(0, srcend), '?autoplay=1', embedVideo.slice(srcend)].join('');
                }
            }

            $('#creatorMarquee').html(embedVideo);
            return false;
        });

            var clickedAdd = $this.attr('add');
            if (clickedRid == undefined)
                return false;

                    // user clicked 'Yes'

                    if (clickedAdd == 1) {
                        $this.find('.waitLink:first').text('You\'re on the waiting list');
                        $this.attr('add', 0);
                    }
                    else {
                        $this.find('.waitLink:first').text('Click to be notified when a spot opens');
                        $this.attr('add', 1);
                    }
                }
                else {
                    // user clicked 'No' so we do nothing
                }
            });
        });
