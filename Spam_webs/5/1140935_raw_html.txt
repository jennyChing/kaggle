 * Allow GA events via `ga-event` attributes on elements, both those
 * static on the page and those added in dynamically.
 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')
 *
 * Elements that have been processed on the page already will have a 'ga-event-added' attribute
 *
 */
$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event
    //  non-interactive (optional) - true for non-interactive events (ie. it shouldn't be used in bounce-rate calcs)
    //
    // var ga_event_handler = function(ev) {
    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';
                
            // Only add a delegated handler once for a given even `type'
            if (!_handlers[type]) {
                // Delegate to handle dynamic elements, track element ref via closure 
                $(document).on(type, selector, (function() {
                    return function(ev) {
                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());
                _handlers[type] = true;
            }

            // mark element as processed
            $(this).attr('ga-event-added', 'true');
        });

        // wait for 100ms and poll again (catch dynmically added elements)
        window.setTimeout(poll_ga_attr_events, 150);
    }


    // let's go, start processing ...
    poll_ga_attr_events();

});

function listen(evnt, elem, func) {
    if (elem.addEventListener) { // W3C DOM

        var r = elem.attachEvent("on"+evnt, func);
        return r;
    }

       </h2>
       <div class="section_body">
        <p>
         Sterling Hayden was one of the stars in some of the classic-era films, like "Asphalt Jungle." He was also the corrupt cop Michael Corleone shot in "The Godfather." Before that, however, Hayden did real espionage work for the OSS, including parachuting into Croatia as well as shipping supplies to Nazi-resistance fighters in Yugoslavia.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="11">
        <img alt="Greta Garbo" class="image" src="//file.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/7/1/3/4/3/71343895/7142f873b8a8b128ff8a1c2e049b72e6f527af8a.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         Greta Garbo
        </span>
       </h2>
       <div class="section_body">
        <p>
         The Swedish-born actress was not just pretty, but talented and intelligent as well. When she retired from acting in 1935, many suspected that it was just an excuse to cover up her activities as a spy for the Allied forces. One of her missions was to transport a man from Copenhagen to Britain. That man happened to be Niels Bohr, who developed the atomic bomb.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="12">
        <img alt="John Ford" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/5/3/1/2/7/531270032/a7964b92b06df0c2a645e032cb9308107db23136.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         John Ford
        </span>
       </h2>
       <div class="section_body">
        <p>
         "Stagecoach" and "Grapes of Wrath" director John Ford was also a Naval Reserve officer before he dabbled in films. His contribution included filming secret footage and documentaries for the OSS where he earned citations for bravery after receiving wounds in one of his assignments. He also alerted the agency about a suspected Japanese presence near the coastal areas of Baja. After the war, he became an advisor to William Donovan.
        </p>
       </div>
      </div>

It originally was a singles bar.Waffle House is now a popular breakfast spot and a late night diner for college students. It all started in 1955 in Decatur, Georgia.Some of you may hardly ever go into one of these restaurants because they now deliver. Pizza Hut started in 1958 in Wichita, Kansas.

             12 Famous Logos with Strange Origins
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1215332">
            The Apple logo may be one of the world's most famous technology logos. An apple with a bite taken out of it is supposed to represent the apple from the Tree of Knowledge in th
            <span class="hellip">
             
            </span>
            <span class="hidden">
             e biblical story of Adam and Eve. Aside from representing knowledge, the fruit from that story also represents sin and a fall from grace.The logo of Trojan Condoms features the profile of a Greek warrior to show virility and masculinity. However, the story of the Trojan horse gives the brand's warrior logo a slightly different meaning. During the attack on Troy in the Trojan War, warriors from a different part of Greece built a large wooden horse and hid their soldiers inside. The people of Troy wheeled the horse into their city where the soldiers burst free from the horse to claim Troy. If you are using a Trojan condom, chances are you don't want its contents to free themselves inside the condom's destination.While the famous golden M logo of McDonald's does in fact stand for McDonald's, the company thought about changing the logo back in the 1960s. While contemplating the proposed logo chain, McDonald's design consultant and psychologist Louis Cheskin insisted they leave the logo as it was, stating that people would subconsciously relate the mounds of the rounded M to a pair of very nourishing breasts. While it is true that McDonald's provides food, the amount of nutritional value in the food calls into question any of the nourishment that is supposedly associated with the logo.All coffee lovers are familiar with the green logo on the Starbucks cups that features a mermaid-like creature in the center of it. The mermaid-esque creature is actually a siren from Greek mythology, and sirens represent obsession, addiction, and death. While obsession and addiction make sense in context of Starbucks, let's hope death never does.The peacock used to represent NBC has strange origins. In Greek mythology, Hermes lulled the hundred-eyed watchman to sleep and killed him. Since he was a servant to Hera, the goddess had the watchman's eyes placed on the tail of her favorite bird, the peacock. The peacock then became symbolic of the all-seeing eye and knowledge.Mobil's logo itself holds no meaning, but the colors do. The red and blue used in the logo echo the red and blue in the flag and are supposed to represent stability. In a world where gas prices are never stable and oil is a hotly contested commodity, stability makes little sense in this company's image.The luxury clothing brand Versace has a well-known symbol of a woman's head inside a circle. What you might not know is the woman in the logo is Medusa, the snake-haired woman from mythology. Legend has it that Medusa was once so beautiful that she seduced Poseidon and the two of them got busy inside the goddess Athena's temple. Athena punished Medusa for her affair and turned her hair into snakes. So the symbol on Versace's garments implies being too attractive can be dangerous.The universal symbol for medical institutions worldwide is a staff wrapped with two snakes surrounded by a pair of wings. This staff, called the Caduceus, is actually Mercury's staff. Mercury was the Roman god of liars and thieves.One of the United States' largest equity firms, Cerberus Capital Management chose its name based on the three-headed mythological dog because the founder wanted to have the company be associated with protecting their clients assets. However, what the company didn't take into account was this mythological beast is famous for eating humans alive after trapping them.The jewelry maker Pandora sells a popular charm bracelet that can add bead to for personalization. The myth of Pandora's Box is the basis for the name of the brand. According to the myth, Pandora introduced evil into the world when she couldn't contain her curiosity and opened her box, making the name a strange moniker for a jewelry brand.Amazon's smiley face logo may be well known, and some may even be aware that the smile itself is an arrow that points from the a to the z of the company name. However, the company named itself after a group of rough and tumble, Greek female warriors. The warrior tribe was fairly violent, making the juxtaposition between the smiling logo and the origins for the company name really stand out.The Droid Eris was the first smartphone carried by Verizon and claimed to have a lot of cool, high tech features. The logo chosen for this line of phones, Eris, was the Greek goddess blamed with starting the Trojan War. She is most often associated with frustration and disaccord, which is quite fitting, given that this line of smart phones had many bugs to work out.

            <a class="title_link black" href="http://www.answers.com/article/1224882/12-famous-tv-restaurants-you-can-actually-visit">
             12 Famous TV Restaurants You Can Actually Visit
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1224882">
            Dine and drink like your favorite television characters. Many of the restaurants you see on TV are inspired by real-life restaurants or have inspired restaurants based on the
            <span class="hellip">
             
            </span>
            <span class="hidden">
             show.There really is a bar named Cheers. In fact, the exteriors you saw in the sitcom, 'Cheers,' were of the real Cheers in Boston.Tom's Restaurant in New York is the real restaurant that inspired Monk's on Seinfeld. On Seinfeld episodes, you will see exterior shots of Tom's Restaurant.In 'Orange is the New Black,' Piper visits a restaurant called Dmitri's. For those scenes, Rosario's, a deli in New York, was the filming location.Before the hit show 'Nashville,' The Bluebird Cafe was already famous as a music venue. However, many of the show's characters have performed at the cafe, making it even more popular for Nashville visitors.Cafe Grumpy in New York is in the television series 'Girls.' Hannah works at the caf? and many scenes take place there.Enjoy a few cocktails at the Campbell Apartment. 'Gossip Girl' made this bar in Grand Central Station famous.McGee's Pub in New York was the inspiration for McClaren's Pub on 'How I Meet Your Mother.' The show's creators Craig Thomas and Carter Bays used this pub, because they used to drink there when they worked together on 'The Late Show with David Letterman.'Have some cherry pie at Twede's Cafe in North Bend, Washington. Twede's Cafe was the filming location for several scenes in Twin Peaks.' The show also made the cafe's cherry pie famous.Holsten's appears in the last scene of 'The Sopranos.' You can visit the ice cream shop in Bloomfield, N.J.In the intro for 'Louie,' you see a shot of Ben's Pizzeria. It is a real pizza joint in New York off of West Third Street.
