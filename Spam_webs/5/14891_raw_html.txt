   var _add_comment_open = false;

		window.onbeforeunload = confirmExit;
		function confirmExit()
		{
			if (!_add_comment_open) return;
			var pm = $('#comments #post_message').val()
			if (trim(pm).length &lt;= 0) return;
			
			return "Do you want to leave without finishing your comment?";
		}
		
		function createCommentCover(node)
		{
			var c = $('&lt;div class="cover" style="width:'+node.width()+'px;height:'+node.height()+'px;position:absolute;"&gt;&lt;/div&gt;');
			node.prepend(c);
			return c;
		}
		
		
		function deleteComment(e)
		{
			var post = $(e).closest('.post');
			var id = post.attr('id').split("_").pop();
			Jigidi.deleteComment(id);

			var cover = createCommentCover(post);
			cover.html('&lt;div&gt;Deleting comment ...&lt;/div&gt;');
		} 
		
		function onCommentDeleted(data)
		{
			if (data.result != 'OK') return;

			$('#comment_count').html(parseInt($('#comment_count').html())-1);				
			$('#post_'+data.id+' .cover').html('&lt;div&gt;&lt;p&gt;Comment deleted&lt;/p&gt;&lt;span class="js_link" onclick="restoreComment(this)"&gt;Undo&lt;/span&gt;&lt;/div&gt;');
		}

		function restoreComment(e)
		{
			var post = $(e).closest('.post');
			var id = post.attr('id').split("_").pop();
			Jigidi.restoreComment(id);
			$('#post_'+id+' .cover').html('&lt;div&gt;Restoring comment ...&lt;/div&gt;');
		}

		function onCommentRestored(data)
		{
			if (data.result != 'OK') return;

			$('#comment_count').html(parseInt($('#comment_count').html())+1);		
			$('#post_'+data.id+' .cover').remove();
		}

		function onCommentAdded(data)
		{
			if (data.result != 'OK') 
			{
				g_error = true;
				$('#comments #post_submit').val("An error occured");
				return;
			}

			$('#comment_count').html(parseInt($('#comment_count').html())+1);
							
			$('#comments #post_submit').val("Comment posted!");

			var message = $('#comments #post_message').val();
			message = message.replace(/\n/g, "&lt;br&gt;");
			
			var html = $('#comments #post_template').html();
			html = html.replace("%message%", message);

			var post = $('&lt;div id="post_'+data.id+'" class="post"/&gt;').html(html);
			
			var placeholder = $('&lt;div style="display:none"/&gt;');
			placeholder.append(post);
			$('#comments').append(placeholder);
			var ph = placeholder.height();
			placeholder.remove();			

			$('#new_post_container').animate({height:ph}, 300, function() {
				$('#new_post_container').hide();
				post.hide();
				$('#comments #post_template').after(post);
				post.fadeIn(300);

				_add_comment_open = false;
				
				$('#comments #add_new_comment').show();
			} );
			$('#new_comment_input').fadeOut(300);
			
			
		}
		
		
		var g_key = 'nrhymh';
		var g_count = 0;
		var g_request_key = false;
		
		function showNewComment(event)
		{
			$('#comments #add_new_comment').hide();
			
			$('#new_comment_input').show();
			_add_comment_open = true;
			
			$('#comments #post_message').val('');
			var e = $('#comments #post_submit');
			e.val("Post comment");
			e.prop('disabled', true);
			$('#new_post_container').css('height','');
			$('#comments #new_post_container').slideDown(200, function(){$('#comments #post_message').focus();});
			g_count++;
			g_request_key = g_key+'|'+g_count;
			event.preventDefault();
			return false;
		}

		var g_timer = false;
		var g_error = false;
		
		function onPostTimer()
		{
			var pm = $('#comments #post_message').val()
			$('#comments #post_submit').prop('disabled', (trim(pm).length &gt; 0) ? false : true);;
		}
		
		function onPostMessageFocus()
		{
			if (g_timer || g_error) return;
			g_timer = setInterval(onPostTimer, 100);
		}

		function postForm()
		{
			var message = $('textarea[name=message]').val();
			if (message.length == 0) return;
			
			clearInterval(g_timer);
			g_timer = false;

			var e = $('#comments #post_submit');
			e.val("... hold on ...");
			e.prop('disabled', true);

			
			Jigidi.addComment('BI9Q0A4A', message, g_request_key);
        }
