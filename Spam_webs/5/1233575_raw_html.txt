       <h2 class="title">
        Art: Superhero Ancestors
       </h2>
       <span>
        by
       </span>
       <a href="http://www.chuknum.com" rel="author external" title="Visit Blues website">
        Blue
       </a>
       <span>
        on
       </span>
       May 10, 2012
      </div>
      <!--END post-meta-->
      <p>
       <a href="http://chuknum.com/wp-content/uploads/2012/05/Superhero-Ancestors-@chuknum.com_.jpg">
        <img alt="" class="size-full wp-image-20188 alignleft" height="290" src="http://chuknum.com/wp-content/uploads/2012/05/Superhero-Ancestors-@chuknum.com_.jpg" title="Superhero Ancestors @chuknum.com" width="290"/>
       </a>
       Ever imagined what the ancestors of superheroes would look like as if they had a father and mother who had the same abilities only they look much older. For instance, we know Batmans father had nothing to do with gadgets and the suit but clever artists
       <a href="http://www.facebook.com/pages/foto-marvellini/299320806756117" onclick="javascript:_gaq.push(['_trackEvent','outbound-article','http://www.facebook.com']);" target="_blank">
        <strong>
         Carlo
        </strong>
        and
        <strong>
         Andrea Marvellini
        </strong>
       </a>
       have given us a fictional glimpse of the past.
      </p>
      <p>
       These two artists who happen to be brothers, have taken old vintage photos and added modern day super heroes onto them, the result is simply great and very believable. Obviously the characters are fictional to start with but these portraits add a lot more depth and history to them.
      </p>
      <p>
       The manipulation of these images are top-notch, they not only replace their head but did tweaking and re-imagination of how suits would look like.Take a look at Darth Vaders family, the helmet has open eye sockets! All in all, I think these are really great.
      </p>

      auto_join_text_ing: "i am",             // [string]   auto tense for present tense: "i was" surfing
      auto_join_text_reply: "i replied to",   // [string]   auto tense for replies: "i replied to" @someone "with"

      loading_text: null,                     // [string]   optional loading text, displayed while tweets load
      query: null,                            // [string]   optional search query
      refresh_interval: null                  // [integer]  optional number of seconds after which to reload tweets
    };
    
    if(o) $.extend(s, o);
    
    $.fn.extend({
      linkUrl: function() {
        var returning = [];
        var regexp = /((ftp|http|https):\/\/(\w+:{0,1}\w*@)?(\S+)(:[0-9]+)?(\/|\/([\w#!:.?+=&amp;%@!\-\/]))?)/gi;
        this.each(function() {
          returning.push(this.replace(regexp,"&lt;a href=\"$1\"&gt;$1&lt;/a&gt;"));
        });
        return $(returning);
      },
      linkUser: function() {
        var returning = [];
        var regexp = /[\@]+([A-Za-z0-9-_]+)/gi;
        this.each(function() {
          returning.push(this.replace(regexp,"&lt;a href=\"http://twitter.com/$1\"&gt;@$1&lt;/a&gt;"));
        });
        return $(returning);
      },
      linkHash: function() {
        var returning = [];
        var regexp = /(?:^| )[\#]+([A-Za-z0-9-_]+)/gi;
        this.each(function() {
          returning.push(this.replace(regexp, ' &lt;a href="http://search.twitter.com/search?q=&amp;tag=$1&amp;lang=all&amp;from='+s.username.join("%2BOR%2B")+'"&gt;#$1&lt;/a&gt;'));
        });
        return $(returning);
      },
      capAwesome: function() {
        var returning = [];
        this.each(function() {
          returning.push(this.replace(/\b(awesome)\b/gi, '&lt;span class="awesome"&gt;$1&lt;/span&gt;'));
        });
        return $(returning);
      },
      capEpic: function() {
        var returning = [];
        this.each(function() {
          returning.push(this.replace(/\b(epic)\b/gi, '&lt;span class="epic"&gt;$1&lt;/span&gt;'));
        });
        return $(returning);
      },
      makeHeart: function() {
        var returning = [];
        this.each(function() {
          returning.push(this.replace(/(&lt;)+[3]/gi, "&lt;tt class='heart'&gt;&#x2665;&lt;/tt&gt;"));
        });
        return $(returning);
      }
    });

    function parse_date(date_str) {
      // The non-search twitter APIs return inconsistently-formatted dates, which Date.parse
      // cannot handle in IE. We therefore perform the following transformation:
      // "Wed Apr 29 08:53:31 +0000 2009" =&gt; "Wed, Apr 29 2009 08:53:31 +0000"
      return Date.parse(date_str.replace(/^([a-z]{3})( [a-z]{3} \d\d?)(.*)( \d{4})$/i, '$1,$2$4$3'));
    }

    function relative_time(time_value) {
      var parsed_date = parse_date(time_value);
      var relative_to = (arguments.length &gt; 1) ? arguments[1] : new Date();
      var delta = parseInt((relative_to.getTime() - parsed_date) / 1000);
      var r = '';
      if (delta &lt; 60) {
	r = delta + ' seconds ago';
      } else if(delta &lt; 120) {
	r = 'a minute ago';
      } else if(delta &lt; (45*60)) {
	r = (parseInt(delta / 60, 10)).toString() + ' minutes ago';
      } else if(delta &lt; (2*60*60)) {
	r = 'an hour ago';
      } else if(delta &lt; (24*60*60)) {
	r = '' + (parseInt(delta / 3600, 10)).toString() + ' hours ago';
      } else if(delta &lt; (48*60*60)) {
	r = 'a day ago';
      } else {
	r = (parseInt(delta / 86400, 10)).toString() + ' days ago';
      }
      return 'about ' + r;
    }

    function build_url() {
      var proto = ('https:' == document.location.protocol ? 'https:' : 'http:');
      if (s.list) {
        return proto+"//api.twitter.com/1/"+s.username[0]+"/lists/"+s.list+"/statuses.json?per_page="+s.count+"&amp;callback=?";
      } else if (s.query == null &amp;&amp; s.username.length == 1) {
        return proto+'//api.twitter.com/1/statuses/user_timeline.json?screen_name='+s.username[0]+'&amp;count='+s.count+'&amp;include_rts=1&amp;callback=?';
      } else {
        var query = (s.query || 'from:'+s.username.join(' OR from:'));
        return proto+'//search.twitter.com/search.json?&amp;q='+encodeURIComponent(query)+'&amp;rpp='+s.count+'&amp;callback=?';
      }
    }

    return this.each(function(i, widget){
      var list = $('&lt;ul class="tweet_list"&gt;').appendTo(widget);
      var intro = '&lt;p class="tweet_intro"&gt;'+s.intro_text+'&lt;/p&gt;';
      var outro = '&lt;p class="tweet_outro"&gt;'+s.outro_text+'&lt;/p&gt;';
      var loading = $('&lt;p class="loading"&gt;'+s.loading_text+'&lt;/p&gt;');

      if(typeof(s.username) == "string"){
        s.username = [s.username];
      }

      if (s.loading_text) $(widget).append(loading);
      $(widget).bind("load", function(){
        $.getJSON(build_url(), function(data){
          if (s.loading_text) loading.remove();
          if (s.intro_text) list.before(intro);
          list.empty();
          var tweets = (data.results || data);
          $.each(tweets, function(i,item){
            // auto join text based on verb tense and content
            if (s.join_text == "auto") {
              if (item.text.match(/^(@([A-Za-z0-9-_]+)) .*/i)) {
                var join_text = s.auto_join_text_reply;
              } else if (item.text.match(/(^\w+:\/\/[A-Za-z0-9-_]+\.[A-Za-z0-9-_:%&amp;\?\/.=]+) .*/i)) {
                var join_text = s.auto_join_text_url;
              } else if (item.text.match(/^((\w+ed)|just) .*/im)) {
                var join_text = s.auto_join_text_ed;
              } else if (item.text.match(/^(\w*ing) .*/i)) {
                var join_text = s.auto_join_text_ing;
              } else {
                var join_text = s.auto_join_text_default;
              }
            } else {
              var join_text = s.join_text;
            };
   
            var from_user = item.from_user || item.user.screen_name;
            var profile_image_url = item.profile_image_url || item.user.profile_image_url;
            var join_template = '&lt;span class="tweet_join"&gt; '+join_text+' &lt;/span&gt;';
            var join = ((s.join_text) ? join_template : ' ');
            var avatar_template = '&lt;a class="tweet_avatar" href="http://twitter.com/'+from_user+'"&gt;&lt;img src="'+profile_image_url+'" height="'+s.avatar_size+'" width="'+s.avatar_size+'" alt="'+from_user+'\'s avatar" title="'+from_user+'\'s avatar" border="0"/&gt;&lt;/a&gt;';
            var avatar = (s.avatar_size ? avatar_template : '');
/*            var date = '&lt;span class="tweet_time"&gt;&lt;a href="http://twitter.com/'+from_user+'/statuses/'+item.id+'" title="view tweet on twitter"&gt;'+relative_time(item.created_at)+'&lt;/a&gt;&lt;/span&gt;';*/
			var date = '&lt;span class="tweet_time"&gt;&lt;a href="http://twitter.com/'+from_user+'" title="view tweet on twitter"&gt;'+relative_time(item.created_at)+'&lt;/a&gt;&lt;/span&gt;';
            var text = '&lt;span class="tweet_text"&gt;' +$([item.text]).linkUrl().linkUser().linkHash().makeHeart().capAwesome().capEpic()[0]+ '&lt;/span&gt;';
   
            // until we create a template option, arrange the items below to alter a tweet's display.
            list.append('&lt;li&gt;' + avatar + join + text + date + '&lt;/li&gt;');
   
            list.children('li:first').addClass('tweet_first');
            list.children('li:odd').addClass('tweet_even');
            list.children('li:even').addClass('tweet_odd');
          });
          if (s.outro_text) list.after(outro);
          $(widget).trigger("loaded").trigger((tweets.length == 0 ? "empty" : "full"));
          if (s.refresh_interval) {
            window.setTimeout(function() { $(widget).trigger("load"); }, 1000 * s.refresh_interval);
          };
        });
      }).trigger("load");
    });
  };
