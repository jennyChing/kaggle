 function ActivateCountDown(strContainerID, initialValue)
 {    _countDowncontainer = document.getElementById(strContainerID);
      if (!_countDowncontainer)
      {
            alert("count down error: container does not exist: "+strContainerID+"\nmake sure html element with this ID exists");

      function CountDownTick()
      {    
      if (_currentSeconds &lt;= 0)
      {
      _countDowncontainer.innerHTML = "Your session has expired.&lt;br /&gt;&lt;br /&gt;Please click the \"OK\" button to return to the home page.";
      return;
      }
      
      SetCountdownText(_currentSeconds-1);
      window.setTimeout("CountDownTick()", 1000);
      }

      {   
       _currentSeconds = seconds;
       var minutes=parseInt(seconds/60);
       seconds = (seconds%60);
       var hours=parseInt(minutes/60);
       minutes = (minutes%60);
       var strText = AddZero(minutes) + ":" + AddZero(seconds);
       _countDowncontainer.innerHTML = "Your session will expire in " + strText + ".&lt;br /&gt;&lt;br /&gt;Click the \"Ok\" button to keep your session active.";
       
       }
