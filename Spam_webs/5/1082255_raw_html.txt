        (function() {
      var form = document.getElementById('mad_mimi_signup_form'),
          submit = document.getElementById('webform_submit_button'),
          validEmail = /.+@.+\..+/,
          isValid;
          
      form.onsubmit = function(event) {
        validate();
        if(!isValid) {
          revalidateOnChange();
          return false;
        }
      };
      
      function validate() {
        isValid = true;
        emailValidation();
        fieldAndListValidation();
        updateFormAfterValidation();
      }
    
      function emailValidation() {
        var email = document.getElementById('signup_email');
        if(!validEmail.test(email.value)) {
          textFieldError(email);
          isValid = false;
        } else {
          removeTextFieldError(email);
        }
      }
    
      function fieldAndListValidation() {
        var fields = form.querySelectorAll('.mimi_field.required');
        for(var i = 0; i &lt; fields.length; ++i) {
          var field = fields[i],
              type  = fieldType(field);
          if(type == 'checkboxes' || type == 'radio_buttons') {
            checkboxAndRadioValidation(field);
          } else {
            textAndDropdownValidation(field, type);
          }
        }
      }
    
      function fieldType(field) {
        var type = field.querySelectorAll('.field_type');
        if(type.length &gt; 0) {
          return type[0].getAttribute('data-field-type');
        } else if(field.className.indexOf('checkgroup') &gt;= 0) {
          return 'checkboxes';
        } else {
          return 'text_field';
        }
      }
    
      function checkboxAndRadioValidation(field) {
        var inputs   = field.getElementsByTagName('input'),
            selected = false;
        for(var i = 0; i &lt; inputs.length; ++i) {
          var input = inputs[i];
          if((input.type == 'checkbox' || input.type == 'radio') &amp;&amp; input.checked) selected = true;
        }
        if(selected) {
          field.className = field.className.replace(/ invalid/g, '');
        } else {
          if(field.className.indexOf('invalid') == -1) field.className += ' invalid';
          isValid = false;
        }
      }
    
      function textAndDropdownValidation(field, type) {
        var inputs = field.getElementsByTagName('input');
        for(var i = 0; i &lt; inputs.length; ++i) {
          var input = inputs[i];
          if(input.name.indexOf('signup') &gt;= 0) {
            if(type == 'text_field') {
              textValidation(input);
            } else {
              dropdownValidation(field, input);
            }
          }
        }
        htmlEmbedDropdownValidation(field);
      }
    
      function textValidation(input) {
        if(input.id == 'signup_email') return;
        var val = input.value;
        if(val == '') {
          textFieldError(input);
          isValid = false;
          return;
        } else {
          removeTextFieldError(input)
        }
      }
    
      function dropdownValidation(field, input) {
        var val = input.value;
        if(val == '') {
          if(field.className.indexOf('invalid') == -1) field.className += ' invalid';
          onSelectCallback(input);
          isValid = false;
          return;
        } else {
          field.className = field.className.replace(/ invalid/g, '');
        }
      }
    
      function htmlEmbedDropdownValidation(field) {
        var dropdowns = field.querySelectorAll('.mimi_html_dropdown');
        for(var i = 0; i &lt; dropdowns.length; ++i) {
          var dropdown = dropdowns[i],
              val      = dropdown.value;
          if(val == '') {
            if(field.className.indexOf('invalid') == -1) field.className += ' invalid';
            isValid = false;
            dropdown.onchange = validate;
            return;
          } else {
            field.className = field.className.replace(/ invalid/g, '');
          }
        }
      }
    
      function textFieldError(input) {
        input.className   = 'required invalid';
        input.placeholder = input.getAttribute('data-required-field');
      }
    
      function removeTextFieldError(input) {
        input.className   = 'required';
        input.placeholder = '';
      }
    
      function onSelectCallback(input) {
        if(typeof Widget != 'undefined' &amp;&amp; Widget.BasicDropdown != undefined) {
          var dropdownEl = input.parentNode,
              instances  = Widget.BasicDropdown.instances;
          for(var i = 0; i &lt; instances.length; ++i) {
            var instance = instances[i];
            if(instance.wrapperEl == dropdownEl) {
              instance.onSelect = validate;
            }
          }
        }
      }
    
      function updateFormAfterValidation() {
        form.className   = isValid ? '' : 'mimi_invalid';
        submit.value     = submitButtonText();
        submit.disabled  = !isValid;
        submit.className = isValid ? 'submit' : 'disabled';
      }
    
      function submitButtonText() {
        var invalidFields = document.querySelectorAll('.invalid'),
            text;
        if(isValid || invalidFields == undefined) {
          text = submit.getAttribute('data-default-text');
        } else {
          if(invalidFields.length &gt; 1 || invalidFields[0].className.indexOf('checkgroup') == -1) {
            text = submit.getAttribute('data-invalid-text');
          } else {
            text = submit.getAttribute('data-choose-list');
          }
        }
        return text;
      }
    
      function revalidateOnChange() {
        var fields = form.querySelectorAll(".mimi_field.required");
        for(var i = 0; i &lt; fields.length; ++i) {
          var inputs = fields[i].getElementsByTagName('input');
          for(var j = 0; j &lt; inputs.length; ++j) {
            inputs[j].onchange = validate;
          }
        }
      }
    })();

        <h1 class="entry-title" itemprop="headline">
         Graduation Party Ideas For All Ages
        </h1>
        <p class="entry-meta">
         <time class="entry-time" datetime="2014-05-14T22:59:30+00:00" itemprop="datePublished">
          May 14, 2014
         </time>
         By
         <span class="entry-author" itemprop="author" itemscope="itemscope" itemtype="http://schema.org/Person">
          <a class="entry-author-link" href="http://www.momsandmunchkins.ca/author/delaine-7810/" itemprop="url" rel="author">
           <span class="entry-author-name" itemprop="name">
            Cheryl
           </span>
          </a>
         </span>
         <span class="entry-comments-link">
          <a href="http://www.momsandmunchkins.ca/2014/05/14/grad-party/#respond">
           Leave a Comment
          </a>
         </span>
        </p>
       </header>
       <div class="entry-content" itemprop="text">
        <p>
         One of the things I love about this collection of Grad Party ideas is that they can be used for all sorts of graduation parties  preschool, kindergarten, elementary school or high school. Im sure that college grad students would even appreciate some of these delicious treats!
        </p>
        <p>
         This is a collection of ideas from other talented bloggers and party planners. To view the full details on any of these ideas, simply click on the title of the party and it will open up in a new window for you.
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-celebrations-at-home.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22334 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-celebrations-at-home.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-celebrations-at-home.jpg" width="481"/>
         </a>
        </p>
        <h2>
         Grad Party Ideas:
        </h2>
        <p>
         These
         <a href="http://celebrationsathomeblog.com/2014/03/diy-graduation-cap-bottle-toppers.html" target="_blank" title="Graduation Cap Bottle Toppers">
          graduation cap bottle toppers
         </a>
         <em>
          (pictured above)
         </em>
         by
         <strong>
          Celebrations At Home
         </strong>
         are simply adorable! Chris shares an easy tutorial for how to make your own.
        </p>
        <p>
         There are so many amazing ideas in this
         <a href="http://www.paisleypetalevents.com/2011/06/amazing-graduation-dessert-table/" target="_blank" title="Graduation party">
          graduation party
         </a>
         by
         <strong>
          Paisley Petal Events
         </strong>
         and I absolutely love these Oreo Cheesecake Push Pops with graduation caps!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-paisley-petal-events.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22340 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-paisley-petal-events.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="424" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-paisley-petal-events.jpg" width="600"/>
         </a>
        </p>
        <p>
         These
         <a href="http://annaandblue.blogspot.ca/2011/05/tutorial-graduation-cupcakes.html" target="_blank" title="Graduation Cupcakes">
          graduation cap cupcakes with an edible diploma
         </a>
         by
         <strong>
          Anna and Blue Paperie
         </strong>
         are incredibly adorable! She walks you through the steps of making your own. Be sure to make extras for taste testing!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cupcakes-anna-and-blue.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22336 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cupcakes-anna-and-blue.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cupcakes-anna-and-blue.jpg" width="401"/>
         </a>
        </p>
        <p>
         These
         <a href="http://pubx.ch/1wauanr" target="_blank" title="Grab Cap Candy Pops">
          graduation cap candy pops
         </a>
         by
         <strong>
          Skip To My Lou
         </strong>
         look pretty easy to create and they would make great centerpieces on your party tables.
        </p>
        <p>
         <strong>
          <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-skip-to-my-lou-candy-graduation-caps.jpg">
           <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22343 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-skip-to-my-lou-candy-graduation-caps.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-skip-to-my-lou-candy-graduation-caps.jpg" width="529"/>
          </a>
         </strong>
        </p>
        <p>
         This
         <a href="http://partypatisserie.com/2014/05/black-white-and-gold-graduation-party/" target="_blank" title="Gold Graduation party">
          Black, White and Gold Graduation Party
         </a>
         by
         <strong>
          Party Patisserie
         </strong>
         is filled with creative ideas. I love the fortune cookies  so perfect for a graduation party!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fortune-cookies-party-patisserie.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22337 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fortune-cookies-party-patisserie.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="400" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fortune-cookies-party-patisserie.jpg" width="600"/>
         </a>
        </p>
        <p>
         The incredible Dawn of
         <strong>
          Not Just A Mommy
         </strong>
         put together this fun
         <a href="http://dawnypoo.blogspot.ca/2014/05/headed-for-bright-future-graduation-party.html#.U3O__HZo-qA" target="_blank" title="Neon Graduation Party">
          Neon Graduation Party
         </a>
         with special touches like shades for a bright future, bright candies and more.
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-bright-future.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22333 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-bright-future.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="286" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-bright-future.jpg" width="600"/>
         </a>
        </p>
        <p>
         There are so many beautiful details in this
         <a href="http://rialeephotography.com/grad-party-details/" target="_blank" title="Graduation Party">
          Graduation Party
         </a>
         by
         <strong>
          Rialee Photography
         </strong>
         . I love how many sweet &amp; personal touches were added.
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-rialee.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22342 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-rialee.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="428" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-rialee.jpg" width="600"/>
         </a>
        </p>
        <p>
         These
         <a href="http://pubx.ch/1x0i4IH" target="_blank" title="Candy Filled Grad Caps">
          candy filled graduation cap cookies
         </a>
         by
         <strong>
          Hungry Happenings
         </strong>
         look delicious! M&amp;Ms are my weakness! Ive always wanted to try making filled cookies or a candy-filled cake. Perhaps Ill try it this summer!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cookies-hungry-happenings.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22335 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cookies-hungry-happenings.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-cookies-hungry-happenings.jpg" width="600"/>
         </a>
        </p>
        <p>
         My mom is obsessed with sayings and inspirational quotes so I know shed love this
         <a href="http://www.lemonsqueezyhome.com/2013/05/words-to-live-by-quote-board-diy-with.html" target="_blank" title="Words To Live By Board">
          Words To Live By
         </a>
         board by
         <strong>
          A Lemon Squeezy Home
         </strong>
         . It would be such a great piece for a graduation party!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-lemon-squeezy.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22339 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-lemon-squeezy.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-lemon-squeezy.jpg" width="426"/>
         </a>
        </p>
        <p>
         These
         <a href="http://pagingsupermom.com/2013/05/mini-diploma-treats-free-printable/" target="_blank" title="Mini Diploma Sweets">
          mini diploma sweets
         </a>
         by
         <strong>
          Paging Supermom
         </strong>
         are so easy to put together  especially with the collection of free treat bag toppers!
        </p>
        <p>
         <a href="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fruit-rollup-diploma.jpg">
          <img alt="Grad Party Ideas" class="aligncenter size-full wp-image-22338 pib-hover-img" data-pin-media="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fruit-rollup-diploma.jpg" data-pin-url="http://www.momsandmunchkins.ca/2014/05/14/grad-party/" height="600" src="http://www.momsandmunchkins.ca/wp-content/uploads/2014/05/graduation-party-fruit-rollup-diploma.jpg" width="431"/>

          graduation wall decorations
         </a>
         here by
         <strong>
          Pear Tree Greetings
