  <script>
   function processUrl(url){	
            url = url.replace(/\/\*.*\*\//, '');
			url = url.replace("$","");	
			url = url.replace("&amp;","&amp;"); 		
			url = url.replace(/ /g,'');		
			url = url.replace(/(&nbsp;)*/g,'');

            var httpNeeded = (url.search(/(http:\/\/|https:\/\/)/i) !== 0);
            if(httpNeeded){
                url = 'http://' + url;
            }

			return url;	
		}
		
		$(function(){
			$('#howitworks').click(function(){
				
				if ($('#demand-mask').hasClass('maskoff-how')){

					$('.how_container').addClass('off');
					$('#flub_main').removeClass('off');
					
					$('#demand-mask').removeClass('maskoff-how');
					$('#demand-footer').removeClass('maskoff');
					
					$('#demand-mask').addClass('maskon');
					$('#howitworks').removeClass('selected');
				}else if($('#demand-mask').hasClass('maskon')){

					$('#flub_main').addClass('off');
					$('.how_container').removeClass('off');

					$('#demand-mask').removeClass('maskon');
					$('#demand-mask').addClass('maskoff-how');
					$('#demand-footer').addClass('maskoff');
					$('#howitworks').addClass('selected');
				}
			
			});

				$('#howitworks').click();

            function openMask(option){
                if($('#demand-mask').hasClass('maskoff-how')){
                    $('#demand-mask').removeClass('maskoff-how');
                    $('.how_container').addClass('off');
                    $('#howitworks').removeClass('selected');
                    $('#demand-mask').addClass('maskoff');
                }else{
                    $('#demand-mask').addClass('maskoff');
                    $('#demand-footer').addClass('maskoff');
                }
                $('#demand-mask').removeClass('maskon');
                $('#flub_main').removeClass('off');

                switch (option){
                    case 'demand':
                        $('#demand_creation').removeClass('off');
                    break;
                    case 'ioffer':
                        $('#ioffer_creation').removeClass('off');
                    break;
                }
            }

            function closeMask(){
                if($('#demand-mask').hasClass('maskoff-how')){
                    $('#howitworks').click();
                }else if($('#demand-mask').hasClass('maskoff')){
                    _gaq.push(['_trackPageview', '/Home']);
                    $('#demand-footer').removeClass('maskoff');
                    $('#demand-mask').removeClass('maskoff');
                    $('#demand-mask').addClass('maskon');
                    $('#demand_creation').addClass('off');
                    $('#ioffer_creation').addClass('off');

                }else if($('#demand-mask').hasClass('maskon')){

                    window.location = '/My-Demands';

                }
            }

            $('#btn-slide, button[tag=ButtonSlide]').click(function(){
                closeMask();
            });
			
			/**
			* demand create start
			* - validate the given demand url
			* - validate url against db website validattions
			* - check if the demand is already exist with this url with-in last 30 minutes
			*
			* - if all goes fine it will show the nex page otherwise show the errors
			*/

            function demandUrl(params){

                var product_title = $('#url').val();

				call_ajax(
					'check_demand_url',
					params,
					function success(response, content){
						if(content.sess.product_title != ''){
							product_title = content.sess.product_title;
						}
						if($('.product_attributes').hasClass('off') == false){
							$('.product_attributes').addClass('off');
						}
						$('.product_attributes .product_attributes_items').html('');
						$.each(content.product_attributes, function(index, attributes){							
							var att_element_id = index.replace(' ', '_');
							var innerhtml = '&lt;div class="row prepend-40 attributes"&gt;';
							innerhtml +=	'&lt;span&gt;'+index+'&lt;/span&gt;&lt;select id="'+att_element_id+'" name="demand_attributes['+index+']" class="form-control input-lg"&gt;';
							var attr_name 	= index;
							var attr_value	= content.sess.demand_attributes[attr_name];
							$.each(attributes, function(index, value){
									var attr_selected = '';
									if(attr_value == value){
										attr_selected = ' selected="selected" ';
									}
								innerhtml += 		'&lt;option value="'+value+'" '+attr_selected+' &gt;'+value+'&lt;/option&gt;';
							});
							innerhtml += 	'&lt;/select&gt;';
							innerhtml += '&lt;/div&gt;';
							$('.product_attributes').removeClass('off');
							$('.product_attributes .product_attributes_items').append(innerhtml);
							innerhtml = ''; // resetting the variable
							
						});			
						//if it is successfull
                        openMask('demand');
		
						$('#demand_creation #url_step_1 #url_value').val(product_title);
						//sending pageview to google analytics						
						/*ga('send', 'pageview', {
						  'page': '/Create-Demand/Quantity',
						  'title': 'About the product'
						});*/
						_gaq.push(['_trackPageview', '/Create-Demand/Quantity']);
						
					},
					function error(response, content){
						if(content.next_step == 'hard_stop'){
							//open hard stop page							
						}else if(content.next_step == 'demand_stop'){
							//open demand stop page
						}
						show_site_messages(response.success, response.messages);
					}		
				);
				return false;
			}


            $('#demand-mask form').submit(function(e){
                var url = processUrl($('#url').val());
                $('#url').val(url);
                var params = $(this).serialize();
                demandUrl(params);
            });


            $('button[tag=Ajax_IofferStartButton]').click(function(){

                var iOffer = new InstantOffer();
                var spinner = new Spinner();
                var url = processUrl($('#url').val());
                $('#url').val(url);


                function iOfferView(view){
                    if(view === 'questions'){
                        $('#ioffer_status_questions .questions_container').html('');
                    }
                    $('#ioffer_creation &gt; div').addClass('off');
                    $('#ioffer_creation &gt; #ioffer_status_'+ view).removeClass('off');
                }

                function iOfferSendAnswer(form, form_values){
                    iOffer.sendAnswer(form_values.name, form_values.value).then(function(questions_available){
                        show_site_messages('success', 'Thanks for your response.');
                        $(form).parent().remove();
                        if(!questions_available){
                            $('#ioffer_creation &gt; div').addClass('off');
                            $('#ioffer_creation &gt; #ioffer_status_processing').removeClass('off');
                            iOfferGetOfferStatus();
                            spinner.updateIntervalTime(iOffer.getTimeLeft());
                            spinner.play();
                        }
                    });
                }

                function iOfferGetOfferStatus(){
                    iOffer.getOfferStatus().then(function(status, content){
                        switch (status){
                            case 'rejected_prohibited':
                                iOfferView('prohibited');
                                $('#url').val('');
                                break;
                            case 'rejected_foundbestprice':
                                iOfferView('rejected');
                                $('#url').val('');
                                break;
                            case 'rejected_cannotdetermine':
                                iOfferView('rejected');
                                $('#url').val('');
                                break;
                            case 'rejected_outofstock':
                                iOfferView('rejected');
                                $('#url').val('');
                                break;
                            case 'rejected_notautomated':
                                iOfferView('notautomated');
                                $('#btn_creation_ioffer_or_demand').attr('data-url', url);
                                break;
                            case 'rejected_rejected':
                                iOfferView('rejected');
                                $('#url').val('');
                                break;
                            case 'questions':
                                iOfferView('questions');
                                for(var i in content){
                                    var question_html = iOffer.getQuestionHtml(content[i]);
                                    $('#ioffer_status_questions .questions_container').append('&lt;div&gt;'+ question_html +'&lt;/div&gt;');
                                }
                                $('form.submit_answer').submit(function(event){
                                    event.preventDefault();
                                    var form_values = $(this).serializeArray()[0];
                                    if(typeof form_values !== 'undefined'){
                                        iOfferSendAnswer($(this), form_values);
                                    }
                                });
                                spinner.pause();
                                break;
                            case 'created':
                                location.href = '/io/' + content;
                                break;
                            case 'timeout':
                                iOfferView('timeout');
                                break;
                            default:
                                iOfferView('timeout');
                                break;
                        }
                    });
                }


                iOffer.createNew(url).then(function(){
                    openMask('ioffer');
                    iOfferGetOfferStatus();
                    spinner.play();
                });

                $('#btn_creation_ioffer_or_demand').click(function(){
                    var product_url = $(this).attr('data-url');
                    var req = $.ajax({
                        global: true,
                        url: '/ajax/check_demand_url',
                        type: "POST",
                        data: {
                            form_post: 'post',
                            current_step: 'start1',
                            demand_url: product_url
                        }
                    });
                    req.done(function(){
                        location.href = '/Create-Demand/Quantity';
                    });
                });


                function iofferDestroy(){
                    iOffer.destroy();
                    spinner.reset();
                    $('#btn_creation_ioffer_or_demand').off('click');
                    $('.ioffer_stop').off('click', iofferDestroy);
                    if($(this).attr('id') !== 'btn-slide'){
                        closeMask();
                    }
                    $('#ioffer_creation &gt; div').addClass('off');
                    $('#ioffer_creation &gt; #ioffer_status_processing').removeClass('off');
                }
                $('#btn-slide, button[tag=ButtonSlide]').addClass('ioffer_stop');
                $('.ioffer_stop').on('click', iofferDestroy);

            });
		});
