	// makes a comment appear when someone clicks "post a comment" or "reply"
	// id - the commentId (0 if "root"), dId - diaryId
	function makeCommentAppear(id, dId) {

		var tempTitle = "";
		var tempText = "";
	
		// if this isn't empty, we were trying to perviously post a comment
		// therefore, we need to clean out all the html in the previous spot
		// so we don't create "nodeLists" for postCommentTextArea and Title
		if (commentId != "") {
			// get the current "postcommenter" element
			var oldElem = document.getElementById('postcomment' + commentId);
	
			if (oldElem) {
				// this will store whatever we might have had before
				if (document.forms['rateForm'].elements['postCommentTitle']) {
					tempTitle = document.forms['rateForm'].elements['postCommentTitle'].value;
				}	
				if (document.forms['rateForm'].elements['postCommentTextArea']) {
					tempText = document.forms['rateForm'].elements['postCommentTextArea'].value;
				}
			
				// clear out the old HTML, so we don't create nodeLists
				oldElem.innerHTML = "";
			}
		}
	
		hideArea("post");

		// store the values globally for what we're currently looking at
	    commentId = id;
	    diaryId = dId;
	    
	    var elementName = 'postcomment' + id;
	    
	    // make sure the element is hidden
		Element.hide(elementName);

		// set the data in it		
	    document.getElementById(elementName).innerHTML = document.getElementById("contentHolder").innerHTML;
		document.forms['rateForm'].elements['postCommentTitle'].value = tempTitle;
		document.forms['rateForm'].elements['postCommentTextArea'].value = tempText;
	
		// make it slide down, take half a second, scaleFrom 3 to avoid "stutter" in IE
		Effect.SlideDown(elementName, { duration: 0.5, scaleFrom: 3});
		window.setTimeout("document.forms['rateForm'].elements['postCommentTitle'].focus()", 501);
		
	    postingComment = true;
	    // since we're starting a new preview cycle, we're going to need
	    // to use the slide down effect.  This ensures that
	    previewingComment = false;

	    var objDiv = document.getElementById(elementName);
		objDiv.scrollTop = objDiv.scrollHeight;
	}

	function cancelPost() {

		var cTextElem = document.forms['rateForm'].elements['postCommentTextArea'];
		var cTitleElem = document.forms['rateForm'].elements['postCommentTitle'];
		
		if ((cTextElem &amp;&amp; cTextElem.value != "")

		{
			// if the don't want to cancel--stop.
			if (!confirm("Cancel this comment?")) {
				return;
			}
		}

		hideArea("cancel");
		// we no longer clear this out so in makeCommentAppear we can see if we have to clean
		// out some hidden HTML
		//commentId = "";
		diaryId = "";
		previewText = "";
		previewTitle = "";
		cTextElem.value = "";
		cTitleElem.value = "";
		previewingComment = false;	
		postingComment = false;

	// trims a string
	function trim(str) {     
		if(!str || typeof str != 'string') {
			return null;     
		}
			
		return str.replace(/^[\s]+/,'').replace(/[\s]+$/,'').replace(/[\s]{2,}/,' '); 
		
	}

	function previewComment(theForm) {
	
		// if (tinyMCE) {
			// tinyMCE.triggerSave();
		// }
		
		// alert(baseUrl);

		var url = baseUrl + "ajaxPreviewComment.do";
		var errorDiv = "errorcomment" + commentId;
		
		// hide any error we might have
		if (hasError) {
			Element.hide(errorDiv);
		}
			
		var cTitleElem = document.forms['rateForm'].elements["postCommentTitle"];
		var cTextElem =  document.forms['rateForm'].elements["postCommentTextArea"];
	
		
		previewTitle = cTitleElem.value;
		previewText = cTextElem.value;
	
	    // trim it so we remove the white sapces
	    previewTitle = trim(previewTitle);

	    // if we have nothing, set the error
	    if (previewTitle == null || previewTitle == "") {
			Effect.Appear(errorDiv, { duration: 0.5 });
			disallowPost();
			hasError = true;
	    	return;
	    }

	    // if we get this far we don't have an error!
	    hasError = false;
	    
	    // apparently using "escape" funcation messes with utf-8 so we just explicity
	    // replace newlines with their url-encoding equavilent.
	    previewText = escape(previewText); 
	    
	    previewText = previewText.replace(/\%u201C/ig, "%22");
	    previewText = previewText.replace(/\%u201D/ig, "%22");
  	    previewText = previewText.replace(/\%u2018/ig, "%27");
  	    previewText = previewText.replace(/\%u2019/ig, "%27");  	    
  	    previewText = previewText.replace(/\%u2013/ig, "-");  
  	    previewText = previewText.replace(/\%u2014/ig, "-");    	      	    
  	    previewText = previewText.replace(/\%u2026/ig, "...");    	
  	    previewText = previewText.replace(/\+/g, "%2B");
	    
	    //previewText.replace(/\n/g, "%0a");
//	    previewText = previewText.replace(/\+/g, "%2b");
  // 	    previewText = previewText.replace(/\%/g, "%25");
//   	    alert(previewText);
	    
		// add the previewText to the URL querystring    
		// have to URL encode it for it to work correctly
	    var params = "t=" + previewText;
	
		// do the fun ajax!    
	    retrieveURL(url, "PREVIEW", commentId, params);
	}

			hasError = false;
		}

	        // if we are canceling, we do it pretty

		    	// otherwise we are posting another comment
		    	// so do it REAL quick
				Element.hide(previewName);	    	
				document.getElementById(elementName).innerHTML = "";
			}
	    }
	}

  function retrieveURL(url, type, id, params) {
    which = id;
    if (window.XMLHttpRequest) { // Non-IE browsers
        req = new XMLHttpRequest();

      
        // pretty sure this adds a listener, which handles communication
        // listeners apparently don't like parameters, either
        if ("RATE" == type) {
            req.onreadystatechange = processRateStateChange;          
        }
        else if ("RECOMMEND" == type) {
            req.onreadystatechange = processRecommendStateChange;          
        } 
        else if ("PREVIEW" == type) {
			req.onreadystatechange = processPreviewComment;                  
        }      
        else {
            req.onreadystatechange = processStateChange;
        }

        try {
            req.open("POST", url, true);
	   		req.setRequestHeader("Content-type", "application/x-www-form-urlencoded; charset=UTF-8");
			req.setRequestHeader("Content-length", params.length);
			req.setRequestHeader("Connection", "close");            	    	
            req.send(params);
        } catch (e) {
            alert(e);
        }
    } 
    else if (window.ActiveXObject) { // IE
        req = new ActiveXObject("Microsoft.XMLHTTP");
		
        if (req) {
            if ("RATE" == type) {
                req.onreadystatechange = processRateStateChange;          
            }
            else if ("RECOMMEND" == type) {
                req.onreadystatechange = processRecommendStateChange;          
            }       
            else if ("PREVIEW" == type) {
				req.onreadystatechange = processPreviewComment;                  
	   		}   
            else {
                req.onreadystatechange = processStateChange;
            }
        
            req.open("POST", url, true);
	   		req.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
			req.setRequestHeader("Content-length", params.length);
			//req.setRequestHeader("Connection", "close");            	    	
            req.send(params);
        }
    }
  }
  

  // listener
  function processStateChange() {
    
    if (req.readyState == 4) { // Complete
      if (req.status == 200) { // OK response
        document.getElementById("theTable").innerHTML = req.responseText;

      } else {

        alert("Problem: " + req.statusText);

      }

    }
  }
  
  
  
	function processPreviewComment() {

		if (req.readyState == 4) { // Complete
			if (req.status == 200) { // OK response
				var postCommentDiv = document.getElementById("postcomment" + commentId);
				var previewCommentDiv = document.getElementById("previewcomment" + commentId);
				
			  	// make it visible
			    // document.getElementById("previewComment").style.display = "block";
			    
				previewText = req.responseText;
				
				var commentError = false;
				
				if (previewText.indexOf("@@@ERROR@@@") != -1) {
					previewText = "&lt;div class='soapbloxError'&gt;" + previewText.substring(11, previewText.length) + "&lt;/div&gt;";						
					commentError = true;
				}
    
		        // set values in the preview area
			    document.getElementById("previewTitle" + commentId).innerHTML = previewTitle;
			    document.getElementById("previewCommentText" + commentId).innerHTML = previewText;
			    
			    // alert(previewingComment); 
    
    			// if we're not already previewing a comment we slide down
    			if (!previewingComment) {
					Effect.SlideDown("previewcomment" + commentId, { duration: 0.5, scaleFrom: 3});
					//window.setTimeout("postButton.focus()", 501);
				}
				// otherwise, we just update the text
				else {
					Element.show("previewcomment" + commentId);
				}
			
				if (commentError) {
					disallowPost();
				}					
				else {
					allowPost();
				}
				
				previewingComment = true;
			}
			else {
				alert("Problem: " + req.statusText);
			}
		}	
	}

	function addTags(textAreaId, startTag, endTag, content) {
		
		var textArea;
		
		if (document.forms['rateForm']) {
			textArea = document.forms['rateForm'].elements[textAreaId];
		}
		else {
			textArea = document.getElementById(textAreaId);
		}
		
		if (typeof textArea.selectionStart != "undefined") {
			textArea.focus();
			var selStart = textArea.selectionStart;
			var selEnd = textArea.selectionEnd;
			var selText = textArea.value.substring(selStart, selEnd);
			var scrlTop = textArea.scrollTop;
			var scrlDown = textArea.value.length &lt;= selStart;
			if (selText.indexOf(startTag) === 0 &amp;&amp; endTag == selText.substring(selText.length - endTag.length, selText.length)) {
				textArea.value = textArea.value.substring(0, selStart) + selText.substring(startTag.length, selText.length - endTag.length) 
					+ textArea.value.substring(selEnd, textArea.value.length);
				textArea.selectionStart = selStart;
				textArea.selectionEnd = selEnd - (startTag.length + endTag.length);
				textArea.scrollTop = scrlTop;
			} else {		
				textArea.value = textArea.value.substring(0, selStart) + startTag + (content ? content : selText)
					+ endTag + textArea.value.substring(selEnd, textArea.value.length);
				textArea.selectionStart = selStart;
				textArea.selectionEnd = selStart + startTag.length + (content ? content.length : selText.length) + endTag.length;
				textArea.scrollTop = (scrlDown ? 10000000 : scrlTop);
			}
		} else if (document.selection) {
			var range = document.selection.createRange();
			if (range.parentElement() != textArea) {
				var textRange = textArea.createTextRange();
				textRange.moveStart('character', textArea.value.length);
				textRange.collapse();
				textRange.select();
				range = document.selection.createRange();
			}
			textArea.focus();
			if (range.text.indexOf(startTag) === 0 &amp;&amp; endTag == range.text.substring(range.text.length - endTag.length, range.text.length)) {
				range.text = range.text.substring(startTag.length, range.text.length - endTag.length);
			} else {
				range.text = startTag + (content ? content : range.text) + endTag;
			}
		} else {
			textArea.focus();
			textArea.value = startTag + (content ? content : "") + endTag;
			if (typeof textArea.scrollTop != "undefined") {
				textArea.scrollTop = 10000000;
			}
		}
	}

           <a class="diaryTitle" href="/diary/18733/the-false-pragmatism-of-a-pragmatic-compromise-with-failure">
            The false pragmatism of a "pragmatic compromise" with failure
           </a>
          </h1>
          <h2 class="author">
           by:
           <a href="/user/Paul Rosenberg">
            Paul Rosenberg
           </a>
          </h2>
          <h3 class="diaryTimestamp">
           <i>
            Sun May 16, 2010 at 14:00
           </i>
          </h3>
          <br/>
         </td>
        </tr>
        <tr>
         <td>
          <script language="Javascript" src="http://openleft.com/upload/js/sharelinks.js" type="text/javascript">
          </script>
          <script language="Javascript" type="text/javascript">
           show_socialmedia_links('The false pragmatism of a &quot;pragmatic compromise&quot; with failure', 'http://openleft.com/showDiary.do?diaryId=18733', true);

          How do you make a "pragmatic" compromise with failure? Pragmatism, after all, is concerned with what works. And failure, by definition, is what doesn't work. You can no more make a pragmatic compromise with failure than you can tell an honest lie. Yet, the idea of a pragmatic compromise with failure is the very essence of the Obama presidency--except when compromise blends imperceptibly into full-scale mimicry.
          <p>
           Bush's war on terrorism was an utter failure, killing hundreds of thousands of innocent Iraqis, Afghans and others for no good reason, just like Osama bin Laden said that America always wanted to-and thereby recruiting more potential terrorists than bin Laden ever dreamed of. Obama's compromise was to kill fewer Iraqis and more Afghans, plus assorted other dark skinned people from Pakistan to Yemen to Somalia. Bush's response to global warming was to ignore it and give oil companies everything they want. Obama's compromise was to pay attention to global warming and
           <i>
            still
           </i>
           give oil companies everything they want. Bush's attitude towards civil liberties was to trample them underfoot gleefully. Obama's compromise is to cut back on the glee, pretending to be serious, even troubled. Bush's response to the Wall Street meltdown was tons of free cash given away by the former head of Goldman Sachs. Obama's response to the Wall Street meltdown was tons of free cash given away by the former head of the New York Fed. Bush's approach to education was to install a fellow Texan with a much-hyped, but ultimately failed educational record running Houston's school system as Secretary of Education, and instituting a narrow, test-driven plan deceptively labeled "No Child Left Behind." Obama's approach to education is to install a fellow Chicagoan (a basketball buddy, in fact), with a much-hyped, but ultimately failed educational record running Chicago's school system as Secretary of Education, and instituting a narrow, test-driven plan deceptively labeled "Race To The Top."
          </p>
          <p>
           It's no accident that conservative policies fail. They're
           <i>
            meant to fail
           </i>
           ....
          </p>
         </td>
        </tr>
        <tr>
         <td class="theFlip">
          <a href="/userDiary.do;jsessionid=C3C52379936176CD569D59F2B61558F4?personId=470">
           Paul Rosenberg
          </a>
          ::
          <a href="/showDiary.do;jsessionid=C3C52379936176CD569D59F2B61558F4?diaryId=18733">
           The false pragmatism of a "pragmatic compromise" with failure
          </a>
         </td>
        </tr>
        <tr>
         <td>
          Conservatism is a philosophy of elite rule, of tradition and privilege, respect for authority, doing what your betters tell you-a philosophy, in short, of enshrining yesterday's answers without even bothering to find out what today's questions are-never mind tomorrow's. Conservatism values things like loyalty-i.e sucking up to the boss, even when he's tragically wrong-and despises things like empathy-i.e. giving a damn about anyone who's
          <i>
           not
          </i>
          above you in the hierarchy of things.
          <p>
           So long as things don't change too much, too fast, it can almost look like it works. Sure, you keep on making the same old mistakes over and over and over again. Failures and successes are repeated endlessly with equal vigor. But as the old Les McCan and Eddie Harris song put it so well, compared to what? There
           <i>
            was
           </i>
           nothing better to compare to conservatism's blind repetition of failure and success. The catastrophes that came from conservatism's failures were just part of life, a test from God for the good, punishment for sin for the not-so-good.
          </p>
          <p>
           This is not to say that better ideas didn't come along. Of course they did. But the whole
           <i>
            idea
           </i>
           of better ideas-which would have made for a real alternative to conservatism,
           <i>
            that
           </i>
           did not exist, at least as far as most people were concerned. There had been a time when the idea of progress
           <i>
            had
           </i>
           existed. That was back in ancient Greece, when the foundations of Western Civilization were being laid. But the West had gotten over that a
           <i>
            long
           </i>
           time ago... until the Crusades accidentally opened up the door to the West's past, reintroducing its classical roots from the safe-keeping of Islamic scholars, and giving rise to the Renaissance, and the birth of modern humanism. In time, this would be followed by the Reformation and the Enlightenment, three profound social transformations that infused the West with new visions of possibility-not to mention a
           <i>
            method
           </i>
           of continuing to create new possibilities, a method known as "science."
          </p>
          <p>
           The dynamism of the modern world made conservatism's failures strikingly evident, as never before. The generation of new ideas made the idea of alternatives inevitable to some, and thus grew the basis for comparisons that could find conservatism wanting. What's more, the pace of change was such that even things that had previously worked no longer necessarily did: feudalism, for example.
          </p>
          <p>
           America was founded on the idea of progress. It combined the philosophy of political liberalism, expounded by the likes of Hume and Locke, with older examples of republican government restructured to work on a scale that previous republics had never imagined. It was a wholesale repudiation of the conservative notion of the "divine right of kings". Indeed, it made no mention of God, and only spoke of religion to ensure that it was entirely separate from government.
          </p>
          <p>
           Over time, of course, America developed its own entrenched elites, and after the Southern elites nearly destroyed the country with their treasonous war of aggression to preserve slavery, its newly unified Northern elites consolidated power as never before. They made their money from technologically advanced industries-railroads, steel mills, oil wells, and eventually automobiles-and the association gave them the deceptive aura of being scientifically advanced as well, as if they knew exactly what they were doing... just as the pre-modern elites could claim, on the basis of being chosen by God. They hit a few bumps along the way, but after WWI, they appeared triumphant, not just over every aspect of American political life, but as the new contenders for world power, as Europe lay shattered. But then, of course, the Great Depression arrived, and showed that their mastery was all a sham. They were so thoroughly discredited that it would be another 70 years before the conservative elites got full control of things again, and within a few short years they produced
           <i>
            another
           </i>
           global financial disaster.
          </p>
          <p>
           At this point, the pragmatic thing to do would be to get rid of them once and for all. They have proven themselves incapable of the very things they promise above all: creating stability, prosperity, and just rewards for diligent hard work. Indeed, they are anything
           <i>
            but
           </i>
           "conservative" in the most appealing sense of the word, the sense they have used to gain the trust of countless millions throughout the ages. They are reckless gamblers whose only interest in conserving anything is their own iron grip on wealth and power. For everyone but themselves, everything they touch turns to ash. Again: the pragmatic thing to do would be to get rid of them once and for all-drive them from power, and consign their failed ideas to the history books. Instead, we've got a President who says that
           <i>
            nothing
           </i>
           is more important than compromising with their abysmal record of failure and deceit.
          </p>
          <p>
           What could
           <i>
            possibly
           </i>
           be more foolish than this? It is a path that's utterly
           <i>
            guaranteed
           </i>
           to prevent us from solving
           <i>
            any
           </i>
           of the problems that now confront us. Indeed, the notion of
           <i>
            pre-
           </i>
           compromising with with failure ensures that we won't even have the
           <i>
            briefest
           </i>
           of discussions of ideas that might actually work. In nominating Elena Kagen to the Supreme Court, Obama praised her for "her openness to a broad array of viewpoints; her habit, to borrow a phrase from Justice Stevens, 'of understanding before disagreeing'; her fair-mindedness and skill as a consensus-builder." There's no doubt this is how he sees himself as well. But what increasingly bothers progressive critics is the flip side of that: the question of whether he has
           <i>
            any clue at all
           </i>
           about understanding those whom he supposedly
           <i>
            agrees
           </i>
           with.
          </p>
          <p>
           My answer to this, quite frankly, is "No!" It's not that he doesn't understand the nature of liberal/progressive thinking--although that's certainly true. The problem is much deeper than that: Just like Dan Quayle, he doesn't understand some of the
           <i>
            most basic facts
           </i>
           in the various major issue areas. He does not understand that a war response to terrorism is like fighting fire with gasoline. He does not understand that massive deficit spending is what finally ended the Great Depression, and that "fiscal responsibility" is what caused the Recession of 1937/38, which considerably prolonged the agony of the Great Depression. He does not understand that global warming is a
           <i>
            physical process
           </i>
           that can't be negotiated away in the Senate. He does not understand the difference between free markets and monopoly capitalism. He does not understand that Glass-Steagall
           <i>
            worked
           </i>
           , and that it was repealed
           <i>
            because
           </i>
           it worked.
          </p>
          <p>
           The list goes on and on. It would be one thing if Obama understood these basic facts, and yet felt that political compromises were necessary to accomplish immediate political goals. In that case, he would stand up and argue the facts for the best possible policy before making the necessary compromises in the end. That's what Ronald Reagan did over and over again--despite the fact that his "facts" were all imaginary.
          </p>
          <p>
           The bottom line here is simple: There's nothing pragmatic about being factually ignorant.  There's nothing pragmatic about getting things done when the things you're getting done only make the problems worse.
          </p>

          </b>
          <span id="rating230095">

         <div>
          <br/>
          I wish I could say I disagreed with your assessment of Obama. For a brief moment there, two years ago, I really thought he was going to be different.
          <p>
           But he isn't. This is the biggest wasted opportunity in American history. And the consequences for the nation, and the world, will be dire. What I find mystifying is that Obama is carrying on the narrow corporate agenda as dutifully as Bush was -- and getting nothing but grief from the right for doing so.
          </p>
          <p>
           The question is, was it inevitable? Have narrow, short-term corporate interests corroded American politics to the point where
           <i>
            nobody
           </i>
           could have emerged as an effective leader?
          </p>
          <p>
           <br/>
           <br/>
           <i>
            "We judge ourselves by our ideals; others by their actions. It is a great convenience."  -- Howard Zinn
           </i>
           <br/>
          </p>

          </a>
          <b>
           good piece
          </b>
          <span id="rating230098">
           (
           <a href="/viewRating.do;jsessionid=C3C52379936176CD569D59F2B61558F4?rateCommentId=230098">
            4.00 / 6
           </a>

          </span>

         <div>
          (I don't understand how you manage to put up so many meaty posts.)
          <p>
           This, though, is somewhat debatable:
          </p>
          <p>
          </p>
          <blockquote>
           Just like Dan Quayle, he doesn't understand some of the most basic facts in the various major issue areas. He does not understand that a war response to terrorism is like fighting fire with gasoline. He does not understand that massive deficit spending is what finally ended the Great Depression, and that "fiscal responsibility" is what caused the Recession of 1937/38, which considerably prolonged the agony of the Great Depression. He does not understand that global warming is a physical process that can't be negotiated away in the Senate. He does not understand the difference between free markets and monopoly capitalism. He does not understand that Glass-Steagall worked, and that it was repealed because it worked.
           <br/>
          </blockquote>
          <p>
           Yes, I think he's internalized neoliberalism and subscribes to its basic tenets, which are simply wrong.
          </p>
          <p>
           But I also think that unlike some pols of his ilk, he may understand on an obscure level, some or all of these things, but this understanding is trumped by a variety of impulses: his need to stay popular; his constricted, fear-based sense of what's possible; his belief that he's moving in the right direction; his temperamental conservatism, which fears dramatic change.
          </p>
          <p>
           I've gained what feels like insight into Obama by observing his supporters. Among them are lefties who can quite fluently describe what the country needs before launching into a defense of Obama on the grounds that he's doing what's possible and that what he's doing is inching us toward the end we all seek. They seem quite sincere, and I think Obama, when he's not simply whoring to get re-elected, probably justifies his decisions to himself on similar ground.
          </p>
          <p>
           The limited idea of what's possible is key. Here's a Kevin Baker piece I keep returning to:
          </p>
          <p>
          </p>
          <blockquote>
           Despite such receptions, the president continues to press for "bipartisanship" and elite consensus. One of the most charismatic politicians of his time, a man who was able to raise the most money and draw the biggest crowds in American political history has apparently decided that his new job is to fluff up the generals and bankers and politicians who not very long ago were in panicked disarray. Armchair psychologists from the Maureen Dowd school of political commentary like to analyze this conversion in terms of the elusive personality of Obama himself. Others prefer to dwell on the surprising ineptitude of his administration. And some simply accept his about-face in terms of the political exigencies of an essentially conservative nation, concluding wistfully that Obama is confronted by so many barriers to change-Republican obstructionism, the treachery of this or that Democratic senator, the nature of the Constitution itself-that the country is now ungovernable.
           <p>
            All of which may be true. But it only skims the surface of a greater tidal shift, one that has little to do with Obama himself and in fact has inundated the whole of our democratic process. This shift, which is subtle and has been many years in the making, might best be understood by considering a design underlying many of the interrogation techniques we employ at the (still-unclosed) prison at Guantnamo or at the black sites we still maintain, wherever they are. That is, bringing about the state known as learned helplessness....
           </p>
           <p>
            For decades now, as our public discourse in general has become more scattered, random, and irrational, Republicans-funded by corporate and other elites in the private sector-have stunned Democrats with absurdist attacks that have proved to be effective at garnering votes and, more important in the long term, at hampering Democrats even when they hold the majority. Democrats have been reduced to a state of psychological helplessness, one in which any political obstacles-ranging from the prevarications of stalking horses like Senators Joe Lieberman and Ben Nelson, to the plaintive cries of the tea-baggers out in the streets, to the sterner demands of the Joint Chiefs of Staff or Big Pharma-are transformed into insurmountable organic obstacles.
           </p>
           <p>
            We have learned to be helpless. And in this state of political depression, it no longer matters how many elections liberals win for the Democrats, or how badly Republican, right-wing policies fail or how much damage they do to the country or the world. There is simply no way to do anything differently.
           </p>
          </blockquote>
          <p>
           <a href="http://harpers.org/archive/2010/04/0082894">
            http://harpers.org/archive/201...
           </a>
          </p>
          <p>

          <b>
           Maybe You're Proving My Point
          </b>
          <span id="rating230112">
           (
           <a href="/viewRating.do;jsessionid=C3C52379936176CD569D59F2B61558F4?rateCommentId=230112">
            4.00 / 5

         <div>
          <blockquote>
           I also think that unlike some pols of his ilk, he may understand on an obscure level, some or all of these things, but this understanding is trumped by a variety of impulses...
          </blockquote>
          <p>
           By bringing in Baker's invocation of learned helplessness you yourself trump all those impulse-level explanations. And, indeed, Obama's promise--
           <i>
            as his mass supporters heard it
           </i>
           --was that we could
           <i>
            overcome
           </i>
           that condition. And that is
           <i>
            precisely
           </i>
           where Obama has most dramatically failed.
          </p>
          <p>
           Now, it's possibly true that Obama undestands the things I ticked off "on an obscure level". But I'm sure that Dan Quayle knew how to spell 'potato' "on an obscure level", too. The problem was, he didn't know it when it mattered most.
          </p>
          <p>
           This gets back to my invocations of Kegan's levels of cognitive development, and how one can simulate the understanding of a higher level when not in a condition of conflict. My favorite example is that of level two teenager promising not to break curfew, and seeming to simulate his parents' level three understanding by understanding and respect their point of view. But he can only do that when their point of view doesn't conflict with his own. He's not being morally bad, he just doesn't have the cognitive capacity to do what they think he's doing. He cannot
           <i>
            actually
           </i>
           grasp in that moment what he
           <i>
            appears to
           </i>
           graps.
          </p>
          <p>
           And in like manner, neither can Obama when he articulates the things that most move his progressive and more-or-less reality-based followers.
           <br/>
           <br/>
           <i>
            "You know what they say -- those of us who fail history... doomed to repeat it in summer school." --
            <i>
             Buffy The Vampire Slayer
            </i>
            , Season 6, Episode 3
           </i>
           <br/>
          </p>

          <b>
           Hoover, Obama and learned shackles
          </b>
          <span id="rating230144">
           (
           <a href="/viewRating.do;jsessionid=C3C52379936176CD569D59F2B61558F4?rateCommentId=230144">
            4.00 / 5
           </a>
           )
          </span>
         </div>
         <div>
          No American President was better trained and prepared to handle the emergency of the Depression than Herbert Hoover. Hoover had played a lead role in keeping Europe going through complex relief efforts after World War I. The relief efforts made Hoover, a gifted engineer with no political background,a national hero. But Hoover was constrained by the idea (and the fact) of deficit spending as his revenues shrunk.
          <p>
           Obama may seem prepared but he too is shackled by notions of deficit spending, incompetent government, and the moral and practical "superiority" of private industry particularly the banker class. When he promised the Blue Dogs to bring the economy back to balanced budgets as soon as possible after the stimulus was spent I cringed.
          </p>
          <p>
           FDR was forced into that crap and we are paying the bill now. "War ended the Depression." Nope, Keynesian spending did by 1941 (9.9+% unemployment for the year) and would have at least two years earlier without the efforts of the Blue Dogs of the day. I heard that mantra for decades from my father and his Republican family as I grew up (and heard my mother's Democratic family argue them down).
          </p>
          <p>
           We really need to know what actually works, not what right wing agitators and their looter corporate buddies pontificate.
           <br/>
          </p>

          <b>
           Progress
          </b>
          <span id="rating230101">
           (
           <a href="/viewRating.do;jsessionid=C3C52379936176CD569D59F2B61558F4?rateCommentId=230101">

           </a>

          </span>
         </div>
         <div>
          <blockquote>
           <i>
            America was founded on the idea of progress.
           </i>
           It combined the philosophy of political liberalism, expounded by the likes of Hume and Locke, with older examples of republican government restructured to work on a scale that previous republics had never imagined. It was a wholesale repudiation of the conservative notion of the "divine right of kings". Indeed, it made no mention of God, and only spoke of religion to ensure that it was entirely separate from government.
          </blockquote>
          <p>
           This right here is what kills me about the sort of Framers' worship that tends to dominate so much of our legal discourse. It is so alien to the very foundations of the idea of America. The idea of borrowing what works, even if it comes from Sweden or France, is at the core of the American identity. Or rather, it used to be.
          </p>
          <p>
           The Constitution's authority depends completely on popular assent. It is not based in divine right, nor in the sovereignty of a long dead elite. Law is supposed to serve human ends, no the other way around.
           <br/>
           <br/>
           <i>
            <i>
             Politics is the art of the possible, but that means you have to think about changing what is possible, not that you have to accept it in perpetuity.
            </i>
           </i>
           <br/>
          </p>
