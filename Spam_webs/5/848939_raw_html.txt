(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

              <h3 class="post-title entry-title">
               Weird Crime: Stop...Cuddle Time
              </h3>
              <div class="post-header">
               <div class="post-header-line-1">
               </div>
              </div>
              <div class="post-body entry-content" id="post-body-2808418406001496891">
               <p>
               </p>
               <div style="text-align: center;">
                <span style="font-size: xx-small;">
                 (photo@
                 <a href="http://happytreefriends.wikia.com/wiki/Cuddles" target="_blank">
                  Happy Tree Friends Wiki
                 </a>
                 )
                </span>
               </div>
               <div class="separator" style="clear: both; text-align: center;">
                <a href="http://3.bp.blogspot.com/-EOU80KA3kCs/UsNYgiXWWUI/AAAAAAAAG8w/8ERbJxb0TiI/s1600/cuddlew.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                 <img border="0" height="400" src="http://3.bp.blogspot.com/-EOU80KA3kCs/UsNYgiXWWUI/AAAAAAAAG8w/8ERbJxb0TiI/s400/cuddlew.png" width="313"/>
                </a>
               </div>
               <div style="text-align: center;">
                <br/>
               </div>
               <br/>
               <br/>
               <b>
                Manatee County, Florida
               </b>
               <br/>
               <b>
                <br/>
               </b>
               In the early hours of December 28, 2013, 31-year-old
               <a href="http://www.thesmokinggun.com/buster/boyfriend-attacked-over-cuddle-refusal-576413" target="_blank">
                Shavonna Rumph
               </a>
               and 33-year-old Henry Price had spent a romantic evening
               <a href="http://www.bradenton.com/2013/12/30/4912343/bradenton-woman-charged-with-battery.html" target="_blank">
                getting intoxicated together
               </a>
               . But as Friday night turned into Saturday morning and the couple
               <a href="http://www.upi.com/Odd_News/Blog/2013/12/31/Florida-woman-charged-with-domestic-violence-against-boyfriend-who-wouldnt-cuddle/6801388508653/" target="_blank">
                headed off to bed
               </a>
               , Shavonna allegedly began to complain about Henry's
               <a href="http://www.inquisitr.com/1079314/woman-allegedly-threatens-boyfriend-with-knife-for-refusing-to-cuddle/" target="_blank">
                aversion to cuddling
               </a>
               .
               <br/>
               <br/>
               Whether this was just a general admonishment on Shavonna's part or the result of Henry's failure to snuggle right then and there is unclear. But whatever caused the disagreement, Henry decided that it was time to
               <a href="http://www.inquisitr.com/1079314/woman-allegedly-threatens-boyfriend-with-knife-for-refusing-to-cuddle/" target="_blank">
                leave the premises
               </a>
               . Shavonna grabbed (and tore) Henry's shirt to keep him from leaving, but he was determined to escape from the bedroom to a cuddle-free zone.
               <br/>
               <br/>
               Unfortunately for him, that's also when Shavonna decided to offer Henry's organs the chance to cuddle up next to some stainless steel. According to the
               <a href="http://i.cdn.turner.com/dr/teg/tsg/release/sites/default/files/assets/cuddlereportfuller.jpg" target="_blank">
                sheriff's report
               </a>
               , Shavonna picked up a kitchen knife and brandished it in Henry's direction.
               <br/>
               <br/>
               <br/>
               <div style="text-align: center;">
                <span style="font-size: xx-small;">
                 <a href="http://mothergoosejuice.wordpress.com/" target="_blank">
                  Mother Goose Juice
                 </a>
                </span>
               </div>
               <div class="separator" style="clear: both; text-align: center;">
                <a href="http://4.bp.blogspot.com/-pyr9LX-Ux-4/UsNZbqaYwSI/AAAAAAAAG84/znpft_IMk_0/s1600/sungglebearevil.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                 <img border="0" height="400" src="http://4.bp.blogspot.com/-pyr9LX-Ux-4/UsNZbqaYwSI/AAAAAAAAG84/znpft_IMk_0/s400/sungglebearevil.jpg" width="324"/>
                </a>
               </div>
               <div style="text-align: center;">
                <span style="font-size: x-small;">
                 "Do ya feel lucky, you non-cuddling punk?"
                </span>
               </div>
               <br/>
               <br/>
               <br/>
               It's not clear from the official report if the altercation moved into the kitchen or Shavonna just kept a kitchen knife the bedroom for occasions like this. But either way, Henry probably regretted not just sucking it up and going with the
               <a href="https://www.youtube.com/watch?v=j0eStJlnNKo" target="_blank">
                Ross Geller 'Hug 'n Roll' maneuver
               </a>
               after Shavonna fell asleep.
               <br/>
               <br/>
               Now to be fair, Shavonna claims that she never picked up a knife. But she did admit to being angry about his refusal to cuddle, ripping his shirt, and even attempting to block the doorway by moving a table in front of it...so yeah, I'm gonna go with Henry's account on this one over the crazy lady's version.
               <br/>
               <br/>
               Shavonna Rumph was arrested and charged with
               <a href="http://www.bradenton.com/2013/12/30/4912343/bradenton-woman-charged-with-battery.html" target="_blank">
                misdemeanor domestic battery
               </a>
               . She was released from jail on Sunday after posting $250 bond.
               <br/>
               <br/>
               <br/>
               <div class="separator" style="clear: both; text-align: center;">
                <a href="http://1.bp.blogspot.com/-H2llBXHNDnM/UsNXYSgdXxI/AAAAAAAAG8o/se0Dh1NnLTs/s1600/Shavonna.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                 <img border="0" height="320" src="http://1.bp.blogspot.com/-H2llBXHNDnM/UsNXYSgdXxI/AAAAAAAAG8o/se0Dh1NnLTs/s320/Shavonna.jpg" width="255"/>
                </a>
               </div>
               <div style="text-align: center;">
                <span style="font-size: x-small;">
                 ...and with that smile on her face, one can hope that maybe
                </span>
               </div>
               <div style="text-align: center;">
                <span style="font-size: x-small;">
                 she got placed with a cell mate who was willing to spoon.
                </span>
               </div>
               <div style="text-align: center;">
                <span style="font-size: x-small;">
                 <br/>
                </span>
               </div>
               <div style="text-align: center;">
                <span style="font-size: x-small;">
                 <br/>
                </span>
               </div>
               <div style="background-color: white; line-height: 20px;">
                <span style="font-family: inherit;">
                 <b style="line-height: 24px;">
                  <i>
                   Please feel free to leave a comment below. If you'd like to sing my praises or tell me how terrible I am more personally, I can also be found
                   <a href="http://twitter.com/NickNafster79" style="background-color: transparent; border: 0px none; margin: 0px; outline: none 0px; padding: 0px; vertical-align: baseline;" target="_blank">
                    on Twitter
                   </a>
                   .
                  </i>
                 </b>
                </span>
               </div>
               <div style="background-color: white; line-height: 20px;">
                <span style="font-family: inherit;">
                 <br/>
                </span>
               </div>
               <div style="background-color: white; line-height: 20px;">
                <span style="line-height: 24px;">
                 <b>
                  <i>
                   <span style="font-family: inherit;">
                    To get updates on when new articles or podcasts are published (and occasional random musings) 'Like' the official
                    <a href="https://www.facebook.com/pages/RamblingBeachCatcom/310287192327965" style="outline: none;" target="_blank">
                     RamblingBeachCat.com Facebook page
                    </a>
                    . Every time someone does, a woman decides there are more important things to threaten her boyfriend with a knife about than his aversion to attempting to sleeping with a
                    <a href="http://www.urbandictionary.com/define.php?term=T-Rex+Arm&amp;defid=4009756" target="_blank">
                     T-Rex arm
                    </a>
                    .
