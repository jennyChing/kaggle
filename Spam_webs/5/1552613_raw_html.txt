    function addFloatingAdvert() {
        var adBox = $("#floating-ad-div")
        var floatingAd = $('.adspot.bottom')
        var bottomLimitElement = $("#page")
                
        if (isAdAllowed(floatingAd, bottomLimitElement.last(),adBox)) { 

            if (isScrolledTo(adBox)) { // floats the ad when scrolled to
                floatingAd.css({
                    position:'fixed',
                    top:'55px'
                });
                if (isElemTopScrolledToAnotherElem(floatingAd, adBox)) { // goes back into position if scrolling up
                    floatingAd.css({
                        position:'relative',
                        top:'auto'
                    });
                }
            } else { // the original position
                floatingAd.css({
                    position:'relative',
                    top:'auto'
                });
            }

            if (isElemBottomScrolledToBottomOf(floatingAd, bottomLimitElement.last())) {
                var topOfTheBottomElement = bottomLimitElement.last().offset().top;
                var bottomOfTheBottomElement = topOfTheBottomElement + bottomLimitElement.last().height();
                // topPosition should be the bottom of the bottomLimitElement + height of the ad - the mysterious 15
                var top = bottomOfTheBottomElement - floatingAd.height() - 15;
                floatingAd.css({
                    position:'absolute',
                    top:top
                });
            }
        }
    }

    function isScrolledTo(elem) {
        if (elem === undefined) {
            return false;
        }

        var docViewTop = $(window).scrollTop(); //num of pixels from the top of the viewable screen to the top of the page
        var elemTop = $(elem).offset().top - 80; //num of pixels from the top of the element to the top of the page
        return (elemTop &lt;= docViewTop);
    }

    function isElemBottomScrolledToBottomOf(elem, bottomLimit) {
        if (elem === undefined || bottomLimit === undefined) {
            return false;
        }

        var elemTop = $(elem).offset().top; //num of pixels from the top of the element to the top of the page
        var elemBottom = elemTop + $(elem).height();

        var bottomLimitTop = $(bottomLimit).offset().top;
        var bottomLimitBottom = bottomLimitTop + $(bottomLimit).height() -15;

        return (elemBottom &gt;= bottomLimitBottom);
    }
    
    function isAdAllowed(elem, bottomLimit, adBox) {
    	if (elem === undefined || bottomLimit === undefined) {
            return false;
        }

        var bottomLimitTop = $(bottomLimit).offset().top;
        var bottomLimitBottom = bottomLimitTop + $(bottomLimit).height() -15;

		var adBoxTop = $(adBox).offset().top
        
        return (bottomLimitBottom &gt;= adBoxTop + $(elem).height())

    }

    function isElemTopScrolledToAnotherElem(elem, otherElem) {
        var elemTop = $(elem).offset().top; //num of pixels from the top of the element to the top of the page
        var otherElemTop = $(otherElem).offset().top; //num of pixels from the top of the element to the top of the page
        var otherElemBottom = otherElemTop + $(elem).height();
        return (elemTop + 10 &lt;= otherElemBottom)
    }
