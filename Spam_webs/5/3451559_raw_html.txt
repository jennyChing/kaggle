   var shrinkToFit = function(element) {
            var parent = element.parent();
            var parentWidth = parent.width();
            var parentHeight = parent.height();
            var lineHeight = parseFloat(element.css('lineHeight'));
            var fontSize = parseFloat(element.css('font-size'));
            var ratio = 1;

            if (!(isNaN(lineHeight) || isNaN(fontSize))) {
                ratio = lineHeight / fontSize;
            }
            if (element.outerWidth() &gt; parentWidth || element.outerHeight() &gt; parentHeight) {
                element.css('line-height', Math.round(100*ratio) + '%');
            }
            while (element.outerWidth() &gt; parentWidth || element.outerHeight() &gt; parentHeight) {
                element.css('font-size', (parseFloat(element.css('font-size')) - 1) + "px" );
                // element.css('line-height', (parseFloat(element.css('line-height')) - ratio) + "px" );
                if (parseInt(element.css('font-size'), 10) &lt;= 9) {
                    break;
                }
            }
        };

        $(function () {
            $('.shrink-to-fit').each(function(i, el) {
                shrinkToFit($(el));
            });
        });
  </script>
