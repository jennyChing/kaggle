function rotate_filmstrip()
{
	var contents =  new Array('gallery','latest','popular');
	if(run_rotater == 1)
	{
		rotate_timer = setTimeout("display_filmstrip('"+contents[start_rotate]+"')", 6631);
	}
}

function display_filmstrip(section)
{
	$$('div#related_video_panel div.active_selection').each( function(obj,index) {
		Effect.Fade(obj.id.substring(0,obj.id.indexOf("_"))+"_filmstrip", { duration: 0.1 });
		$(obj.id).removeClassName('active_selection');
	});
	$(section+'_tab').addClassName('active_selection');
	setTimeout("display_thumbs('"+section+"')", 200);
}

function display_thumbs(section)
{
	Effect.Appear(section+'_filmstrip', { duration: 0.5 });	
	start_rotate = start_rotate + 1;
	if(start_rotate &gt; 2) start_rotate = 0;
	rotate_filmstrip();
}
