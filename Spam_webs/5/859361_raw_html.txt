(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

                  Forgotten Memory Card Password

                 <div class="post-body entry-content" id="post-body-9004166750686659368" itemprop="articleBody">
                  <div dir="ltr" style="text-align: left;" trbidi="on">
                   We are all use memory card in our mobile for extra storage. Some people use password protection for safety And many of them has forget their memory card password .
                   <br/>
                   If you are a Symbian device user then no need to worry about forgotten memory card password.
                   <br/>
                   <br/>
                   <div class="separator" style="clear: both; text-align: center;">
                    <a href="http://4.bp.blogspot.com/-0-Al6zZ91VA/VMJOBuEt_SI/AAAAAAAAARM/GfANuZvjw4s/s1600/Memory-Cards.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                     <img border="0" height="320" src="http://4.bp.blogspot.com/-0-Al6zZ91VA/VMJOBuEt_SI/AAAAAAAAARM/GfANuZvjw4s/s1600/Memory-Cards.jpg" width="313"/>
                    </a>
                   </div>
                   <br/>
                   <br/>
                   In this post I am going to teach you how to reset or hack your memory card password in easy way.
                   <br/>
                   Before we apply the step you need to have X-Plore application which is used to explore your devices system files and folders even the hidden folders in your device
                   <br/>
                   <br/>
                   <span style="background-color: white;">
                    <span style="color: red;">
                     <b>
                      Mark 1 :
                     </b>
                    </span>
                   </span>
                   At first install the ''X-Plore'' application in your device. So search on the internet or play store and download and install this application. In this case you can download trail version of this application.
                   <br/>
                   <br/>
                   <b>
                    <span style="color: red;">
                     Mark 2 :
                    </span>
                   </b>
                   After install the application open the X-plore application and press 0 (zero ) and check which is your marked in the system files.
                   <br/>
                   <br/>
                   <div class="separator" style="clear: both; text-align: center;">
                    <a href="http://1.bp.blogspot.com/-xKGtSkuwOfU/VMJN3uWkN3I/AAAAAAAAARE/FsikQ_jbCH4/s1600/How-To-Unlock-The-Password-of-Locked-Memory-Card.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                     <img border="0" src="http://1.bp.blogspot.com/-xKGtSkuwOfU/VMJN3uWkN3I/AAAAAAAAARE/FsikQ_jbCH4/s1600/How-To-Unlock-The-Password-of-Locked-Memory-Card.jpg"/>
                    </a>
                   </div>
                   <br/>
                   <br/>
                   <b>
                    <span style="color: red;">
                     Mark 3 :
                    </span>
                   </b>
                   After done this step then go for
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    following path C:/System/Data/Mmcstore
                   </span>
                   <br/>
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    <br/>
                   </span>
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    <span style="color: red;">
                     <b>
                      Mark 4 :
                     </b>
                    </span>
                    When you reached this path then just press (3)
                   </span>
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    under option to set it in the Hex-viewer
                   </span>
                   <br/>
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    <br/>
                   </span>
                   <span style="background-color: white;">
                    <span style="font-family: Georgia, Times New Roman, Times, serif;">
                     <span style="font-size: 14px; line-height: 21px;">
                      <b>
                       <span style="color: red;">
                        Mark 5 :
                       </span>
                      </b>
                      Then see on the thirdcolumn then you will able to see a line of codes like ''
                     </span>
                    </span>
                   </span>
                   <span style="background-color: white; font-family: Georgia, 'Times New Roman', Times, serif; font-size: 14px; line-height: 21px;">
                    TMSD02G (c???x???5?5?5?5?5). Just check the character between the ? it is your password is:
                    <span style="color: red;">
                     <b>
                      55555
                     </b>
                    </span>
