    function downloadFile(link, userClick, alreadyLoaded) {
        if (userClick)
            return $.release.fn.downloadFile(link);

        if (!alreadyLoaded) {
            var downloadId = $getQuerystring("DownloadId");
            if (!downloadId)
                downloadId = getIdFromFragment();
            if (downloadId) {
                var clickOncePath = $("a[fileId='" + downloadId + "']").attr('d:clickOncePath');
                var clickOnceUrl = 'http://dropboxifier.codeplex.com/downloads/get/clickOnce/*REPLACE*'.replace('downloads/get/clickOnce/*REPLACE*', 'downloads/get/clickOnce/' + clickOncePath);
                var fileUrl = 'http://dropboxifier.codeplex.com/downloads/get/0'.replace('downloads/get/0', 'downloads/get/' + downloadId);
                
                window.location = clickOncePath ? clickOnceUrl : fileUrl;
            }
        }

        return false;
    }

    function getIdFromFragment() {
        var path = document.location.toString();

        if (path.match('#')) {
            var fileID = '#' + path.split('#')[1];

            if (fileID.toLowerCase().indexOf("downloadid=") &gt; 0) {
                fileID = fileID.split("=");

                if (fileID[1].length &gt; 0) {
                    return fileID[1];
                }
            }
        }
    }

      function saveSubscriptionDataDirectly(type) {

    function saveSubscriptionData() {
        var callingComponent = $('#subscriptionSettings #callingComponent').val();
        closeSubscriptionDialog();

        var type = $('input[name=EmailSubscriptionTypeList]:checked').val();
        if (type === '')
            return false;
        var removeIndividual = $('input[name=StopNotificationsCheckBox]:checked').val();
        if (removeIndividual === undefined)
            removeIndividual = false;
        $('#subscriptionSettings').remove();
        return saveSubscriptionDataCore(type, removeIndividual, callingComponent);
    }

    function saveSubscriptionDataCore(type, removeIndividual, callingComponent) {

        var isSubscribed = $('#SubscribeCheckBox').is(':checked') &amp;&amp; $('#SubscribeCheckBox').is(':enabled');
        CodePlex.UpdateProgress.show();
        $('#subscriptionContent').load('http://dropboxifier.codeplex.com/subscriptions/Release/save',
                                       { isIndividualSubscribed: isSubscribed, deliveryStatus: type, displayType: 'Full', removeIndividualSubscriptions: removeIndividual },
                                       function () {
                                           CodePlex.UpdateProgress.hide();
                                           if (callingComponent) {
                                               $('#actionBar_placeholder').load('http://dropboxifier.codeplex.com/workitem/ActionBar',
                                                   { activeView: 'AdvancedView' });
                                           }
                                       });
        if (type === '1' || type === '2') {
            $('#unsubscribeSection').show();
        }
        else {
            $('#unsubscribeSection').hide();
            $('#stopNotificationsDiv').hide();
        }

        return false;
    }

        return false;
    }

    function adjustSubscriptionLinkPlacement() {
        var $changeSubLink = $("#ShowSubscriptionSelection"),
            $changeSubPlaceHolder = $("#ShowSubscriptionSelectionPlaceHolder"),
            currentOffsetLeft,
            offsetLeft,
            currentOffsetTop,
            offsetTop,
            ieOffset = $.browser.msie ? 1 : 0,
            lineHeight = parseInt($changeSubPlaceHolder.css('line-height'));
        
        if ($changeSubLink.length != 0 &amp;&amp; $changeSubPlaceHolder.length != 0) {
            currentOffsetLeft = parseInt($changeSubLink.css("left")),
            offsetLeft = $changeSubPlaceHolder.offset().left - $changeSubLink.offset().left + 1, // the +1 just centers the text between the parens more evenly
            
            currentOffsetTop = parseInt($changeSubLink.css("top")),
            offsetTop = $changeSubPlaceHolder.offset().top - $changeSubLink.offset().top - lineHeight + 5, // the -5 just centers the text between the parens more evenly

            $changeSubLink.css("left", currentOffsetLeft + offsetLeft + ieOffset);
            $changeSubLink.css("top", currentOffsetTop + offsetTop + ieOffset);
        }
    }

    function adjustEmailLinkPlacement() {
        var $changeEmailLink = $("#ChangeProfileLink"),

        }
    }
