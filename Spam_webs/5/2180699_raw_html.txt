		$(document).ready(function() {
			  var AJAX = 0;
			 
			 $('.submitButton').click(function(){				
				if(!AJAX) {
					AJAX = 1;
					$('#ajax-wait').show();
					$.ajax({
						type: "POST",
						url: "/email_submit.php",
						data: 'email='+ $('#MobileAppEmail').val(),
						dataType: "json",
						success: function(data){
							AJAX = 0;
							if(data.valid == 1) {
								$('#mobileOS_page_bottom').html("&lt;span style='color: #E46419; font-size: 14'&gt;Thanks! We will notify you as soon as we release a version of Draw a Stickman for your device.&lt;/span&gt;");
							}else{
								alert(data.message);
							}
							$('#ajax-wait').hide();
						},
						error: function(msg){
							alert('We encountered an unknown error. Please try again!');
							AJAX = 0;
							$('#ajax-wait').hide();
						}
					});
				}
				
				return false;
			  });
		});
 </script>
