        $(function(){

            var $topBar              = $('#topbar');
            var $fixedShareContainer = $('.fixed-share-container');
            var $cta                 = $('#cta-container');
            var shareOffset          = 105;

            var slideCount           = $scrollAboutTemplate.find('.slide').length;
            var $scrollBody;

            var $closeIcons;
            var fixed                = true;
            var sharePosAbsolute     = false;
            var $window              = $(window);
            var onScroll;
            var onResize;
            var onTopBarClick;
            var InsideNav            = Imgur.InsideNav.getInstance();
            var newSlideViewPercent  = 0.2;
            var hashes               = [];
            var eventsFired          = {
                slideViews               : -1,
                scrollSlideInvoked       : false
            };

            //if user signed in, remove the "Become an Imgurian" slide
            if (InsideNav._.signed){
                $scrollAboutTemplate.find('.become-imgurian').remove();
                slideCount--;
            }

            $scrollAboutTemplate.find('.slide').each(function(){
                hashes.push($(this).data('hash'));
            });
            hashes = hashes.reverse();


            //prepend "scroll up to learn more" slides to the rest of the content
            $body.wrapInner('&lt;div id="scroll-body"&gt;&lt;/div&gt;').prepend($scrollAboutTemplate[0].outerHTML);

            $closeIcons = $('#scroll-about .icon-x, #scroll-about .keep-browsing');
            $scrollBody = $('#scroll-body');
            $scrollBody.css('margin-top', viewportHeight * slideCount - topBarHeight);
            $body.scrollTop(curTop + viewportHeight * slideCount - topBarHeight);
            $scrollAbout = $('#scroll-about');
            $scrollAbout.css('top', - viewportHeight * (slideCount - 1));
            $fixedShareContainer.css({
                'top': shareOffset + $cta.height()
            });

            $('video').each(function(){
                this.play();
            });

            onScroll = _.throttle(function(){
                var pos = $body.scrollTop();
                if (pos &lt; viewportHeight * (slideCount - 1) &amp;&amp; fixed){
                    //set scroll-about position to absolute so we can scroll through the slides
                    fixed = false;
                    $scrollAbout.css({
                        'position': 'absolute',
                        'top': 0
                    });
                    $topBar.css({
                        'position': 'fixed',
                        'bottom': 0
                    });
                } else if (pos &gt; viewportHeight * (slideCount - 1) &amp;&amp; !fixed){
                    //reset to fixed so body content hovers back over.
                    fixed = true;
                    $scrollAbout.css({
                        'position': 'fixed',
                        'top': - viewportHeight * (slideCount - 1)
                    });
                    $topBar.css({
                        'position': 'relative',
                        'bottom': 'auto'
                    });
                }

                var scrollAboutHeight = viewportHeight * slideCount - topBarHeight;
                if (!sharePosAbsolute &amp;&amp; pos &lt; scrollAboutHeight){
                    $fixedShareContainer.css({
                        'position': 'absolute',
                        'top': 45
                    });
                    sharePosAbsolute = true;
                } else if (sharePosAbsolute &amp;&amp; pos &gt;= scrollAboutHeight){
                    $fixedShareContainer.css({
                        'position': 'fixed',
                        'top': shareOffset + $cta.height()
                    });
                    sharePosAbsolute = false;
                }

                //fire event when user invokes the test
                if (pos &lt; scrollAboutHeight &amp;&amp; !eventsFired.scrollSlideInvoked){

                    Imgur.Util.jafoLog({
                        event: 'scrollSlideInvoke'
                    });
                    eventsFired.scrollSlideInvoked = true;
                }

                //fire an event (only once) for each consecutive slide the user sees
                for(var i=0; i &lt; slideCount; i++){
                    var threshold = (slideCount - i - newSlideViewPercent) * viewportHeight;
                    if (pos &lt; threshold &amp;&amp; eventsFired.slideViews &lt; i){
                        var hash = $('hash');
                        Imgur.Util.jafoLog({
                            event: 'scrollSlideView',
                            meta: {
                                slidePosition: (i+1), //1-indexed for analytics purposes
                                hash: hashes[i]
                            }
                        });
                        eventsFired.slideViews = i;
                    }
                }

            }, 150);

            onResize = _.throttle(function(){

                }

                viewportHeight = $(window).height();

                }
                $scrollBody.css('margin-top', viewportHeight * slideCount);
                InsideNav.positonScrollPanel();
                if (atTop){
                    $window.scrollTop(viewportHeight * slideCount);
                }
            }, 150);

            onTopBarClick = function(){
                $body.scrollTop(viewportHeight * slideCount - topBarHeight);
            };

            $window.on('scroll', onScroll);
            $window.on('resize', onResize);
            $topBar.on('click', onTopBarClick);

            });

                    meta: {

                        hash: hash,
                        accountId: accountId
                    }
                });
            });

                        accountId: accountId
                    }
                });
            });

                $scrollBody.css('margin-top', 0);
                $window.scrollTop(0);

                $window.off('scroll', onScroll);
                $window.off('resize', onResize);
                $topBar.off('click', onTopBarClick);

                $topBar.css({
                    'position': 'relative',
                    'bottom': 'auto'
                });

                $fixedShareContainer.css({
                    'position' : 'fixed',

                        button: button
                    }
