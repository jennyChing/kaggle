    // Open Get Zipping Overlay
    function GetZippingOpen() {
        zc.popup.show("getzipping");
        // Change the position of the overlay depending on the viewport
        if (Modernizr.mq('(max-width: 440px)')) {
            zc.popup.position("getzipping", {my: "right top", at: "right bottom", of: ".getzipping-btn", offset: "6 14" });

        }
        // Animate the interior of the Overlay
        jQuery(".getzipping-interior-animate").slideDown({duration: getZippingSlideSpeed});
    }

    // Close Get Zipping Overlay
    function GetZippingClose() {
        var getZippingOverlay = jQuery('#overlay-getzipping');

        if (getZippingOverlay.is(":visible")) {
            getZippingOverlay.jqmHide();
        } else {
            // turn off the animation for smaller viewports
            if (Modernizr.mq('(min-width: 640px)')) {
                getZippingOverlay.show(); // Show the Overlay again so we can animate

                var getZippingHeight = getZippingOverlay.outerHeight();
                getZippingHeight = getZippingHeight - 8; // subtract 8px for the green border at the bottom of the button
                getZippingOverlay.animate({ top: "-" + getZippingHeight + "px" }, getZippingSlideSpeed, function(){
                    jQuery(this).hide(); // hide the overlay after animation
                    jQuery(".getzipping-interior-animate").hide(); // hide the interior content so it can animate when opened again.
                });
            } else {
                jQuery(".getzipping-interior-animate").hide(); // hide the interior content so it can animate when opened again.
            }
        }
    }

    // GetZipping Overlay
    jQuery('.getzipping-btn').click(function() {
        GetZippingOpen();
        return false;
    });

    // replace the Get Zipping button with a mobile nav icon
    var ziptopia_nav_btn = '/sites/default/files/images/btn-ziptopia-mobile-nav.svg';

            swapBtn();

            putbackBtn();

            <h2>
             <strong>
              SUN, SUITCASES, AND SWEETS IN KITSILANO
             </strong>
            </h2>
            <p>
             Home to the first
             <a href="http://www.lululemon.com/about/" target="_blank">
              Lululemon
             </a>
             (the beloved brand of yoga attire),
             <a href="http://en.wikipedia.org/wiki/Kitsilano" target="_blank">
              Kitsilano
             </a>
             (called Kits for short) runs along the coast of English Bay, a short drive from downtown Vancouver. Look for the red "
             <a href="http://www.yattermatters.com/2009/03/vancouvers-kitsilano-finds-its-roots/" target="_blank">
              Kitsilano Memories Project
             </a>
             " signs sprinkled throughout the neighborhood to read about long-time residents' memories of the area.
            </p>
            <p>
             <a href="http://cfapp.vancouver.ca/parkfinder_wa/index.cfm?fuseaction=FAC.ParkDetails&amp;park_id=112" target="_blank">
              Kitsilano Beach
             </a>
             <span>
              (1499 Arbutus St.;
             </span>
             <a href="http://vancouver.ca/parks-recreation-culture/kitsilano-beach.aspx" target="_blank">
              <span>
               <span>
                vancouver.ca/kitsilano-beach
               </span>
              </span>
             </a>
             <span>
              ) Spend a sunny afternoon here, where trim twentysomethings play beach volleyball, families lounge on the sandy beach overlooking Burrard Inlet, and kids climb on the playground's jungle gym. Paddleboard rentals, basketball courts, and a 137-meter heated outdoor salt-water pool add to your recreational options. No wonder "National Geographic" mentions Vancouver  and specifically Kits  as one of the top 10 beach cities in the world.
             </span>
            </p>
            <p>
             <strong>
              <span>
               Wanderlust
              </span>
             </strong>
             <span>
              (1929 West 4th Ave., 604-739-2182;
             </span>
             <a href="http://www.wanderlustore.com/" target="_blank">
              <span>
               <span>
                wanderlustore.com
               </span>
              </span>
             </a>
             <span>
              ) If journeys beyond B.C. are in your future, then a stop here is a must. The 25-year-old (and aptly named) travel shop carries an exhaustive selection of maps
              <span>
               and guidebooks covering virtually every corner of the globe. Here you'll also find luggage and accessories like suitcase organizers, money belts, and toiletry kits for your next adventure (with or without a Zipcar). You can even buy gift-wrap printed with maps of the world or ask the staff for tips on your next trip.
              </span>
             </span>
            </p>
            <p>
             <span>
              <span>
               <span class="caption caption-none">
                <span class="caption-inner" style="width: 720px;">
                 <img alt="Its delicious, local, farm-to-table food. Its dreamed up by a celebrity chef. Its Fable." height="340" src="http://dru-cdn.zipcar.com/sites/default/files/3_vancouver_fable_4.jpg" title="Its delicious, local, farm-to-table food. Its dreamed up by a celebrity chef. Its Fable." width="720"/>
                 Its delicious, local, farm-to-table food. Its dreamed up by a celebrity chef. Its Fable.
                </span>
               </span>
              </span>
             </span>
            </p>
            <p>
             <strong>
              <span>
               Fable
              </span>
             </strong>
             <span>
              (1944 West 4th Ave., 604-732-1322;
             </span>
             <a href="http://fablekitchen.ca/" target="_blank">
              <span>
               <span>
                fablekitchen.ca
               </span>
              </span>
             </a>
             <span>
              ) Now that you've worked up an appetite, cross the street to Fable, a cozy spot that serves unfussy farm-to-table fare. Helmed by former Top Chef Canada contestant Trevor Bird, Fable's name is a portmanteau of farm to table and its seasonally changing menu uses local ingredients like Sawmill Bay mussels and B.C. gem tomatoes prepared in an open kitchen.
             </span>
            </p>
            <p>
             <strong>
              <span>
               Rain or Shine Ice Cream
              </span>
             </strong>
             <span>
              (1021926 West 4th Ave., 604-428-7246;
             </span>
             <a href="http://rainorshineicecream.com" target="_blank">
              <span>
               <span>
                rainorshineicecream.com
               </span>
              </span>
             </a>
             <span>
              ) Still have room for dessert? Order one of Fable's daily macarons or stop into Rain or Shine, which opened last fall just a few doors away. Their seasonally rotating selection of flavors like honey lavender or sea salted chocolate rosemary make it tough to choose just one. Fortunately, flights containing small servings of four different flavors help ease indecisiveness. (Warning: may cause brain freeze!)
             </span>
            </p>
            <h2>
             <strong>
              <span>
               DOUGHNUTS, DOG PARKS, AND DINERS IN SOUTH MAIN
              </span>
             </strong>
            </h2>
            <p>
             <span>
              South Main (or SoMa for short) makes up a subsection of Mount Pleasant, an up-and-coming area of East Van that has traditionally attracted its share of artists, designers, and authors. South Main has no official boundaries, but extends from roughly 6
              <sup>
               th
              </sup>
              to 33
              <sup>
               rd
              </sup>
              Avenue.
             </span>
            </p>
            <p>
             <span>
              <span class="caption caption-none">
               <span class="caption-inner" style="width: 720px;">
                <img alt='With expert-roasted Fair Trade coffee and drool-worthy pastries, its no wonder Vancouver Magazine recently named 49th Parallel Best Coffee House."' height="340" src="http://dru-cdn.zipcar.com/sites/default/files/6_vancouver_49thparallel_4.jpg" title='With expert-roasted Fair Trade coffee and drool-worthy pastries, its no wonder Vancouver Magazine recently named 49th Parallel Best Coffee House."' width="720"/>
                With expert-roasted Fair Trade coffee and drool-worthy pastries, its no wonder Vancouver Magazine recently named 49th Parallel Best Coffee House.
               </span>
              </span>
             </span>
            </p>
            <p>
             <strong>
              <span>
               49
               <sup>
                th
               </sup>
               Parallel
              </span>
             </strong>
             <span>
              (2902 Main St., 604-420-4900;
             </span>
             <a href="http://www.49thparallelroasters.com/" target="_blank">
              <span>
               49thparallelroasters.com
              </span>
             </a>
             <span>
              ) Get your morning java fix with a cup of Fair Trade custom-roasted coffee and a handcrafted doughnut in flavors like apple maple bacon or strawberry rhubarb. Creative types congregate with their MacBooks at the shop's long communal table (overlooking the kitchen where Lucky's Doughnuts are made), while the outdoor patio encourages latte-drinkers to linger on warm, sunny days.
             </span>
            </p>
            <p>
             <strong>
              <span>
               Queen Elizabeth Park
              </span>
             </strong>
             <span>
              (46000 Cambie St.,
              <a>
              </a>
              <a href="http://vancouver.ca/parks-recreation-culture/queen-elizabeth-park.aspx" target="_blank">
               vancouver.ca/parks-recreation-culture/queen-elizabeth-park.aspx
              </a>
              ) Once you're satisfactorily fed and caffeinated, head a few blocks west of Main Street for a jog or stroll. Here, young couples push strollers or trail dogs on leashes and pensioners soak up the scenery. In addition to tennis courts, picnic areas, a rose garden, and some of the best views of downtown Vancouver's skyline, the park has ample free (and some paid) parking (score!). Plus, i
              <span>
               f it starts raining (as it tends to do in B.C.), there's always the indoor
               <a href="vancouver.ca/parks-recreation-culture/bloedel-conservatory.aspx" target="_blank">
                Bloedel Floral Conservatory
               </a>
               .
              </span>
             </span>
            </p>
            <p>
             <span>
              <span>
               <span class="caption caption-none">
                <span class="caption-inner" style="width: 720px;">
                 <img alt="Whether youre an omnivore, carnivore, vegan, vegetarian, or celiac, youre sure to fill up at the artsy Wallflower." height="340" src="http://dru-cdn.zipcar.com/sites/default/files/4_vancouver_wallflower_4.jpg" title="Whether youre an omnivore, carnivore, vegan, vegetarian, or celiac, youre sure to fill up at the artsy Wallflower." width="720"/>
                 Whether youre an omnivore, carnivore, vegan, vegetarian, or celiac, youre sure to fill up at the artsy Wallflower.
                </span>
               </span>
              </span>
             </span>
            </p>
            <p>
             <strong>
              <span>
               The Wallflower Modern Diner
              </span>
             </strong>
             <span>
              (2420 Main St., 604-568-7554
             </span>
             ) After working up an appetite in the park, head back to Main Street, where the Wallflower serves brunch to the neighborhood's hipsters until 4 p.m., seven days a week, and later turns into a bar. The menu is vegan and celiac-friendly with comfort foods like nachos (served vegan or regular), pierogies, vegan mac 'n cheese, and, of course, poutine. Funky mismatched mugs and a Ms. Pac-Man machine by the washroom lend retro style to the space.
            </p>
            <p>
             <span>
              <span class="caption caption-none">
               <span class="caption-inner" style="width: 720px;">
                <img alt="Printed matter matters, and it doesnt get hipper than at letter-centric Regional Assembly of Text." height="340" src="http://dru-cdn.zipcar.com/sites/default/files/5_vancouver_assemblyoftext_4.jpg" title="Printed matter matters, and it doesnt get hipper than at letter-centric Regional Assembly of Text." width="720"/>
                Printed matter matters, and it doesnt get hipper than at letter-centric Regional Assembly of Text.
               </span>
              </span>
             </span>
            </p>
            <p>
             <strong>
              <span>
               Regional Assembly of Text
              </span>
             </strong>
             <span>
              (3934 Main St., 604-877-2247,
             </span>
             <a href="http://assemblyoftext.com/" target="_blank">
              <span>
               assemblyoftext.com
              </span>
             </a>
             <span>
              ) The upper part of Main Street offers indie shops and boutiques selling everything from vinyl to vintage fashion. Among them is this charming stationery shop opened
             </span>
             <span>
              by two graduates of the Emily Carr Institute of Art and Design. Regional Assembly of Text aims to make letter-writing cool again with letter-pressed cards, a typewriter you can rent by the hour, an in-store button-making station, and a monthly letter-writing club. Old-school lockers and filing cabinets lend a cool vintage vibe to the shop, and a cozy reading library lets you browse 'zines and self-published books.
             </span>
            </p>
            <h2>
             <strong>
              <span>
               FRUIT, FASHION, AND GASTOWN
              </span>
             </strong>
            </h2>
            <p>
             <strong>
              <span>
               <span class="caption caption-none">
                <span class="caption-inner" style="width: 720px;">
                 <img alt="For a refreshing, nutritious treat on the run, theres no better wheeled wonder than the popular Juice Truck." height="340" src="http://dru-cdn.zipcar.com/sites/default/files/7_vancouver_juicetruck_4.jpg" title="For a refreshing, nutritious treat on the run, theres no better wheeled wonder than the popular Juice Truck." width="720"/>
                 For a refreshing, nutritious treat on the run, theres no better wheeled wonder than the popular Juice Truck.
                </span>
               </span>
              </span>
             </strong>
            </p>
            <p>
             <strong>
              The Juice Truck
             </strong>
             (Abbott St. &amp; Water St.;
             <a href="http://thejuicetruck.ca/" target="_blank">
              thejuicetruck.ca
             </a>
             ) Owners Zach Berman and Ryan Slater started the Juice Truck in July, after a trek through the Himalayas, where they noticed locals getting most of their nutrients from the juice of
             <a href="http://en.wikipedia.org/wiki/Hippophae" target="_blank">
              seabuckthorn berries
             </a>
             . That experience inspired the pair to create one of Canada's first mobile juice and smoothie bars. Now, you can start the day off right (without hauling out to the Himalayas) with a cold-pressed juice or smoothie from the Juice Truck. Combinations include apple juice pressed with carrot, celery, and mint or pineapple juice pressed with cucumber, kale, mint, and ginger.
            </p>
            <p>
             <strong>
              <span>
               Meadow
              </span>
             </strong>
             <span>
              (104 Water St., 604-620-5802,
             </span>
             <a href="http://www.shopmeadow.ca" target="_blank">
              <span>
               <span>
                shopmeadow.ca
               </span>
              </span>
             </a>
             <span>
              ) Once youre fully juiced, stroll Water Street and browse the shops, including Meadow, opened by sisters Sasha Freeman and Casey Myrfield in November 2012. The sibling duo stocks the charming shop with a curated, charming selection of handcrafted items, including quirky cards, whimsical iPhone cases, flowered dresses, and flasks engraved with ironic sayings.
             </span>
            </p>
            <p>
             <strong>
              <span>
               Salt Tasting Room
              </span>
             </strong>
             <span>
              (45 Blood Alley, 604-633-1912,
              <a href="http://salttastingroom.com/" target="_blank">
               salttastingroom.com
              </a>
              ) Tucked away in a quiet back alley, the Salt Tasting Room has just 8 tables and one long communal table where you can pair artisan meats, cheeses, and condiments to customize your own charcuterie plate. The floor-to-ceiling chalkboard menu lists Salt's selection of delicacies from around the world, such as Talleggio cheese from Italy, Guinness mustard, and B.C. hazelnuts. Also notable are the Pinot Blanc and Gamay Noir, two B.C. wines served on tap instead of from a bottle.
             </span>
            </p>
            <p>
             <span>
              <span class="caption caption-none">
               <span class="caption-inner" style="width: 720px;">
                <img alt="Stanley Park makes way for bikers, plus boasts gorgeous views of water, mountains, trees, and wildlife. All without leaving the city. " height="340" src="http://dru-cdn.zipcar.com/sites/default/files/8_vancouver_seawall_4.jpg" title="Stanley Park makes way for bikers, plus boasts gorgeous views of water, mountains, trees, and wildlife. All without leaving the city. " width="720"/>
                Stanley Park makes way for bikers, plus boasts gorgeous views of water, mountains, trees, and wildlife. All without leaving the city.
               </span>
              </span>
             </span>
            </p>
            <p>
             <strong>
              <span>
               Stanley Park
              </span>
             </strong>
             (
             <span>
              1166 Stanley Park Drive,
              <a href="http://vancouver.ca/parks-recreation-culture/stanley-park.aspx" target="_blank">
               vancouver.ca/parks-recreation-culture/stanley-park.aspx)
              </a>
              Next, work off the cheese, meat, and wine with a stroll along Vancouver's
              <a href="http://vancouver.ca/parks-recreation-culture/seawall.aspx" target="_blank">
               <strong>
                Seawall
               </strong>
              </a>
              , which starts just beyond the edge of Gastown in neighboring Coal Harbor and extends through Stanley Park all the way to Kitsilano Beach. Stanley is a 1,001-acre public park that includes an 18-hole
              <a href="http://vancouver.ca/parks-recreation-culture/stanley-park-pitch-putt.aspx" target="_blank">
               <strong>
                Pitch &amp; Putt
               </strong>
              </a>
              golf course, an aquarium, gardens, and walking or jogging trails. If you want to drive instead of walking, the park has paid parking through EasyPark and several Zipcars parked nearby. (Its like they knew you were coming.)
             </span>
            </p>
