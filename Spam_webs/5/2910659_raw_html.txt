function startsearch(textbox) {
        
        var searchterm = textbox.value;
        
        if (searchterm == null)
            return;


        searchterm = searchterm.replace(/#/g, "");
        searchterm = searchterm.replace("/", " ");
        searchterm = searchterm.replace("\\", " ");
        searchterm = searchterm.replace(/%/g,"");
        searchterm = searchterm.replace(/&amp;/g,"%26");
        searchterm = searchterm.replace(/=/g,"");
        searchterm = searchterm.replace(/&gt;/g,"");
        searchterm = searchterm.replace(/&lt;/g,"");
        
        
        if(searchterm.length  &lt; 3){
            alert("Please enter at least 3 characters for a search term");
            return;
        }
        if(searchterm.length &gt; 43){
            alert("Please enter lesser than 50 characters for a search term");
            return;
        }

        if(searchHost == null)
            return;

        if(searchHost.length == 0)
            return            

        window.location.href = searchHost +  searchterm;
