      <p>
       Parents of newborns should be first in line for a relaxing getaway, but being homebound tends to go with the territory of caring for a new baby. Architect
       <a href="http://www.manuelvillaarq.com/">
        Manuel Villa
       </a>
       had the best of both worlds in mind when he designed his
       <a href="http://www.archdaily.com/35128/habitable-polyhedron-manuel-villa/">
        Habitable Polyhedron
       </a>
       , a place of respite in the garden of a familys home in Bogota, Columbia. An independent space from day to day house activities, the
       <a href="http://www.archdaily.com/35128/habitable-polyhedron-manuel-villa/">
        Habitable Polyhedron
       </a>
       is a dwelling place in nature designed to embrace a child and his parents.
      </p>
      <p>
       <!--more-->
      </p>
      <p>
       Villas unique geometric aesthetic for this structure is inspired in the shapes perception processes children develop in their first years of life. The basic shapes of things and their differences are key elements in the development of knowledge, and specifically in acquiring reading skills and geometric basic concepts. One face of
       <a href="http://www.archdaily.com/35128/habitable-polyhedron-manuel-villa/">
        Habitable Polyhedron
       </a>
       s pine structure opens onto a teak deck leading to the outside, and small windows and a skylight allow for ample natural light. A workspace and a sofa round out the internal features. As a child grows up, the
       <a href="http://www.archdaily.com/35128/habitable-polyhedron-manuel-villa/">
        Habitable Polyhedron
       </a>
       may serve as a serene setting for hobbies,
       <a href="http://www.inhabitots.com/2009/09/17/two-great-reads-for-design-aware-kids/">
        reading
       </a>
       and
       <a href="http://www.inhabitots.com/2009/09/21/green-building-blocks-review-citiblocs/">
        play
       </a>
        in the same vein as a
       <a href="http://www.inhabitots.com/2009/08/28/bauraums-froschkoenig-treehouse/">
        treehouse
       </a>
       or
       <a href="http://www.inhabitots.com/2009/03/30/qb-playhouses-kids-can-design-their-dream-home/">
        playhouse
       </a>
       .
      </p>

function mce_success_cb(resp){
    $('#mce-success-response').hide();
    $('#mce-error-response').hide();
    if (resp.result=="success"){
        $('#mce-'+resp.result+'-response').show();
        $('#mc_input_div').hide();
        $('#mc-embedded-subscribe').hide();
        $('#mce-'+resp.result+'-response').html('&lt;h3 style="margin:0; text-align:left; color:#000000;" &gt;THANK YOU FOR SUBSCRIBING!&lt;/h2&gt;'+resp.msg);
        $('#mc-embedded-subscribe-form').each(function(){
            this.reset();
    	});
    } else {
        var index = -1;
        var msg;
        try {
            var parts = resp.msg.split(' - ',2);
            if (parts[1]==undefined){
                msg = resp.msg;
            } else {
                i = parseInt(parts[0]);
                if (i.toString() == parts[0]){
                    index = parts[0];
                    msg = parts[1];
                } else {
                    index = -1;
                    msg = resp.msg;
                }
            }
        } catch(e){
            index = -1;
            msg = resp.msg;
        }
        try{
            if (index== -1){
                $('#mce-'+resp.result+'-response').show();
                $('#mce-'+resp.result+'-response').html(msg);            
            } else {
                err_id = 'mce_tmp_error_msg';
                html = '&lt;div id="'+err_id+'" style="'+err_style+'"&gt; '+msg+'&lt;/div&gt;';
                
                var input_id = '#mc_embed_signup';
                var f = $(input_id);
                if (ftypes[index]=='address'){
                    input_id = '#mce-'+fnames[index]+'-addr1';
                    f = $(input_id).parent().parent().get(0);
                } else if (ftypes[index]=='date'){
                    input_id = '#mce-'+fnames[index]+'-month';
                    f = $(input_id).parent().parent().get(0);
                } else {
                    input_id = '#mce-'+fnames[index];
                    f = $().parent(input_id).get(0);
                }
                if (f){
                    $(f).append(html);
                    $(input_id).focus();
                } else {
                    $('#mce-'+resp.result+'-response').show();
                    $('#mce-'+resp.result+'-response').html(msg);
                }
            }
        } catch(e){
            $('#mce-'+resp.result+'-response').show();
            $('#mce-'+resp.result+'-response').html(msg);
        }
    }
}
