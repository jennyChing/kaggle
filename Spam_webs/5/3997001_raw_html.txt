    <script>
     var carousel_autorun = 1;
var carousel_prev_index = 0;
var carousel_index = 0;
var carousel_count = 7;
var pix_gray_src  =  'http://www.poetryinternationalweb.net/pi/graphics/icons/pix_gray.png';
var pix_black_src =  'http://www.poetryinternationalweb.net/pi/graphics/icons/pix_black.gif';
function carousel_prev(){
	carousel_prev_index = carousel_index;
	carousel_index--;
	if(carousel_index &lt; 0){
		carousel_index = carousel_count-1;
	}
	carousel_show_index();
}
function carousel_next(){
	carousel_prev_index = carousel_index;
	carousel_index++;
	if(carousel_index &gt; carousel_count-1){
		carousel_index = 0;
	}
	carousel_show_index();
}
function carousel_show_index(){
	for(var i=0; i &lt; carousel_count; i++){
		document.getElementById('carousel_item_'+i).style.display='none';
	}
	document.getElementById('carousel_item_'+carousel_index).style.display='block';
	
	document.getElementById('carousel_nav_'+carousel_prev_index).src = pix_gray_src;
	document.getElementById('carousel_nav_'+carousel_index).src = pix_black_src;

}

function carousel_show_random(){
	if(carousel_autorun == 1){
		carousel_prev_index = carousel_index;
		carousel_index = Math.floor((Math.random()*carousel_count));
		//alert(carousel_index);
		carousel_show_index();
	}
}

function carousel_goto_index(index){
	carousel_autorun = 0;
	carousel_prev_index = carousel_index;
	carousel_index = index;
	carousel_show_index();
}

carousel_show_random();
