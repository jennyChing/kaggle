		var PromoID = $(this).data("ga_promo_id");
		var PromoHref =  $(this).attr("href");  
		onPromoClickGA(PromoID, PromoHref);
	}); 
	
	$(".cart_items_link.remove").click(function() {  
	
		var ProdSku = $(this).data("product_sku");
		var ProdName = $(this).data("product_name");
		var ProdCategory = $(this).data("category");
		var ProdBrand = $(this).data("brand");
		var ProdPrice = $(this).data("price");
		var ProdQuantity = $(this).data("quantity");
		 
		RemoveCartItemGA(ProdSku, ProdName, ProdCategory, ProdBrand, ProdPrice, ProdQuantity);  
	}); 
	
	$(".cart_items_bottom .update-all").click(function() {
		$(".cart_item_list .quantity").each(function() {
			
			var RemoveLink = $(this).siblings(".cart_items_link.remove");
			var ProdQuantity = $(RemoveLink).data("quantity");
			var ProdSku = $(RemoveLink).data("product_sku");
			var ProdName = $(RemoveLink).data("product_name");
			var ProdCategory = $(RemoveLink).data("category");
			var ProdBrand = $(RemoveLink).data("brand");
			var ProdPrice = $(RemoveLink).data("price");
			
			if ($(this).val() != ProdQuantity) { 
				if (ProdQuantity &gt; $(this).val()) {
					var QtyDiff = ProdQuantity - $(this).val();
					RemoveCartItemGA(ProdSku, ProdName, ProdCategory, ProdBrand, ProdPrice, QtyDiff);  
				} else {
					var QtyDiff = $(this).val() - ProdQuantity;
					AddCartItemGA(ProdSku, ProdName, ProdCategory, ProdBrand, ProdPrice, QtyDiff); 
				}
				 
			}
		});
		//return false;
	});
});

      // 5. The API calls this function when the player's state changes.
      //    The function indicates that when playing a video (state=1),
      //    the player should play for six seconds and then stop. 
    function onPlayerStateChange(event, currplayer) {
			 
	   var VideoUrl = currplayer.getVideoUrl();       // Get the URL that the event targes
	   var VideoId  = VideoUrl.match( /[?&amp;]v=([^&amp;#]*)/ )[ 1 ];  // Extract the ID    
		 
	  // console.log('YT Event: YouTube ' + VideoId + ' ' + event.data) 
		
		if (event.data == YT.PlayerState.PLAYING) { 
			  ga('send', 'event', 'Social', 'YouTube Playing', VideoId);  
			  ELF_Cosmetics_Play_Button()
			  
        } else if (event.data == YT.PlayerState.ENDED) { 
		
			if ($('#hp_promo_pop_close').length) {  
				$(".close-btn-" + VideoId).click();
			}
				
			if ($('.play-btn-' + VideoId).length) {  
				$('.play-btn-' + VideoId).show();
				$('.play-btn-' + VideoId).parent().find(".clip").removeClass("open"); 
				if ($(".hp_slides").lenght &gt; 1) { $('#hp_slide_steps').show() };
			} 
					
			ga('send', 'event', 'Social', 'YouTube Ended', VideoId); 
			
			ELF_Cosmetics_Video_Completion()

        } else if (event.data == YT.PlayerState.PAUSED) {
			if (currplayer.getCurrentTime() != currplayer.getDuration()) { 
				ga('send', 'event', 'Social', 'YouTube Paused', VideoId); 
			}
		} else if (event.data == -1) {  
		} else {
			  ga('send', 'event', 'Social', 'YouTube '  + event.data, VideoId);  
		}
      }
	  
 
   function TrackYouTubeEvents(YTPlayer) {  
			
		var player = new YT.Player(YTPlayer, { 
			events: {
				'onReady': function() { 
								 
					var VideoUrl = player.getVideoUrl();       // Get the URL that the event targes
	  				var VideoId  = VideoUrl.match( /[?&amp;]v=([^&amp;#]*)/ )[ 1 ];  // Extract the ID    
		
					if ($(".play-btn-" + VideoId).length) {  
						var playButton = $(".play-btn-" + VideoId); 
						playButton.click(function() {    
							player.seekTo(0,false);
							player.playVideo();
						}); 
					}
					
					if ($(".close-btn-" + VideoId).length) {  
						var closeButton = $(".close-btn-" + VideoId); 
						closeButton.click(function() {   
							player.seekTo(0,false);
							player.stopVideo(); 
						}); 
					}
					
				},
				'onStateChange': function(event) { 
					onPlayerStateChange(event, player)
				}
			}
		}); 
    }
