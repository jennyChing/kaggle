function navTopTutorials() {
	var self = this;

	self.stories = [];
	$.ajax({
		url: "/handlers/RelatedStories.ashx?nodeType=" + "tutorial&amp;numPerPage=5",
		dataType: 'JSON',
		async: false,
		success: function (data) {

			if (data !== null) {
				//self.stories = data;
				if (data) {

					var niceUrls;
					$.ajax({
						url: '/handlers/global/getniceurls.ashx',
						data: { 'ids': data.map(function (el) { return el.Id; }).join(',') },
						dataType: 'JSON',
						async: false,
						success: function (data) {
							niceUrls = data;
						}
					});


					$.each(data, function (index, item) {
						item.niceUrl = niceUrls[index];
						self.stories.push(item);
					});
					
				}
			}
		}
	});
}

function navTopUpcomingRacesModel() {
	var self = this;

	self.upcoming = ko.observableArray([]);
	

	$.ajax({
		url: '/handlers/EquibaseProxy.ashx',
		data: { url: equibaseHostDomain + '/Data.cfm/Races/Upcoming', dataParams: 'ct=4' },        
		dataType: 'JSON',
		//async: false,
		success: function (data) {

		self.upcoming(data);
			$.ajax({
				url: '/handlers/global/getniceurls.ashx',
				data: { 'equibaseIds': data.map(function (el) { return el.track.identity; }).join(',') },
				dataType: 'JSON',
				//async: false,
				success: function (data2) {
					$.each(data, function (index, val) {
						data[index].niceUrl = data2[index];
					});
					
					self.upcoming(data);

				}
			});
		}
	});
}
