        function slider(selector, selector_child, sub_selector) {
            var timer = {};
            var is_changing = false;
            var promo_div = $(selector);
            var page = 1;
            function change(id) {
                if (is_changing) {
                    return;
                }

                if (page == id) {
                    return page;
                }
                var max_page = pages();
                if (id &lt; 1) {
                    id = max_page;
                }
                if (id &gt; max_page) {
                    id = 1;
                }
                is_changing = true;
                if (id == max_page) {
                    $(promo_div).find(selector_child).stop().animate({"margin-left": max_promo_margin()}, 2000, "easeOutExpo", function() {
                        is_changing = false;
                    });
                } else {
                    $(promo_div).find(selector_child).stop().animate({"margin-left": -(id - 1) * 100 + "%"}, 2000, "easeOutExpo", function() {
                        is_changing = false;
                    });
                }
                return (page = id);
            }
            function pages() {
                var promos = $(promo_div).find(sub_selector);
                return Math.ceil((20 + promos.size() * promos.outerWidth(true)) / $(promo_div).width());
            }
            function max_promo_margin() {
                var promos = $(promo_div).find(sub_selector);
                var margin = Number($(promos[1]).css("margin-left").replace("px", ""));
                return -(promos.size() * (promos.outerWidth(true) + margin) - $(promo_div).width()) + margin;
            }
            return {
                next: function() {
                    change(page + 1);
                },
                prev: function() {
                    change(page - 1);
                }
            }
        }

        var comp_slider = slider(".slides", "ul", "li");
        setInterval(function() {
            comp_slider.next();
        }, 4000);
    });
