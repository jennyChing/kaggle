		set_language: function(language) {

		},
		get_values: function() {

				country: this.el.find('dl.country dd a.selected').attr('rel'),
				language: this.el.find('dl.language dd a.selected').attr('rel'),
			};
		},
		get_signs: function() {
			return {
				flag: this.el.find('dl.country dd a.selected').children('img').attr('src'),

			};
		}
	};

	});

		    var result = {};

		        result[unescape(p1)] = unescape(p3);
		        return str;
		    });
		    return result;

  <script type="text/javascript">
   var slider;

$(function() {
	// SLIDER

	slider = {
		api: undefined,

		resize_defer: undefined,
		is_resizing: false,

		container: $('#gameSlider &gt; div.sliderContent'),
		arrows: $('#gameSlider div.sliderCarousel &gt; a'),
		thumbs: $('#gameSlider ul.sliderNav a'),

		enable: function(index) {
			var thumbs = this.thumbs;
			var arrows = this.arrows;

			this.api = this.container.switchable({
				triggers: false,
				panels: 'li',
				initIndex: index, // display the first panel
				effect: 'scrollLeft', // taking effect when autoplay == true
				easing: 'cubic-bezier(.455, .03, .515, .955)', // equal to 'easeInOutQuad'
				loop: true, // not taking effect, because end2end == true
				prev: arrows.eq(0),
				next: arrows.eq(1),
				autoplay: true,
				onSwitch: function(event, index) {
					thumbs.removeClass('current');
					thumbs.eq(index).addClass('current');
				},
				interval: 5,
				api: true // if set to true, Switchable returns API
			})

			var api = this.api;

			this.thumbs.each(function(index, a) {
				$(a).hover(function() {
					api.pause();

					var local_index = index;
					api.switchTo(local_index);
					api.play();
					return false;
				});
			});
		},

		adjust: function() {
			// adjust visual elements
			var sight = document.body.clientWidth;
			if (sight &gt; 1920) {
				sight = 1920;
			}

			this.container.find('ul &gt; li').andSelf().css('width', sight);

			var nearest_wrapper = this.container.children('ul');
			nearest_wrapper.css('width', sight * 3);

			// avoid black background
			if (this.api &amp;&amp; this.api.index != 0) {
				nearest_wrapper.css('left', '-' + (sight * this.api.index).toString() + 'px');
			}
		},

		onresize: function() {
			// resize on second time, third time and ...
			if (this.is_resizing === true) {
				clearTimeout(this.resize_defer);
			} else {
				this.is_resizing = true;
				this.api.pause();
			}

			this.resize_defer = setTimeout(function() {
				window.slider.after_resize();
			}, 300);

			this.adjust();
		},

		after_resize: function() {
			this.is_resizing = false;

			// 
			this.arrows.unbind('click');
			this.thumbs.unbind('click');

			this.enable(slider.api.index);
		}
	}

	slider.adjust();
	slider.enable(0);
	
	$(window).resize(function() {
		slider.onresize();
	});
	
	// NEWS
	var news_items = $('#wrapper &gt; div.homeNews &gt; div.homeNewsItem');
	news_items.children('a').click(function() {
		news_items.removeClass('expanded');
		$(this).parent().addClass('expanded');
		return false;
	});
});
