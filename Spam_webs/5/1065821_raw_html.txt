  //this same updateFreeShipping script is copied from and exist also in blockextendedcart/default-2013.html
  var updateFreeShipping = function() {
    var cartTotal = 0;
    // calculate cartTotal excluding pampers products
    var diapersPresent = false;
    var diapersMessage = "";

    var productRow =$(".cart_product_line");

    //adds all product values excluding pampers
    for(var i = 0; i &lt; productRow.length; i++){
      var productTitle = $(productRow[i]).find(".item&gt;a").text();
      var productValue = $(productRow[i]).find(".price &gt; .price").text();

      //check if row is pampers product
      var regex = /pampers/i;
      var isPampers = regex.test(productTitle);
      if(isPampers){
        diapersPresent = true;
        productValue = 0;
      } else {
        productValue = productValue.replace(/&nbsp;/g, ''); //Remove spaces
        productValue = productValue.replace(/[\s\.]/g, ''); //Remove spaces
        productValue = productValue.replace('',''); //Remove euro symbol
        productValue = productValue.replace('\,','\.'); //Replace decimal point
        productValue = parseFloat(productValue);
      }
      cartTotal = cartTotal + productValue;
    }

    if(diapersPresent){
      diapersMessage = "&lt;br/&gt;&lt;em class='fusnote'&gt; * Bl&#246;jor ing&#229;r inte i fri frakt-kampanjen.&lt;/em&gt;";
    } else {
      diapersMessage = "";
    }

    cartTotal = parseFloat(cartTotal);
    cartTotal = Math.floor(cartTotal);

    var untilFreeShipping = 1000 - cartTotal;
    var untilFreeshippingText;
    var untilFreeShippingWithCurrency;

    // adding currency for message
    if("kr" == 'eur' || "kr" == 'Euro' || "kr" == ''){
      untilFreeShippingWithCurrency = " " + untilFreeShipping;
    } else {
      untilFreeShippingWithCurrency = untilFreeShipping + " kr";
    }

    // check if it is free shipping or not
    if(untilFreeShipping &lt;= 0){
      untilFreeshippingText = "Du har nu fri frakt!";
    } else {
      untilFreeshippingText = untilFreeShippingWithCurrency + " kvar till fri frakt! " + diapersMessage;
    }

    $("#until-free-shipping-message").html(untilFreeshippingText);

  }

function UpdateSizeDependencies() {
  var selectedSize = $("#size-select");
  if (selectedSize.length == 0) return;

  sizeId				= $(selectedSize).val(),
    info				= selectedSize.attr('class').split("erpseparator", 2),
    number				= +info[0],
    erpId				= info[1],
    isBookable			= $("#product-is-bookable").val() == "true",
    maxQuantitySetting	= parseInt($("#setting-max-quantity").val());

  //---Set data for cart add handler---
  $('input#product-size-id').val(sizeId);
  $('input#product-size-erpid').val(erpId);

  //---Show correct button (buy/book/monitor)---
  if(number &gt; 0) {
    //Buy mode
    $("#buy-button").show();
    $("#book-button, #monitor-button, #block-monitor-product").hide();
  } else if(isBookable) {
    //Book mode
    $("#book-button").show();
    $("#buy-button, #monitor-button, #block-monitor-product").hide();
  } else {
    //Monitor mode
    $("#monitor-button").show();
    $("#buy-button, #book-button").hide();
    $("#block-monitor-product-size").val(sizeId);
  }

  //---Correct quantity options according to the number in stock---
  var quantityLimit = 0; //Monitor mode
  if(number &gt; 0) { //Buy mode
    quantityLimit = Math.min(number, maxQuantitySetting);
  } else if(isBookable) { //Book mode
    quantityLimit = maxQuantitySetting;
  }

  var quantitySelect = $("#quantity-select");
  $(quantitySelect).find("option").remove();

  if(quantityLimit == 0) {
    $("#quantity-select").parent().hide();
  } else {
    $("#quantity-select").parent().show();
    for(var i = 1; i &lt;= quantityLimit; i++) {
      $(quantitySelect).append("&lt;option value='" + i + "'&gt;" + i + "&lt;/option&gt;")
    }
  }
}

	var highligtTxt = function( txt, findTxt ){
		var spanTxt 	= "&lt;span class='highlight'&gt;",
			spanEndTxt 	= "&lt;\/span&gt;",
			txtLC = txt.toLowerCase(),
			findTxtLC = findTxt.toLowerCase(),
			indx = txtLC.indexOf( findTxtLC );

		if( indx!=-1 ){
			var findTxtLength = findTxt.length;
			txt = txt.substring( 0, indx ) + spanTxt + txt.substr( indx, findTxtLength ) + spanEndTxt + txt.substring( findTxtLength + indx, txt.length )
		}

		return txt;
	},
	itemsLoop = function(items, currentValue) {
		var html = '',brand,
			product;

		html = '&lt;ul&gt;&lt;li&gt;';

		// SUGGESTIONS
		var suggestions = items.suggestions;

		if(suggestions &amp;&amp; suggestions.length&gt;0) {

			}
			html+='&lt;/ul&gt;';
		}

		// CORRECTIONS
		

		// BRAND RESULTS 
		brands = items.brandResults;

		if(brands &amp;&amp; brands.length&gt;0) {

		// CATEGORY RESULTS

			html+='&lt;li class="heading"&gt;&lt;h2&gt;Kategorier (' + categories.length + ')&lt;/h2&gt;&lt;/li&gt;';

				html+='&lt;li class="link"&gt;';
