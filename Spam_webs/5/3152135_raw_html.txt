future of the next generation seems bleak.

    $('a[href^="/subscriptions/subscribe"]').click(function (e) {
        // check to see current page is not "subscriptions"
        var pattern = /^\/subscriptions/;
        if( ! pattern.test(window.location.pathname) ) {
            e.preventDefault();

            // check to see if there are any params on goto url
            var qpattern  = /\?/;
            var q = ( qpattern.test( $(this).attr('href') ) ) ? '&amp;' : '?';
            window.location = $(this).attr('href') + q +'cf=' + window.location.pathname;
        }
    });

    $('.account-trigger').click(function(e) {
        e.preventDefault();
        var mTitle = ($(this).attr('title')) ? $(this).attr('title') : 'My Account';
        //forums
        if ( $(this).hasClass('url-redirect') ){
            openAccountWorkflow('/accounts/usernameautogen?r=' + $(this).attr('href'), 'Create a user name', acmgt_dialog_close, show_close);
        } else {
            var url = $(this).attr('url');
            // switch to ssl
            if ( window.location.protocol == "http:" &amp;&amp; url.match(/^\/subscriptions/i) ) {
                var redirurl = "https://" + window.location.hostname + window.location.pathname + "#!" + url;
                window.location = redirurl;
                return false;
            }
            openAccountWorkflow(url, mTitle, acmgt_dialog_close, show_close);
        }
        return false;
    });

    var urlHash = window.location.hash;
    if ( urlHash ) {
        var path = urlHash.substring(2);
        if ( path.match(/^\/accounts/i) ) {
            var title = "";
            var verifyType = urlHash.split('/');
            switch(verifyType[3]){
                case 'cp':
                    title = "Create a free account";
                    // hide close button in upper right corner
                    show_close = false;
                    break;
                case 'ce':
                    title = "Edit your e-mail address";
                    close_redirect = true;
                    break;
                case 'n':
                    title = "Newsletter sign-up";
                    break;
                case 'e':
                    title = "Log in";
                    close_redirect = true;
                    break;
                case 's':
                    title = "Subscribe to &lt;em&gt;The Chronicle&lt;/em&gt;";
                    show_close = false;
                    break;
            }
            if ( path.match(/^\/accounts\/signupnewsletter/) ) {
                title = "Newsletter sign-up";

        $('a.ui-dialog-titlebar-close').hide();
    }
