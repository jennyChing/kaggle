   function loadAfterAd () {

                bWidth = $banner.find('iframe').width();

            if ( bWidth !== null &amp;&amp; bWidth &lt;= 728 ) {
                $after.empty();
                $after.dfp();
            }
        }
    }

        $('.adunit').each(function() {
            var $element = $(this);
            var $parent = $element.closest('.advertisement');
            if ($parent.length == 0 || $parent.css('display') !== 'none') {
                $element.addClass('visible');
            }
        });
        $('.adunit.visible').dfp({

                } else {

    /**
     * Add a script inside the interstitial div, that initialize
     * the slot to show the GPT ad; this function also save the
     * slot's instance in window, to refresh the ad every time
     * this function is called.
     * ad-gallery-container is the div's id
     */
    function interstitialAd(adunit, adsize, divId) {
        var gtag = document.createElement('script');
        gtag.type = 'text/javascript';
        gtag.innerHTML = 'googletag.cmd.push(function() {' +

                    + adsize[1] + '],"'+divId+'");' +
                'window.adInterstitial = slot1;' +
            '}' +
            'else{' +
                'slot1 = window.adInterstitial;' +
            '}' +
            'slot1.addService(googletag.pubads());googletag.'+
                'enableServices();googletag.display("'+divId+'");});';
        //add the script tag inside the div
        $('#'+divId).append(gtag);
        //refresh only the instance of the interstitial ad
