function selectSize(sizeId, elementId, name) {
    var size = document.getElementById(elementId);
    size.value = sizeId;
    var sizelabel = document.getElementById('defaultSize');
    if (sizelabel)
    {

    }
    
    var sizeBoxes = document.getElementById('sizeBoxes');
	if (sizeBoxes)
	{
		var liArray = sizeBoxes.childNodes;
		for(var i=0;i&lt;liArray.length ; i++){

function refreshSizeDropDown(sizeDropDownId, colorDropDownId) {
	sizeVisibleCount = 0 ; //reset the size selection count

	var sizeDropDown = document.getElementById(sizeDropDownId);
	var colorDropDown = document.getElementById(colorDropDownId);
	var colorVal = colorDropDown != null ? colorDropDown.options[colorDropDown.options.selectedIndex].value : '1';
	var attributeVal = 0;
	var attributeDropDown = document.getElementById('productAttribute');
	if(attributeDropDown != null)
	{
		var e = attributeDropDown.children[0];
		attributeVal = e.options[e.selectedIndex].value
	}
	var isFulfillmentElement = document.getElementById('isFulfillment');
	var isFulfillment =false;
	if(isFulfillmentElement != null){
		isFulfillment=isFulfillmentElement.value;
	}
	var productSizeArray = stockAvailabilityMap[colorVal][attributeVal];
	if (sizeDropDown != null &amp;&amp; sizeDropDown.options!=null) {
		if(sizeDropDown.options.length &gt; 0)
		{
			var defaultValue = sizeDropDown.options[sizeDropDown.options.selectedIndex].value;
		}else
		{
			var defaultValue=0;
		}
		sizeDropDown.options.length=0;
		var x;
		for (x in productSizeArray) {
			sizeDropDown.options[x] = new Option(productSizeArray[x].name, productSizeArray[x].id, productSizeArray[x].isDefault, productSizeArray[x].id==defaultValue);
			
			sizeVisibleCount++; //increment the number of sizes visible
		}
	}
	else if (sizeDropDown != null)
	{
		var sizeBoxes = document.getElementById('sizeBoxes');
		var previouslySelectedSize = sizeDropDown.value;
		if (sizeBoxes)
		{
			var liArray = sizeBoxes.childNodes;
			var i=0;
			var needToUpdateSize = true;
			var firstAvailableSize;
			for (i=0; i&lt;liArray.length; i++)
			{
				if(liArray[i].id != undefined)
				{
					var x;
					if (!isFulfillment)
					{
						liArray[i].style.display='none';
						for (x in productSizeArray)
						{
							if (productSizeArray[x].id== liArray[i].id)
							{
								liArray[i].style.display='block';
								if (!firstAvailableSize) { firstAvailableSize = i; }
								
								sizeVisibleCount++; //increment the number of colors
								
								break;
							}
						}
					} else {
						if (!firstAvailableSize) { firstAvailableSize = i; }						
					}
					if (liArray[i].style.display=='block' &amp;&amp; liArray[i].id == previouslySelectedSize)
					{
						needToUpdateSize = false;
					}
				}
			}
			if (needToUpdateSize &amp;&amp; firstAvailableSize &amp;&amp; firstAvailableSize &lt; liArray.length)
			{
				selectSize(liArray[firstAvailableSize].id, sizeDropDownId, liArray[firstAvailableSize].title);
			}
		}
	}
	//alert("sizeVisibleCount="+sizeVisibleCount);
}

function refreshAvailabilityMessage() {
	var sizeDropDown = document.getElementById('size_19355069');
	var colorDropDown = document.getElementById('color_19355069');
	var attributeVal = 0;
	var attributeDropDown = document.getElementById('productAttribute');
	if(attributeDropDown != null)
	{
		var e = attributeDropDown.children[0];
		attributeVal = e.options[e.selectedIndex].value
	}
	var colorVal = colorDropDown != null ? colorDropDown.options[colorDropDown.options.selectedIndex].value : '1';
	var sizeVal = 0;
	if(sizeDropDown != null &amp;&amp; sizeDropDown.options!= null &amp;&amp; sizeDropDown.options.length &gt; 0)
	{
		sizeVal = sizeDropDown.options[sizeDropDown.options.selectedIndex].value;
	}
	else if(sizeDropDown != null &amp;&amp; sizeDropDown.value != null &amp;&amp; sizeDropDown.value != '')
	{
	    sizeVal = sizeDropDown.value
	}
	
	if(true) {
		var availabilityMsg = document.getElementById("availabilityMessage");
		if (availabilityMsg != null) {
				var inStockAvailability = stockAvailabilityMap[colorVal][attributeVal];
				var outOfStockAvailability = outOfStockAvailabilityMap[colorVal][attributeVal];
				//set default value when there is no valid stock information
				if(inStockAvailability &amp;&amp; inStockAvailability.length == 0 &amp;&amp; outOfStockAvailability &amp;&amp; outOfStockAvailability.length == 0) {
					setAddToCartVisibility(true);
				}
				else {
						var selectionInStock = false;
		    			for (x in inStockAvailability) {
							if (inStockAvailability[x].id == parseInt(sizeVal)) {
									availabilityMsg.innerHTML = "Availability: " + inStockAvailability[x].stockAvailabilityMessage ;
									selectionInStock = true;
									availabilityMsg.style.color = "";
							}							
						}
					if(!selectionInStock) {
						for (x in outOfStockAvailability){
							if (outOfStockAvailability[x].id == parseInt(sizeVal)) {
									availabilityMsg.innerHTML = "Availability: " + outOfStockAvailability[x].stockAvailabilityMessage ;
									availabilityMsg.style.color = "red";
							}							
						}
				}
				setAddToCartVisibility(selectionInStock);
				}
		}
	}
	else
	{
		if (sizeVisibleCount != 0) {
		var key = createMessageMapKey(colorVal, sizeVal);
	} else {
		var key = createMessageMapKey(colorVal, 0);
	}
	//alert("sizeVisibleCount key="+key);
	//look it up
	if (stockMsgMap) {
		var stockOverride = stockMsgMap[key];
		var availabilityMsg = document.getElementById("availabilityMessage");
		if (availabilityMsg != null) {
			if (stockOverride != null) {
				availabilityMsg.innerHTML = "Availability: " + stockOverride.message;
			} else {
				availabilityMsg.innerHTML = "Availability: " + "In Stock | Most Orders Ship In 24 Hrs";
			}
			if ((stockOverride != undefined &amp;&amp; stockOverride != null &amp;&amp; !stockOverride.isAddToCartable) ||
			(stockOverride == null &amp;&amp; 'True' == 'False')) {
				availabilityMsg.style.color = "red";
				if (sizeDropDown)
				{
					sizeDropDown.disabled = true;
				}
			} else {
				availabilityMsg.style.color = "#666666";
				if (sizeDropDown)
				{
					sizeDropDown.disabled = false;
				}
			}
		}
		var addToCartSection = document.getElementById("addToCartSection");
		if (addToCartSection != undefined) {
			if (stockOverride != undefined &amp;&amp; stockOverride != null &amp;&amp; !stockOverride.isAddToCartable) {
				addToCartSection.style.visibility = "hidden";
			} else {
				addToCartSection.style.visibility = "visible";
			}
		}
	}
	}
}

				addToCartSection.style.visibility = "visible";
			}
		}
}
