var isFirstLoad = true;

 /* preload images */
function preload() {
    if (play){
        for ( var i = 0; i &lt;= frameCount; i++ ) {
            if (frameArray[i] != undefined &amp;&amp; frameArray[i] != "") {
                imgArray[i] = new Image();
                imgArray[i].src = frameArray[i];
                if(isFirstLoad){
                    isFirstLoad = false;
                    calculateCompletion();
                }
            }
            
        } 
    }else{
        play = true;
        animateNot360();
    }    
}



// Move to next image
function next() {
    current++; 
}

function previous() {
    current--; 
}



function calculateCompletion(){
   if(completion != 100){
        if( imgArray[current] == undefined || imgArray[current].complete ){
                completion = Math.round(current / frameCount * 100);
                document.getElementById('completion').innerHTML = completion + "%";
                next();
                calculateCompletion();
        }else{
            //console.log(current + " not ready");
            var x = setTimeout("calculateCompletion()", 10); 
        }
    }else{
       $("#completion").fadeOut();
       current = 1;
       if (is360) animate360(); else animateNot360();
    }   
}



function animateNot360(){
    try{
        if( current != imgArray.length ){
            next();
        }else{
            imgArray.reverse();
            current = 1;
        }
        frameArrayObject.src = imgArray[current].src;
        var t = setTimeout("animateNot360()", frameDuration);  
    }catch(err){
        if (is360) animate360(); else animateNot360();
    }

}

function reverse(){
    if(completion == 100){
        current = frameCount - current;
        imgArray.reverse();
    }
}


function animate360(){
    if (play){
        try{
            if( current != imgArray.length ){
                next();
            }else{
                current = 1;
            }
            frameArrayObject.src = imgArray[current].src;
            var t = setTimeout("animate360()", frameDuration);
        }catch(err){
            animate360();
        }
    }
}



function speedUp(speedIncrement) {
    frameDuration += speedIncrement;
    if (frameDuration &lt; 10) {
        frameDuration = 10;
    }

}

    frameCount = frameArray.length;

            play = false;

             $(".btn-play").css("display","block");
            return false; });

    }

        $(".btn-reverse").fadeIn().click(function () { reverse(); return false; });
    }
    else {
        $(".btn-reverse").fadeOut();
    }
    $(".btn-slow").click(function () { speedUp(20); return false; });
    $(".btn-fast").click(function () { speedUp(-20); return false; });


    /* show the stuff ! */
    $(".livepx-player").fadeIn(); 

});
