  	function inidrawWheel(lastRes) {
	//lastRes=5;
	clearCanvas(ctx);
		var mynumb=new Array();
		mynumb[1]=2; mynumb[2]=5; mynumb[3]=6; mynumb[4]=10; mynumb[5]=8; mynumb[6]=4; mynumb[7]=3; mynumb[8]=7; mynumb[9]=1; mynumb[10]=11;

			ctx.lineWidth = 2;

				ctx.fillStyle = colors[i];

				ctx.arc(65, 65, insideRadius, angle + arc, angle, true);
				ctx.stroke();

				ctx.shadowOffsetY = -1;

				ctx.rotate(angle + arc / 2 + Math.PI / 2);
				var text = numbers[i];
				//console.log(i+" "+ (angle + arc));

		}

	function wheelspin(spinforno) {
		//spinforno=10;
		//1:9,2:5,3:6,4:10,6:8,7:7,9:4,10:3,12:1,13:2
		var mynumb=new Array();
		mynumb[1]=12; mynumb[2]=13; mynumb[3]=10; mynumb[4]=9; mynumb[5]=2; mynumb[6]=3; mynumb[7]=7; mynumb[8]=6; mynumb[9]=15; mynumb[10]=4;
		spinAngleStart = mynumb[spinforno] + 10;
		spinTime = 0;
		spinTimeTotal = 3 * 1000;
		rotateWheel();
	}
  
	function rotateWheel() {
		spinTime += 30;
		if(spinTime &gt;= spinTimeTotal) {
			stopRotateWheel();
			return;
		}
		var spinAngle = spinAngleStart - easeOut(spinTime, 0, spinAngleStart, spinTimeTotal);
		startAngle += (spinAngle * Math.PI / 180);
		drawRouletteWheel();
		spinTimeout = setTimeout('rotateWheel()', 30);
	}
  
	function stopRotateWheel() {
		clearTimeout(spinTimeout);
		var degrees = startAngle * 180 / Math.PI + 90;
		var arcd = arc * 180 / Math.PI;
		var index = Math.floor((360 - degrees % 360) / arcd);
		ctx.save();
		ctx.font = 'bold 20px sans-serif';
		var text = numbers[index]
		ctx.fillText(text, 65 - ctx.measureText(text).width / 2, 65 + 10);
		ctx.restore();
		if(windowNotfirstLoad &amp;&amp; RemainTime&gt;0 &amp;&amp; !$("#aDiv").is(":visible") &amp;&amp; !$("#lossDiv").is(":visible"))
		{
			spin();
		}
	}
  
	function easeOut(t, b, c, d) {
		var ts = (t/=d)*t;
		var tc = ts*t;
		return b+c*(tc + -3*ts + 3*t);
	}
	
	function clearCanvas(context) {
		var canvas = document.getElementById("wheelcanvas");
		startAngle = 0;
		arc = Math.PI / 5;
		spinTimeout = null;
		spinArcStart = 10;
		spinTime = 0;
		spinTimeTotal = 0;
		context = canvas.getContext("2d");
		context.clearRect(0,0,130,130);
		context.clearRect(0, 0, canvas.width, canvas.height);
		var w = canvas.width;
		canvas.width = 1;
		canvas.width = w;
	}

	function pad(d) {
		return (d &lt; 10 &amp;&amp; d &gt; -1) ? '0' + d.toString() : d.toString();
	}

	var presentTime=0;
	var RemainTime=0;

	function countdown(){
		var spintime=0;
		var currentdate=new Date();
	
		if(RemainTime==0 || presentTime==0)
		{
			$.post("http://www.gomroo.com/index.php?route=module/PlayGame/getLastSpinTime",function(lastspintimedata){
				var rsltStr=$.parseJSON(lastspintimedata);
				if(rsltStr.num_rows&gt;0)
				{
					RemainTime=0;
					presentTime=currentdate.getTime()/1000;
					var lastspintimedata=new Date(rsltStr.row['date_time']*1000);
					var lastclienttime=lastspintimedata.getTime()/1000;
					RemainTime=60-parseInt(presentTime-Number(lastclienttime));
					if(RemainTime&lt;0){
						RemainTime=0;
					}
					
					if(windowfirstLoad)
					{
						inidrawWheel(rsltStr.row['result']);
						
					}
					else
					{
						clearCanvas(ctx);
						wheelspin(rsltStr.row['result']);
					}
				}
			});
		}
		else
		{
			presentTime=currentdate.getTime()/1000;
			if(RemainTime&gt;0){
				RemainTime=RemainTime-1;
				windowfirstLoad=0;
			}
			if(RemainTime &lt;= 10 &amp;&amp; RemainTime &gt;= 0){
				if($("#play").length)
				{	
					if($("#play").prop("disabled")==false)
					{
						$(".rad").children("div").hide();
						$("#Orange").removeAttr("checked");
						$("#Green").removeAttr("checked");
						$("#Odd").removeAttr("checked");
						$("#Even").removeAttr("checked");
					}				
					$("#play").prop("disabled",true);
					disable_rad();
				}
			}
			if(RemainTime &lt;= 0)
			{
				if($("#play").length)
				{
				$("#play").prop("disabled",false);
				enable_rad();
				}
			}
		}
	
		document.getElementById("timer").innerHTML="00:"+pad(Number(RemainTime));
		t=setTimeout(function(){countdown()},1000);
	}
