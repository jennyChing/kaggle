			var l = $(this).parent().position().left;

function validateMe(){
	var res = true;
	var patt = /^([0-9a-zA-Z]([-.\w]*[0-9a-zA-Z])*@([0-9a-zA-Z][-\w]*[0-9a-zA-Z]\.)+[a-zA-Z]{2,9})$/;
	var name = $('#formEmail').find('#name'), email = $('#formEmail').find('#email'),  message = $('#formEmail').find('#message'), phone = $('#formEmail').find('#phone');
	//alert(name.val());
	if(name.val() ==""){
		name.parent().find('.error').text('Please Enter Required Fields');	res = false; }
	else
		name.parent().find('.error').text('');	
	if(phone.val() ==""){
		phone.parent().find('.error').text('Please Enter Required Fields');	res = false; }
	else
		phone.parent().find('.error').text('');	
			
	if(email.val() ==""){
		email.parent().find('.error').text('Please Enter Required Fields');
		res = false; }
	else{
		email.parent().find('.error').text('');
		if(patt.test(email.val()) == false){
			email.parent().find('.error').text('Invalid Email address.');res = false;  }
		else
			email.parent().find('.error').text('');	
	}
	
	if(message.val() ==""){
		message.parent().find('.error').text('Please Enter Required Fields').show();	res=false; }
	else
		message.parent().find('.error').text('');	
	//alert(res);
	return res;
}
