function checkStrength(password){
  var err=0;
  //initial strength
    var strength = 0
    //length is ok, lets continue.
  //if length is 8 characters or more, increase strength value
  if (password.length &gt; 7) strength += 1
  //if password contains both lower and uppercase characters, increase strength value
  if (password.match(/([a-z].*[A-Z])|([A-Z].*[a-z])/))  strength += 1
  //if it has numbers and characters, increase strength value
  if (password.match(/([a-zA-Z])/) &amp;&amp; password.match(/([0-9])/))  strength += 1 
  //if it has one special character, increase strength value
    if (password.match(/([!,%,&amp;,@,#,$,^,*,?,_,~])/))  strength += 1
  //if it has two special characters, increase strength value
    if (password.match(/(.*[!,%,&amp;,@,#,$,^,*,?,_,~].*[!,%,&amp;,@,#,$,^,*,?,_,~])/)) strength += 1
  //now we have calculated strength value, we can return messages
  //if value is less than 2
  if(password.match(/([a-z\d])\1\1/i)){ 
    $('#status1').html('&lt;div class="errorMsg" &gt;&lt;i&gt;&lt;/i&gt;&lt;span&gt;3 consecutive characters not allowed!&lt;/span&gt;&lt;/div&gt;');
    err++;checkerr(err);
  }else if (strength &lt; 2 ) {
    $('#status1').html('&lt;div class="pwWMsg"&gt;&lt;i&gt;&lt;/i&gt;&lt;span&gt;Password could be more secure.&lt;/span&gt;&lt;/div&gt;');
  } else if (strength == 2 ) {
    $('#status1').html('&lt;div class="pwMMsg" &gt;&lt;i&gt;&lt;/i&gt;&lt;span&gt;Password is okay.&lt;/span&gt;&lt;/div&gt;');
  }else {
    $('#status1').html('&lt;div class="okMsg" &gt;&lt;i&gt;&lt;/i&gt;&lt;/div&gt;');
  }
}
