function windowshade_get_data(action,key,sku,qty, giftwrapremove) {
    var data = "action=" + action;
    var items = $('#windowshade_items').val().split("|");
    if (items.length &gt; 1) {
        for (i = 1; i &lt; items.length; i++) {
            details = items[i].split(";");
            if (details.length == 3) {
                if (details[1] == sku)
                    data += "&amp;sku" + details[0] + "=" + details[1] + "&amp;qty" + details[0] + "=" + qty;
                else
                    data += "&amp;sku" + details[0] + "=" + details[1] + "&amp;qty" + details[0] + "=" + details[2];
            }
        }
        if (giftwrapremove != "")
        	 data += "&amp;" + giftwrapremove + "=0";
    }
    if (data == "action=" + action)
        data = "";
    
    return data;
}
