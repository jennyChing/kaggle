                    <h3 class="post-title entry-title" itemprop="name">
                     Galactic Manicure
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-8758209726021729170" itemprop="articleBody">
                     <div class="separator" style="clear: both; text-align: center;">
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://4.bp.blogspot.com/-uHMQ5AGfWsE/T08-I1GIzpI/AAAAAAAAIbc/Y4L8CwS0aCk/s1600/galactic+manicure.jpg" imageanchor="1" style="">
                       <img border="0" src="http://4.bp.blogspot.com/-uHMQ5AGfWsE/T08-I1GIzpI/AAAAAAAAIbc/Y4L8CwS0aCk/s1600/galactic+manicure.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     I loved how my
                     <a href="http://www.itsbecauseithinktoomuch.com/search/label/nails">
                      glittery gradient mani
                     </a>
                     turned out so much that I decided to do a variation on it, using the night sky as my inspiration!
                     <br/>
                     <br/>
                     Here's how I did it:
                     <br/>
                     <br/>
                     <b>
                      1. Prep
                     </b>
                     <br/>
                     I grabbed
                     <a href="http://rstyle.me/d2637r8m">
                      deep navy
                     </a>
                     ,
                     <a href="http://rstyle.me/hz2tr8m">
                      silver glitter
                     </a>
                     , and
                     <a href="http://rstyle.me/kdcfkr8m">
                      gold glitter polish
                     </a>
                     along with
                     <a href="http://rstyle.me/wv4kr8m">
                      my favorite top coat
                     </a>
                     . I also grabbed an orange stick and some
                     <a href="http://rstyle.me/vgvtr8m">
                      nail art rhinestones
                     </a>
                     .
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://3.bp.blogspot.com/-2fVLhK-Bm8Q/Tj8PHnwIGHI/AAAAAAAADRg/9CljDgj7M0k/s1600/IMAG0948e.jpg" imageanchor="1" style="">
                       <img border="0" src="http://3.bp.blogspot.com/-2fVLhK-Bm8Q/Tj8PHnwIGHI/AAAAAAAADRg/9CljDgj7M0k/s640/IMAG0948e.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      2. Base
                     </b>
                     <br/>
                     Paint a base coat with the navy polish, reminiscent of the night sky!
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://4.bp.blogspot.com/-Pjgc45L779w/Tj8PJ8orsMI/AAAAAAAADRk/1qgq2F-2qrQ/s1600/IMAG0952e.jpg" imageanchor="1" style="">
                       <img border="0" src="http://4.bp.blogspot.com/-Pjgc45L779w/Tj8PJ8orsMI/AAAAAAAADRk/1qgq2F-2qrQ/s640/IMAG0952e.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      3. Sparkly Stars
                     </b>
                     <br/>
                     Swipe the smaller sized glitter polish on the top third of each nail.
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://2.bp.blogspot.com/-YyodvnTzkw0/Tj8PMQfmjYI/AAAAAAAADRo/ch2t3Oy2pb8/s1600/IMAG0953e.jpg" imageanchor="1" style="">
                       <img border="0" src="http://2.bp.blogspot.com/-YyodvnTzkw0/Tj8PMQfmjYI/AAAAAAAADRo/ch2t3Oy2pb8/s640/IMAG0953e.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      4. More Sparkly Stars
                     </b>
                     <br/>
                     Cover the top two-thirds of each nail with the larger sized glitter. Use the tip of the brush to place glitter along the very tip of each nail to deepen the gradient effect.
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://1.bp.blogspot.com/-eSsJvBl-8KE/Tj8POm4IeGI/AAAAAAAADRs/Nko6GEGqm9A/s1600/IMAG0954e.jpg" imageanchor="1" style="">
                       <img border="0" src="http://1.bp.blogspot.com/-eSsJvBl-8KE/Tj8POm4IeGI/AAAAAAAADRs/Nko6GEGqm9A/s640/IMAG0954e.jpg" width="90%"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                     </div>
                     <br/>
                     <br/>
                     <b>
                      5. Rhinestones
                     </b>
                     <br/>
                     Choose a two or three nails to accent with rhinestones.Wipe a thin layer of your top coat on the first nail in the spot you want to place the gem. Use the moistened point of your orange stick to pick up the gem and place it on the wet polish. Repeat with nail you wish to accent.
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://1.bp.blogspot.com/-Bk7NfRwB2tE/Tj8TSzdorII/AAAAAAAADSY/eWGNdlomeHU/s1600/IMAG0964.jpg" imageanchor="1" style="">
                       <img border="0" src="http://1.bp.blogspot.com/-Bk7NfRwB2tE/Tj8TSzdorII/AAAAAAAADSY/eWGNdlomeHU/s640/IMAG0964.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      6. Make it Last
                     </b>
                     <br/>
                     To secure the rhinestones and make the manicure last as long as possible, finish up with a coat of a good top coat. I recommend
                     <a href="http://rstyle.me/wv4kr8m">
                      Seche Vite
                     </a>
                     .
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://2.bp.blogspot.com/--xAFukxcjXs/Tj8PRmxDXBI/AAAAAAAADRw/mh21dy_X7_E/s1600/IMAG0958e.jpg" imageanchor="1" style="">
                       <img border="0" src="http://2.bp.blogspot.com/--xAFukxcjXs/Tj8PRmxDXBI/AAAAAAAADRw/mh21dy_X7_E/s640/IMAG0958e.jpg" width="90%"/>
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      7. Show it Off
                     </b>
                     <br/>
                     Go out and parade that out-of-this-universe mani around! Or snap a photo and send it my way! ;)
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://1.bp.blogspot.com/-NTqB17iFF_Q/Tj8PT3xAuiI/AAAAAAAADR0/3ttygrhBaQc/s1600/IMAG0956e.jpg" imageanchor="1" style="">
                       <br/>
                      </a>

            }
            comment.deleteclass = 'item-control blog-admin';
            if (entry.gd$extendedProperty) {
              for (var k in entry.gd$extendedProperty) {
                if (entry.gd$extendedProperty[k].name == 'blogger.itemClass') {
                  comment.deleteclass += ' ' + entry.gd$extendedProperty[k].value;
                } else if (entry.gd$extendedProperty[k].name == 'blogger.displayTime') {
                  comment.displayTime = entry.gd$extendedProperty[k].value;
                }
              }
            }
            comments.push(comment);
          }
        }
        return comments;
      };

      var paginator = function(callback) {
        if (hasMore()) {
          var url = config.feed + '?alt=json&amp;v=2&amp;orderby=published&amp;reverse=false&amp;max-results=50';
          if (cursor) {
            url += '&amp;published-min=' + new Date(cursor).toISOString();
          }
          window.bloggercomments = function(data) {
            var parsed = parse(data);
            cursor = parsed.length &lt; 50 ? null
                : parseInt(parsed[parsed.length - 1].timestamp) + 1
            callback(parsed);
            window.bloggercomments = null;
          }
          url += '&amp;callback=bloggercomments';
          var script = document.createElement('script');
          script.type = 'text/javascript';
          script.src = url;
          document.getElementsByTagName('head')[0].appendChild(script);
        }
      };
      var hasMore = function() {
        return !!cursor;
      };
      var getMeta = function(key, comment) {
        if ('iswriter' == key) {
          var matches = !!comment.author
              &amp;&amp; comment.author.name == config.authorName
              &amp;&amp; comment.author.profileUrl == config.authorUrl;
          return matches ? 'true' : '';
        } else if ('deletelink' == key) {
          return config.baseUri + '/delete-comment.g?blogID='
               + config.blogId + '&amp;postID=' + comment.id;
        } else if ('deleteclass' == key) {
          return comment.deleteclass;
        }
        return '';
      };

      var replybox = null;
      var replyUrlParts = null;
      var replyParent = undefined;

      var onReply = function(commentId, domId) {
        if (replybox == null) {
          // lazily cache replybox, and adjust to suit this style:
          replybox = document.getElementById('comment-editor');
          if (replybox != null) {
            replybox.height = '250px';
            replybox.style.display = 'block';
            replyUrlParts = replybox.src.split('#');
          }
        }
        if (replybox &amp;&amp; (commentId !== replyParent)) {
          document.getElementById(domId).insertBefore(replybox, null);
          replybox.src = replyUrlParts[0]
              + (commentId ? '&amp;parentID=' + commentId : '')
              + '#' + replyUrlParts[1];
          replyParent = commentId;
        }
      };

      var hash = (window.location.hash || '#').substring(1);
      var startThread, targetComment;
      if (/^comment-form_/.test(hash)) {
        startThread = hash.substring('comment-form_'.length);
      } else if (/^c[0-9]+$/.test(hash)) {
        targetComment = hash.substring(1);
      }

      // Configure commenting API:
      var configJso = {
        'maxDepth': config.maxThreadDepth
      };
      var provider = {
        'id': config.postId,
        'data': items,
        'loadNext': paginator,
        'hasMore': hasMore,
        'getMeta': getMeta,
        'onReply': onReply,
        'rendered': true,
        'initComment': targetComment,
        'initReplyThread': startThread,
        'config': configJso,
        'messages': msgs
      };

      var render = function() {
        if (window.goog &amp;&amp; window.goog.comments) {
          var holder = document.getElementById('comment-holder');
          window.goog.comments.render(holder, provider);
        }
      };

      // render now, or queue to render when library loads:
      if (window.goog &amp;&amp; window.goog.comments) {
        render();
      } else {
        window.goog = window.goog || {};
        window.goog.comments = window.goog.comments || {};
        window.goog.comments.loadQueue = window.goog.comments.loadQueue || [];
        window.goog.comments.loadQueue.push(render);
      }
    })();
// ]]&gt;
                     </script>
