Perfect for a weekday supper!&lt;/em&gt;&lt;/p&gt;
&lt;p style="text-align: center;"&gt;&lt;a href="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-2f.jpg"&gt;&lt;img class="aligncenter size-full wp-image-7435" src="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-2f.jpg" alt="Pierogies with Kiebasa and Cabbage" width="1200" height="1200"&gt;&lt;/a&gt;&lt;/p&gt;
&lt;h3 style="text-align: center;"&gt;Pierogies with Kiebasa and Cabbage&lt;/h3&gt;
&lt;p style="text-align: justify;"&gt;This is pure comfort food to me and I love that it can be on the table in 30 minutes! I&#8217;m sharing this today over at WeekdaySupper where all week we are sharing recipes for dinner that can be made in 30 minutes or less. This is so full of flavor and alsoeasy to make. I used the reduced calorie kielbasa.&lt;/p&gt;
&lt;p style="text-align: justify;"&gt;I start off by cooking the pierogies, while they are cooking, I add the kielbasa and cabbage to my Dutch oven pan. I let it cook/steam for about 15 minutes stirring it a couple of times. When the pierogies float it is time to drain them. When the cabbage is cooked I brown the pierogies quickly in a little butter and when brown add them to the pan with the kielbasa and cabbage.&lt;/p&gt;
&lt;p style="text-align: center;"&gt;&lt;a href="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-5f.jpg"&gt;&lt;img class="aligncenter size-full wp-image-7436" src="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-5f.jpg" alt="Pierogies with Kiebasa and Cabbage" width="1200" height="1200"&gt;&lt;/a&gt;&lt;/p&gt;
&lt;p style="text-align: justify;"&gt;When I&#8217;m not in a hurry I make my own &lt;a href="http://thatsmyhome.com/momsdiner/cheddar-and-potato-pierogies/" target="_blank"&gt;pierogies&lt;/a&gt;. They are so fun to make and you can change the fillings to whatever you desire. They can also be frozen. If you do I would place the uncooked pierogies on a cookie sheet in a single layer to freeze. Once they are frozen you can put them in a plastic bag. I&#8217;ve been making my own for years and I love the dough recipe I use. I usually make them around holiday time.&lt;/p&gt;
&lt;p style="text-align: center;"&gt;&lt;a href="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-6f.jpg"&gt;&lt;img class="aligncenter size-full wp-image-7437" src="http://recipesfoodandcooking.com/wp-content/uploads/2015/04/pierogies-cabbage-kielbasa-6f.jpg" alt="Pierogies with Kiebasa and Cabbage" width="1200" height="1200"&gt;&lt;/a&gt;&lt;/p&gt;

Author:&nbsp;&lt;span itemprop="author"&gt;Mary Ellen&lt;/span&gt;&lt;/div&gt; &lt;div class="ERSHead"&gt; Recipe type:&nbsp;&lt;span itemprop="recipeCategory"&gt;Main Dish&lt;/span&gt;&lt;/div&gt; &lt;div class="ERSClear"&gt;&lt;/div&gt; &lt;div class="ERSTimes"&gt; &lt;div class="ERSHead"&gt; Prep time:&nbsp; &lt;time itemprop="prepTime" datetime="PT10M"&gt;10 mins&lt;/time&gt; &lt;/div&gt; &lt;div class="ERSHead"&gt; Cook time:&nbsp; &lt;time itemprop="cookTime" datetime="PT20M"&gt;20 mins&lt;/time&gt; &lt;/div&gt; &lt;div class="ERSHead"&gt; Total time:&nbsp; &lt;time itemprop="totalTime" datetime="PT30M"&gt;30 mins&lt;/time&gt; &lt;/div&gt; &lt;/div&gt; &lt;div class="ERSClear"&gt;&lt;/div&gt; &lt;div class="ERSHead"&gt; Serves:&nbsp;&lt;span itemprop="recipeYield"&gt;serves 4&lt;/span&gt;&lt;/div&gt; &lt;div class="ERSClear"&gt;&nbsp;&lt;/div&gt; &lt;/div&gt; &lt;div class="ERSIngredients"&gt; &lt;div class="ERSIngredientsHeader ERSHeading"&gt;Ingredients&lt;/div&gt; &lt;ul&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 package frozen pierogies&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 package kielbasa (I used lite)&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 medium head of cabbage&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 onion - sliced&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;salt and pepper&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;&frac14; cup water&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 tablespoon oil&lt;/li&gt; &lt;li class="ingredient" itemprop="ingredients"&gt;1 tablespoon butter&lt;/li&gt; &lt;/ul&gt; &lt;div class="ERSClear"&gt;&lt;/div&gt; &lt;/div&gt; &lt;div class="ERSInstructions"&gt; &lt;div class="ERSInstructionsHeader ERSHeading"&gt;Instructions&lt;/div&gt; &lt;ol&gt; &lt;li class="instruction" itemprop="recipeInstructions"&gt;Put a pan large enough for the pierogies on the stove filled with water. Bring to a boil. Add the pierogies. When they start to float remove from heat and drain.&lt;/li&gt; &lt;li class="instruction" itemprop="recipeInstructions"&gt;Cut the kielbasa into pieces. Add with a little oil to the skillet.&lt;/li&gt; &lt;li class="instruction" itemprop="recipeInstructions"&gt;Cut the cabbage into large chunks and add to the pan with the kielbasa and onions. Add the water. Cover the pan and let the cabbage steam for 12 - 15 minutes.&lt;/li&gt; &lt;li class="instruction" itemprop="recipeInstructions"&gt;Brown the pierogies in a no stick pan with the butter. Brown on both sides and add to the pan with the cabbage and kielbasa.&lt;/li&gt; &lt;/ol&gt; &lt;div class="ERSClear"&gt;&lt;/div&gt; &lt;/div&gt; &lt;div class="endeasyrecipe" title="style001" style="display: none"&gt;3.2.2925&lt;/div&gt; &lt;/div&gt;

          removeErrorMessage(input);
        }

  function listsValidation() {
    var mainListsDiv = document.getElementById("signup_audience_lists");
    if(mainListsDiv &amp;&amp; mainListsDiv.className.indexOf("required") &gt;= 0) {

      for(var i = 0; i &lt; form.elements.length; ++i) {
        var input = form.elements[i];
        if(input.type != "checkbox") continue;
        if(input.checked) listSelected = true;
      }
      if(!listSelected) {
        if(mainListsDiv.className.indexOf("invalid") == -1) mainListsDiv.className += " invalid";
        isValid = false;
      } else {
        mainListsDiv.className = mainListsDiv.className.replace(/ invalid/g, "");
      }
    }
  }

  function errorMessage(input) {
    input.className   = "required invalid";
    input.placeholder = input.getAttribute("data-required-field");
  }

  function removeErrorMessage(input) {
    input.className   = "required";
    input.placeholder = "";
  }

  function updateFormAfterValidation() {
    form.className = isValid ? "" : "mimi_invalid";
    submit.value = submitButtonText();
    submit.disabled = !isValid;
    submit.className = isValid ? "submit" : "disabled";
  }

        text = submit.getAttribute("data-choose-list");
      } else {
        text = submit.getAttribute("data-invalid-text");
      }
    }
    return text;
  }

  function checkIfAllFieldsAreValid() {
    var allFieldsValid = true
    for(var i = 0; i &lt; form.elements.length; ++i) {
      var input = form.elements[i];
      if(input.placeholder &amp;&amp; input.placeholder.length &gt; 0) allFieldsValid = false;
    }
    return allFieldsValid;
  }

  function revalidateOnChange() {
    for(var i = 0; i &lt; form.elements.length; ++i) {
      var input = form.elements[i];
      if(input.className.indexOf("required") &gt;= 0 || input.type == "checkbox") {
        input.onchange = validate;
      }
    }
  }
