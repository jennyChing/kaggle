    function getPageXY(e)
    {
        var xPos = e.originalEvent.touches[0].pageX;
        var yPos = e.originalEvent.touches[0].pageY;
        return xPos+':'+yPos;

    function productQuickView(pid){
        var $product = jQuery('#product-' + pid);
        var $activeColorSwatch = $product.find('.list-colorswatch-active');
        var postData = {
            id: pid
        };
        if ($activeColorSwatch.length) {

                    success: function(data, textStatus, jqXHR){
                        jQuery.fancybox({
                            autoSize: true,
                            autoCenter: true,
                            content: data,
                            height: 455,
                            modal: false,
                            padding: 0,
                            scrolling: 'auto',
                            width: 700
                        });
                        jQuery('.cc-popup #qty').uniform();

                        if (swatchOptionId) {
                            jQuery('#attribute1034-quickview').val(swatchOptionId);
                        }
                        var colorLabel = jQuery('#attribute1034-quickview option:selected').text();

                        if (typeof(colorLabel) != 'undefined') {
                            colorLabel = colorLabel.toLowerCase();
                            jQuery('.colorswatch').each(function(index, swatch){
                                var $swatch = jQuery(swatch);
                                if ($swatch.attr('alt').toLowerCase() == colorLabel) {
                                    /**
                                     * Play nice with the existing Prototype code and fire a native click event on the
                                     * swatch link
                                     */
                                    var el = $swatch.get(0);
                                    if (jQuery.browser.msie &amp;&amp; parseInt(jQuery.browser.version) &lt; 9) { // IE
                                        (el.fireEvent('onclick'));
                                    } else { // Standard browsers
                                        var evObj = document.createEvent('Events');
                                        evObj.initEvent('click', true, false);
                                        el.dispatchEvent(evObj);
                                    }

                                    var event = jQuery.Event('click');
                                    event.target = $swatch.children('img').get(0);
                                    $swatch.parent().trigger(event);
                                }
                            });
                        }
                    }
                });
            }
        });
        return false;
    }

    function changeCatalogProductImage(configurable_product_id, simple_product_id, label){

        });
