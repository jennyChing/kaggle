         <a class="external" href="https://www.google.com/maps/preview#!data=!1m8!1m3!1d3!2d-0.192862!3d51.492132!2m2!1f291.8!2f71.14!4f75!2m4!1e1!2m2!1sc9UMhWP_MWm9U0L48xEjYw!2e0" onclick="_gaq.push(['_trackEvent', 'outbound-article', 'https://www.google.com/maps/preview#!data=!1m8!1m3!1d3!2d-0.192862!3d51.492132!2m2!1f291.8!2f71.14!4f75!2m4!1e1!2m2!1sc9UMhWP_MWm9U0L48xEjYw!2e0', 'going here']);" rel="nofollow" target="_blank" title="Doctor Who Google">
