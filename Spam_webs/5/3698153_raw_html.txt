       <h1 itemprop="headline name">
        Discover the health benefits of smiling
       </h1>
       <h2 itemprop="alternativeHeadline description">
        A smile can go a long way  especially when it comes to your health. Flashing 
  a timely grin will not only win over more friends, your mood and stress 
  levels will improve, too
       </h2>

          <p>
           <sup>

             <strong>
              Do More
             </strong>
            </a>
            <strong>
             by seeing, trying or learning something new:
            </strong>
           </sup>
          </p>
         </div>
         <div class="secondPar">
          <p>
           Everyone can fit a smile in to their day, no matter how packed the schedule. And smiling is good for you, even if you dont much feel like smiling.
          </p>
         </div>
         <div class="thirdPar">
          <p>
           Cognitive behavioural therapy suggests that our minds will often follow the lead of our bodies  so if you smile, youll assure your brain that everythings OK, thus improving your mood.
          </p>
         </div>
         <div class="fourthPar">
          <p>
           However, a big toothy grin might feel out of place on a rush-hour train, so follow the lead of the Buddhist practice of the half-smile: allow the corners of your mouth to upturn, stopping when you feel any tension. Focus on maintaining this for 10 minutes once a day  it will relax the muscles of your face, easing tension in both mind and body.
          </p>
         </div>
         <div class="fifthPar">
          <p>
           You will enjoy a pleasing shift in your mood, allowing you to meet the demands of the day calmly and positively.
          </p>
         </div>

          <p>

            for more self-improvement ideas. Produced in association with Hyundai, Do More aims to help you get more out of 2014 than you ever thought possible.
           </strong>
          </p>

            };

            var closeArticleContainer = function () {
                if (settings.panelOpen) {
                    $(settings.articleContainer).toggle('slide', {direction: 'left'}, function () {
                        //animation complete
                    });
                    settings.panelOpen = !settings.panelOpen;
                }
                return false;
            };


            var isScrolledIntoView = function(elem) {
                if($(elem).length) { //check if element exists
                    var docViewTop = $(window).scrollTop();
                    var docViewBottom = docViewTop + $(window).height();

                    var elemTop = $(elem).offset().top;
                    var elemBottom = elemTop + $(elem).height();

                    return ((elemBottom &lt;= docViewBottom) &amp;&amp; (elemTop &gt;= docViewTop));
                } else {
                    return false;
                }
            };

            var unique = function(array) {
                return array.filter(function (el, index, arr) {
                    return index == arr.indexOf(el);
                });
            }

            var createCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        articleListToSkip[0] = currentArticleId;
                        $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
                        return true;
                    }
                });
            }

            var updateCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        updateCookie(currentArticleId, articleListToSkip);
                        return true;
                    }
                });
            }

            var updateCookie = function(articleIdToSkip, articleListToSkip) {
                articleListToSkip.push(articleIdToSkip);
                articleListToSkip = unique(articleListToSkip);
                $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
            }

            var chooseNextArticle = function (currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if ((article.id != currentArticleId) &amp;&amp; ($.inArray(article.id, articleListToSkip) === -1)) {

                        //set article
                        settings.articleTitle = article.title;
                        settings.articleUrl = article.url;
                        settings.articleId = article.id;
						settings.articleImage = article.imgUrl;
                        return true;
                    }
                });
            }

            //public API
            return {
                settings: settings,
                init: init,
                display: display,
                openArticleContainer: openArticleContainer,
                closeArticleContainer: closeArticleContainer
            }
        })();

    };
