				menuDelay, menuOpen = true, overMenu = false;
			
			// animate global menu open &amp; reset auto-close timer
			function openMenu() {
				menuOpen = true;
				clearTimeout(menuDelay);
				$globalMenu.animate({
					height: 38
				}, 250, function() {
					$(this).css('overflow', 'visible'); // make sure overflow is visible post-animation
				}).css('overflow', 'visible'); // animate() hides overflow during animation, so undo that
				resetMenuDelay();
			}
			
			// animate global menu closed
			function closeMenu() {
				$globalMenu.animate({
					height: '1'
				}, 250, function() {
					menuOpen = false;
					$(this).css('overflow', 'hidden'); // make sure overflow is hidden post-animation
				}).css('overflow', 'visible'); // animate() hides overflow during animation, so undo that
			}
			
			// hide global bar after 3 seconds
			function resetMenuDelay() {
				clearTimeout(menuDelay);
				menuDelay = setTimeout(closeMenu, 2000);
			}
			
			// open global menu on scroll &amp; continuously reset auto-close timer
			$window.bind('scroll', function() {
				if (!menuOpen) {
					openMenu();
				} else {
					resetMenuDelay();
				}
			});
			
			// open global menu when mouse moves in lower third of viewport
			$window.bind('mousemove', function(e) {
				if ((e.pageY - $window.scrollTop()) &gt; (windowHeight - windowHeight / 3)) {
					if (!menuOpen) {
						openMenu();
					} else if (!overMenu) {
						resetMenuDelay();
					}
				}
			});
			
			// reset stored window height if resized
			$window.bind('resize', function() {
				windowHeight = $window.height();
			});
			
			// make sure global menu remains open when mouse is over it or any children
			$globalMenu.bind('mouseover', function() {
				overMenu = true;
				clearTimeout(menuDelay);
			}).bind('mouseleave', function() {
				overMenu = false;
			});

			resetMenuDelay();
