  function calculatePrice(qty, term){
    var remainingQty = qty;
    var levelQty = 0;

    var total = 0

    var i = 1;
    while(remainingQty &gt; 0){
      
      if(remainingQty &lt;= 5){
        levelQty = remainingQty
      } else {
        levelQty = 5
      }

      if(i == 1){
          if(term == 12) total += levelQty*38;
          if(term == 3) total += levelQty*14;
      } else
      if(i == 2){
          if(term == 12) total += levelQty*32;
          if(term == 3) total += levelQty*12;
      } else
      if(i &gt;= 3){
          if(term == 12) total += levelQty*26;
          if(term == 3) total += levelQty*10;
      }

      remainingQty = remainingQty - levelQty;
      i++;
    }
    return total;
  }

  function qtyCheck(qty){
    if(qty &gt; 15){

        return false;
    }
