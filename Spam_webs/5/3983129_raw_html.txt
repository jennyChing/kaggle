function getURLParam(str)
{
   var query=this.location.search.substring(1);
    if (query.length &gt; 0){
        var params=query.split("&amp;");
        for (var i=0 ; i&lt;params.length ; i++){
            var pos = params[i].indexOf("=");
            var name = params[i].substring(0, pos);
            var value = params[i].substring(pos + 1);
			if (name == str)
				return value;
        }
    }
	return -1;
}

{
	setHTML();

}			

function completeDone()
{
	qlocks[0].city_name = cityList[findIndex][0];
	qlocks[0].gmt_offset = cityList[findIndex][2];

	if (cityList[findIndex][3] &gt;= 0)
		dstSet = dstList[cityList[findIndex][3]];
	else
		dstSet = noDST;
		
	qlocks[0].dst_week1 = dstSet[0];
	qlocks[0].dst_dow1 = dstSet[1];
	qlocks[0].dst_month1 = dstSet[2];

	qlocks[0].dst_week2 = dstSet[3];
	qlocks[0].dst_dow2 = dstSet[4];
	qlocks[0].dst_month2 = dstSet[5];
	
	qlocks[0].update();
	setHTML();
}				

function findChange()
{
	setHTML();
}


function bgColorChange(ctl)
{
	var clr = new RGBColor(ctl.value);
	if (clr.ok)
	{
		qlocks[0].bg_color = clr.toHex();
		qlocks[0].update();
		setHTML();
	}
	
}

function textColorChange(ctl)
{
	var clr = new RGBColor(ctl.value);
	if (clr.ok)
	{
		qlocks[0].text_color = clr.toHex(); 
		qlocks[0].update();
		setHTML();
	}
}

{

	html += 'qlock_dst_week2='+qlocks[0].dst_week2+';\n';	
	html += 'qlock_dst_dow2='+qlocks[0].dst_dow2+';\n';
