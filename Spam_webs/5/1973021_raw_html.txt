		//declare addMarker function
		function addMapMarker(latitude, longitude, title, description, map, cat, z) {
			var latLng = new google.maps.LatLng(latitude, longitude);
			var contentString = '&lt;div style="width:200px; min-height:50px"&gt;' + '&lt;strong&gt;' + title + '&lt;/strong&gt;' + '&lt;p&gt;' + description + '&lt;/p&gt;' + '&lt;/div&gt;';
			
			var infowindow = new google.maps.InfoWindow({
				content: contentString,
					maxWidth: 400
			});
		
			var marker = new google.maps.Marker({
				position: latLng,
				title: title,
				category: cat,
				icon: cat,
				map: map,
				zIndex: z,
				draggable: false				
			});
			
			marker.myCategory = cat;
			marker.setZIndex(z);
			gmarkers.push(marker);
	
			google.maps.event.addListener(marker, 'click', function () {
				// close previously opened info window
				if (lastOpenInfoWindow) lastOpenInfoWindow.close();
					
				if (lastOpenInfoWindow != infowindow) {
					// open current infowindow
					infowindow.open(map, marker);
						
					// set handle to currently opened infowindow
					lastOpenInfoWindow = infowindow;
				} else {
					lastOpenInfoWindow = null;
				}
			});	
		}

		//shows all markers of a particular category, and ensures the checkbox is checked ==
		function show(category) {
			for (var i=0; i&lt;gmarkers.length; i++) {
				if (gmarkers[i].myCategory == category) {
					gmarkers[i].setVisible(true);
				}
			}
		}

		//hides all markers of a particular category, and ensures the checkbox is cleared ==
		function hide(category) {
			for (var i=0; i&lt;gmarkers.length; i++) {
				if (gmarkers[i].myCategory == category) {
					gmarkers[i].setVisible(false);
				}
			}
		}

		function findAddress(address) {
			var addressStr=document.getElementById("searchState").value;
			if (!address &amp;&amp; (addressStr != '')) 
				address = "State of "+addressStr;
			else 
				address = "US";
				
			if ((address != '') &amp;&amp; geocoder) {
				geocoder.geocode( { 'address': address}, function(results, status) {
					if (status == google.maps.GeocoderStatus.OK) {
						if (status != google.maps.GeocoderStatus.ZERO_RESULTS) {
							if (results &amp;&amp; results[0]
								&amp;&amp; results[0].geometry &amp;&amp; results[0].geometry.viewport) 
								map.fitBounds(results[0].geometry.viewport);
						}
					} else {
						console.log("Geocode was not successful for the following reason: " + status);
					}
				});
			}
		}

   function getParameterByName(name) {
    name = name.replace(/[\[]/, "\\[").replace(/[\]]/, "\\]");

        results = regex.exec(location.search);
    return results == null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));
}
