                   <div class="post hentry" itemprop="blogPost" itemscope="itemscope" itemtype="http://schema.org/BlogPosting">
                    <meta content="http://upload.wikimedia.org/wikipedia/commons/thumb/4/4a/Roastbeef_with_yorkshire_puddings.jpg/702px-Roastbeef_with_yorkshire_puddings.jpg" itemprop="image_url"/>

                    <h3 class="post-title entry-title" itemprop="name">
                     Rosemary Roasted Potatoes
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-6079309738966453671" itemprop="description articleBody">
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <strong>
                       </strong>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <img alt="File:Roastbeef with yorkshire puddings.jpg" height="341" src="http://upload.wikimedia.org/wikipedia/commons/thumb/4/4a/Roastbeef_with_yorkshire_puddings.jpg/702px-Roastbeef_with_yorkshire_puddings.jpg" width="400"/>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <strong>
                        <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                         What you need:
                        </span>
                       </strong>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                        2 pounds red potatoes
                       </span>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                        3 tablespoons olive oil
                       </span>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                        2 tablespoons dried Rosemary
                       </span>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                        <br/>
                       </span>
                      </div>
                     </div>
                     <div style="border-bottom: medium none; border-left: medium none; border-right: medium none; border-top: medium none;">
                      <div style="text-align: center;">
                       <span style="font-family: Georgia, Times New Roman, serif; font-size: large;">
                        Preheat oven to 350. Wash potatoes and peel a narrow strip around centers. Place potatoes in sprayed 9x13 pan. Drizzle with olive oil and sprinkle rosemary over the potatoes. Stir to coat potatoes. Bake uncovered for 1 hour or until the skins are crispy and the potatoes are fork tender.
                       </span>
                      </div>
