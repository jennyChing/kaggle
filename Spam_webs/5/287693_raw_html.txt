        this.params.dc = 198;
        break;
      case 'la-romana':
        this.params.id_albergo = 10580;
        this.params.dc = 427;
      break;
      case 'punta-cana':
        this.params.id_albergo = 10581; 
        this.params.dc = 139;
        break;
      case 'palm-beach':
        this.params.id_albergo = 10582; 
        this.params.dc = 517;
        break;
      case 'huatulco':
        this.params.id_albergo = 10583; 
        this.params.dc = 549;
        break;
      case 'los-cabos':
        this.params.id_albergo = 10584; 
        this.params.dc = 126;
        break;
      case 'villamagna':
        this.params.id_albergo = 10586;
        this.params.dc = 538;
        break;
      case 'sands':
        this.params.id_albergo = 12069;
        this.params.dc = 9602;
        break;
      case 'sugar-bay':
        this.params.id_albergo = 12361;
        this.params.dc = 2534;
        break;
    }
  } 

  this.setRange = function (from,to) {
    var nights = Math.max(1,Math.round((to-from)/ONE_DAY));
    this.params.notti_1 = nights;
    this.params.aa = from.getFullYear();
    this.params.mm = from.getMonth()+1;
    this.params.gg = from.getDate();
  }

  this.submit = function() {
  
    if(this.params.id_albergo &gt; 0) {  
        var url = 'https://booking.dreamsresorts.com/reservation_hotel.htm',s = '';
    } else {
        var url = 'https://booking.dreamsresorts.com/reservation_gruppo.htm',s = '';
        this.params.id_gruppo = 9437; 
        this.params.dc_gruppo = 1151; 
        this.params.mostra_loc = "no"; 
        this.params.id_brand = 4; 

    }
  jq("form",this.b).find('.params').remove();
    if (1) {
      for (var i in this.params) { 
        //console.log(i + " : " + this.params[i]);
        // s += (s==''?'?':'&amp;') + i+'='+this.params[i];
    jq("form",this.b).append('&lt;input type="hidden" name="' + i + '" value="' + this.params[i] + '"  class="params" /&gt;');
      }  
    }
    jq("form",this.b).attr({'action': url, 'method': 'POST'}).submit();
  }

  this.getDate = function (id) {
    var date = new Date(jq(id,this.b).attr('value'));
    return 'Invalid Date'==date || isNaN(date) ? null:date;
  }

  this.setDate = function (id,v) {
    jq(id,this.b).attr('value',jq.datepicker.formatDate("mm/dd/yy",v)); 
  }
  
  this.fade = function (v) {
    if (that.options &amp;&amp; that.options.index) 
      return {opacity:v?1:0,'z-index':v?1:-1};
    else  
      return {opacity:v?1:0,'z-index':0};
  }

  this.show = function(id,on,time) {
    var el = jq(id,that.b);
    if (on) el.css('display','inline-block');
    if (on) el.animate(that.fade(on), time, function() { el.css('display','inline-block') } ); 
      else  el.animate(that.fade(on), time, function() { el.css('display','none') } ); 
  }

  this.update = function (tip) {
    setTimeout(function() {
      var from = that.getDate('.booking_depart');
      var to   = that.getDate('.booking_return');
 
      var v_sel  = that.sel ? jq(that.sel,that.b).attr('data-val'):'';

      }
     // var dimmed = 'RESORTALIAS';
      if (from &amp;&amp; to) {
        if (from&gt;to) that.setDate('.booking_return',to = from); 
        var nights = Math.max(1,Math.round((to-from)/ONE_DAY));
    
        if (nights&gt;maxNights) {
          nights = maxNights;
          to = new Date(from.getTime() + nights*ONE_DAY);
          that.setDate('.booking_return',to); 
        }
      }
  
      var time = tip===0 ? 0:300;
      if (tip!=null || !b_from) that.show(".booktip_depart",b_from, time);
      if (tip!=null || !b_to  ) that.show(".booktip_return",b_to  , time);
      if (tip!=null || !b_sel ) that.show(".booktip_sel"   ,b_sel , time);
  
      jq(".booking_depart",that.b).css(           b_from? {border:'1px solid white'}:{border:""});
      jq(".booking_return",that.b).css(           b_to  ? {border:'1px solid white'}:{border:""});
      if (that.seltip) jq(that.seltip,that.b).css(b_sel ? {border:'1px solid white'}:{border:""});
    },0);
  }
  
  jq('.booktip_depart, .booktip_return, .booktip_sel',this.b).die().live('hover', function(){ that.show(this,0,300) })
  jq(".booking_bar_icon",this.b).live('click', function() { jq(this).prev().trigger('click'); });
  
  jq(".booking_bar_date",this.b).each(function() {
    jq(this).datepicker('destroy');  
  });
  
  this.update(0);

    if (!that.sel) jq.datepicker.dpDiv.css({'bottom':'auto'});

          jq(this).datepicker('destroy');

      beforeShow: function(input, inst) {
              
          if (!that.sel || that.options &amp;&amp; that.options.index) return;
          jq.datepicker._pos = jq.datepicker._findPos(input);
          jq.datepicker._pos[1] -= jq(document).scrollTop()+225;
              
      }

        jq(this).attr('selected', 'selected')
      }
    });
  }


  jq(".booking_btn.air",this.b).die().live('click', function() {
    that.check = false;
    that.update();
    if (!that.dimmed) {
      jq("form",that.b).attr('action','https://partners.kcdataservices.com/amr/custom/buildMyOwn.jsp?agn_id=67296').submit();
