            var topSpace = 40;

            var bottomMargin = 0;

            var getColumnHeight = function ()
            {
                var height = 0;
                var totalVisibleDivs = $thinColumn.children('div').not(':hidden');
                var divs = totalVisibleDivs.each(function (index, value)
                {
                    height += $(this).outerHeight(true);
                });

                topMargin = parseInt($(totalVisibleDivs[0]).css('margin-top'), 10);
                bottomMargin = parseInt($(totalVisibleDivs[$(totalVisibleDivs).length - 1]).css('margin-bottom'), 10);

                return height;
            };

            $window.on('scroll resize ', function ()
            {
                var columnHeight = getColumnHeight();
                var columnOffsetTop = scrollTop($primaryThinColumn);
                var left = scrollLeft($primaryThinColumn) + parseInt($primaryThinColumn.css('padding-left'), 10);

                if (columnOffsetTop - topSpace &lt; 0)
                {
                    $thinColumn.css({ 'position': 'fixed', 'top': (topSpace - topMargin) + 'px', 'left': left + 'px' });
                    $chatDiv.css({ 'left': left + 'px' });
                }

                var footerOffsetTop;
                if ($footer.length &gt; 0)
                {
                    footerOffsetTop = scrollTop($footer);
                }
                else
                {
                    footerOffsetTop = columnOffsetTop + $primaryThinColumn.height();
                }

                if (footerOffsetTop - columnHeight - bottomSpace - topSpace + bottomMargin &lt; 0)
                {
                    $primaryThinColumn.css({ 'position': 'relative' });
                    $thinColumn.css({ 'position': 'absolute', 'top': $footer.offset().top - $primaryThinColumn.offset().top - columnHeight - bottomSpace + bottomMargin, 'left': 'auto' });
                    $chatDiv.css({ 'top': '40px' });
                }
                else if (columnOffsetTop &gt; topSpace)
                {
                    $thinColumn.css({ 'position': 'relative', 'top': 'auto', 'left': 'auto' });
                    $chatDiv.css({ 'top': 'auto' });
                }
            });
        }
