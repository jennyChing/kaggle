function removeHtmlTag(strx, chop) {
    var s = strx.split("&lt;");
    for (var i = 0; i &lt; s.length; i++) {
        if(s[i].indexOf("&gt;")!==-1){
            s[i] = s[i].substring(s[i].indexOf("&gt;") + 1, s[i].length);
        }
    }
    s = s.join("");
    s = s.substring(0, chop - 1);
    return s;
}
