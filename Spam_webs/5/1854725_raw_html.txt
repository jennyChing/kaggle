function FixPosStatue(element, value)
{
    $(element).each(function()
    {
        $(window).scroll(function()
        {
            var offset = $('.bg-temple').offset();
            var elementHeight = $(this).height();
            var calcTop = offset.top - (elementHeight / 3);

            if ($(this).scrollTop() &gt; calcTop)
                $(element).css({ 'position': 'absolute', 'top': 'initial', 'bottom': '10%' });
            else
                $(element).css({ 'position': 'fixed', 'top': '30%' });
        });

        $(window).load(function()
        {
            var offset = $('.bg-temple').offset();
            var elementHeight = $(this).height();
            var calcTop = offset.top - (elementHeight / 3);

            if ($(this).scrollTop() &gt; calcTop)
                $(element).css({ 'position': 'absolute', 'top': 'initial', 'bottom': '10%' });
            else
                $(element).css({ 'position': 'fixed', 'top': '30%' });
        });

        $(window).resize(function()
        {
            var offset = $('.bg-temple').offset();
            var elementHeight = $(this).height();
            var calcTop = offset.top - (elementHeight / 3);

            if ($(this).scrollTop() &gt; calcTop)
                $(element).css({ 'position': 'absolute', 'top': 'initial', 'bottom': '10%' });
            else
                $(element).css({ 'position': 'fixed', 'top': '30%' });
        });
    });
}
