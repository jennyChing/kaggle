	function addEmailSuccess(responseText, statusText, xhr, $form) {
		// A session timeout returns a single json object having status=session_expired
		var status = responseText.status,
			$html = $('html');

					fireTealiumEvent({

						"lead_type" : "email header popup phone signup"

				}

					fireTealiumEvent({

						"lead_type" : "email header popup signup"
					});
				}
				$html.removeClass('first-visit');
				$('#emailSubscribeAddressModal').val('');
			}
			else {
				if ($html.hasClass('small-screen')) {
					fireTealiumEvent({
						"e_header_email_p" : "t",
						"lead_type" : "email header phone signup"
					});
