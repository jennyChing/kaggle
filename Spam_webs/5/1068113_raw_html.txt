  <script type="text/javascript">
   $('.hideFooter').click(function(){
    $('#stickyFooter').hide();
  });

  $('form#emailFootSubmit').submit(function(e) {
    //get form data email_address
    var formData = $(this).serialize();
    //add loading gif 
    $('#submitFooter').slideUp();
    //$('#displayFootError').slideUp();
    $('#displayFootLoading').show();
      //send ajax call
      $.ajax({
        url: '/ajax/emailpop_join_scroll.php',
        data: formData,
        type: 'POST',
        //handle successful ajax call
        success: function(data) {
          //hide loading gif
          $('#displayFootLoading').slideUp();
          //display thank you message for new entry
          data = $.trim(data);
          if (data == "created") {
             _gaq.push(['_trackEvent', 'Email_Overlay_Sitewide', 'Sticky Footer Email Created', 'Sticky Footer Email Test_814']);
            $('#displayFootThankYou').show();
          }
          //display already exists message for duplicate entry
          if (data == 'exists') {
            _gaq.push(['_trackEvent', 'Email_Overlay_Sitewide', 'Sticky Footer Email Exists', 'Sticky Footer Email Test_814']);
            $('#displayFootExists').show();
          }
          //handle error and give option to retry
          if (data == 'error') {
            _gaq.push(['_trackEvent', 'Email_Overlay_Sitewide', 'Sticky Footer Email Error', 'Sticky Footer Email Test_814']);
            $('#displayFootError').show();
          }
        }
      });
    return false;
  });
  </script>
