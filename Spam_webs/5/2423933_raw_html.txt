    var ajaxurl = "http://www.yummly.com/dish/wp-admin/admin-ajax.php";

    function yummly_load_more_labs(post_type,skip,limit){

        var data = {
            "action":"yummly_get_labs_posts",
            "post_type":post_type,
            "skip":skip,
            "limit":limit
         };

        $.post(ajaxurl,data,function(result){
            var $load_more = $("#load-more");
            $load_more.toggleClass("loading");
            var result = JSON.parse(result);
            var handlebars_template;

            //console.log(result);
            /*
            *
            * when you get results back you need to
            * compare the result.count value
            * and the length of a $("WHATEVER").length of the items
            * already in the list to see if it is time to hide the
            * load more
            *
            */

            handlebars_template = post_type == "labs" ? labs_template : post_template;

            var len = result.posts.length;
            for (var i = 0; i &lt; len; i++) {
                var html = handlebars_template(result.posts[i]);
                $("#post-list").append(html);
            }

            var count;
            if(post_type == "post"){
              count = $(".post").length;
            }else{
              count = $("#post-list &gt; li").length;
            }
            if($(".labs").length &gt; 0){
                count += 1;
            }
            if(count &gt;= result.total){
                $load_more.hide();
            }

        });
    }

    function yummly_load_more_posts(skip,limit,taxonomy,handlebars_template,$container){
      skip = parseInt(skip);
      var data = {
          "action":"yummly_get_posts",
          "skip":skip,
          "limit":limit,
          "post_type":"post"
       };

      if(taxonomy.hasOwnProperty("type")){
        data.taxonomy_type = taxonomy.type;
        data.taxonomy_slug = taxonomy.slug;
      }
      //console.log(data);
      //console.log("Outgoing data____________________");
      //console.log(data);
      $.post(ajaxurl,data,function(result){
        // console.log(result);
        var result = JSON.parse(result);
        var len = result.posts.length;
        //console.log("Container____________________");
        // console.log($container.html());
        $container.html(" ");
        $('#load-more-posts').toggleClass("loading");

        for (var i = 0; i &lt; len; i++) {
          var html = handlebars_template(result.posts[i]);
          $("#load-more-container").append(html);
          $container.append(html);
        }

        var count = $(".post").length;
        if(count &gt;= parseInt(result.total)){
            $("#load-more-posts").hide();
        }
        $("#load-more-container .post").each(function( index ) {
            if ($(this).find('img').attr('src') == "") {
              $(this).find('img').attr('src', styleUri+"/img/post-placeholder.png");

            }
        });
        // console.log(result);
      })
    };
