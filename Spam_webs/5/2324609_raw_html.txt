   function setGoogleTracking() {

  var field_label = false,
      $self       = false;

  $('.start-button').bind( 'click', function() {

    $self = $(this);

    if ( $self.closest('#header').length ) {
      field_label = 'header';
    }
    else if ( $self.closest('#intro-title').length ) {
      field_label = 'body';
    }
    else if ( $self.closest('.sign-up-footer').length ) {
      field_label = 'footer';
    }

    if ( typeof _gaq !== 'undefined' ) {
      _gaq.push(['_trackEvent', 'Sign up', 'Sign up', field_label]);
    }

  });


} // setGoogleTracking

setGoogleTracking();
