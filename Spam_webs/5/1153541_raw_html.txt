	function sendIosLink(obj) {

		var valueDefault = phone.attr('data-default-value');

		val = phone.val();

		if (typeof (val) == "undefined" || val == "" || val == valueDefault || val == "(___) ___-____") {
			errorMessage.show();
			errorMessage.text(valueDefault);
			phone.addClass('error');
			return;
		} else {
			phone.removeClass('error');
			val = phone.val().trim().replace(/[^\d.]/g, "");
			var agentid = 0;
			if ($j.getQueryString) {
				var qauid = $j.getQueryString("a");

				contentType: "application/json; charset=utf-8",

				processData: false,
				success: function (response) {

						var confirm = subscribe.siblings('div.confirmation');

				},
