      <p>
       <span style="font-size: 10pt;">
        LOUISVILLE, Ky.  Whats in a name? Money, for one thing. Especially if you have a Kentucky Derby starter named Itsaknockout, who will go to post just hours ahead of the richest boxing match of all time.
       </span>
      </p>
      <p>
       This Louisville-based Starlight Racing-owned colt had one of his more disappointing races of the season, beaten by 21 lengths by his Todd Pletcher-trained stablemate Materiality in the Florida Derby the previous time out.
      </p>
      <p>
       But Pletcher said the colt has trained well in advance of the Derby, and for his owners, circumstances made running him in the race a marketing no-brainer.
      </p>
      <p>
       He might face long odds in the race, but his sponsorship deal guarantees he will get plenty of pre-race attention.
      </p>
      <p>
       Jack Wolf, founder and co-managing partner of Starlight Racing, reached out to some partners in Starlight about using the colts name to strike some kind of sponsorship with promoters for the welterweight fight between Floyd Mayweather Jr. and Manny Pacquiao, which will take place Saturday in Las Vegas with a purse expected to be in excess of $200 million.
      </p>
      <p>
       The result of those discussions could be seen draped over Itsaknockouts back after a workout Friday morning, a blanket that bears the names Mayweather and Pacquiao and the message live on pay-per-view.
      </p>
      <p>
       Wolf chuckled about it on the backside.
      </p>
      <p>
       Im the luckiest guy going, he said. Id rather be lucky than good.
      </p>
      <p>
       Terms of the marketing deal have not been disclosed. In addition to the blanket, jockey Luis Saez will wear pants that have Mayweather on one leg and Pacquiao on the other.
      </p>
      <p>
       Not bad for a colt whose name was a bit of a mystery until recently. According to Drew Deener, morning host for Louisvilles ESPN 680 AM, a bunch of the owners were having dinner after Itsaknockout won the Fountain of Youth at Gulfstream Park and started talking about his name.
      </p>
      <p>
       Nobody knew how he got the name, Deener said. It was funny, because everyone was asking, Did you name him? I didnt name him, did you?
      </p>
      <p>
       The answer to the name came from Wolf, who said his wife Laurie named him at the last minute because she thought the name of his sire, Lemon Drop Kid, sounded like a boxers nickname.
      </p>
      <p>
       Wolf said he hasnt been a huge fan of boxing, But I am now  Whats the line on that? Is Mayweather favored? I dont know.
      </p>
      <p>
       The high-definition pay-per-view price tag on the fight is $100 million. Wolf said he expects to have some people over to watch after the Derby.
      </p>
      <p>
       And thanks to a hastily but fortuitously picked name, he will have a lot more eyes on his colt when it goes to post. Another reminder that you never know what will draw the publics attention on Derby Day  or why.
      </p>

      function xmlToJson(xml) {
		// By dw
		// Create the return object
		var obj = {};

		if (xml.nodeType == 1) { // element
			// do attributes
			if (xml.attributes.length &gt; 0) {
			obj["@attributes"] = {};
				for (var j = 0; j &lt; xml.attributes.length; j++) {
					var attribute = xml.attributes.item(j);
					obj["@attributes"][attribute.nodeName] = attribute.nodeValue;
				}
			}
		} else if (xml.nodeType == 3) { // text
			obj = xml.nodeValue;
		}

		// do children
		if (xml.hasChildNodes()) {
			for(var i = 0; i &lt; xml.childNodes.length; i++) {
				var item = xml.childNodes.item(i);
				var nodeName = item.nodeName;
				if (typeof(obj[nodeName]) == "undefined") {
					obj[nodeName] = xmlToJson(item);
				} else {
					if (typeof(obj[nodeName].push) == "undefined") {
						var old = obj[nodeName];
						obj[nodeName] = [];
						obj[nodeName].push(old);
					}
					obj[nodeName].push(xmlToJson(item));
				}
			}
		}
		return obj;
	}

 	// Check if node of multidimensional array exists
 	// Example call: issetmulti(array, "items", "myitem", "title")
	function issetmulti(arr) {
	    var i, max_i;
	    for (i = 1, max_i = arguments.length; i &lt; max_i; i++) {
	        arr = arr[arguments[i]];
	        if (arr === undefined) {
	            return false;
	        }
	    }
	    return true;
	}

						itemcountbystyle["Has Teaser"] = 0;
						itemcountbystyle["Headline Only"] = 0;
					var newstories = $();
					var elemsformasonry = [];
					iscroll.logstar("Iterating over each item.")
					iscroll.log(jsondata);
					$.each(iscroll.stories, function(i, item) {	// Cycle through each item
						var newstory = $('&lt;a class="recent-story"&gt;');
						newstory.attr("href", item["link"]["#text"]);
						newstory.append($('&lt;span class="recent-story-title"&gt;' + item["title"]["#text"] + '&lt;/span&gt;'));
						 if (issetmulti(item, "images", "image", "link", "#text")) {
								var imagelink = item["images"]["image"]["link"]["#text"];
								newstory.attr("data-itemstyle", "Has Image");
								newstory.append($('&lt;img src="' + imagelink + '"/&gt;'));
								itemstyle = "Has Image";
						} else if (typeof item["teaser"] !== "undefined"
							&amp;&amp; typeof item["teaser"]["#text"] !== "undefined") {
							newstory.append($('&lt;span class="recent-story-teaser"&gt;' + item["teaser"]["#text"] + '&lt;/span&gt;'));
							itemstyle = "Has Teaser";
						} else {
							// newstory.append($('&lt;span class="recent-story-teaser"&gt;' + item["title"]["#text"] + '&lt;/span&gt;'));
							newstory.addClass("recent-story-no-teaser");
							itemstyle = "Headline Only";
						}
						itemcountbystyle[itemstyle]++;
						newstory.attr("data-itemstyle", itemstyle); // Used in analytics event when story clicked
						// $(".recent-stories").append(newstory);
						newstories = newstories.add(newstory);
						elemsformasonry.push(newstory);
					});
					iscroll.logstar("Done iterating")
					iscroll.logstar("Waiting for images to load...");
			
					newstories.imagesLoaded()
						.always( function() {
							iscroll.logstar("Always fired");
						})
						.done( function() {
							iscroll.logstar("Images loaded");
							iscroll.refreshMasonry(newstories, elemsformasonry);
							iscroll.logstar("Masonry Refreshed");
							// $(document).scrollTop( iscroll.scrolltop.value );
							iscroll.fireItemStyleTrackingEvents(itemcountbystyle);
							// iscroll.seeIfNewStoriesLoadedYet();
							iscroll.spinner.hide();
							iscroll.alreadyloading=false;
							iscroll.logstar("Checking to see if we need new stories...");
							iscroll.seeIfWeNeedNewStories();
						})
						.fail( function() {
							iscroll.logstar("IMAGES FAILED!");
						});
				})
				.fail(function () {
					if (iscroll.failedEnoughAlready()) return; // don't keep trying forever
					iscroll.start += 20;
				   	iscroll.getNewStoriesAJAX();
				   	return;

				});
		},
		failedEnoughAlready: function() {
			iscroll.failcount++;
			return (iscroll.failcount &gt; 10);
		},
		seeIfNewStoriesLoadedYet: function () {
			if ($(".recent-stories").height() &gt; iscroll.oldstoriesheight.value) {
				iscroll.alreadyloading=false;
				iscroll.spinner.hide();
				iscroll.seeIfWeNeedNewStories();
			}
		},
		spinner: {
			selector: ".slideshowSpinnerBox",
			show: function() {
				$(iscroll.spinner.selector).css("display", "block");
			},
			hide: function() {
				$(iscroll.spinner.selector).css("display", ""); // Defaults to none as set via CSS
			}
		},
		fireItemStyleTrackingEvents: function(itemcountbystyle) {
			Object.keys(itemcountbystyle).forEach( function(key) {
				var keysvalue = itemcountbystyle[key];
				if (keysvalue &gt; 0) {
					ga('itTracker.send',
						'event',
						'Recent Stories -infinitescroll|style -desktop',
						param1 + " -displayed",
						key,
						keysvalue,
						{'nonInteraction': 1}
					);
				}
			});
		},
		refreshMasonry: function(newstories, elemsformasonry) {
			$(".recent-stories").append(newstories);
			if (iscroll.masonryalreadyinitialized == false) {
				iscroll.masonryalreadyinitialized = true;
				$(".recent-stories").masonry({itemSelector: '.recent-story', gutter:10});
				
			} else {
				$(".recent-stories").masonry("appended", newstories, true);
			}
			Stickyboxes.updateAll();
		}
	};
