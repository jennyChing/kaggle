    var i = 5;
    jQuery('#setcdown').text('5 S');

        if (i &gt; 0) {
            i--;
            jQuery('#setcdown').text(i + ' S');

            jQuery('#blurmask').hide();
            window.clearInterval(timer);
        }

    var timer = window.setInterval(tim, 1000);
