        function createSimpleSlider(args){
    // Load all data from settings
    $(document).ready(function(){ 
        var settings = args.settings;
        var slides = args.settings.slidesObj;
        var transitionTime = args.settings.transitionTime;
        var transitionType = args.settings.transitionType;
        var domStructure = args.settings.domStructure;
        var navPrevTxt = args.settings.navPrevTxt;
        var navNextTxt = args.settings.navNextTxt;
        var responsive = args.settings.responsive;
        var autoAdvanceSlides = args.settings.autoAdvanceSlides;
        var autoAdvanceDelay = args.settings.autoAdvanceDelay;
        var autoAdvanceReset = args.settings.autoAdvanceReset;
        var slideshowConatiner = $('#'+args.elementId);
        var simpleSlider = '#'+args.elementId+' #simpleSlider';
        var partNumber = args.elementId.split('_')[0];
        var callbackBuildComplete = args.settings.callbackBuildComplete;
        
        // Do math and reset time in objects
        // Need to reset the data in object to milliseconds
        autoAdvanceDelay = parseInt(autoAdvanceDelay) * 1000;
 
        // Build
        function domBuilder(){
            // Vars
            var slideshowHTML = '';
            var counter = 1;
            // Build the dom structure
            if(domStructure === 'strict'){
                slideshowHTML = '&lt;div id=simpleSlider&gt;&lt;div id="slides"&gt;&lt;div class="slideGutter"&gt;&lt;/div&gt;&lt;/div&gt;&lt;div id="navigation"&gt;&lt;div class="prev navigation"&gt;&lt;a class="'+partNumber+'_prev" href="#"&gt;'+navPrevTxt+'&lt;/a&gt;&lt;/div&gt;&lt;div class="next navigation"&gt;&lt;a class="'+partNumber+'_next" href="#"&gt;'+navNextTxt+'&lt;/a&gt;&lt;/div&gt;&lt;/div&gt;&lt;div id="pagination"&gt;&lt;ul id="'+partNumber+'_pagination" class="pagination"&gt;&lt;/ul&gt;&lt;/div&gt;&lt;/div&gt;';
            } else if(domStructure === 'loose'){
                slideshowHTML = '&lt;div id=simpleSlider&gt;&lt;div class="prev navigation"&gt;&lt;a class="'+partNumber+'_prev" href="#"&gt;'+navPrevTxt+'&lt;/a&gt;&lt;/div&gt;&lt;div class="next navigation"&gt;&lt;a class="'+partNumber+'_next" href="#"&gt;'+navNextTxt+'&lt;/a&gt;&lt;/div&gt;&lt;ul id="'+partNumber+'_pagination" class="pagination"&gt;&lt;/ul&gt;&lt;div id="slides"&gt;&lt;div class="slideGutter"&gt;&lt;/div&gt;&lt;/div&gt;&lt;/div&gt;';
            } else{
                slideshowHTML = '&lt;div id=simpleSlider&gt;&lt;div id="slides"&gt;&lt;div class="slideGutter"&gt;&lt;/div&gt;&lt;/div&gt;&lt;div id="navigation"&gt;&lt;div class="prev navigation"&gt;&lt;a class="'+partNumber+'_next" href="#"&gt;'+navPrevTxt+'&lt;/a&gt;&lt;/div&gt;&lt;div class="next navigation"&gt;&lt;a class="'+partNumber+'_next" href="#"&gt;'+navNextTxt+'&lt;/a&gt;&lt;/div&gt;&lt;/div&gt;&lt;div id="pagination"&gt;&lt;ul id="'+partNumber+'_pagination" class="pagination"&gt;&lt;/ul&gt;&lt;/div&gt;&lt;/div&gt;';
            }
            // Append a parent container to hold the slides
            slideshowConatiner.append(slideshowHTML);
            // Loop object literal
            $.each(slides, function(){
                // DOM elements 
                var slideHTML = '';
                var imageHTML = '';
                var titleHTML = '';
                var captionHTML = '';
                var paginationHTML = '';
                // Loop inner object literal
                $.each(this, function(key, value){
                    // Loop array and build DOM elements for img, caption and title
                    if(key == 'src'){
                    	// domain swap, try to fix broken src path(s)
                    	try{
	                    	protocolInSlide = value.split('/')[0];
	                    	protocolInURL = window.location.protocol;
	                    	hostInSlide = value.split('/')[2];
	                    	hostInUrl = window.location.host;
	                    	// test domain
	                    	if(hostInSlide !== undefined &amp;&amp; hostInSlide !== hostInUrl){
								value = value.replace(hostInSlide, hostInUrl);
	                    	}
	                    	// test protocol
	                    	if(protocolInURL !== protocolInSlide){
		                    	value = value.replace(protocolInSlide, protocolInURL);
	                    	}
                    	}
                    	catch(error){
	                    	return error;
                    	}
                    	
                        imageHTML = '&lt;div class="image"&gt;&lt;img src="'+value+'"&gt;&lt;/div&gt;';
                    } else if(key == 'title'){
                        titleHTML = '&lt;div class="'+key+'"&gt;'+value+'&lt;/div&gt;';
                    } else if(key == 'caption'){
                        captionHTML = '&lt;div class="'+key+'"&gt;'+value+'&lt;/div&gt;';
                    }
                    // Set first slide to active
                    if(counter == 1){
                        slideHTML = '&lt;div id="slide'+counter+'" class="slide active first"&gt;&lt;div class="gutter"&gt;'+imageHTML + titleHTML + captionHTML +'&lt;/div&gt;&lt;/div&gt;';
                        paginationHTML = '&lt;li&gt;&lt;a href="#" class="active first"&gt;'+counter+'&lt;/a&gt;&lt;/li&gt;';
                    } else if(counter === slides.length ){
                        slideHTML = '&lt;div id="slide'+counter+'" class="slide last"&gt;&lt;div class="gutter"&gt;'+imageHTML + titleHTML + captionHTML +'&lt;/div&gt;&lt;/div&gt;';
                        paginationHTML = '&lt;li&gt;&lt;a href="#" class="last"&gt;'+counter+'&lt;/a&gt;&lt;/li&gt;';
                    } else {
                        slideHTML = '&lt;div id="slide'+counter+'" class="slide"&gt;&lt;div class="gutter"&gt;'+imageHTML + titleHTML + captionHTML +'&lt;/div&gt;&lt;/div&gt;';
                        paginationHTML = '&lt;li&gt;&lt;a href="#"&gt;'+counter+'&lt;/a&gt;&lt;/li&gt;';
                    }
                });
                // Add the object literal contnets to the DOM
                $(simpleSlider +' .slideGutter').append(slideHTML);
                $(simpleSlider +' ul.pagination').append(paginationHTML);
                // Update the counter
                counter = counter + 1;
            });
        }

        function resizeElements(){
            var orginalWidth = $(simpleSlider).width();
            // Set the img and image containter width
             $(simpleSlider +' .image img').each( function(){
                // Run check for IE 8 &lt;, hasLayout needs width set inline
                if (navigator.userAgent.match('MSIE 8.0') || navigator.userAgent.match('MSIE 7.0')){
                                         //alert(navigator.userAgent);
                                         $(this).attr('style="width:"'+orginalWidth+'"');
                } else{
                        $(this).width(orginalWidth); // setting the img width necessary to avoid margin bewteen images
                                }
                                $(this).closest('.slide').width(orginalWidth);
            });
            // Reset orginal width  
            orginalWidth = $(simpleSlider).width();
            // Set the slideGutter width
            var slideTotal = $(simpleSlider +' .slide').length;
            var slideTotalWidth = slideTotal * orginalWidth;
            $(simpleSlider +' .slideGutter').css('width', slideTotalWidth);

        // Actions based on user input
        function inputActions(element){
            // Vars
            var activeSlide = $(simpleSlider +' .slide.active');
            var activePagination = $('ul#'+partNumber+'_pagination a.active');
            var containerWidth = $(simpleSlider).width();
            var slideWidth = $(simpleSlider +' .slide').width();
            var slideWidthNeg = parseInt($(simpleSlider +' .slideGutter').width()) * -1;
            var slideGutter = $(simpleSlider +' .slideGutter');
            var slideIndexPrev = $(simpleSlider +' .slide.active').prev('.slide').index();
            var curSlideGutterPos = parseInt($(simpleSlider +' .slideGutter').css('left'));
            // Loop Vars
            var curSlideGutterClonePos = parseInt($(simpleSlider +' .slideGutter').not('.slideGutter.clone').css('left'));
            var posToMoveCloneNeg = (Math.abs(containerWidth) + Math.abs(curSlideGutterClonePos)) * -1;
            var posToMoveClone = (Math.abs(containerWidth) + Math.abs(curSlideGutterClonePos));
            // Loop specific methods 
            if(transitionType == 'loop'){
                if($(element).hasClass('next')){
                    if(activeSlide.hasClass('last')){
                        $(activeSlide).removeClass('active');
                        // need to move the clone left dynamicly, we do this by appending the clone to the left 
                        slideGutter.first().clone().addClass('clone').css('left', slideWidth).insertBefore(slideGutter);
                        // Move the slide gutter
                        animateSlides(slideGutter, posToMoveCloneNeg);
                        var slideGutterClone = $(simpleSlider +' .slideGutter.clone');
                        animateSlides(slideGutterClone, 0, true);
                        // Add active class to slide and pagnation
                        $(simpleSlider +' .slide.first').addClass('active');
                        activePagination.removeClass('active');
                        $('ul#'+partNumber+'_pagination a.first').addClass('active');
                    }
                    else{
                        // Get the simple slider container width and set to negative
                        var posToMove = (Math.abs(containerWidth) + Math.abs(curSlideGutterPos)) * -1;
                        // Move the slide gutter
                        animateSlides(slideGutter, posToMove);
                        // Set current slide and pagnation to active
                        activeSlide.removeClass('active').next('.slide').addClass('active');
                        activePagination.removeClass('active').parent().next('li').find('a').addClass('active');
                    }
                }
                else if($(element).hasClass('prev')){
                    if(activeSlide.hasClass('first')){
                        $(activeSlide).removeClass('active');
                        activePagination.removeClass('active');
                        // need to move the clone left dynamicly, we do this by appending the clone to the left
                        slideGutter.first().clone().addClass('clone').css('left', slideWidthNeg).insertBefore(slideGutter);
                        // Move the slide gutter
                        animateSlides(slideGutter, posToMoveClone);
                        var slideGutterClone = $(simpleSlider +' .slideGutter.clone');
                        var slideGutterCloneLastSlide = $(simpleSlider +' .slideGutter.clone .slide.last').width();
                        var slideGutterCloneMovePos = (($(simpleSlider +' .slideGutter.clone').width()) * -1) + slideGutterCloneLastSlide;
                        animateSlides(slideGutterClone, slideGutterCloneMovePos, true);
                        $(simpleSlider +' .slide.last').addClass('active');
                        $('ul#'+partNumber+'_pagination a.last').addClass('active');
                    }
                    else{
                        var newPos = (Math.abs(slideIndexPrev) * Math.abs(slideWidth) * -1);
                        // Set current slide and pagnation to active
                        activeSlide.removeClass('active').prev('.slide').addClass('active');
                        activePagination.removeClass('active').parent().prev('li').find('a').addClass('active');
                        animateSlides(slideGutter, newPos);
                    }
                }
                else {
                    // Pagination 
                    var elementText = parseInt($(element).text()) -1;
                    var newPos = (Math.abs(elementText) * Math.abs(slideWidth) * -1);
                    // Set current slide to the active slide
                    activeSlide.removeClass('active');
                    $(simpleSlider +' .slide').eq(elementText).addClass('active');
                    // Update Pagination
                    $(element).addClass('active');
                    $(element).parent().siblings().children().removeClass('active');
                    animateSlides(slideGutter, newPos);
                }
            }
            // Timeline specific methods
            else if(transitionType == 'timeline'){
                // Navigation
                if($(element).hasClass('next')){
                    if(activeSlide.hasClass('last')){
                        // Tranisition type of 
                        animateSlides(slideGutter, 0);
                        activeSlide.removeClass('active');
                        $(simpleSlider +' .slide.first').addClass('active');
                    }
                    else{
                        // Get the simple slider container width and set to negative
                        var posToMove = (Math.abs(containerWidth) + Math.abs(curSlideGutterPos)) * -1;
                        // Move the slide gutter
                        animateSlides(slideGutter, posToMove);
                        // Set current slide to the active slide
                        activeSlide.removeClass('active').next('.slide').addClass('active');
                    }
                } else if($(element).hasClass('prev')){
                    // Previous Slide
                    if(activeSlide.hasClass('first')){
                        activeSlide.removeClass('active');
                        $(simpleSlider +' .slide.last').addClass('active');
                        var slideIndex = $(simpleSlider +' .slide.active').index();
                        var newPos = (Math.abs(slideIndex) * Math.abs(slideWidth) * -1);
                        animateSlides(slideGutter, newPos);
                    }
                    else{   
                        var newPos = (Math.abs(slideIndexPrev) * Math.abs(slideWidth) * -1);
                        // Set current slide to the active slide
                        activeSlide.removeClass('active').prev('.slide').addClass('active');
                        animateSlides(slideGutter, newPos);
                    }
                } else {
                    // Pagination 
                    var elementText = parseInt($(element).text()) -1;
                    var newPos = (Math.abs(elementText) * Math.abs(slideWidth) * -1);
                    // Set current slide to the active slide
                    activeSlide.removeClass('active');
                    $(simpleSlider +' .slide').eq(elementText).addClass('active');
                    // Update Pagination
                    $(element).addClass('active');
                    $(element).parent().siblings().children().removeClass('active');
                    animateSlides(slideGutter, newPos);
                }
            }
        } // end inputActions
        // Function that animates slides
        function animateSlides(element, pos, need){
            $(element).animate({
                left : pos
            }, transitionTime,
            function() {
                if(need){
                    $(simpleSlider +' .slideGutter').not('.slideGutter.clone').remove();
                    $(simpleSlider +' .slideGutter.clone').removeClass('clone');
                    bindSwipeEvents();
                }
            });
        } // end animateSlides
        function slideshowClickEvents(){
            // Navigation Events
            $('a.'+partNumber+'_prev, a.'+partNumber+'_next').click( function(event) {
                var element = $(this);
                if(!element.hasClass('disabled')){
                    $(this).addClass('disabled');
                    setTimeout(function(){
                        $(element).removeClass('disabled');
                     }, transitionTime);
                    inputActions($(this).parent());
                }
                event.preventDefault();
             });
            // Pagaination Events
            $('ul#'+partNumber+'_pagination a').click( function(event) {
                inputActions(this);
                event.preventDefault();
            });
        } // end slideshowClickEvents
        // User-Invoked Swipe Events
        function bindSwipeEvents(){
            // Vars for swipe events
            var downPos = 0;
            var downPosSlide = 0;
            var upPos = 0;
            $(simpleSlider).bind('touchstart', function(e){
                downPos = e.originalEvent.touches[0].pageX;
                parseInt(downPos);
                e.preventDefault();
                return false;
            });
            // Watch this,might have add chaining worng.
            $(simpleSlider +'  .slideGutter,  '+simpleSlider+' .slideGutter.clone').bind('touchmove', function(e){
                upPos = e.originalEvent.touches[0].pageY;
                downPosSlide = e.originalEvent.touches[0].pageX;
                parseInt(upPos);
            });
            $(simpleSlider +' .slideGutter').bind('touchend', function(e){
                swipe(downPos, upPos, downPosSlide);
            });
        } // end bindSwipeEvents
        function swipe(downPos, upPos, downPosSlide){
            var divPos = downPos - downPosSlide;
            var simplerSliderWidth = $(simpleSlider +' .slide.active').width();
            var swipedPrcnt = simplerSliderWidth / Math.abs(divPos);
            // Define a total number of pixels to scroll
            if( upPos &gt; 0 &amp;&amp; swipedPrcnt &lt;= 1.8 ){
                if(divPos &lt; 0){
                // Slide to previous slide
                    $('a.'+partNumber+'_prev').click();
                } else{
                    // Slide to next slide
                    $('a.'+partNumber+'_next').click();
                }
            }
            // Zero out for next round
            downPos = 0;
            downPosSlide = 0;
            upPos = 0;
            totalPos = 0;
        } // end swipe
        // Function that advances slides when page loads
        function autoAdvance(){
            if(autoAdvanceSlides == 'true'){
                var setInit = '';
                setInit = window.setInterval(function(){
                    // Run advance function with element to advance
                    var elementToAdvance = $(simpleSlider +' .next');
                    inputActions(elementToAdvance);
                }, autoAdvanceDelay);
                // Actions to prefrom based on jQuery version 
                var jqueryVersion = jQuery.fn.jquery;   
                // if &lt; jQuery version 1.5.1
                if(jqueryVersion &lt;= '1.5.1'){
                    $(simpleSlider).mouseenter(function(){
                        clearTimeout(setInit);
                    });
                    $(simpleSlider).mouseleave(function() {
                      need = 'clear';
                      setInit = window.setInterval(function(){
                            // Run advance function with element to advance
                      var elementToAdvance = $(simpleSlider +' .next');
                      inputActions(elementToAdvance);
                      }, autoAdvanceDelay);
                    });
                } else {
                    $(simpleSlider).on({
                        mouseenter: function(){
                            clearTimeout(setInit);
                        },
                        mouseleave: function(){
                            need = 'clear';
                            setInit = window.setInterval(function(){
                            // Run advance function with element to advance
                            var elementToAdvance = $(simpleSlider +' .next');
                            inputActions(elementToAdvance);
                            }, autoAdvanceDelay);
                        }
                    });
                }
            }
        } // end autoAdvance
        // Actions to preform on window resize e.g., effected elements are img, slide and slideGutter
        function dynamicActions(){
            if(responsive == 'true'){ 
                window.onresize = function() {
                    // Image and Image conatainer
                    // resize the image to fit the simple slider container
                    var slideshowConatinerWidth = $(simpleSlider).width();
                    // resize the img
                    $(simpleSlider +' .image img').width(slideshowConatinerWidth);
                    // resize slide
                    $(simpleSlider +' .slide').width(slideshowConatinerWidth);
                    // resize slideGutter to hold the images
                    var slideTotal = $(simpleSlider +' .slide').length;
                    var slideTotalWidth = slideTotal * slideshowConatinerWidth;
                    $(simpleSlider +' .slideGutter').css('width', slideTotalWidth);
                    // keep active slide centered by repositing slideGutter
                    // active slide index * width
                    var slideIndex = $(simpleSlider +' .slide.active').index();
                    var slideWidth = $(simpleSlider +' .slide').width();
                    var newPos = (Math.abs(slideIndex) * Math.abs(slideWidth)) * -1;
                    $(simpleSlider +' .slideGutter').css('left', newPos);
                    // Set height
                    var heightActiveSlide = $(simpleSlider +' .slide.active .image').height();
                    var newHeight = heightActiveSlide;
                    $(simpleSlider +' .slideGutter').height(heightActiveSlide);
                    $(simpleSlider).height(newHeight);
                    $(simpleSlider +' .slideGutter .gutter').height(heightActiveSlide);
                };
            }
        } // end dynamicActions
        // Call functions
        domBuilder();
        resizeElements();
        slideshowClickEvents();
        autoAdvance();
        dynamicActions();
        bindSwipeEvents();
        //Callback
        if(callbackBuildComplete != 'undefined'){
                        var F=new Function (callbackBuildComplete);
                        return(F());
        }
    }); // end document ready
