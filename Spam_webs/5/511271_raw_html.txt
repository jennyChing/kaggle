		function getObjWhenPropertyEquals(prop, val, myArray)
		{
		    for (var i = 0; i &lt; myArray.length; i++)
		    {
		        // check the obj has the property before comparing it
		        if (typeof myArray[i][prop] === 'undefined') {continue;}
		        // if the obj property equals our test value, return the obj
		        if (myArray[i][prop] === val)
		        {
		        	// add array key to the object
		        	myArray[i].arrayKey = i;
		        	return myArray[i];
		        }
		    }
		    // didn't find an object with the property

		}

		 function preprocessMainMenuSubNavItems (menuItem)

				var subnavItems = menuItem.subitems;

		/**
		 * Preprocess Stories
		 */
		function preprocessMainMenuStories (menuItem) {
			var maxStories = 4;
			if (menuItem.stories &amp;&amp; menuItem.stories.length &gt; 0)
			{
				menuItem.featured = [];
				var featImageCount = 0;
				var stories = menuItem.stories;
				for (var j = 0; j &lt;= stories.length-1; j++) {
					// BUILD FEATURED STORIES
					// copy stories to "featured" property, grab stories with images if they exist.
					if(stories[j].image &amp;&amp; featImageCount&lt;2)
					{
						featImageCount++;
						menuItem.featured.push(menuItem.stories[j]);
						menuItem.stories.splice(j,1);
					}
				};
				// BUILD FEATURED STORIES (cont.)
				// Didn't find images for all items
				if (featImageCount == 1)
				{
					// Only 1 image, add the first story available
					menuItem.featured.push(menuItem.stories[0]);
					menuItem.featured.sort(compareStoryDates);
					menuItem.stories.splice(0,1);
				}
				else if (!featImageCount)
				{
					// No images, add the first 2 stories from "stories"
					menuItem.featured = [menuItem.stories[0],menuItem.stories[1]];
					menuItem.stories.splice(0,2);
				}
				// LIMIT TO MAX ITEMS
				if (menuItem.stories.length &gt; maxStories)
				{
					menuItem.stories.splice(maxStories,menuItem.stories.length);
				}
			}
				return menuItem;
		}
