        <p>
         Austin is all grown up: Its youthful restaurant scene has changed dramatically in the past decade, with ramen and banh mi slowly becoming as common as chicken-fried steak and enchiladas (which are still beloved, don't get us wrong). Despite its growth, the city remains a bastion of Southern hospitality: Upscale restaurants still feel as friendly as the food trucks on every corner, where you can find some of the city's best barbecue.     function initializeMap() {
		// Create the map in the "mapcanvas" DIV
		var myOptions = {

			center: new google.maps.LatLng(0, 0),
			mapTypeId: google.maps.MapTypeId.ROADMAP
		};

			var point = createEntryCenterPointV3(val);
			
			
			// Create/Expand a boundary that encompasses all marker points
			bounds.extend(point);
			
			// Create a Marker at the center point for the Entry

			infoHolder[i] = [point, marker, val.selector];
			// Create a popup Info window for the Entry Marker
			addEntryInfoToMarkerV3(map,val, marker, false, false);
			c++;
		});
		
		fitPointsWithBounds(map,bounds,c,16);
		
		$('.map-check').change(function() {
			editionChangeMap();
		});
	}
	function editionChangeMap() {
		var bounds=new google.maps.LatLngBounds();
		$('.map-check').each(function(){
			var edition = $(this).attr('data-switch');
			var checked = $(this).is(":checked");
			for (var i = 0; i &lt; infoHolder.length; i++) {
				if(infoHolder[i][2] == edition &amp;&amp; $('.map-check:checked').length != 0)
				{
					if(checked){
						infoHolder[i][1].setMap(map);
						bounds.extend(infoHolder[i][0]);
					} else {
						infoHolder[i][1].setMap(null);
					}
				} else if($('.map-check:checked').length == 0)  {
					infoHolder[i][1].setMap(map);
					bounds.extend(infoHolder[i][0]);
				}
			}
		});
		google.maps.event.trigger(map, 'resize');
		map.fitBounds(bounds);
	}
	google.maps.event.addDomListener(window, 'load', initializeMap);
