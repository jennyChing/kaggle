   var Open_Close_Fix = function( button_classname, open_text, close_text )
{
	this.open_state = false;
	this.open_text = '.' + open_text;
	this.close_text = '.' + close_text;
	this.button_classname = '.' + button_classname;
	
	// Setup a pointer back to this class for use in events
	var self = this;
	
	// Bind the click event on our buttons to this class's click() function
	$(this.button_classname).unbind( 'click' ).bind( 
		'click', 
		function() {
			self.click( open_text, close_text );
		}
	);
	
}

Open_Close_Fix.prototype.click = function( open_text, close_text )
{
	if( !this.open_state )
	{
		this.classname = '.' + 'pane';
		this.open_all_tabs();
		this.classname = '.' + 'pane-title';
		this.open_all_titles();
		var allp = document.getElementsByTagName('p');
		var text = "";
		var pchildren; 
		for (var i=0; i&lt;allp.length; i++) {
			pchildren = allp[i].innerHTML;
			if (pchildren == open_text) {
				document.getElementsByTagName('p')[i].innerHTML = close_text;
			}
		}
	}
	else
	{
		this.classname = '.' + 'pane';
		this.close_all_tabs();
		this.classname = '.' + 'pane-title';
		this.close_all_titles();
		var allp = document.getElementsByTagName('p');
		var text = "";
		var pchildren; 
		for (var i=0; i&lt;allp.length; i++) {
			pchildren = allp[i].innerHTML;
			if (pchildren == close_text) {				
				document.getElementsByTagName('p')[i].innerHTML = open_text;
			}
		}		
	}
	
	this.open_state = !this.open_state;
}


Open_Close_Fix.prototype.open_all_tabs = function()
{
	$(this.classname).css( 'display', 'block' );
}

Open_Close_Fix.prototype.open_all_titles = function()
{
	$(this.classname).addClass("current");
}

Open_Close_Fix.prototype.close_all_tabs = function()
{
	$(this.classname).css( 'display', 'none' );
}
Open_Close_Fix.prototype.close_all_titles = function()
{
	$(this.classname).removeClass("current");

}
