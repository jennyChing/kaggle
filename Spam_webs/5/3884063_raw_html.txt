    function debounce(fn, delay) {
      var timer = null;
      return function () {
        var context = this, args = arguments;
        clearTimeout(timer);
        timer = setTimeout(function () {
          fn.apply(context, args);
        }, delay);
      };
    }
    function renderCards(){
        //viewport thingies
        console.log('resixe');
        var cols = Math.floor($(window).width() / 440);
        var rows = Math.floor(($(window).height() - 200) / 208);
        var itemsToShow = cols * rows;
        var cards = $('.channel-card');
        var padding = ($(window).width() % 440)/2;
        $('.row').css('padding-left', padding+'px');
        $('.row').css('padding-right', padding+'px'); 
        cards.each(function(i){
            if (i &gt;= itemsToShow) $(cards[i]).hide();
            else $(cards[i]).show();
        })
    }

        $(window).on('resize', debounce(renderCards, 250));
