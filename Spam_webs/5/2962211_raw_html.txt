            function swap_random_face(){
                // get a new face from the list
                rand_index = getRandomArbitrary(0, face_list.length - 1);
                new_face = face_list[rand_index];
                // get an old face to replace
                rand_index = getRandomArbitrary(1, 26);
                old_face = $('#face_' + rand_index);
                old_face_img = $('#face_' + rand_index + ' img');
                
                // don't replace with face that's already there
                replace = true;
                $(".face-link").each(function(index) {
                    if($(this).attr("href") == new_face['url']){
                        replace = false;
                    }
                });
                // replace the face
                if(replace){
                    old_face_img.fadeOut(1000, function() {
                        // Animation complete.
                        old_face.attr('href', new_face['url']);
                        old_face_img.attr('src', new_face['img']);
                        old_face_img.fadeIn(1000);
                    });                    
                }
            }

            function getRandomArbitrary(min, max) {
                return Math.round(Math.random() * (max - min) + min);
            }
