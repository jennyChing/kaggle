            isAnimate = false;
        });

    {
        if(isAnimate)
        {
            return;
        }
        isAnimate = true;
        $(".comment-" + nowTestimonials).fadeOut("slow", function () {
            nowTestimonials--;

            if ( nowTestimonials &lt; 1) {
                nowTestimonials = $(".testimonialsBox").length;
            }

            //console.log("In: " + nowTestimonials);
            $(".comment-" + nowTestimonials).fadeIn("slow");
					
            isAnimate = false;
        });
    }
