   function callonload(){
	// Handles the 'Remember my User ID/SSO' functionality
	

	  var remember = $.cookie('remember');

	  if ( remember == 'true' ) {
	  	var username = $.cookie('usernamesso');
	  	// autofill the fields
	  	$('#usernamesso').attr("value", username);
	  	$('#rememberssooruserid').attr('checked','checked');
	  	$('#password').focus();
	  }
	  else{
		  $('#usernamesso').focus();
	  }


	  $('#rememberssooruserid').click (function ()
	  {
	  if ($(this).is (':checked') &amp;&amp; $("#usernamesso").val() != '')
	  {
	  var username = $('#usernamesso').attr("value");
	  	$('#usernamesso').attr("value", username);
	  	// set cookies to expire in a year
	  	$.cookie('usernamesso', username, { path: '/', expires: 365 });
	  	$.cookie('remember', true, { path: '/', expires: 365 });
	  	} else {
	  	// reset cookies and clear input
	  	$('#usernamesso').val(''); 
	  	$('#rememberssooruserid').removeAttr("checked");
	  	$.cookie('usernamesso', null, { path: '/', expires: -10 });
	  	$.cookie('remember', null, { path: '/', expires: -10 });
	  }
	  });

	  $('#usernamesso').change(function(){
	  		if( ($(this).val() != username)) {
	  			$.cookie('usernamesso', null, { path: '/', expires: -10 });
	  			$.cookie('remember', null, { path: '/', expires: -10 });
	  			$('#rememberssooruserid').removeAttr("checked");
	  		}
	  	});
	  
	}
  </script>
