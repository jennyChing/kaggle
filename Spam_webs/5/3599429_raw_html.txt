    function highlightTweet(element) {
      element.focus();

    function checkLength(txtEl, countEl) {
      textElement = document.getElementById(txtEl);
      counterElement = document.getElementById(countEl);
      maxLen = 100; // max number of characters allowed
      if (textElement.value.length &gt;= maxLen) {
        //var msg = "You have reached your maximum limit of characters allowed";
        //alert(msg);
        textElement.value = textElement.value.substring(0, maxLen);
        counterElement.innerText = 'maximum characters reached';
      }
      else {
        charsLeft = maxLen - textElement.value.length;
        counterElement.innerText = charsLeft + ' character' + (charsLeft &gt; 1 ? 's' :'') + ' left';
      }
    }
