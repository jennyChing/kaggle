        prep_reply = function(leader_id, is_an_edit) {
            is_an_edit = typeof is_an_edit !== 'undefined' ? is_an_edit : false;
            proto_form = $('div.reply_form.prototype')
            post_form = proto_form.clone()
                .removeClass('prototype')
            leader = $('.comment_' + leader_id + ':visible:last')
            replies = leader.find('.replies:first')
            post_button = post_form.find('.action.post')
            post_button.attr('leader', leader_id)
            if (leader_id == 0) {
        //      post_form.prependTo(replies)
                post_form.insertBefore(leader)
            } else {
                if (replies.length &gt; 0) {
                    post_form.prependTo(replies)
                } else {
                    leader.append(post_form)
                }
            }
            textarea = wire_textarea(post_form.find('textarea.message :first'))
            post_button.click( function(){
                post_form = $(this).closest('div.replier')
                textarea = post_form.find('textarea :first')
                if (post_form.find('textarea').val().length &gt;= 1) {
                    switch (leader_id) {
                        case 0:
                            make_comment(post_form, is_an_edit, leader_id)
                            wire_textarea(textarea)
                            textarea.expanding('destroy').expanding()
                            break;
                        default:
                            make_comment(post_form, is_an_edit, leader_id)
                            post_form.find('textarea').focus()
                            post_form.hide()
                    }
                }
            })
            trash = textarea.find('a.comment_cancel.trash')
            switch (leader_id) {
                case 0:
                    textarea.bind('input propertychange', function() {
                        length = $(this).val().length
                        if (length &gt;= 1) {
                            trash.fadeIn(250)
                        } else {
                            trash.fadeOut(250)
                        }
                    })
                    post_form.show()
                    break
                default:
                    post_form.find('a.comment_cancel').click(function(){
                        post_form.hide()
                    })
                    post_form.slideDown(200 , function(){
                        textarea.focus()
                        trash.show()
                    })
            }
            post_form.find('a.comment_cancel').click(function(){
                $(this).hide()
            })
            return post_form
        }
