                $(this).find('.toggle_visibility')
                    .click( function(){
                        visibility_link = $(this)
                        if (visibility_link.closest('.comment_header').hasClass('unpopular'))   {
                            visibility_link.html('hide comment')
                                .closest('.comment_header')
                                .removeClass('unpopular')
                                .next('.comment_body').removeClass('unpopular')
                        } else {
                            visibility_link.html('show comment')
                                .closest('.comment_header')
                                .addClass('unpopular')
                                .next('.comment_body').addClass('unpopular')
                        }
                    })
            })
        }


        var wire_votes = function(elements, vote){
            $(elements).each(function() {
                $(this).click(function() {
                    if(!runUserCheck()){
                        return false
                    }
                    comment_id = $(this).closest('.cb_comment').attr('id')
                    container = $(this).closest('.votes_container')
                    $.post("comment_actions.php", {
                        'op': 'cast_vote',
                        'cid': comment_id,
                        'vote': vote
                    }, function(votes_now) {
                        votes_now = $.parseJSON(votes_now)
                        container.find('.cast').removeClass('cast').addClass('uncast')
                        vote_sum = container.find('.vote_sum').html(votes_now['up'] - votes_now['down'])
                        switch(votes_now['yours']) {
                            case -1:
                                vote_sum.removeClass('voted_up')
                                container.find('.downvote').removeClass('uncast').addClass('cast')
                                break;
                            case 0:
                                vote_sum.removeClass('voted_up')
                                break;
                            case 1:
                                vote_sum.addClass('voted_up')
                                container.find('.upvote').removeClass('uncast').addClass('cast')
                                break;
                        }
                        vote_sum.removeClass('is_zero')
                    })
                })
            })
        }

        hide_comments_after = function(number) {
            clone = $('.more_comments.prototype').clone()
            clone.removeClass('prototype')
            last_leader = $('#allComments').find('.comment:not(.prototype):not(.threaded)').get(number - 1)
            first_hidden = $(last_leader).nextAll('.comment :not(.threaded)').first()
            other_hidden = first_hidden.nextAll('.comment')
            total_hidden = other_hidden.not('.threaded').length + 1
            if (total_hidden &gt; 1) {
                first_hidden.hide()
                other_hidden.hide()
                clone.find('.count').html(total_hidden)
                clone.insertBefore(first_hidden)
                clone.find('a.more').click( function() {
                    first_hidden.prev('.more_comments').fadeOut( 200, function(){
                        first_hidden.fadeIn()
                        other_hidden.fadeIn()
                    })
                })
            }
        }

        make_comment = function(comment_form, is_an_edit, leader_id) {
            if(runUserCheck()) {
                $('.hidden_reply_link').show()
                button = comment_form.find('.action.post :first')
                message = comment_form.find('textarea.message').val()
                leader = button.attr('leader');
                // console.log('leader ', leader)
                hid = $('#creationPage').attr('hid')
                botCheckField = $('#commentField3').val();
                unprocessed_comment = {
                    "hid": hid,
                    "inputComment": message,
                    "returnData": false,
                    "notABot": true,
                    "botCheck" : botCheckField
                }
                if (leader != '') {
                    unprocessed_comment['threadid'] = leader
                }
                fake_comment = {
                    "message": message,
                    "id": leader,
                    "leader": leader
                }
                if (is_an_edit) {
                    $.post('/processCommentEdit', {
                        "cid": leader_id,
                        "new_message": message
                    })
                    displayed = display_comment(fake_comment, true, is_an_edit).fadeIn()
                    render_deletable($('.cb_comment.fake'))
                    render_editable($('.cb_comment.fake'))
                } else {
                    spinner = $('.spinner.prototype').clone().removeClass('prototype')
                    leader_el = $('.comment_' + leader + ':visible:last');
                    replies_el = leader_el.find('.replies:first');
                    if (replies_el.length == 0) {
                        replies_el = $('&lt;div&gt;&lt;/div&gt;');
                        replies_el.addClass('replies');
                        replies_el.addClass('to_comment');
                        leader_el.find('.comment_block').append(replies_el);
                    }
                    spinner.prependTo(replies_el);
                    $.post('/processComment', unprocessed_comment).done( function(data) {
                        fake_comment['id'] = data.CID;
                        displayed = display_comment(fake_comment, true, is_an_edit).fadeIn()
                        render_deletable($('.cb_comment.fake'))
                        render_editable($('.cb_comment.fake'))
                    })
                }
                comment_form.find('textarea').blur().val('')
            } else {
                $.cookies.set( 'patCom', comment_form.find('textarea.message').val() ); // set cookie message in case we lose it
                return false;
            }
        }

        window.post_form = {}
        prep_reply = function(leader_id, is_an_edit) {
            is_an_edit = typeof is_an_edit !== 'undefined' ? is_an_edit : false;
            proto_form = $('div.reply_form.prototype')
            post_form = proto_form.clone()
                .removeClass('prototype')
            leader = $('.comment_' + leader_id + ':visible:last')
            replies = leader.find('.replies:first')
            post_button = post_form.find('.action.post')
            post_button.attr('leader', leader_id)
            if (leader_id == 0) {
        //      post_form.prependTo(replies)
                post_form.insertBefore(leader)
            } else {
                if (replies.length &gt; 0) {
                    post_form.prependTo(replies)
                } else {
                    leader.append(post_form)
                }
            }
            textarea = wire_textarea(post_form.find('textarea.message :first'))
            post_button.click( function(){
                post_form = $(this).closest('div.replier')
                textarea = post_form.find('textarea :first')
                if (post_form.find('textarea').val().length &gt;= 1) {
                    switch (leader_id) {
                        case 0:
                            make_comment(post_form, is_an_edit, leader_id)
                            wire_textarea(textarea)
                            textarea.expanding('destroy').expanding()
                            break;
                        default:
                            make_comment(post_form, is_an_edit, leader_id)
                            post_form.find('textarea').focus()
                            post_form.hide()
                    }
                }
            })
            trash = textarea.find('a.comment_cancel.trash')
            switch (leader_id) {
                case 0:
                    textarea.bind('input propertychange', function() {
                        length = $(this).val().length
                        if (length &gt;= 1) {
                            trash.fadeIn(250)
                        } else {
                            trash.fadeOut(250)
                        }
                    })
                    post_form.show()
                    break
                default:
                    post_form.find('a.comment_cancel').click(function(){
                        post_form.hide()
                    })
                    post_form.slideDown(200 , function(){
                        textarea.focus()
                        trash.show()
                    })
            }
            post_form.find('a.comment_cancel').click(function(){
                $(this).hide()
            })
            return post_form
        }

        wire_textarea = function(textarea){
            textarea.val('')
            textarea.expanding('destroy').expanding()
            form = $(textarea).closest('div.replier')
            post_verb = form.find('.action.post')//.hide()
            form.find('.action_icon').fadeIn()
            trash_verb = form.find('div .action.trash :first').hide()
            post_button = textarea.closest('.floater').find('input.action.post')
            textarea.removeClass('watermark').expanding()
            textarea.focus(function(){
                $(this).css('overflow','hidden');
                $(this).keydown(function (e) {
                    if (e.keyCode == 13) {
                        if (e.ctrlKey) {
                            post_button.click()
                        }
                    }
                });
                container = $(this).closest('.floater')
                post = container.find('.action.post').fadeIn(250)
                trash = container.find('a.comment_cancel')
                trash.click(function() {
                    textarea = $(this).closest('.floater').find('textarea :first')
                    wire_textarea(textarea)
                })
            })
            return textarea
        }

        //edit your own comments
        is_comment_editable = function(comment) {
            your_user_id = $('.you.user.details :first').attr('user_id')
            comment_user_id = comment.closest('.cb_comment').attr('user_id')
            if ( comment.find('.comment_header').hasClass('deleted')) {
                return false
            } else if ( your_user_id == comment_user_id) {
                return true
            }
            return false
        }

        prep_edit = function(comment_id) {
            to_edit = $('.comment_' + comment_id + ':visible')
                .not('.edit_hidden')
                .addClass('edit_hidden')
            to_edit.find('.comment_header:first').hide()
            to_edit.find('.comment_body:first').hide()
            to_edit.find('.comment-avatar:first').hide()
            to_edit.find('.user_bar:first').hide()
            message = to_edit.find('.comment_text:first').html()
            reply_form = prep_reply(comment_id, true)
            reply_form.css('margin-left', '-60px')
            reply_form.find('textarea:first').val(message)
            reply_form.find('a.comment_cancel').click(function(){
                to_edit.find('.comment_header:first').show()
                to_edit.find('.comment_body:first').show()
                to_edit.find('.comment-avatar:first').show()
            })
        }

        render_editable = function(comments) {
            $(comments).each(function() {
                if (is_comment_editable($(this))) {
                    edit_button = $(this).find('span.action.edit :first')
                    edit_button.click(function(){
                        cid = $(this).closest('.cb_comment').attr('id')
                        prep_edit(cid)
                    })
                    $(this).find('.comment_block:first').hover(function(){
                        cid = $(this).closest('.cb_comment').attr('id')
                        edit = $(this).find('span.action.edit :first')
                        edit.show()

                        edit = $(this).find('span.action.edit :first')
                        edit.hide()

        }

        is_comment_deletable = function(comment) {
            your_user_id = $('.you.user.details :first').attr('user_id')
            creator_id = $('#creationPage').attr('sid')

        render_deletable = function(comments) {
            console.log('rendering deletables')
            $(comments).each(function() {
                if (is_comment_deletable($(this))){
                    trash = $(this).find('span.action.trash :first')
                    trash.click(function(e){
                        cid = $(this).closest('.cb_comment').attr('id')
                        console.log('delete clicked', cid)
                        e.preventDefault()
                        // console.log('trash clicked')
                        trash = $(this)
                        apprise('Are you sure you want to delete this comment? You cannot undo once deleted.', {'verify':true}, function(r)
                           {
                                if(r)
                                {
                                    $.post('/processCommentDelete', {
                                        "cid": cid })
                                    remove_comment(cid)
                                }
                                else
                                {
                                    // user clicked 'No' so we do nothing
                                }
                           });
                    })
                    $(this).find('.comment_block:first').hover(function(){
                        trash = $(this).find('span.action.trash :first')
                        trash.show()
                    }, function() {
                        trash = $(this).find('span.action.trash :first')
                        trash.hide()
                    })
                }
            })
        }

        display_comment = function(details, fake, is_an_edit) {
            console.log('display_comment()', details)
            fake = typeof fake !== 'undefined' ? fake : false;
            proto_comment = $('div.cb_comment.prototype')
            comment = proto_comment.clone()
                .removeClass('prototype')
                .addClass('comment_' + details.id)
                .attr('id', details.id)
            if (fake) {
                comment.addClass('fake')
            } else {
                comment.find('span.user')
                    .attr('user_id', details.user.id)
                comment.find( 'a.name' )
                    .html( details.user.name )
                comment.find( 'a.icon' )
                    .attr( 'href', '/user?u=' + details.user.id )
                comment.find( 'img.icon' )
                    .attr( 'src', details.user.icon )
            }
            comment.find('.comment_text')
                .html(details.message.replace(/&amp;/g, '&amp;').replace(/&lt;/g, '&lt;').replace(/&gt;/g, '&gt;').replace(/"/g, '&quot;'))
            reply = comment.find( 'a.reply' )
                .attr( 'cid', details.id)
                .click(function(){
                })
            wire_replies([reply])
            now = moment()
            if (is_an_edit){
                comment.find('span.comment_time').html('edited just now')
            } else {
                comment.find('span.comment_time').html('just now')
            }
            replies = $('.comment_' + details.leader + ':visible:last').find('.replies:first')
            comment.find('a.action.trash :first').hide()
            trash = comment.find('a.action.trash :first').attr('leader',details.id)

            render_deletable(comment)

            date_comment = function(comment) {
                comment.find('span.comment_time').fadeIn()
            }

            replies.each(function() {
                clone = comment.clone()
                if (is_an_edit) {
                    clone.css('margin-left', '-80px')
                    clone.find('.comment-avatar')//.css('margin-left', '-20px')
                }
                comment_spinner = $(this).find('.spinner' ).first()
                if (comment_spinner.length) {
                    comment_spinner.replaceWith(clone)
                } else {
                    clone.prependTo($(this))
                }
                setTimeout( date_comment, 3000, clone);
            })

            //fade in the comment's posting time
            if (fake) {
                comment.find('span.comment_time').html('just now').fadeIn()
            } else {
                merit_user( details.user )
            }

            return comment
        }

        merit_user = function(details) {
            $('.user[user_id="' + details.id + '"] .flair').not('.prototype').each(
                function(index, flair){
                    merit = $(flair).attr('merit')
                    if ($.inArray(merit, details.merits) == -1) {
                        flair.remove()
                    }
            })
            details.merits.forEach(function(merit) {
                proto_flair = $('i.flair.prototype[merit="' + merit + '"]')
                flair = proto_flair.clone()
                    .removeClass('prototype')
                    .attr('merit', merit)
                switch (merit) {
                    case 'pledged':
                        flair.attr('title', 'pledges $' + (details.pledge.cents / 100))
                    break;
                }
                $('span.user[user_id="' + details.id +'"]').each(function(index, user){
                    if ($(user).find('i.flair[merit="' + merit + '"]').length == 0) {
                        flair.clone()
                            .tipsy({fade: true, gravity: 's'})
                            .insertAfter($(user).find('i.flair.prototype'))
                    }
                })
            })
        }

function sharePage(url,type)
{
var shareHeight, shareWidth, shareX, shareY, shareCount;
switch(type)
{
case 'facebook':
shareWidth = 600;
shareHeight = 380;
break;
case 'twitter':
shareWidth = 800;
shareHeight = 320;
break;
case 'tumblr':
shareWidth = 450;
shareHeight = 435;
break;
case 'pin':
shareWidth = 600;
shareHeight = 300;
break;
case 'google':
shareWidth = 500;
shareHeight = 320;
break;
default:
return;
}

shareX = ((window.screen.width-shareWidth)/2);
shareY = ((window.screen.height - shareHeight)/2);
newwindow=window.open(url,'shareWindow','height='+shareHeight+',width='+shareWidth+',left='+shareX+',top='+shareY);
if (window.focus) {newwindow.focus()}
}
