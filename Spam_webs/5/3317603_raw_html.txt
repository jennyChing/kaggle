  }

  function textAndDropdownValidation(field, type) {
    var inputs = field.getElementsByTagName('input');
    for(var i = 0; i &lt; inputs.length; ++i) {
      var input = inputs[i];
      if(input.name.indexOf('signup') &gt;= 0) {
        if(type == 'text_field') {
          textValidation(input);
        } else {
          dropdownValidation(field, input);
        }
      }
    }
    htmlEmbedDropdownValidation(field);
  }

  function textValidation(input) {
    if(input.id == 'signup_email') return;
    var val = input.value;
    if(val == '') {
      textFieldError(input);
      isValid = false;
      return;
    } else {
      removeTextFieldError(input)
    }
  }

  function dropdownValidation(field, input) {
    var val = input.value;
    if(val == '') {
      if(field.className.indexOf('invalid') == -1) field.className += ' invalid';
      onSelectCallback(input);
      isValid = false;
      return;
    } else {
      field.className = field.className.replace(/ invalid/g, '');
    }
  }

  function htmlEmbedDropdownValidation(field) {
    var dropdowns = field.querySelectorAll('.mimi_html_dropdown');
    for(var i = 0; i &lt; dropdowns.length; ++i) {
      var dropdown = dropdowns[i],
          val      = dropdown.value;
      if(val == '') {
        if(field.className.indexOf('invalid') == -1) field.className += ' invalid';
        isValid = false;
        dropdown.onchange = validate;
        return;
      } else {
        field.className = field.className.replace(/ invalid/g, '');
      }
    }
  }

  function textFieldError(input) {
    input.className   = 'required invalid';
    input.placeholder = input.getAttribute('data-required-field');
  }

  function removeTextFieldError(input) {
    input.className   = 'required';
    input.placeholder = '';
  }

  function onSelectCallback(input) {
    if(typeof Widget != 'undefined' &amp;&amp; Widget.BasicDropdown != undefined) {
      var dropdownEl = input.parentNode,
          instances  = Widget.BasicDropdown.instances;
      for(var i = 0; i &lt; instances.length; ++i) {
        var instance = instances[i];
        if(instance.wrapperEl == dropdownEl) {
          instance.onSelect = validate;
        }
      }
    }
  }

  function updateFormAfterValidation() {
    form.className   = isValid ? '' : 'mimi_invalid';
    submit.value     = submitButtonText();
    submit.disabled  = !isValid;
    submit.className = isValid ? 'submit' : 'disabled';
  }

  function submitButtonText() {
    var invalidFields = document.querySelectorAll('.invalid'),
        text;
    if(isValid || invalidFields == undefined) {
      text = submit.getAttribute('data-default-text');
    } else {
      if(invalidFields.length &gt; 1 || invalidFields[0].className.indexOf('checkgroup') == -1) {
        text = submit.getAttribute('data-invalid-text');
      } else {
        text = submit.getAttribute('data-choose-list');
      }
    }
    return text;
  }

  function revalidateOnChange() {
    var fields = form.querySelectorAll(".mimi_field.required");
    for(var i = 0; i &lt; fields.length; ++i) {
      var inputs = fields[i].getElementsByTagName('input');
      for(var j = 0; j &lt; inputs.length; ++j) {
        inputs[j].onchange = validate;
      }
    }
  }
})();

        <p>
         I dont often get a chance to get out without my husband or one of the kids. Im sure you know how that is. It just seems that quality alone time is tough to find. I spent time this week at an event in Chicago and had a bit of free time where I didnt have anything scheduled. I decided that a ladies night out outfit was definitely needed. I wanted my outfit to be casual but not too casual.
        </p>
        <p>
         <a href="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/necklace-top.jpg">
          <img alt="My style: Ladies Night Out" class="aligncenter size-full wp-image-4831" height="491" src="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/necklace-top.jpg" width="736"/>

        <h2>
         My Style: Ladies Night Out
        </h2>
        <p>
         Since I live in jeans, I knew that a nice pair of dark jeans would definitely be included. I wanted a shirt that was a bit dressier than a t-shirt without being too dress. This gauzy peasant top over a white tank top worked perfectly for me. I added this beautiful gold love necklace and I was all set.
        </p>
        <p>
         The
         <a href="http://www.redenjewelry.com/shop/love-necklace-gold/" rel="nofollow" target="_blank">
          gold love necklace
         </a>
         is one of the handmade pieces from Reija Eden Jewelry. She offers necklaces, bracelets and earrings made for every day wear that will get you from day to night time with ease. This love necklace is made in the USA and is gold plated brass. It has a 16 inch chain which is the perfect length for most of my shirts.
        </p>
        <p>
         <a href="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/love-necklace-full.jpg">
          <img alt="My style: Ladies Night Out" class="aligncenter size-full wp-image-4832" height="777" src="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/love-necklace-full.jpg" width="736"/>
         </a>
        </p>
        <p>
         I love that this necklace is simple enough to wear with a casual outfit but is also very elegant looking and could be worn with a dressier outfit instead. It even arrived in a beautiful gift box. This exact necklace was actually seen at the Cannes Film Festival 2014. I love knowing that Im wearing the same jewelry as celebrities wore. Even though celebrities wore this necklace, its affordable enough for your average person to wear at only $75.
        </p>
        <p>
         <a href="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/necklace-bag.jpg">
          <img alt="My style: Ladies Night Out" class="aligncenter size-full wp-image-4833" height="491" src="http://www.the-socialites-closet.com/wp-content/uploads/2014/08/necklace-bag.jpg" width="736"/>
         </a>
        </p>
        <p>
         So, if you were lucky enough to have a ladies night out with your friends, where would you go?
        </p>
