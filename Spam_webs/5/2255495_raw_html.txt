        <div style="width: 79%; float: left;  ">
         Level 10 was not easy to reach on Steamtris. It was so hard even for people that played the game for years. Legends were made about some players going that far, there was even a story about a kid that reached Level 10 after playing the game for only few months. Nobody believed in those myths for long time until one day there was a big announcement in local newspapers that rumors were actually true and that the little boy will actually demonstrate his skills on the square in our town. I still remember that moment, that Saturday, hundreds of people at our local square and him playing that game like nobody before. It was so magical, untrue, it was the moment when I started believing that in order to do big things in life you just need one substance, love for what you do.

function commentValidate() {
   jQuery.post('newsite/scripts/form_validation.php',{ name: jQuery('#name').val(), system: '0', email: jQuery('#email').val(), cvalid: jQuery('#cvalid').val(), comment: jQuery('#comment').val(), validate: '2', comment_post_ID: jQuery('#comment_post_ID').val(), rating: jQuery('#rating').val() }, function(data) {


var errorMessage = "";

if(data.name != undefined) {
        jQuery('#name').css({'background':'#343940', 'border': '2px solid #fff'});
        errorMessage = errorMessage + data.name + '&lt;br/&gt;';
    } else {
        //alert('NAME:' + data.name);
        jQuery('#name').css({'background':'none', 'border':'1px solid #2b2f35'});
    }
    if(data.email != undefined) {
        jQuery('#email').css({'background':'#343940', 'border': '2px solid #fff'});
        errorMessage = errorMessage + data.email + '&lt;br/&gt;';
    } else {
       jQuery('#email').css({'background':'none', 'border':'1px solid #2b2f35'});

    }

    if(data.cvalid != undefined) {
        jQuery('#cvalid').css({'background':'#343940', 'border': '2px solid #fff'});
        errorMessage = errorMessage + data.cvalid + '&lt;br/&gt;';
    } else {

        jQuery('#cvalid').css({'background':'none', 'border':'1px solid #2b2f35'});
    }

   if(data.comment != undefined) {
        jQuery('#comment').css({'background':'#343940', 'border': '2px solid #fff'});
        errorMessage = errorMessage + data.comment + '&lt;br/&gt;';
    } else {
        jQuery('#comment').css({'background':'none', 'border':'1px solid #2b2f35'});
    }

        if (data.name === undefined &amp;&amp; data.email === undefined &amp;&amp; data.cvalid === undefined
        &amp;&amp; data.comment === undefined) {
            errorMessage = "Thank you. Your comment is scheduled for admin approval.";
            jQuery('#cform').slideUp(2000, function() {
                jQuery(this).css({'visibility' : 'hidden'});
            });
            jQuery('#validation_response').css({'display' : 'block'});
            jQuery('#validation_response').removeClass('error');
            jQuery('#validation_response').addClass('success');
            jQuery('#validation_response').html(errorMessage);

            } else {
                jQuery('#validation_response').html(errorMessage);
                jQuery('#validation_response').css({'display' : 'block'});
                jQuery('#validation_response').addClass('error');
            }
    //jQuery('#validation_response').html(data);
}, "json");

}
