function createXHR()
{
	if (window.ActiveXObject)
	{
		xhr = new ActiveXObject("Microsoft.XMLHTTP");
	}
	else if (window.XMLHttpRequest)
	{
		xhr = new XMLHttpRequest();
	}
}

	xhr.send(null);
