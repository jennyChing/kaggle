		var searchPhrase = document.getElementsByName("searchPhrase")[0].value.toLowerCase();
		var listOfSearchPhrase = ["jobs","job","recruitment","vacancy","vacancies"];
		if(listOfSearchPhrase.indexOf(searchPhrase) &gt; -1){
			myForm.action = jobUrl;
		} else {
			myForm.action = "/search/search.html";
		}
		myForm.submit();
	}
