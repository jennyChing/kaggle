            var $objectVideos  = $("object, embed");

            resizeVideos($iframeVideos, 'iframe');
            //resizeVideos($objectVideos, 'object');


            function resizeVideos( videos, method ) {
                // Figure out and save aspect ratio for each video
                videos.each(function() {
                    if ( method === "iframe" ) {
                        $(this).data('aspectRatio', this.height / this.width);
                    } else {
                        // cant assign data to an OBJECT tag
                        $(this).attr('data-aspectRatio', this.height / this.width);

                    }

                    $(this)
                        // and remove the hard coded width/height
                        .removeAttr('height')
                        .removeAttr('width');
                });

                // When the window is resized
                $(window).resize(function() {
                    var newWidth = $fluidEl.width();

                    
                    // Resize all videos according to their own aspect ratio
                    videos.each(function() {
                        var $el = $(this);
                        $el.width(newWidth);
                        
                        if ( method === "iframe" ) {
                            $el.height(newWidth * $el.data('aspectRatio'));
                        } else {
                            $el.height(newWidth * $el.attr('data-aspectRatio'));
                        }

                    });

                // Kick off one resize to fix all videos on page load
                }).resize();

            }
        });

function signupRedirect() {
	
	/*if (is_mustang == 1) {
		alert('signupRedirect signup_redirect_url = ' + signup_redirect_url);
	}*/
	
	
	
	// do we have a redirect url
	if (signup_redirect_url != '') {
		
		
		// set some vars that are strings we will search for
		var replace_site_url = '[site_url]';
		var replace_email = '[email]';
		var replace_enc_email = '[e]';
		
		// go through each string, if it is not empty
		if (replace_site_url != '') {
			// add the home page url for this site, urlencoded
			signup_redirect_url = signup_redirect_url.replace(replace_site_url, escape('http://www.allfreecrochetafghanpatterns.com'));
		}
		
		if (replace_email != '') {
			// add the email address the user signed up with
			signup_redirect_url = signup_redirect_url.replace(replace_email, signup_redirect_qry_params['email']);
		}
	
        if (replace_enc_email != '') {
			// add the email address the user signed up with
			signup_redirect_url = signup_redirect_url.replace(replace_enc_email, signup_redirect_qry_params['enc_email']);
		}		
		
				if(post_newsletter_signup.is_new_signup == true){
			var addl_qry_str = '?';
			if(signup_redirect_url.indexOf("?") &gt; 0){
				// the url already has a query string, so just tack on to it
				addl_qry_str = '&amp;';
			}
			
			signup_redirect_url += addl_qry_str + 'pp=1';
		}
		
		
				
		if (is_mustang == 1) {
			var confirm_str = 'signupRedirect' + "\n\n";
			confirm_str += 'replace_site_url = ' + replace_site_url + "\n\n";
			confirm_str += 'replace_email = ' + replace_email + "\n\n";
			confirm_str += 'post_newsletter_signup.is_new_signup = ' + post_newsletter_signup.is_new_signup + "\n\n";
			
			confirm_str += 'signup_redirect_qry_params[email] = ' + signup_redirect_qry_params['enc_email'] + "\n\n\n\n";
			
			confirm_str += 'signup_redirect_url = ' + signup_redirect_url + "\n\n";
			confirm_str += 'http_host = http://www.allfreecrochetafghanpatterns.com' + "\n\n";
			
			confirm_str += 'signup_redirect_id = ' + signup_redirect_id + "\n\n";
			
			confirm_str += 'Click OK to stay on this page.' + "\n";
			confirm_str += 'Click Cancel to go to Flow URL.' + "\n";
			
			if (confirm(confirm_str)) {
				return;
			}
		}
		
		// redirect the user
		window.location = signup_redirect_url;
		
		
		
	}
	
	
}
