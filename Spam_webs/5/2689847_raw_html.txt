           <div class="tCustomization tCustomization_post_description">
            <p>
             Dressage is the teamwork and systematic development of a horse and rider. Initially it helps the rider and horse communicate with one another to achieve balance, strength, flexibility and accuracy. Many riders use dressage as a great way to train the horse in order to better their performance and at its highest levels, produce a horse that is light and balanced with the ability to float effortlessly through the arena. Eventually, the communication between the horse and rider is so subtle its almost as if the horse is moving off its own accord without the rider instructing it to do so. As the horse and rider become more advanced in dressage, they will then perform the more spectacular movements such as the lateral movement (where the horse travels diagonally and sideways) and collected work such as pirouettes and piaffe (where the horse trots in place). There are many other terms worth noting when it comes to dressage.

function loadSocialWidgetInAPost(post) {
	if (post.length &gt; 0) {
		//facebook
		if (typeof FB != 'undefined') {
			FB.XFBML.parse(jQuery(".share-button-fb",post)[0]);
			post.addClass("loaded-fb");
		}
		//twitter
		if (typeof twttr != 'undefined') {
			jQuery(".share-button-tw a", post).addClass("twitter-share-button");
			twttr.widgets.load(jQuery(".share-button-tw",post)[0]);
			post.addClass("loaded-tw");
		}
		//G+
		if (typeof gapi != 'undefined') {
			gapi.plusone.go(jQuery(".share-button-gp",post)[0]);
			post.addClass("loaded-gp");
		}
	}
}


function loadVisibleSharers() {
	jQuery(".post").each(function(){
		var elt = jQuery(this);
		if (elt.attr("id") == "topic-tags-widget") {
			return;
		}
		if (elt.hasClass("loaded-tw") &amp;&amp; elt.hasClass("loaded-gp") &amp;&amp; elt.hasClass("loaded-fb")) {return;}
		if (elt.isOnScreen()) {
			loadSocialWidgetInAPost(elt);
		}
	});
}

jQuery(window).bind("resize", function(event) {
	loadVisibleSharers();
});

var scrollingChecker = 0;

	clearTimeout(scrollingChecker);

}

function onStopScrolling() {
	loadVisibleSharers();
}
