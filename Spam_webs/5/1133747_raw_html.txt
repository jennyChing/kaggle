                  }
              }
          });
  });
  
  
  function toggleTelekomBar() {
  // telekom list toggles
	  $('.telekom-list a').click(function(){
		  var url = $(this).attr('name');
		  if ($(this).is('.selected')) {
			  $(this).removeClass('selected');
			  $('#'+url).slideToggle('slow');
		  } else {
			  $('.telekom-list a').removeClass('selected').end();
			  $(this).addClass('selected');
			  $('#'+url).slideToggle('slow').siblings('.bar').slideUp();
			  $('#mobile-services').slideUp('slow');
		  }
	  });
	
	  //telekom list toggles close button
	  $('.close-btn').click(function(){
		  $('.bar').slideUp();
		  $('#telekom-bar a').removeClass('selected');
	  });
	
	  // mobile services toggle
	  $('.mobile-services a').click(function(){
		  if ($('.telekom-list a').is('.selected')) {
			  $('.bar').slideUp();
			  $('#telekom-bar a').removeClass('selected');
		  }
		  $('#mobile-services').slideToggle('medium');
	  });
	
	  $('.close-btn-mobile').click(function(){
		  $('#mobile-services').slideUp();
	  });
  }
