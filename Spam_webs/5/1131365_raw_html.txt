function sel(item, val) {
	if (install_status.length &lt;= item) install_status[item] = 0;
	if (install_status[item] != val) {
		if (val) {
			update_count++;
			update_size += update_sizes[item];
		} else {
			update_count--;
			update_size -= update_sizes[item];
		}
	}
	install_status[item] = val;
}

function _isDriver(n) {
	var i;
	var count = driverinfo.length;
	for(i=0;i&lt;count;i++) {
		if (n == driverinfo[i]) return 1
	}
	return 0;
}

function GetAllUpdates() {
	var res = '';
	var i;
	var count = install_status.length;
	for(i=0;i&lt;count;i++) {
		if (i &gt; 0) res = res + ',';
		if (_isDriver(i)) {
			res = res + '0';
			install_status[i] = '0';
		} else {
			res = res + '1';
			install_status[i] = '1';
		}
	}
	return res;
}

function getCurrentSelection() {
	var res = '';
	var count = install_status.length;
	for(i=0;i&lt;count;i++) {
		if (i &gt; 0) res = res + ',';
		res = res + install_status[i]; 
	}
	return res;
}
