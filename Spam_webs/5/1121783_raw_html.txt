  function convertCP2DecNCR ( textString ) {
  var outputString = "";
  textString = textString.replace(/^\s+/, '');
  if (textString.length == 0) { return ""; }
  textString = textString.replace(/\s+/g, ' ');
  var listArray = textString.split(' ');
  for ( var i = 0; i &lt; listArray.length; i++ ) {
    var n = parseInt(listArray[i], 16);
    outputString += ('&amp;#' + n + ';');
  }
  return( outputString );

  function convertChar2CP ( textString ) { 
    var haut = 0;
    var n = 0;
    CPstring = '';
    for (var i = 0; i &lt; textString.length; i++) {
        var b = textString.charCodeAt(i); 
        if (b &lt; 0 || b &gt; 0xFFFF) {
            CPstring += 'Error ' + dec2hex(b) + '!';
        }
        if (haut != 0) {
            if (0xDC00 &lt;= b &amp;&amp; b &lt;= 0xDFFF) {
                CPstring += dec2hex(0x10000 + ((haut - 0xD800) &lt;&lt; 10) + (b - 0xDC00)) + ' ';
                haut = 0;
                continue;
            }else {
                CPstring += '!erreur ' + dec2hex(haut) + '!';
                haut = 0;
            }
        }
        if (0xD800 &lt;= b &amp;&amp; b &lt;= 0xDBFF) {
            haut = b;
        }
        else {
            CPstring += dec2hex(b) + ' ';
        }
    }
        CPstring = CPstring.substring(0, CPstring.length-1);
