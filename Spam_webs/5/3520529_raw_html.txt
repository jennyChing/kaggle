                        resultWidth = image.getAttribute('data-w');
                        resultHeight = image.getAttribute('data-h');
                    }
                    image.setAttribute("src", resultSrc);

                    if (resultWidth) {
                        image.setAttribute("width", resultWidth);
                    } else {
                        if (window.console &amp;&amp; window.console.warn) {
                            console.warn("No width found for image: "+resultSrc+" and layout '"+(layout || "default")+"'.");
                        }
                    }

                    if (resultHeight) {
                        image.setAttribute("height", resultHeight);
                    } else {
                        if (window.console &amp;&amp; window.console.warn) {
                            console.warn("No height found for image: "+resultSrc+" and layout '"+(layout || "default")+"'.");
                        }
                    }
                }
            }
        }
    }
    function getCurrentLayout() {
        var i, L = layouts.length;
        if (bootstrapVars.isIE8 || bootstrapVars.isIE9) {

            // Special treatment for IE 8 and 9 which do not natively support matchMedia.
            // There is a polyfill for matchMedia for IE9, however, after reviewing performance metrics on jsperf,
            // the fallback approach is being used there instead.

            var windowWidth = document.body.clientWidth;
            for (i = 0; i &lt; L; i++) {
                if (windowWidth &gt;= layouts[i].minWidth) {
                    return layouts[i].layout;
                }
            }
        } else if (window.matchMedia) {
            for (i = 0; i &lt; L; i++) {
                if (matchMedia('(min-width: '+layouts[i].minWidth+'px)').matches) {
                    return layouts[i].layout;
                }
            }
        } else {
            if (window.console &amp;&amp; window.console.warn) {
                console.warn("Unsupported browser is being used. getCurrentLayout() will return undefined.");
                return 'undefined';
            }
        }
    }

    // Throttle has been extracted and paired down slightly from underscore.js here in order
    // to utilize it before any resources (including underscore) have been downloaded.
    var nowFn = Date.now || function() { return new Date().getTime();},
        throttle = function(func, wait) {
            var context, args, result;
            var timeout = null;
            var previous = 0;
            var later = function() {
                previous = nowFn();
                timeout = null;
                result = func.apply(context, args);
                context = args = null;
            };
            return function() {
                var now = nowFn();
                var remaining = wait - (now - previous);
                context = this;
                args = arguments;
                if (remaining &lt;= 0) {
                    clearTimeout(timeout);
                    timeout = null;
                    previous = now;
                    result = func.apply(context, args);
                    context = args = null;
                } else if (!timeout) {
                    timeout = setTimeout(later, remaining);
                }
                return result;
            };
        };

    function domReadyHandler () {
        bootstrapImages(document.body);
    }

    var resizeHandler = throttle(function () {
        var newLayout = getCurrentLayout();
        if (newLayout != layout) {
            layout = newLayout;
            bootstrapImages(document.body);
        }
    }, 50);

    if (bootstrapVars.isIE8) {
        document.attachEvent("onreadystatechange", domReadyHandler);
        window.attachEvent("onresize", resizeHandler);
    } else if (window.addEventListener) {
        window.addEventListener('DOMContentLoaded', domReadyHandler);
        window.addEventListener("resize", resizeHandler);
    }

    window.bootstrapImages = bootstrapImages;
})();
