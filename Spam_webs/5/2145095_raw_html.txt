                 <header>
                  <h3>
                   What Does 100% of Your Daily Value of Cholesterol Look Like?
                  </h3>
                  <p>
                   It's no secret that eating fatty foods raises your bad cholesterol level, also known as LDL. An elevated LDL clogs up your arteries and makes it difficult for your heart to do its job. Potentially, it could lead to heart disease.
                  </p>
                  <p>
                   The USDA recommends consuming no more than 300 mg of cholesterol a day. While a deep-fried Twinkie at the county fair is an obvious no-no, other high cholesterol culprits may be sneaking into your diet. Check out what that number looks like in terms of everyday food items.
                  </p>
                  <p style="margin-bottom:15px !important;">
                   <span style="font-weight: bold !important;">
                    Warning:
                   </span>
                   you may need to revise your grocery listand your eating habits!
                  </p>
                 </header>

                 <footer>
                  <p>
                   The amount of food in each photo represents your entire daily recommended value of cholesterol. These photos were shot with a Canon 7D DSLR. The plate shown is 10.25 in. (26 cm).
                  </p>
                  <p>
                   The USDA recommends no more than 300 mg a daybut that's not a number you should strive for. Saturated and trans fats are not part of a balanced diet. You should limit them as much as possible.
                  </p>
                  <p>
                   Replace saturated and trans fats with healthy fats, such as those found in mono- and polyunsaturated fats food sources. For example, cook with olive oil instead of butter. Drink fat-free milk instead of whole. Eat more fish and less red meat.
                  </p>

   var fireOnce = false;
  require(['base/async'], function( Async ) {

    var waitForSignal = false;

    function GPTRefreshAllAds() {
      // KIMPS
      for( var i = 0; i &lt; adsArrayForKIMP.length; i++) {
        if( window.HL_ga_sendTrackEvent &amp;&amp; adsArrayForKIMP[i] ) {
          try {
            HL_ga_sendTrackEvent({
              gaEventCategory : 'Any Page',
              gaEventAction : 'Key Impression',
              gaEventLabel: adsArrayForKIMP[i] + ": hn.us.hl.lc.x.x.x",
              gaEventNonInteraction: true
            });
          } catch (e) {
            // if an undefined value in array exists, do no log in GA
            // this does not affect ad refresh
          }
        }
      }
      // THIS IS THE ONE CALL FOR ENTIRE SITE
      googletag.cmd.push( function() {
        googletag.pubads().refresh( adsArrayToRefresh );
      });
    }

    Async( arrayPromiseFunctions, function( PromiseFunction, callback) {
      PromiseFunction().then( function( dataObj ){
        if( dataObj ){
          adsArrayToRefresh.push( dataObj.slotObj );
          adsArrayForKIMP.push( dataObj.slotName );
        } else {
          waitForSignal = true;
        }
        callback();
      }, function() {
        callback();
      });
    }, function(){
      if ( waitForSignal ) {
        $( document ).on( 'wslAdNotShow', function() {
          if( !fireOnce ) {
            GPTRefreshAllAds();
            fireOnce = true;
          }
        });
      } else {
        GPTRefreshAllAds();
      }
    });
  });
