            var $max_select  = $('.has-hrsb').length &gt; 0 ? $(this).parents('.search-refine-element-panel').find('select[name*=_max]').not('.none') : $(this).parents('li').find('select[name*=_max]').not('.none'),

                $max_chosen  = parseFloat($max_select.find('option:selected').val()),
                $min_chosen  = parseFloat($(this).val()),
                $clone       = $(this).find('option:gt(0)').clone(false);                         // clone the min list (excluding the first option)

            if (!isNaN($min_chosen)) {
                // minimum amount selected

                if (!$max_select.find('option[value="' + $min_chosen + '"]').val()) {               // max option list have these values

                    $max_select.html($clone).prepend($max_options.eq(0));                           // recreate max list
                    $max_select.find('option[value="' + $max_chosen + '"]').prop('selected', true); // retain the selected max option
                }

                // this removes all options that are less than the minimum selected
                $max_select.find('option').each(function() {
                    if ($(this).val() === '') return true;
                    if ($(this).val() &lt; $min_chosen) $(this).remove();
                }).end();

                if ($max_chosen) {
                    $max_select.find('option[value="' + $max_chosen + '"]').prop('selected', true);
                }

            } else {

                // no min selected - min and max should show full lists
                $max_select.html($clone).prepend($max_options.eq(0));

                if (!isNaN($max_chosen)) {
                    $max_select.find('option:selected').removeAttr('selected');
                    $max_select.find('option[value="' + $max_chosen + '"]').prop('selected', true);
                }

            }

        });
