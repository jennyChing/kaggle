function update_postage_details(elem){

   var frm = elem.form;
   var ship_to = $.trim(frm.postage_available_to.value);
   var zip_to = $.trim(frm.zip_to.value);
   var quantity = 1;

   if(! form_changed){
        var ship_to_selection = frm.postage_available_to;
        var ship_to_string = ship_to_selection.options[ship_to_selection.options.selectedIndex].text;
        var msg1 = "Alerta eBid\n&gt;&gt; Enviar para";
        var msg2 = "j est seleccionado e apresentado acima."
        var msg = msg1 + " " + ship_to_string + " " + msg2;        
        alert(msg);
        return false;
   }  
   form_changed = false;



   if( frm.quantity ) 
   {
      quantity = frm.quantity.value;
   }

   var pattern = /^\d+$/;
   if( '' != 'closed' &amp;&amp;  ! pattern.test( quantity ) )
   {
       alert( "Alerta eBid\n&gt;&gt; Por favor introduza apenas nmeros inteiros. Mximo disponvel  20.");
       return false;
   }

   if( '' != 'closed' &amp;&amp; quantity &gt; 20 ) 
   {
           alert("Valor mximo  20");
           return false;
   }


//   if( ! (ship_to == 'pt' &amp;&amp; 0 || ship_to != 'pt' &amp;&amp; 0) )
//   {
//      quantity = 1; 
//   }

   if( ! ship_to )    {
        ship_to = shipping_home_country;
   }

   var move_to_home_site = false;
   var link_to_move ;

   if( is_zip_need_for_country( ship_to ) &amp;&amp; ! valid_zip( ship_to, zip_to ))   {
      form_changed = true;
      return false;
   }

   if( ship_to != current_site )    {
      for( var i = 0; i &lt; shipping_available.length; i++)      {
         if( shipping_available[i] == ship_to &amp;&amp; current_site != shipping_continent[i] ) 
         {
             move_to_home_site = confirm( "Alerta eBid\n&gt;&gt; Para mostrar preos de custos de envio na moeda local, voc pode ser transferido para eBid "+unescape(shipping_name[i])+" ao clicar Ok. Clique CANCELAR para voltar atrs");
             link_to_move = shipping_link[i]+'&amp;quantity='+quantity+'#ship';
             break;
         }
      }
   }



/*
   if( is_zip_need_for_country( ship_to )  || quantity &gt; 1 )    {
      //if( ship_to !=  shipping_home_country || zip_to != shipping_home_zip || quantity &gt; 1)       {
        SetCookie( 'zip_to' , zip_to , 0, 'ebid.net');
        if( ! move_to_home_site )        {
          if( ship_to ==  shipping_home_country )           {
             return send_postage_ajax_request(elem);
          }          else          {
             return send_postage_ajax_international_request(elem);
          }
        }
      //}
      //else      {
      //  SetCookie( 'zip_to' , '' , 0, 'ebid.net');
      //}
   }
*/

   SetCookie( 'ship_to' , ship_to , 0, 'ebid.net' );
   if( is_zip_need_for_country( ship_to ) )
   {
        SetCookie( 'zip_to' , zip_to , 0, 'ebid.net');
   }
   else
   {
       SetCookie( 'zip_to' , '' , 0, 'ebid.net');
   }

   if( ship_to == shipping_home_country )
   {
       if( 0 || quantity &gt; 1 )
       {
             return send_postage_ajax_request(elem);
       }
   }
   else
   {
      if( 0 || quantity &gt; 1 )
      {
          return send_postage_ajax_international_request(elem);
      }
   }


   if( move_to_home_site ) { window.location = link_to_move; return false;}

   for( var i=0; i &lt; shipping_available.length; i++)
   {
      if( shipping_available[i] == ship_to &amp;&amp; shipping_code[i])
      {
//          document.getElementById('shipping_info').innerHTML = shipping_code[i];

         $("#shipping_info").html(shipping_code[i]);
         $("#shipping_country").html(shipping_name_html[i]);
         $("#shipping_lowest_postage").html(shipping_lowest_postage[i]);
         $("#shipping_lowest_bid").html(shipping_lowest_bid[i]);

         $("#bid_quantity").val(quantity);
         $("#buynow_quantity").val(quantity);

         $("#shipping_country2").html(shipping_name_html[i]);
         $("#shipping_lowest_postage2").html(shipping_lowest_postage[i]);
         $("#shipping_lowest_buynow").html(shipping_lowest_buynow[i]);

          return false;
      }
   }

   if( 0 &amp;&amp; ship_to != shipping_home_country ) //intenational to ajax
   {
       return send_postage_ajax_international_request(elem);
   }
   return false;
}

function send_postage_ajax_international_request(elem){
   var mo = 'int-postage-calc-json';
   var now=new Date();
   var frm = elem.form;
   var ship_to = frm.postage_available_to.value;
   var zip_to = frm.zip_to.value;
   var r  = now.getTime() +"-"+Math.random();

   var quantity = 1;
   if( frm.quantity ) 
   {
      quantity = frm.quantity.value;
   }


   var pars = 'mo=' + mo +'&amp;ship_to='+ship_to+'&amp;zip_to='+zip_to+'&amp;quantity='+quantity+'&amp;live='+'&amp;auction=136350449'+'&amp;r='+r;
   document.getElementById('shipping_info').innerHTML ="&lt;br /&gt;&lt;div align=center&gt;&lt;img src=http://pt.ebid.net/icons/twirl.gif height=32 width=32&gt;&lt;/div&gt;";

     $.ajax({
       type: "GET",
       url: AjaxUrl,
       data: pars,
       timeout: 10000,
       dataType: 'json',
       success: function(html){
         $("#shipping_info").html(html.html);
         $("#shipping_country").html(html.country_name);
         $("#shipping_lowest_postage").html(html.lowest_postage);

       },
       error: function reportError(request)
       {
          alert("Desculpe. Ocorreu um erro na transferncia de dados.");
       }
     });
 return false;
}
