         <a href="http://www.monoprice.com/products/search.asp?keyword=vga" onclick="_gaq.push(['_trackEvent', 'outbound-article', 'http://www.monoprice.com/products/search.asp?keyword=vga', 'VGA,']);" rel="nofollow" title="VGA">
