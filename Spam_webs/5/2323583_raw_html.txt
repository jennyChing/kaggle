		function savePage (){
		
			var pageTitle = jQuery('#page_title').val();
			
			if (pageTitle == "Name your Nation" || pageTitle == ""){
			
				alert ("Plase enter a page title");
				return false;
			
			}
			
			var curButton = document.getElementById("add_page_button");
			curButton.value = "Saving...";
			curButton.onclick = "";
		
			jQuery.ajax({
	
				url:"http://www.egmnow.com/wp-admin/admin-ajax.php",
				type:'POST',
				data:'action=save_new_page&amp;title='+pageTitle,
				success:function(results){
				
					var endStr = results.substr (0,(results.length-1));
					
					if (endStr.substr (endStr.length-1,1) == "-") {
					
						endStr = endStr.substr (0,endStr.length-2);
					
					}
	
					window.location = endStr;
	
				}
	
			});
		
		}	
		
		function switchGameTabs (theOp,pageId,subOp){
		
			var children = document.getElementById("game_page_tabs").getElementsByTagName("li");
		
			for (var i = 0; i &lt; children.length; i++){
			
				if (children[i].id != "game_page_tab_"+theOp){
				
					if (children[i].id != "build_a_nation"){
				
						children[i].className = "sub_tab";
						
					}
				
				} else {
				
					children[i].className = "sub_tab sub_tab_on";
				
				}
			
			}
	
			jQuery.ajax({
	
				url:"http://www.egmnow.com/wp-admin/admin-ajax.php",
				type:'POST',
				data:'action=switch_game_tabs&amp;op='+theOp+'&amp;page_id='+pageId+"&amp;sub_op="+subOp,
				success:function(results){
	
					var curObj = document.getElementById ("game_content");
					
					if (curObj){
					
						curObj.innerHTML = results.substr (0,(results.length-1));
						
						/*
						if (theOp == "photos"){
						
							jQuery('#gallery a').lightBox();
						
						}
						*/
					
					}
	
				}
	
			});
		
		}

            <div class="margin_ten clear single_content" id="single_content">
             <p>
              <img alt="" class="alignnone size-large wp-image-247972" height="354" src="http://d1vr6n66ssr06c.cloudfront.net/wp-content/uploads/2015/01/FRIDAY-THE-13TH-THE-VIDEO-GAME-LOGO-ON-ART3-630x354.jpg" title="FRIDAY THE 13TH - THE VIDEO GAME - LOGO ON ART3" width="630"/>
             </p>
             <p>
              <em>
               Friday the 13th
              </em>
              , the genre-defining blockbuster horror franchise, is set to thrill and terrify a whole new generation of digital fans as the incomparable, unstoppable Jason Voorhees returns to videogames for his first solo outing in over 25 years, EGM has exclusively learned.
             </p>
             <p>
              Currently in development,
              <em>
               Friday the 13th
              </em>
              will reportedly be an asymmetrical, co-operative and competitive multiplayer predator/prey horror experience that pits a small group of resourceful survivors against a single player in control of the immortal, relentless slasher icon.
             </p>
             <p>
              This will be the first official
              <em>
               Friday the 13th
              </em>
              licensed game to feature the hockey maskwearing, machete-wielding psychopath since players last faced off against the infamous Jason on the NES back in 1989.
             </p>
             <p>
              <a href="http://www.egmnow.com/?attachment_id=247973" rel="attachment wp-att-247973">
               <img alt="" class="alignright size-medium wp-image-247973" height="300" src="http://d1vr6n66ssr06c.cloudfront.net/wp-content/uploads/2015/01/FRIDAY-THE-13TH-ANNOUNCE-ART-V2-252x300.jpg" style="margin-left: 10px; margin-bottom: 10px;" title="FRIDAY THE 13TH ANNOUNCE ART - V2" width="252"/>
              </a>
              The new game, tentatively titled
              <em>
               Friday the 13th
              </em>
              , is in development for multiple platforms for an October 2015 release. The games lore will not only reference elements from the classic film franchise, but also incorporate content and themes from the upcoming
              <em>
               Friday the 13th
              </em>
              television series.
             </p>
             <p>
              With a new television series on the horizon and plans announced for the thirteenth installment of the film franchise, we felt the timing was right to finally explore Jason, Camp Crystal Lake, and the rest of the Friday the 13th story in interactive form, explains Sean S. Cunningham, creator of the original film that is credited by many as the genesis of the slasher/horror genre. We have some exciting new ideas for a game that supplies plenty of replay value, while delivering the kinds of thrills and scares that fans of the franchise have come to expect.
             </p>
             <p>
              Cunningham hints that an experienced game studio has taken the reins of the titles development and will be announced in the coming weeks.
             </p>
