var date = jQuery('#custom_date').val();
if(validateDate(date)) {
jQuery('.pick-btn').css('display', 'inline-block');
} else {
jQuery('.pick-btn').css('display', 'none');
}
});
function validateDate(date) {
var filter = new RegExp("(20[123][0-9])([-])([123456789]|0[123456789]|10|11|12)([-])([1-9]|[0-3][0-9])");
if (filter.test(date)) {
return true;
} else {
return false;
}
}
function selectDeliveryDate() {
var date = (jQuery(this).attr('rel'));
var dateFormat = jQuery("#date_format").val();
var formattedDate = formatDate(date, dateFormat);
jQuery('#4212_display').val(formattedDate);
jQuery('#4212').val(date);
jQuery('.date-enabled').removeClass('active');
// jQuery(this).addClass('active');
jQuery('#custom_date').val(date);
// jQuery('.pick-btn').css('display', 'inline-block');
jQuery('#delivery-select').fadeOut(125);
jQuery(".alternate-pick-btn").show();
jQuery('html, body').css('overflowY', 'auto');
}
function formatDate(date, dateFormat) {
var dateSplit = date.split("-");
var day = parseInt(dateSplit[2]);
var month = parseInt(dateSplit[1]);
var year = dateSplit[0]
var formattedDate = '';
switch(dateFormat) {
case '1':
formattedDate = month + '/' + day + '/' + year;
break;
case '2':
formattedDate = month + '/' + day + '/' + year.toString().substring(2);
break;
case '3':
formattedDate = month + '-' + day + '-' + year;
break;
case '4':
formattedDate = month + '-' + day + '-' + year.toString().substring(2);
break;
case '5':
formattedDate = year.toString().substring(2) + '/' + month + '/' + day;
break;
case '6':
formattedDate = year + '/' + month + '/' + day;
break;
case '7':
formattedDate = year.toString().substring(2) + '-' + month + '-' + day;
break;
case '8':
formattedDate = year + '-' + month + '-' + day;
break;
case '9':
formattedDate = day + '/' + month + '/' + year.toString().substring(2);
break;
case '10':
formattedDate = day + '/' + month + '/' + year;
break;
case '11':
formattedDate = day + '-' + month + '-' + year.toString().substring(2);
break;
case '12':
formattedDate = day + '-' + month + '-' + year;
break;
default:
formattedDate = month + '/' + day + '/' + year;
break;
}
return formattedDate;
}
function loadSmallCalendar() {
Ajax.Responders.register({
onCreate: function(){
jQuery('#spinner').show();
},
onComplete: function(){
jQuery('#spinner').hide();
}
});
var selectedDate = jQuery('#4212').val();
if (selectedDate) {
var dateArray = selectedDate.split("-");
var startDate = dateArray[1] + "-" + dateArray[0];
} else {
var startDate = "7-2015";
}
var params = {date: startDate, productId: 5086, selectedDate: selectedDate};
var url = (window.location.protocol === 'https:') ? 'https://www.flowermuse.com/advdelivery/index/changeMonth/' : 'http://www.flowermuse.com/advdelivery/index/changeMonth/';
new Ajax.Request(url, {
parameters:params,
method:'get',
onSuccess: function(transport) {
if (transport.responseText != 'false') {
jQuery('#mini-delivery-select').replaceWith(transport.responseText);
jQuery('#mini-delivery-select').show();
}
},
onFailure: function() {
alert('Something went wrong...');
}
});
}
