	function insertParam(url, parameterName, parameterValue, atStart){
	    replaceDuplicates = true;
	    if(url.indexOf('#') &gt; 0){
	        var cl = url.indexOf('#');
	        urlhash = url.substring(url.indexOf('#'),url.length);
	    } else {
	        urlhash = '';
	        cl = url.length;
	    }
	    sourceUrl = url.substring(0,cl);

	    var urlParts = sourceUrl.split("?");
	    var newQueryString = "";

	    if (urlParts.length &gt; 1)
	    {
	        var parameters = urlParts[1].split("&amp;");
	        for (var i=0; (i &lt; parameters.length); i++)
	        {
	            var parameterParts = parameters[i].split("=");
	            if (!(replaceDuplicates &amp;&amp; parameterParts[0] == parameterName))
	            {
	                if (newQueryString == "")
	                    newQueryString = "?";
	                else
	                    newQueryString += "&amp;";
	                newQueryString += parameterParts[0] + "=" + (parameterParts[1]?parameterParts[1]:'');
	            }
	        }
	    }
	    if (newQueryString == "")
	        newQueryString = "?";

	    if(atStart){
	        newQueryString = '?'+ parameterName + "=" + parameterValue + (newQueryString.length&gt;1?'&amp;'+newQueryString.substring(1):'');
	    } else {
	        if (newQueryString !== "" &amp;&amp; newQueryString != '?')
	            newQueryString += "&amp;";
	        newQueryString += parameterName + "=" + (parameterValue?parameterValue:'');
	    }
	    return urlParts[0] + newQueryString + urlhash;
	};

       <h2>
        <a href="http://hisdesignonline.com/twitter-etiquette-101/">
         Twitter Etiquette 101
        </a>
       </h2>
       <div class="post-content">
        <div id="getsocialmain">
         <p>
          Weve all read the articles on how to respond to people online, but its always good to revisit proper manners.
         </p>
         <p>
          No matter what social network youre on and using, alway remember the Golden Ruletreat others the way you want to be treated.
         </p>
         <p>
          Its also a good idea to remember not to ever post, tweet, or share something that you wouldnt want your kids, spouse, wife, boss, parents, or anyone else important to see. Remember social media is an open, wild frontier where any post can go viral. Many jobs have been lost and countless friendships have been hurt by a careless tweet or post.
         </p>
         <p>
          Below is a video with some simple reminders for Twitter etiquette. Whether or not you have been on Twitter since the beginning or have just started using social media, its a good idea to remind yourself that there are real people reading those 140 characters.
         </p>
         <p>
         </p>
