					function searchRef (selector, site) { 
						// Get the reference to search.
						var ref = $('.reference_navigation .active')[0].getAttribute('data-ref-id');
						
						if ( $(selector)[0].query.value !== '' ) {
							if ( site == 'spanish' ) {
								document.location = 'http://www.spanishcentral.com/translate/' + encodeURIComponent($(selector)[0].query.value.trim());

								document.location = '/scrabble/finder/' + encodeURIComponent($(selector)[0].query.value.trim());

								document.location = '/' + ref + '/' + encodeURIComponent($(selector)[0].query.value.trim());
							};
						} else if ( site == 'spanish' ) {
							document.location = 'http://www.spanishcentral.com/';
						} else if ( site == 'scrabble' ) {
							document.location = '/scrabble/';
						};
							
						return false;
					};

function trendWatchPromoShowSlide (node, slide) {
    var entry = $(node).parents('.tw_entry')[0];

    // Initialize the data.
    if ( typeof(slide) == 'undefined' ) {
        slide = parseInt(entry.dataset.slide, 10) - 1;
    };

    // Hide all of the slides.
    $(entry).find('.slide').css('display', 'none');

    // Get he slides.
    var slides = $(entry).find('.slide');

    // Make sure the slide index is within bounds.
    slide = Math.max(0, Math.min(slide, slides.length - 1));

    // Show the appropriate slide.
    if ( slide &lt; slides.length ) {
        $($(entry).find('.slide')[slide]).css('display', 'block');
    };

    $(entry).removeClass('first-slide').removeClass('last-slide');

    if ( slide == 0 ) {
        $(entry).addClass('first-slide');
    };

    if ( slide == (slides.length - 1) ) {
        $(entry).addClass('last-slide');
    };

    entry.dataset.slide = slide;
};
