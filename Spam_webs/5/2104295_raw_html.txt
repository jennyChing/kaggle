             <h3>
              Raw Caramel Chews Directions
             </h3>
             <div>
              <span class="instructions">
               <ol class="numbering">
                <li class="numbering">
                 <div>
                  <span style="color: #008000;">
                   <strong>
                    Base
                   </strong>
                  </span>
                 </div>
                </li>
               </ol>
               <ul>
                <li>
                 Blend the nuts into a powder
                </li>
                <li>
                 Place all ingredients into a food processor and process using the S blade until well combined
                </li>
                <li>
                 Shape the mixture into desired shape
                </li>
                <li>
                 Place in freezer for at least two hours
                </li>
               </ul>
               <li class="numbering">
                <div>
                 <span style="color: #008000;">
                  <strong>
                   Caramel
                  </strong>
                 </span>
                </div>
               </li>
               <ul>
                <li>
                 Blend all ingredients together until smooth
                </li>
               </ul>
               <li class="numbering">
                <div>
                 <span style="color: #008000;">
                  <strong>
                   Chocolate
                  </strong>
                 </span>
                </div>
               </li>
               <ul>
                <li>
                 Mix all ingredients together with a whisk
                </li>
               </ul>
               <li class="numbering">
                <div>
                 <span style="color: #008000;">
                  <strong>
                   Assembly
                  </strong>
                 </span>
                </div>
               </li>
               <ul>
                <li>
                 Place the bases on a non stick sheet
                </li>
                <li>
                 Put a layer of caramel on the base and place in freezer to firm up
                </li>
                <li>
                 Put the bars on a fork and place into the chocolate and cover well
                </li>
                <li>
                 Place onto a non stick sheet for at least two hours
                </li>
               </ul>

                <div>
