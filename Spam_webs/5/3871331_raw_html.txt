		function CheckAll()
		{
		    var v = document.frmResults.checkIt;
		    var bChecked = v.value == "1" ? true : false;
			
			// This will modify the JobInfo value as necessary
			var arrInfo;
			var lchkIdx;
			var lchkCount;
			var chkBoxes;

			if (typeof (document.frmResults.chkJobClientIds.length) != 'undefined') {
			    chkBoxes = document.frmResults.chkJobClientIds;
			    lchkCount = chkBoxes.length - 1; // one for hidden
			    for (lchkIdx = 0; lchkIdx &lt;= lchkCount; lchkIdx++) {
			        chkBoxes[lchkIdx].checked = bChecked;
			        chkBoxes[lchkIdx].setAttribute("aria-checked", bChecked);
			        if (bChecked == false) {
			            chkBoxes[lchkIdx].removeAttribute("checked");
			        }
			        arrInfo = chkBoxes[lchkIdx].value.split("|");
			        onCheckJob(lchkIdx, arrInfo[0], arrInfo[1], arrInfo[2], 0, true);
			    }
			}
			else {
			    document.frmResults.chkBoxes.checked = bChecked;
			    document.frmResults.chkBoxes.setAttribute("aria-checked", bChecked);
			    if (bChecked == false) {
			        document.frmResults.chkBoxes.removeAttribute("checked");
			    }
			    arrInfo = document.frmResults.chkJobClientIds.value.split("|");
			    onCheckJob(0, arrInfo[0], arrInfo[1], arrInfo[2], 0, true)
			}
			
			// build aria-label to keep the table cell label in sync with checkbox
			var selectAllLbl = document.frmResults.cmdSelect.title;
			if (bChecked == true) {
			    selectAllLbl += " " + document.getElementById("ctl00_MainContent_hChecked").value;
			}
			else {
			    selectAllLbl += " " + document.getElementById("ctl00_MainContent_hUnchecked").value;
			}
			
			// now find the parent TG to set the aria label
			var parent = document.frmResults.cmdSelect.parentNode;
			while (parent != null &amp;&amp; parent != undefined &amp;&amp; parent.tagName != "TH") {
			    parent = parent.parentNode;
			}
			if (parent.tagName == "TH") {
			    parent.setAttribute("aria-label", selectAllLbl);
			}
			
			if (v.value == "1") {
			    v.value = 0;
			}
			else {
			    v.value = 1;
			}

        }

		function setcolor() 
		{		    
		    if (!Dom.getElementsByClassName) {
		        Dom.getElementsByClassName = function(cn) {
		            var rx = new RegExp("(?:^|\\s)" + cn + "(?:$|\\s)");
		            var allT = Dom.getElementsByTagName("*"), allCN = [], ac = "", i = 0, a;
		            while (a = allT[i = i + 1]) {
		                ac = a.className;
		                if (ac &amp;&amp; ac.indexOf(cn) !== -1) {
		                    if (ac === cn) { allCN[allCN.length] = a; continue; }
		                    rx.test(ac) ? (allCN[allCN.length] = a) : 0;
		                }
		            }
		            return allCN;
		        }
		    }		    
		    var array = Dom.getElementsByClassName("yui-dt0-col-fjobs");
		    if (array.length &gt; 0) 
		    {
		        var bodystl = document.getElementById("ctl00_ContenBody");
		        var bgclr = getStyle(bodystl, 'background-color');

		       if(typeof(bgclr) == 'undefined')
		            bgclr = "white"; 
		        
		        for (var i = 0; i &lt; array.length; i++) {
		            if (i == 1)
		                array[i].setAttribute("style", "border-top-color:transparent; background-color:" + bgclr);
		            else
		                array[i].setAttribute("style", "background-color:" + bgclr);
		        }
		    }	    
		}
		
		function getStyle(x, styleProp) {
		    if (x.currentStyle)
		        var y = x.currentStyle[styleProp];
		    else if (window.getComputedStyle)
		        var y = document.defaultView.getComputedStyle(x, null).getPropertyValue(styleProp);
		    return y;
		}
		
		function SetcmdSelectText()
		{ // For setting the cmdSelect text
			var lchkCount;
			var chkBoxes;
			var cmdSelect;
			var checkIt;
			var bAllSelected = true;
			var lChkIdx;

			if (typeof (document.frmResults.chkJobClientIds.length) != 'undefined') {
			    chkBoxes = document.frmResults.chkJobClientIds;
			    lchkCount = chkBoxes.length - 1; // one for hidden
			    for (lChkIdx = 0; lChkIdx &lt;= lchkCount; lChkIdx++) {
			        if (!chkBoxes[lChkIdx].checked) {
			            bAllSelected = false;
			            break;
			        }
			    }
			}
			else {
			    if (!document.frmResults.chkJobClientIds.checked) {
			        bAllSelected = false;
			    }
			}		
			
			cmdSelect = document.frmResults.cmdSelect;
			checkIt = document.frmResults.checkIt;
			// need to keep grid columnheader label in sync with checkbox state
			var selectAllLbl = cmdSelect.title;

			if (bAllSelected) {
			    checkIt.value = 0;
			    cmdSelect.checked = true;
			    cmdSelect.setAttribute('aria-checked', true);
			    selectAllLbl += " " + document.getElementById("ctl00_MainContent_hChecked").value;

			}
			else {
			    checkIt.value = 1;
			    cmdSelect.checked = false;
			    cmdSelect.removeAttribute("checked");
			    cmdSelect.setAttribute('aria-checked', false);
			    selectAllLbl += " " + document.getElementById("ctl00_MainContent_hUnchecked").value;
			}
			// now find the parent TH to set the aria label
			var parent = document.frmResults.cmdSelect.parentNode;
			while (parent != null &amp;&amp; parent != undefined &amp;&amp; parent.tagName != "TH") {
			    parent = parent.parentNode;
			}
			if (parent.tagName == "TH") {
			    parent.setAttribute("aria-label", selectAllLbl);
			}
        }
