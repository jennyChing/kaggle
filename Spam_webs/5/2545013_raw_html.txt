        <h1>
         Fotballreiser Barcelona
         <br/>
         <img alt="stadion kart barcelona" height="576" src="http://www.reiseshop.no/media/wysiwyg/Barcelona-camp-Nou-plasser-ny.20.02.2015.jpg" style="width: 384px; height: 294px; float: right;" title="Stadion kart Barcelona fotballreise" width="600"/>
        </h1>
        <p>
         Grunnpakkeninkluderer 3 stjernes hotell og billett i kategori 2
         <strong>
          , mellomblfarge p stadionkartet
         </strong>
         . Dere fr plasser samlet. Disse plassene eri frste eller andre etasje bak det ene mlet.
         <br/>
         <br/>
         Du kan oppgradere til langsiden p
         <strong>
          oransje
         </strong>
         felt eller
         <strong>

     function setGridItemsEqualHeight($)
	{
		var SPACING = 20;
		if ($(window).width() &gt;= 480)
		{
			$('.category-products-grid').removeClass("auto-height");
			
			var gridItemMaxHeight = 0;
			$('.category-products-grid &gt; .item').each(function() {
				$(this).css("height", "auto");
				
				////////////////////////////////////////////////////////////////
									var actionsHeight = $(this).find('.actions').height();
					$(this).css("padding-bottom", (actionsHeight + SPACING) + "px"); //Set new padding
								////////////////////////////////////////////////////////////////
				
				gridItemMaxHeight = Math.max(gridItemMaxHeight, $(this).height());
			});
			
			//Apply max height
			$('.category-products-grid &gt; .item').css("height", gridItemMaxHeight + "px");
		}
		else
		{
			$('.category-products-grid').addClass("auto-height");
			$('.category-products-grid &gt; .item').css("height", "auto");
			$('.category-products-grid &gt; .item').css("padding-bottom", "20px");
		}
	}
