	var targets = pathjs.customTargets.join(";");

pathjs.getPageFormatIds = function(pageId){

	var formatIdArr = [];
	for (slot in myAdMeta){
		if (myAdMeta.hasOwnProperty(slot)){
			//if this is an array, add all members to formatIdArr
			if (pathjs.isArray(myAdMeta[slot])){
				formatIdArr.concat(myAdMeta[slot]);
			//otherwise just add single value
			} else {
				formatIdArr.push(myAdMeta[slot]);
			}
		}
	}
	//make the array a CSV string
	return formatIdArr.join();
};
