function changeLocation(){

	if(city != "")
	{lctnURL = lctnURL +"/"+ city}
	window.location.href = lctnURL;
}
function selectPopCity(state,city)
{
	$("#showPopCity").colorbox.close();

		type: "POST",

			updateSubLocation(html);
		}
	});
