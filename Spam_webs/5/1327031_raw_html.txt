						update:function() {
							$("#banners_off_list").style.marginTop = map(animation.frames.shift(), 0, 100, 0, -56)+"px";

								animation.complete();
								animation=null;
							}
						},

							$("#banners_off_list").insertBefore(document.getElementById("banners_off_list_item"+ths.banners[0]),null);

					animation.update();

					ths.animation = animation;

			function foldMenu(itemId) {

			   // define var scopes
			   var easing_arr = [13,24,35,44,53,61,69,75,81,86,90,94,97,98,100]; // quad
			   // var easing_arr = [28,49,65,76,85,90,94,97,98,99,100,100,100,100,100]; // quint

			   var animDuration = easing_arr.length, count = 0;
			   var startHeight = {};

			   if(active != itemId) { // if not active, animate; else: bugger off.

			   // var waarde = 1;
				// proto.tween(this, {prop:"waarde", start:1, end:100, duration:8, ease:"out", debug:true, round:true })

			      active = itemId;

			      // get the initial heights forthe duration of the animation
			      for(tempItemId in blocks) {
			         startHeight[tempItemId] = getCss($('#'+tempItemId), 'height');
			      }

			      // animation
			      clearInterval(interval);
			      interval = setInterval(function() {
			            // define var scopes
			            var easedCount, endHeight, newHeight, startMargin, endMargin, newMargin;

			            for(tempItemId in blocks)
			            {
			               // what's the final height? folded out or in. If it's folded in,
			               // calculate available height depending on the active folded out height
			               endHeight = itemId == tempItemId ? 166 : 67;
			               easedCount = easing_arr[count];


			               // map the new height to the progress of the animation
			               newHeight = map(easedCount, 1, 100, startHeight[tempItemId], endHeight);
			               $('#'+tempItemId).style.height = newHeight+'px';

			               if(tempItemId != 'first-item' &amp;&amp; newHeight != endHeight)
			               {
			                  startMargin = itemId == tempItemId ? 11 : 21;
			                  endMargin = itemId == tempItemId ? 21 : 11;
			                  newMargin = map(easedCount, 1, 100, startMargin, endMargin);
			                  $('#'+tempItemId+' h2')[0].style.marginTop = newMargin+'px';
			               }
			            }

			            count++;
			            if(count &gt;= animDuration) {
			               clearInterval(interval);
			               clickActive = itemId;
			            }
			         },
			         (1000/60)
			      );
			   }
			}
