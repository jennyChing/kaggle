    ToTop.prototype = {
        windowScroll: function (self) {
            if (!self.bSys) {
                clearTimeout(self.timer);
            }
            self.bSys = false;
            var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
            var t = scrollTop + document.documentElement.clientHeight - self.oDiv.offsetHeight;
            if (scrollTop &gt; 30) {
                this.oDiv.style.display = "block";
            } else {
                this.oDiv.style.display = "none";
            }
            self.startMove(parseInt(t), self);
        },
        clickToTop: function (self) {
            self.timer = setInterval(function () {
                var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
                self.iSpeed = Math.floor(-scrollTop / 8);
                self.bSys = true;
                document.documentElement.scrollTop = scrollTop + self.iSpeed;
                document.body.scrollTop = scrollTop + self.iSpeed;
                if (scrollTop == 0) {
                    clearInterval(self.timer);
                }
            }, 15);
        },
        startMove: function (iTarget, self) {
            clearInterval(self.timer1);
            self.timer1 = setInterval(function () {
                var iSpeed = (iTarget - self.oDiv.offsetTop) / 4;
                iSpeed = iSpeed &gt; 0 ? Math.ceil(iSpeed) : Math.floor(iSpeed);

                if (iTarget == self.oDiv.offsetTop) {
                    clearInterval(self.timer1);
                } else {
                    self.oDiv.style.top = self.oDiv.offsetTop + iSpeed + "px";
                }
            }, 30);
        }
    }

            var quantity = "1";

                quantity = $("#product_quantity").val();
            }
            if (library.validate.isPositiveInt(quantity) == false) {
                quantity == "1";

            if (parseInt(quantity) &gt; 999) {
                quantity = "999";
            }
            var productId = sku;

                productId += "-" + quantity;

            library.shoppingcart.add(productId);
