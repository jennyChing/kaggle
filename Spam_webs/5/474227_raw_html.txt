    function isEmailValidSignup(email) {
        return /^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w{2,}$/.test(email);
    }
    function ValidateSignup() {
       // document.getElementById("mailfromSignup").setAttribute("Class", "");
        var isValid = true;
        if ($('#mailfromSignup').val() == '') {
            document.getElementById("mailfromSignup").setAttribute("Class", "search-fieldRequired");
            isValid = false;
        }
        if (isValid) {
            if (!isEmailValidSignup($('#mailfromSignup').val())) {
                document.getElementById("mailfromSignup").setAttribute("Class", "search-fieldRequired");
                isValid = false;
            }
        }
        return isValid;
    }
