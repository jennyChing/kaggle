		function make_dropdown_menu($icon, $menu, width, othericons, left_side) {
			
			// State
			var hovering_in_icon = false;
			var hovering_in_menu = false;
			var touched_to_open = false;
			var touched_to_close = false;
			var will_close_soon = false;

			var handle_menu = function() {
				if (touched_to_open || touched_to_close) {
					// Touch device
					if (touched_to_open) {

						// Ensure we close other menus, if they've been touched open
						if (othericons) {
		                    for (var i = 0; i &lt; othericons.length; i++) {
		                        othericons[i].trigger('force_touch_open_to_close');
		                    }
		                }
						if (left_side) {
							$menu.css({
								'top': 50 + 'px',
								'left': ($icon.offset().left) + 'px',
								'width': width + 'px'
							}).slideDown('fast');
						} else {
							$menu.css({
								'top': 50 + 'px',
								'left': ($icon.offset().left - width + $icon.width()) + 'px',
								'width': width + 'px'
							}).slideDown('fast');
						}
					} else {
						$menu.slideUp('fast');
					}	
				} else {
					// Mouse/pointer
					if (hovering_in_icon || hovering_in_menu) {
						if (left_side) {
							$menu.css({
								'top': 50 + 'px',
								'left': ($icon.offset().left) + 'px',
								'width': width + 'px'
							}).clearQueue().slideDown('fast');
						} else {
							$menu.css({
								'top': 50 + 'px',
								'left': ($icon.offset().left - width + $icon.width()) + 'px',
								'width': width + 'px'
							}).clearQueue().slideDown('fast');
						}
					} else {
						will_close_soon = true;
						setTimeout(function () {
							if (will_close_soon) {
								$menu.slideUp('fast');
							}
						}, 150);
					}
				}
			}

			$icon.bind('force_touch_open_to_close', function() {
				if (touched_to_open) {
					touched_to_open = false;
					touched_to_close = true;
					$icon.removeClass('active');
					$menu.slideUp('fast');
				}
			});

			$icon.hover(function () {
				hovering_in_icon = true;
				will_close_soon = false;
				$icon.addClass('active');
				handle_menu();
			}, function () {
				hovering_in_icon = false;
				$icon.removeClass('active');
				handle_menu();
			});

			$menu.hover(function () {
				hovering_in_menu = true;
				will_close_soon = false;
				handle_menu();
			}, function () {
				hovering_in_menu = false;
				handle_menu();
			});

			$icon.on("touchstart", function(e) {
				e.preventDefault();
				if (touched_to_open) {
					touched_to_open = false;
					touched_to_close = true;
					$icon.removeClass('active');
				} else {
					touched_to_open = true;
					touched_to_close = false;
					$icon.addClass('active');
				}
				handle_menu();
			});
		}
	});
	})(jQuery);
