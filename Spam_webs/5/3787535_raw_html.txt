    <h1>
     VoIP Comparison Tool
    </h1>
    <p align="justify" class="mainbody">
     <strong>
      VoIP Comparison
     </strong>
     is a critical activity to perform prior to selecting a Business VoIP provider. If you have ever tried comparing VoIP providers, you likely have an appreciation of how tough a job it is to perform a comparison that actually yields useful results. Here are just a few reasons why a VoIP comparison is so difficult:
    </p>
    <ul class="round_blue_tick">
     <li>
      <strong>
       Pricing
      </strong>
      - Often tiered based on number of user seats requried. Not always cleared stated and instead vague statements are made such as "From $XX.YY per month".
     </li>
     <li>
      <strong>
       Features
      </strong>
      - Different terminology is often used by different providers to describe the same feature. No clear statements made on unsupported features.
     </li>
     <li>
      <strong>
       Add-ons
      </strong>
      - Information on additional costs for "premium" features is often hard to find.
     </li>
     <li>
      <strong>
       Support
      </strong>
      - Information is not always available.
     </li>
    </ul>
    <div class="spacer_ten">
    </div>
    <p align="justify" class="mainbody">
     WhichVoIP.com has made this job much easier for you, allowing you to perform a true apples-for-apples comparison using our
     <strong>
      VoIP comparison tool
     </strong>
     . We believe this is one of the most powerful tools on the market today for VoIP comparison shopping and we hope you find it useful.
    </p>
    <p align="justify" class="mainbody">
     VoIP providers that wish to be added to this tool or update existing plan information, should
     <a href="http://www.whichvoip.com/common/contact_us.htm" title="Contact WhichVoIP.com">
      Contact Us
     </a>
     for details.
    </p>

var error = "";
  if (strng.length == 0) {
     error = "Please enter your NAME.\n"
  }
return error;	  
}

function checkEmail(strng) {
var error = "";
	if (strng == "") {
   		error = "Please enter your CONTACT EMAIL in the appropriate entry box.\n";
	}
    var emailFilter=/^.+@.+\..{2,3}$/;
    if (!(emailFilter.test(strng))) { 
       error = "Please enter a valid EMAIL ADDRESS.\n";
    }
    else {
	//test email for illegal characters
       var illegalChars= /[\(\)\&lt;\&gt;\,\;\:\\\"\[\]]/
         if (strng.match(illegalChars)) {
          error = "The EMAIL ADDRESS you entered contains illegal characters.\n";
       	 }
    }
return error;    
}


function checkmessage(strng) {
var error = "";
  if (strng.length == 0) {
     error = "Please enter a COMMENT.\n"
  }
return error;	  
}

function checkcaptcha(strng) {
var error = "";
  if (strng.length == 0) {

  }
