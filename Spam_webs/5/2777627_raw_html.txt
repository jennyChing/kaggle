          <div class="static-body">
           <div class="windows">
            <h2>
             Windows
            </h2>
            <div class="ie">
             <p>
              <strong>
               <span class="internet-explorer">
                Enabling Cookies in Internet Explorer 7, 8, 9 +
               </span>
              </strong>
             </p>
             <ol>
              <li>
               Open the Internet
               <a id="_GoBack" name="_GoBack">
               </a>
               Browser
              </li>
              <li>
               Click Tools (or "gear" icon at top right hand corner) &gt; Internet  Options &gt; Privacy &gt; Advanced
              </li>
              <li>
               Check Override automatic cookie handling
              </li>
              <li>
               For First-party Cookies and Third-party Cookies  click Accept
              </li>
              <li>
               Click OK and OK
              </li>
             </ol>
            </div>
            <div class="firefox">
             <p>
              <strong>
               <span class="firefox">
                Enabling Cookies in Firefox
               </span>
              </strong>
             </p>
             <ol>
              <li>
               Open the Firefox browser
              </li>
              <li>
               Click Tools &gt; Options &gt; Privacy &lt; Use custom settings for history
              </li>
              <li>
               Check Accept cookies from sites
              </li>
              <li>
               Check Accept third party cookies
              </li>
              <li>
               Select Keep until: they expire
              </li>
              <li>
               Click OK
              </li>
             </ol>
            </div>
            <div class="chrome">
             <p>
              <strong>
               <span class="chrome">
                Enabling Cookies in Google Chrome
               </span>
              </strong>
             </p>
             <ol>
              <li>
               Open the Google Chrome browser
              </li>
              <li>
               Click Tools icon Or type in Go to chrome://settings/ to the URL window, hit enter
              </li>
              <li>
               Click 'Advanced settings' &gt; Select Privacy &gt; Content settings
              </li>
              <li>
               Check 'Allow local data to be set (recommended)'
              </li>
              <li>
               Click 'Done'
              </li>
             </ol>
            </div>
           </div>
           <div class="mac">
            <h2>
             Mac
            </h2>
            <div class="firefox">
             <p>
              <strong>
               <span>
                Enabling Cookies in Firefox
               </span>
              </strong>
             </p>
             <ol>
              <li>
               Open the Firefox browser
              </li>
              <li>
               Firefox &gt; Preferences
              </li>
              <li>
               Click 'Privacy' tab
              </li>
              <li>
               Under 'History' select Firefox will: 'Use custom settings for history'
              </li>
              <li>
               Check 'Accept cookies from sites' and then check 'Accept third-party cookies'
              </li>
              <li>
               Click OK
              </li>
             </ol>
            </div>
            <div class="chrome">
             <p>
              <strong>
               <span>
                Enabling Cookies in Google Chrome
               </span>
              </strong>
             </p>
             <ol>
              <li>
               Open the Google Chrome browser
              </li>
              <li>
               Chrome &gt; Preferences
              </li>
              <li>
               Click 'Show advanced settings' at the bottom.
              </li>
              <li>
               Under Privacy select 'Content settings'
              </li>
              <li>
               Under 'Cookies' select 'Allow local data to be set (recommended)'
              </li>
              <li>
               Click 'OK'
              </li>
             </ol>
            </div>
            <div class="safari">
             <p>
              <strong>
               <span>
                Enabling Cookies in Safari
               </span>
              </strong>
             </p>
             <ol>
              <li>
               In Safari
              </li>
              <li>
               Safari &gt; Preferences
              </li>
              <li>
               Click 'Privacy' tab
              </li>
              <li>
               Under 'Block cookies' check 'Never'
              </li>
             </ol>
            </div>
            <div class="mobile">
             <p>
              <strong>
               Enabling Cookies in Mobile Safari (iPhone, iPad)
              </strong>
             </p>
             <ol>
              <li>
               Go to the Home screen by pressing the Home button or by unlocking your phone/iPad
              </li>
              <li>
               Select the Settings icon.
              </li>
              <li>
               Select Safari from the settings menu.
              </li>
              <li>
               Select 'accept cookies' from the safari menu.
              </li>
              <li>
               Select 'from visited' from the accept cookies menu.
              </li>
              <li>
               Press the home button to return the the iPhone home screen.
              </li>
              <li>
               Select the Safari icon to return to Safari.
              </li>
              <li>
               Before the cookie settings change will take effect, Safari must restart. To restart Safari press and hold the Home button (for around five seconds) until the iPhone/iPad display goes blank and the home screen appears.
              </li>
              <li>
               Select the Safari icon to return to Safari.
              </li>
             </ol>
            </div>
