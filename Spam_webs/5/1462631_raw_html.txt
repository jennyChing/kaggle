function filter_wheels(wheels_type, ele)
{
	if ($('select[name="selectYear"]').val() == "Select Year" )
	{
		alert('please select the year of your car');
	}
	else if ($('select[name="selectMake"]').val() == "Select Make")
	{
		alert('please select the make of your car');
	}
	else if ($('select[name="selectModel"]').val() == "Select Model")
	{
		alert('please select the Model of your car');
	}
	else if ($('select[name="selectBody"]').val() == "Select Body")
	{
		alert('please select the Body for your car');
	}
	else if ($('select[name="selectOption"]').val() == "Select Option")
	{
		alert('please select the Option of your car');
	}
	else
	{
        var id = $(ele).attr("id");
        $("#"+id+" button").text("Loading...");
        $("#non-staggered-buttons .loading").slideDown('slow');
        //
		var year = $('select[name="selectYear"]').val();
        year = encodeURIComponent(year).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
		var make = $('select[name="selectMake"]').val();
        make = encodeURIComponent(make).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
		var model = $('select[name="selectModel"]').val();
        model = encodeURIComponent(model).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
		var body = $('select[name="selectBody"]').val();
        body = encodeURIComponent(body).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
		var option = $('select[name="selectOption"]').val();
        option = encodeURIComponent(option).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
        var zones_id = $('select[name="zones_id"]').val();
        zones_id = encodeURIComponent(zones_id).replace(/[!'()]/g, escape).replace(/\*/g, "%2A");
        
        var thread_front = $('input[name="tire_thread_front"]').val();
        var thread_rear = $('input[name="tire_thread_rear"]').val();
		
        if ( typeof year !== "undefined" &amp;&amp; typeof make !== "undefined" &amp;&amp; typeof model !== "undefined" &amp;&amp; typeof body !== "undefined" &amp;&amp; typeof option !== "undefined" )
		{
		  wheels_type = "all";
		  if (thread_rear &gt; thread_front) { //if OEM is staggered
    //wheels_type = "";
    }  
			if (wheels_type == "all")
			{
				var url = "index.php?route=product/category/getCarData&amp;type=wheel&amp;year=" + year + "&amp;make=" + make + "&amp;model=" + model + "&amp;body=" + body + "&amp;option=" + option + "&amp;sort=p.price&amp;order=ASC&amp;resp=" + resp + "&amp;zone_id="+zones_id;
				window.location.href = url;
			}
			else
			{
				var url = "index.php?route=product/category/getCarData&amp;type=wheel&amp;year=" + year + "&amp;make=" + make + "&amp;model=" + model + "&amp;body=" + body + "&amp;option=" + option + "&amp;sort=p.price&amp;order=ASC&amp;staggered=1&amp;resp=" + resp + "&amp;zone_id="+zones_id;
				window.location.href = url;
			}
		}
	}
}
