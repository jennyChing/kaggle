(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

            <h3 class="post-title entry-title">
             <a href="http://www.theretiredbabyboomer.net/2015/01/your-retirement-pension-plan.html">
              Your Retirement Pension Plan
             </a>
            </h3>
            <div class="post-header-line-1">
            </div>
            <div class="postmeta-primary">
             <span class="meta_date">
              1/04/2015 06:39:00 AM
             </span>
             <span class="meta_author">
              David Weldy
             </span>
             <span class="meta_comments">
              <a href="http://www.theretiredbabyboomer.net/2015/01/your-retirement-pension-plan.html#comment-form" onclick="">
               4 comments
              </a>
             </span>
            </div>
            <div class="post-body entry-content">
             <p>
              <br/>
             </p>
             <div class="separator" style="clear: both; text-align: center;">
              <a href="http://2.bp.blogspot.com/-QX546E-uOWo/VKYHoKr6_iI/AAAAAAAAGRc/339BWjFQnaU/s1600/pensionfund.jpg" imageanchor="1" style="clear: right; float: right; margin-bottom: 1em; margin-left: 1em;">
               <img border="0" src="http://2.bp.blogspot.com/-QX546E-uOWo/VKYHoKr6_iI/AAAAAAAAGRc/339BWjFQnaU/s1600/pensionfund.jpg"/>
              </a>
             </div>
             <br/>
             <br/>
             <br/>
             <br/>
             For a lot of us Baby Boomers, we were blessed in that we worked with the same employer most of our lives. We will probably be the last generation that can make that statement. We may well be the last generation that has a retirement pension from our employer as well. That is going to be more and more true in the private sector. The government employer will probably follow shortly in the years to come. Future generations will have to save for their own retirement with a 401k or something similar. But, for now, let's focus on the baby boomer generation.
             <br/>
             <br/>
             If you receive a pension that is vetted
             <i>
              (meaning it is your money and can't be touched by your former)
             </i>
             then you have nothing to worry about. But, that is not true with all pensions. Some companies, such as the Washington Post, Kodak, Lockheed Martin
             <i>
              (to name just three)
             </i>
             have either frozen their pension plans (
             <i>
              meaning some or all the employees covered by the plan stop earning some or all of the benefits from the start of the freeze going forward)
             </i>
             or terminated the pension plans altogether. Now, the three I just mentioned have changed their pension in some form. By no means are they the only employers to announce changes to their pension plans for new or current retirees. To my knowledge, the three mentioned have not terminated their pension plans. But, this is all beside the point. If you have a retirement pension plan from a private sector employer, you need to do some checking on the solvency of the plan itself. The first place I would start would be with your Human Resources Office. You may not get a lot of information there if they are worried about the reaction of employees.
             <br/>
             <br/>
             The best place to check on the internet I know to check on your employer pension plan would be at the
             <a href="http://www.pensionrights.org/publications/fact-sheet/pension-freezes">
              Pension Rights Center
             </a>
             . They have a wealth of information there. They could give you more information on whether or not your former employer is having retirement pension problems. The city of Detroit canceled or reduced retirement benefits to many of its retirees. They are caught between a rock and a hard place now. They could have gotten help before it was too late. Now, it is too late. All they will have is Social Security for retirement. As we have discussed before, if Social Security is all you have for retirement, you are in trouble.
