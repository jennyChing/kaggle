load_work = function(attrs)
{
	var es_id = $('#featured-list a:last-child').data('es-id');
	var limit = (profile_type == 'tailor' ? 10:5);
	
	$('#action-menu .notif-work-count').hide();		
	
	if( $('#featured-list').length == 0 )
	{

		if( typeof attrs != 'undefined' &amp;&amp; attrs.quick_open )
			$('#side-container').addClass('work-open quick-open');
		
		var data = {
			avatar: profile_avatar,
			load_more: ( profile_type=='tailor' ? 1:0 ),
			promote_premium: ((profile_type!='tailor' &amp;&amp; work_count &gt; 5 &amp;&amp; zg_id == profile_id) ? 1:0)
		}
		
		var html = '&lt;a href="#" class="work-avatar br-3 btn-public-work"&gt;&lt;img src="{{avatar}}" class="br-2 trans"&gt;&lt;/a&gt; \
					&lt;h1&gt;My Recent Work&lt;/h1&gt; \
					&lt;div id="featured-list" class="clearfix"&gt;&lt;div class="load-more-container"&gt;&lt;div id="loading-more" class="wrap loader clearfix br-5"&gt;&lt;img SRC="'+Core.baseUrl+'img/ooc/loader-z-black.gif"&gt;&lt;/div&gt;&lt;/div&gt;&lt;/div&gt; \
					{{#if load_more}} \
					&lt;div class="load-more-container"&gt; \
						&lt;div id="load-more" class="wrap br-5"&gt;&lt;a href="#" class="butt"&gt;Load More&lt;/a&gt;&lt;/div&gt; \
						&lt;div id="loading-more" class="wrap loader clearfix br-5"&gt;&lt;img SRC="'+Core.baseUrl+'img/ooc/loader-z-black.gif"&gt;&lt;/div&gt; \
					&lt;/div&gt; \
					{{/if}} \
					{{#if promote_premium}}{{/if}}';
		

		var template = Handlebars.compile(html);
		$('#side-content').html(template(data));
		
		var saved_work_views = $.jStorage.get("pprofile_work_views")||{};
		saved_work_views[profile_id] = work_count;
		$.jStorage.set("pprofile_work_views",saved_work_views);
	}

	$.ajax({
			url: "/profile/work/",
			global: false,
			type: "POST",
			data: ({ id: profile_id, es_id: es_id, limit: limit, origin: 'public_profile' }),
			dataType: "json",
			beforeSend: function()
			{
				$('#action-menu a.work').addClass('trans loading');	
				
				$('.load-more-container #load-more').fadeOut('fast', function(){
					$('.load-more-container #loading-more').fadeIn('fast').css('display','inline-block');
				});

			},
			success: function(result)
			{

				if (is_error(result, true))
					return;
				
				$('#side-container').addClass('work-open');
				$('#action-menu a.work').text('*');	
				
				if( $('#featured-list .work').length == 0 )
					$('#featured-list').html(render_work(result.data));
				else
					$('#featured-list').append(render_work(result.data));

				
				if(result.has_more)
				{
					/*track_page_end(function(){
						load_work();
					});*/
					$('.load-more-container #load-more').show();
					
				}
				else
				{
					$('.load-more-container').hide();
				}
			},
			complete: function(r,x)
			{
				$('.load-more-container #loading-more').hide();
				$('#action-menu a.work').removeClass('trans loading');
			}
		});	
}

append_work_notification = function()
{
	saved_work_views = $.jStorage.get("pprofile_work_views")||{};

	{
		var show_count = work_count - ( typeof saved_work_views[profile_id] != 'undefined'  ? saved_work_views[profile_id] : 0);
		if(show_count &gt; 0)
			$('.p-icon.work').after('&lt;span class="btn-public-work notif-work-count butt-orange br-3 animated fadeInUp '+ (show_count&lt;10 ? 'small':'') +'"&gt;'+ (show_count&gt;20 ? '20+' : show_count) +'&lt;/span&gt;');
	}
}
