function getQuadrant(map, marker)
		{
		  var bounds = map.getBounds();
          var sw = bounds.getSouthWest();
          var ne = bounds.getNorthEast();
          var east = ne.lng();
          var north = ne.lat();
          var west = sw.lng();
          var south = sw.lat();
          var width = west - east; 
          var offsetAmount = width/3;
          var cmin = west - offsetAmount;
          var cmax = west - offsetAmount*2;
          //DrawNSLine(map,cmin,"#FF0000");
          //DrawEWLine(map,cmax,"#0000FF");
          var i = east + (width/2);
          var j = south + (north - south)/3;
          var quadrant;
          var position = marker.getPosition();
          if (position.lat() &gt; j ){
          	quadrant = "N";
          }else{
          	quadrant = "S";
          }
          var lng = position.lng();
          if(cmin &gt; lng || lng &gt; cmax){
	          if(lng&lt;cmin){
	          	quadrant += "W";
	          }else if(lng&gt;cmax){
	          	quadrant += "E";
	          }
          }
          return quadrant;
		}
