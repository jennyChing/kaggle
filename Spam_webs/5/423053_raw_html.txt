  function refreshData()
{
    x = 1;  // x = seconds
 	var d = new Date()
 	var h = d.getHours();
 	var m = d.getMinutes();
 	var s = d.getSeconds();
 	
 	if (h&lt;=9) {h = '0'+h};
 	if (m&lt;=9) {m = '0'+m};
	if (s&lt;=9) {s = '0'+s};
	
 	var	color = '#'+h+m+s;

    $("p#hex").text(color);
     
    setTimeout(refreshData, x*1000);
}
