      <h2 id="post-2552">
       <a href="http://ajaxian.com/archives/iphone-native-looking-skin" rel="bookmark" title="Permanent Link to iPhone Native Looking Skin">
        iPhone Native Looking Skin
       </a>
      </h2>
      <p class="meta">
       Category:
       <a href="http://ajaxian.com/by/topic/iphone" rel="category tag" title="View all posts in iPhone">
        iPhone
       </a>
       <a href="http://technorati.com/tag/iPhone" rel="tag" title="View the technorati tag: iPhone">
        <img alt="View the technorati tag: iPhone" border="0" class="technorati-image" height="16" src="http://technorati.com/favicon.ico" width="16"/>
       </a>
       ,
       <a href="http://ajaxian.com/by/topic/mobile" rel="category tag" title="View all posts in Mobile">
        Mobile
       </a>
       <a href="http://technorati.com/tag/Mobile" rel="tag" title="View the technorati tag: Mobile">
        <img alt="View the technorati tag: Mobile" border="0" class="technorati-image" height="16" src="http://technorati.com/favicon.ico" width="16"/>
       </a>
      </p>
      <p>
       The default look of a web page is pretty bad. Simple white background with ugly black lettering. Nothing nice.
      </p>
      <p>
       With the iPhone though, it doesnt look half as bad, and now, thanks to Joe Hewitt (again) there is an even better base.
      </p>
      <p>
       You can simply use his
       <a href="http://joehewitt.com/files/iphone/navigation.html">
        native looking navigation skin
       </a>
       to make your applications look and feel like a native iPhone application.
      </p>
      <p>
       The magic sits in:
      </p>
      <ul>
       <li>
        <a href="http://joehewitt.com/files/iphone/iphonenav.js">
         iphonenav.js
        </a>
        : The JavaScript handles getting rid of the browser url bar, the back button (Safari finally supports changing the hash nicely!), changing based on the orientation, swiping a page across, and more.
       </li>
       <li>
        <a href="http://joehewitt.com/files/iphone/iphonenav.css">
         iphonenav.css
        </a>
        : The CSS makes the standard elements look like real iPhone elements thanks to magic like -webkit-border-image: url(iPhoneButton.png) 0 8 0 8;
       </li>
      </ul>
      <p>
       There is still more to do of course. It would be great to give developers simple access to the red deletes, the blue buttons, the twisty round deletes, and more. This allows anyone to build an app that will look decent and just like the iPhone apps that users are used too.
      </p>
      <p>
       Thanks again Joe!
      </p>

		/**
		 * Usage:
		 *
		 *  stringToBreak - string to break up (required arg)
		 *  url - true or false (if true, string will be broken at maxLength value without regard to word spacing)
		 *  maxLength - maximum number of characters per line (optional, defaults to 25)
		 *  threshold - number of characters under/over the maxLength a line could possibly be (optional, defaults to 5)
		 *  separator - value to use to break up lines (optional, defaults to '&lt;br /&gt;')
		 **/
		function breakString(stringToBreak, url, maxLength, threshhold, separator) {

			// boolean for if we are parsing a URL or text - urls can be parsed at maxLength regardless of whitespace
			var url = (arguments[1] === undefined) ? false : url;

			// default to break at 15 chars if maxLength is invalid
			var maxLength = ((arguments[2] === undefined) || isNaN(maxLength) || (maxLength &lt; 1)) ? 25 : maxLength;

			// default to 5 characters of "padding" above or below the maxLength per line
			var threshhold = ((arguments[3] === undefined) || isNaN(threshhold) || (threshhold &gt; maxLength) || (typeof(threshhold) == undefined)) ? 5 : threshhold;

			// default to using a BR for the separator
			var separator = (arguments[4] === undefined) ? '&lt;br /&gt;' : separator;

			// save the contents of the new broken string
			var brokenString = new String();

			// just do this calculation once
			var minLength = (maxLength - threshhold);

			// if the string is a URL, then break it at maxLength
			if (url) {
				if (stringToBreak.length &gt; maxLength) {
					var loopNum = Math.ceil(stringToBreak.length / maxLength);
					var currentEndPoint = 0;
					for (var k = 0; k &lt; loopNum; ++k) {
						var beginPoint = currentEndPoint;
						currentEndPoint = (currentEndPoint + maxLength);
						brokenString += (stringToBreak.substring(beginPoint,currentEndPoint) + separator);
					}
					brokenString += stringToBreak.substring(currentEndPoint,stringToBreak.length);
				} else {
					brokenString = stringToBreak;
				}
			}
			// if the string is text, break up words on whitespace
			else {
				// break up the string into an array, based on space
				var pieces = stringToBreak.split(' ');

				for (var k=0; k&lt;pieces.length; ++k) {

					var curPiece = new String();

					while ( curPiece.length &lt;= minLength ) {
						if ( k &gt;= (pieces.length-1)) {
							break;
						} else {
							if ( (pieces[k+1].length + curPiece.length) &gt;= minLength ) {
								break;
							} else {
								curPiece += (pieces[k++] + ' ');
							}
						}
					}
					curPiece += (pieces[k] + ' ' + separator);
					brokenString += curPiece;
				}
			}
			return brokenString;
		}
