                        }, (i * 300));
                    });
                
                    setTimeout(function() {
                        $('#newsletter-signup-form .legend').fadeIn(200);
                    }, ($("#newsletter-signup-form .required-icon").length * 300));
                });
            });
                      
            
            // Check for proper email formatting
            $('#newsletter-email').blur(function() {
                if (!$(this).val().match(/.+\@.+\..+/i) &amp;&amp; $(this).val().length &gt; 0) {
                    if (!$(this).hasClass('error')) {
                        $(this).addClass('error');     
                        var p = $(this).position();           
                        $(this).after('&lt;div class="field-error" style="margin-left: ' + p.left + 'px"&gt;&lt;div class="field-error-arrow"&gt;&lt;/div&gt;&lt;div class="field-error-text"&gt;Not a valid email format. Did you forget something?&lt;/div&gt;&lt;/div&gt;');
                    }
                    $('#newsletter-submit').attr('disabled','disabled');
                    $('#newsletter-signup-form form').attr('disabled','disabled');
                    
                } 
            });
            $('#newsletter-email').keyup(function(){
                if ($(this).val().match(/.+\@.+\..+/i)) {
                    if ($(this).hasClass('error')) {
                        $(this).removeClass('error');
                        if ($(this).next().hasClass('field-error'))
                            $(this).next().remove();
                    }
                    $('#newsletter-submit').removeAttr('disabled');
                    $('#newsletter-signup-form form').removeAttr('disabled');              
                }
            });
            
            // let a person know when the emails don't match
            $('#newsletter-re-type-email').keyup(function(){
                var pattern = new RegExp('^' + $(this).val(), 'i');
            
                if (!$('#newsletter-email').val().match(pattern)) {
                    if (!$(this).hasClass('error')) {
                        $(this).addClass('error');                
                        var p = $(this).position();           
                        $(this).after('&lt;div class="field-error" style="margin-left: ' + p.left + 'px"&gt;&lt;div class="field-error-arrow"&gt;&lt;/div&gt;&lt;div class="field-error-text"&gt;Email doesn\'t match&lt;/div&gt;&lt;/div&gt;');
                    }
                    $('#newsletter-submit').attr('disabled','disabled');
                    $('#newsletter-signup-form form').attr('disabled','disabled');
                } else {
                    if ($(this).hasClass('error')) {
                        $(this).removeClass('error');
                        if ($(this).next().hasClass('field-error'))
                            $(this).next().remove();
                    }
                    $('#newsletter-submit').removeAttr('disabled');
                    $('#newsletter-signup-form form').removeAttr('disabled');
                }
            });

        <h1 class="posttitle">

        </h1>

       <p>
        America, WAKE UP! Things are happening right in front of our noses, and we dont seem to grasp their significance, their danger to our way of life!
       </p>
       <p>
        Get this: Just five days ago, Robert Gates, former director of the CIA and former secretary of defense, and now president of the Boy Scouts of America,
        <a href="http://www.wnd.com/2015/05/boy-scouts-cave-only-the-beginning-warns-author/">
         made a declaration to the assembled executives of that wonderful, time-honored youth organization.
        </a>
       </p>
       <div class="left_prom_160 desktop-only">
        <div class="desktop-only" id="div-gpt-ad-story160LeftSide">
         <script type="text/javascript">
          googletag.cmd.push(function() { googletag.display('div-gpt-ad-story160LeftSide'); });
         </script>
        </div>
       </div>
       <p>
        We must deal with the world as it is, not as we might wish it to be.
       </p>
       <p>
        And what was he talking about? The cascading events, the incessant discrimination lawsuits, and the impending Supreme Court decision on same-sex marriage had led him to conclude that the hallowed rules that define the whole Scouting program cannot be sustained.
       </p>
       <p>
        Think about it. Although the Boy Scouts officially began in 1909, and the Girl Scouts a year later, as a voluntary, nonprofit, non-governmental organization, dedicated by oath and internal law to honor and morality and duty to God and country, this man who personally directed our national defense and investigative agencies is advocating surrender of the Scouts founding principles to the encroaching, virulent campaign of militant homosexual activists!
       </p>
       <p>
        Dear God, Bob Gates, what are you thinking?
       </p>
       <p>
        Is this how you see the world as it is? Has the United States of America become so weak, so lost, so unable to defend our very moral underpinnings, our national DNA, our constitutional defenses and rights, that we must cave in to relentless pressure on Scouting to embrace homosexuality, not just in boys and girls enrolled in Scouting but in the leaders who will be lasting influences on their young, trusting lives?
       </p>
       <p>
        How do you think America itself came to be? By accepting the world as it was? By kowtowing to the imperial dictates of the king of England? By continuing to be a colony of Great Britain? Even by listening to the 30 percent or more of the colonists who actually
        <i>
         preferred
        </i>
        subservience to the king rather than risk the dangers of revolution?
       </p>
       <p>
        No, Bob, just in case youve forgotten, our ancestors laid their lives, their fortunes, and their sacred honor on the line to create, under God, a new nation, conceived in liberty,
        <i>
         dedicated to Judeo-Christian biblical principles
        </i>
        , a new kind of country where people, all people, could be free to live and serve God however they chose. Remember?
       </p>
       <div class="module widget_wnd_prom_widget mobile-only">
        <div class="promo-300x250 mobile-only" id="wndm_story_midcontent_300x250">
         <script type="text/javascript">
          googletag.cmd.push(function() { googletag.display('wndm_story_midcontent_300x250'); });
         </script>
        </div>
       </div>
       <p>
        It wasnt capitulation, a meek surrender to things as they were. It was a moral stand, a scary but necessary commitment to do what was
        <i>
         right
        </i>
        , not what was expedient, or popular, or less trouble. It was exactly the reverse of what you just advised. It was a sacrificial commitment to create a society that existed as they wished it to be, so help them God.
       </p>
       <p>
        And thats exactly what happened. God helped them to create a better society than had ever existed before! Thats our heritage. Thats Americas DNA, our core values, the republic envisioned and codified by that incredible Constitution that decreed, among so many other things, FREEDOM  of religion, speech, the right to peacefully assemble, even to peacefully dissent from rules and regulations a Congress may have bound on us.
       </p>
       <p>
        The First Amendment simply and forcefully states, Congress shall make no laws respecting (or concerning) an establishment of religion, nor restricting the free exercise thereof ! Bob, though Scouting is not officially or primarily a religious organization, its founder, Lord Baden Powell, and his associates
        <i>
         presupposed
        </i>
        and consciously created an oath and Scout law that were infused with and founded on biblical principles. To do ones duty to God and Country is as clearly Bible-related as our Declaration of Independence: all men are
        <i>
         created
        </i>
        equal and endowed
        <i>
         by their Creator
        </i>
        with certain inalienable rights.
       </p>
       <p>
        And though Scouting does benefit from the same privileges other nonprofits like churches, schools and hospitals enjoy, in whatever ways their localities provide, it is not a government program. It is a private, voluntary, community-supported organization that can make its own rules about who can  and who cannot  be included in it programs, awards and leadership.
       </p>
       <p>
        Bob, I know youre a good man faced with tough decisions. You want to be fair and compassionate and still keep the Scouts what they were created to be. Right? You only signed on for two years as president; you dont want to leave Scouting permanently marred, totally changed into something the founders would be appalled by  and that will inevitably bring about the collapse and demise of Scouting itself.
       </p>
       <p>
        Tell the militants, who are obviously well-funded and dedicated to their causes and lifestyles, Create your own Scouting organization! Welcome homosexual leaders. Let parents who think their kids would benefit by that lifestyle  or suspect their kids are already homosexual  know they can have a welcoming, supportive group of like-minded people, committed to providing most, if not all, of original Scouting experiences!
       </p>
       <p>
        Why should the militants have to force their way and persuasions on a long honored, proven youth organization that is based on biblical foundations that are antithetical to what they stand for? It is a
        <i>
         free country, is it not?
        </i>
        Live like free and independent people 
        <i>
         and let others do the same.
        </i>
       </p>
       <p>
        Bob, dear Bob, the real issue is far deeper and more pervasive than just what you propose for Scouting. Our whole society, our very way of life, is on the line. Are we just to totally succumb to the world as it is and not continue to try to make it what we wish it to be? Do we just give up and become a pot-ridden populace, with all the mounting consequences? Do we just cave in to abortion advocates, who already account for a million or more infant deaths a year, here in the land of the brave and home of the free? Do you endorse the decisions of our president and former attorney general
        <i>
         not
        </i>
        to enforce immigration laws on our books?
       </p>
       <p>
        Do you side with those who want to remove Gods name from all public life, prayer from schools, even under God from our Pledge of Allegiance?
       </p>
       <p>
        In 1954, Special Counsel Joseph Welch confronted Sen. Joe McCarthy: At long last, have you left no sense of decency? No sense of decency?
       </p>
       <p>
        On issue after issue, Bob, our courts and our leaders are caving in, surrendering to political correctness and abandoning, not just now but perhaps for our kids future, all of the moral guidelines, the DNA that made America who we are.
       </p>
       <p>
        At long last, have we left no decency?
       </p>
