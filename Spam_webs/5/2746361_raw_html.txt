	function equalHeightColumns(group) {
		tallest = 0;
		width = jQuery(window).width();							
		group.each(function() {
			jQuery(this).removeAttr('style');			
			thisHeight = jQuery(this).height();
			if(thisHeight &gt; tallest) {
				tallest = thisHeight;
			}
		});
		if(width &gt; 767) {	
			group.height(tallest);	
		}
	}
