function MS_change() {
	// Set the country ISO code
	select_number = document.forms[1].ms.selectedIndex;
	if (select_number == 0) {
		document.forms[1].iso.value = "---";
	} else {
		document.forms[1].iso.value = document.forms[1].ms.options[select_number].value;
	}
	document.forms[1].vat.focus();
	return true;
}

function check(country,number) {
	$j('#result').html('validating VAT number...');
	$j.getJSON('http://isvat.appspot.com/'+$j('#iso').val() +'/' +$j('#vat').val() +'/?callback=?', function(data){ 
	//$j.getJSON('/order/vat-validation/' + $j('#iso').val() +'/' +$j('#vat').val() , function(data){ 
		if (data==false){
			$j('#result').html('Sorry, wrong VAT number.');  
			// only put to 21 if the outside EU is not selected
			if (!$j('#outside_EU').attr('checked')){
				simpleCart.taxRate=0.21;
				simpleCart.update();
				EU_vat=0;
			}
		}
		if (data==true){
			$j('#result').html('VAT number is correct.');
			if ($j('#iso').val()=="NL") $j("#result").html("VAT number is correct... but you are living in Holland.&lt;br&gt;So no 0 % VAT for you.");
			if ($j('#iso').val()!="NL")
			simpleCart.taxRate=0;
			simpleCart.update();
			EU_vat=1;
		}
	})
	.error(function() { $j("#result").html('Sorry something went wrong'); });
}
