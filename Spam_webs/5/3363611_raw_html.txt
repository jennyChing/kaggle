    var avail_days = Object();
    function load_days(year, month) {
        var key = year + '_' + month;
        if (avail_days[key] == 'undefined' || avail_days[key] == null) {
            $.get('/calendar/speedbump/' + year + '/' + month, function (data) {
                avail_days[key] = data;
                $('.gocomics_calendar .cal').datepicker('refresh');
            }, 'json');
        }
    }
