function createIconsDefinitions() {
	var hotJobIcon = new iconDefinition('hot-job', 
						RESOURCES.getResource("icons.urgentNeedJob"),
						function(job, isUserLogged, isAddToJobCartAction) {
							if(job.hotJob) {
								return true;
							}
							return false;
						});
	var addedToJobCart = new iconDefinition('added-to-job-cart', 
						RESOURCES.getResource("icons.addedToJobCartResource"),
							function(job, isUserLogged, isAddToJobCartAction) {
								if(isUserLogged == true &amp;&amp; isAddToJobCartAction &amp;&amp; job.addedToJobCart) {
									return true;
								}
								return false;
							});
	var draft = new iconDefinition('draft', 
						RESOURCES.getResource("icons.draftSubmission"),
							function(job, isUserLogged, isAddToJobCartAction) {
								if(isUserLogged  == true &amp;&amp; job.draft) {
									return true;
								}
								return false;
							});
	var alreadyAppliedOn = new iconDefinition('already-applied-on', 
						RESOURCES.getResource("icons.completedSubmission"),
							function(job, isUserLogged, isAddToJobCartAction) {
								if(isUserLogged  == true &amp;&amp; job.alreadyAppliedOn) {
									return true;
								}
								return false;
							});
	
	return new Array(hotJobIcon, addedToJobCart, draft, alreadyAppliedOn);
}
