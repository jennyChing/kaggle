    /**
   * Experiments class,
   * contain basic logic for all experiments.
   * Experiments enable if cookie value more than zero
   * @param options
   * @constructor
   */
  function Experiment(options) {
    if (!options) {
      console &amp;&amp; console.error('no Experiment options');
      return;
    }

    this.options = options;
    this.expValue = this._getCookieValue();

    if (!this.expValue) {
      this.expValue = this._generateExperimentValue();
      this._setCookie();
    }
  }

  Experiment.prototype = {

    /**
     * Return experiment`s cookie value
     * @returns {*}
     */
    getValue: function () {
      return this.expValue;
    },

    /**
     * Checks whether the passed value
     * in the value range of the experiment
     * @param expValue
     * @returns {boolean}
     */
    inRange: function (expValue) {
      var options = this.options;

      return expValue &gt;= options.min &amp;&amp; expValue &lt;= options.max;
    },

    logs: function () {
      window.debug &amp;&amp; console &amp;&amp; console.log('exp value: %s', expValue);
    },

    /**
     * Set experiment`s cookie value
     * @private
     */
    _setCookie: function () {
      var name = this.options.name;

      if (!name &amp;&amp; window.debug &amp;&amp; window.console) {
        window.console.error('Experiment can`t set cookie, because options not is set the name');
      }

      cookie.set(name, this.expValue.toString(), 30, window._sf.cookieDomain);
    },

    /**
     * Return experiment`s cookie value
     * @returns {*}
     * @private
     */
    _getCookieValue: function () {
      var val = cookie.get(this.options.name);

      if (val === '')
        return val;

      val = parseInt(val);

      if (isNaN(val))
        return '';

      return val;
    },

    /**
     * Validate experiment`s cookie value
     * @param expValue
     * @returns {boolean}
     * @private
     */
    _isNotCookieValue: function (expValue) {
      return expValue &lt; 0 || expValue === 0;
    },

    /**
     * Generate experiment`s value
     * If random value more than options proporsion
     * experiments not switch on.
     * @returns {*}
     * @private
     */
    _generateExperimentValue: function () {
      var options = this.options,
        chance = this._getRandom(1, 100);

      // skip experiments
      if (chance &gt; options.proporsion)
        return 0;

      // enable experiments
      return this._getRandom();
    },

    /**
     * Get random value between min and max
     * If arguments min, max not passed,
     * get value from options
     * @param min - {Number} minimal value
     * @param max - {Number} maximum value
     * @returns {*}
     * @private
     */
    _getRandom: function (min, max) {

      if (!min) {
        min = this.options.min;
      }

      if (!max) {
        max = this.options.max;
      }

      return Math.floor(Math.random() * (max - min + 1)) + min;
    }
  };
