/*
 * Allow GA events via `ga-event` attributes on elements, both those
 * static on the page and those added in dynamically.
 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')
 *
 * Elements that have been processed on the page already will have a 'ga-event-added' attribute
 *
 */
$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event
    //  non-interactive (optional) - true for non-interactive events (ie. it shouldn't be used in bounce-rate calcs)
    //
    // var ga_event_handler = function(ev) {
    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';
                
            // Only add a delegated handler once for a given even `type'
            if (!_handlers[type]) {
                // Delegate to handle dynamic elements, track element ref via closure 
                $(document).on(type, selector, (function() {
                    return function(ev) {
                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());
                _handlers[type] = true;
            }

            // mark element as processed
            $(this).attr('ga-event-added', 'true');
        });

        // wait for 100ms and poll again (catch dynmically added elements)
        window.setTimeout(poll_ga_attr_events, 150);
    }


    // let's go, start processing ...
    poll_ga_attr_events();

});

function listen(evnt, elem, func) {
    if (elem.addEventListener) { // W3C DOM
        elem.addEventListener(evnt,func,false);
    }
    else if (elem.attachEvent) { // IE DOM
        var r = elem.attachEvent("on"+evnt, func);
        return r;
    }
}

     <div class="module citizen_brand_piqa_question ">
      <div class="hgroup">
       <h2 class="heading">
        Q&amp;A with Nick Boutelier
       </h2>
      </div>
      <ul class="questionlist">
       <li asg-card="">
        <div class="qa">
         <div class="question fs14 fw600">
          <a href="http://www.answers.com/nickboutelier/Q/474997_what-is-isidewith-com">
           What is iSideWith.com?
          </a>
         </div>
         <div class="answer fs12">
          iSideWith is an interactive, non-partisan website that helps voters track how their political views compare to their friends, political parties, candidates and state ballot measures. Users answer a series of questions on important national issues including taxes, Medicare and Gay Marriage and local issues such as term limits and eliminating state income taxes. After they have completed the questionnaire, a complex algorithm provides a ranking, by percentage, of which friends, parties, candidates and issues they side with.
         </div>
         <li asg-card="">
          <div class="qa">
           <div class="question fs14 fw600">
            <a href="http://www.answers.com/nickboutelier/Q/474992_what-motivated-you-to-create-isidewith">
             What motivated you to create iSideWith?
            </a>
           </div>
           <div class="answer fs12">
            Taylor (Peck) and I were roommates after college and we both had very different backgrounds when it came to politics. I am a web engineer and artist who didn't have time to follow politics while Taylor was a political news junkie from a very young age. We didn't have a medium to compare and contrast how our views aligned. We created iSideWith as a tool to compare (issue by issue) our political beliefs. We then saw the opportunity for how this could save other voters time and how engaging the social aspect of it was. We believe that if voters are reminded which political issues are important to them and are shown how their views align with political candidates, they will be more likely to vote and influence their friends and family to do so. We designed our quiz to appeal to everyone from political junkies to undecided voters. Users can spend 3 minutes or 1 hour on the quiz. If you feel passionately about a topic (the economy for example) we allow you to expand the questions in that section to see more advanced questions. Politics is not always black and white, so we also allow users the ability to choose expanded versions of "yes, and..." or "no, but..." answers and even write in their own personal answer.
           </div>
           <li asg-card="">
            <div class="qa">
             <div class="question fs14 fw600">
              <a href="http://www.answers.com/nickboutelier/Q/474987_how-do-you-decide-what-questions-to-ask-on-isidewith-com">
               How do you decide what questions to ask on iSideWith.com?
              </a>
             </div>
             <div class="answer fs12">
              Through a combination of data analysis and user feedback. The news section of our site analyzes thousands of articles per day to find issues that are trending by region. We use that data combined with daily suggestions from users to add new questions. If we see a local or national issue explode in popularity during the day, we will usually have it up on the quiz later that day. Since we display the live results of all of our data, it's really interesting to instantly see how America is dividing on a new issue that didn't exist earlier.
             </div>
             <li asg-card="">
              <div class="qa">
               <div class="question fs14 fw600">
                <a href="http://www.answers.com/nickboutelier/Q/474982_what-changes-are-you-making-to-isidewith-for-the-2014-elections">
                 What changes are you making to iSideWith for the 2014 elections?
                </a>
               </div>
               <div class="answer fs12">
                For the 2014 midterm elections, we focused on five important questions voters will ask in November: "Who" should they vote for, "What" can they vote on, "When" can they vote, and "Where" can they vote. We then designed a voter guide that organizes all the answers to those questions into one highly personalized and localized resource that will save voters hours of researching which local candidates and ballot measures they side with all the way down to their city officials. We launched a beta version of our voter guide in mid July covering most of their state officials and will be adding city officials on a daily basis starting in August.
               </div>
               <li asg-card="">
                <div class="qa">
                 <div class="question fs14 fw600">
                  <a href="http://www.answers.com/nickboutelier/Q/474977_what-has-been-the-most-surprising-thing-about-the-results-you-find-with-isidewith">
                   What has been the most surprising thing about the results you find with iSideWith?
                  </a>
                 </div>
                 <div class="answer fs12">
                  It was surprising to see there are some issues where voters strongly divide with their party leaders. This is more apparent on a city and county level, but it's very interesting to see that leaders can still be very out of touch with their constituents on specific issues. Last year we started segmenting answers by U.S. Census data, so it's also interesting to dig through the polling results pages to see see how users of different income, education, and race divide on issues.
                 </div>
                 <li asg-card="">
                  <div class="qa">
                   <div class="question fs14 fw600">
                    <a href="http://www.answers.com/nickboutelier/Q/474972_according-to-the-center-for-the-study-of-the-american-electorate-only-57-5-of-all-eligible-voters-cast-a-ballot-for-the-2012-presidential-election-do-you-think-social-media-sites-like-isidewith-will-help-increase-the-number-of-americans-who-vote-i">
                     According to the Center for the Study of the American Electorate, only 57.5% of all eligible voters cast a ballot for the 2012 Presidential Election. Do you think social media sites like iSideWith will help increase the number of Americans who vote in the 2016 election?
                    </a>
                   </div>
                   <div class="answer fs12">
                    <p>
                     Yes, we feel that discussing issues with friends helps voters organize and clarify their beliefs as well as getting them more engaged and passionate about voting. The challenge for social media, in terms of politics, is creating an environment that fosters productive political conversations. Some social media sites are notorious for unproductive political discussion which turns many people off to politics.
                    </p>
                    <p>
                     Earlier this year we tested a forum concept with two separate groups of our users. The first group of users were allowed to see how they sided with each user they entered a conversation with. The second group was not allowed to see how they sided with the user they were having a conversation with. It was amazing to see that the first half of users resulted in much more productive conversations, even on issues they had polarizing viewpoints on. Users were seeking out users with opposing beliefs, wanting to hear their point of view. The second group of users had a much greater number of conversations that died quickly or ended in flame wars. We noticed that when users enter a conversation knowing where the other person is coming from, it creates a buffer of understanding, intrigue, and productive conversation.
                    </p>
                   </div>
                   <li asg-card="">
                    <div class="qa">
                     <div class="question fs14 fw600">
                      <a href="http://www.answers.com/nickboutelier/Q/474967_why-do-you-think-so-many-americans-choose-not-to-vote-in-national-mid-term-elections">
                       Why do you think so many Americans choose not to vote in national mid-term elections?
                      </a>
                     </div>
                     <div class="answer fs12">
                      In the past I didn't vote during midterm elections because it was too difficult. It took a lot of time and it was difficult to gather all the information. Once I found the information, it was filled with rhetoric designed to mislead instead of give a straight stance on policy issues. Our goal for the midterm elections is to organize information so users can make decisions based on how a policy or candidate aligns with their political beliefs, instead of rhetoric.
                     </div>
                     <li asg-card="">
                      <div class="qa">
                       <div class="question fs14 fw600">
                        <a href="http://www.answers.com/nickboutelier/Q/474962_have-you-received-any-feedback-from-visitors-who-took-the-political-party-quiz">
                         Have you received any feedback from visitors who took the Political Party Quiz?
                        </a>
                       </div>
                       <div class="answer fs12">
                        Yes, our whole model and processes of the site is built around analyzing and implementing user feedback. It's absolutely critical for monitoring our accuracy and developing new features. On an average day, we receive over 1,500 suggestions for adding additional answers to questions on our quiz. We eventually had to build internal tools to help us process all the feedback we receive to find common threads so we can quickly respond and act on user feedback.
                       </div>
                       <li asg-card="">
                        <div class="qa">
                         <div class="question fs14 fw600">
                          <a href="http://www.answers.com/nickboutelier/Q/474957_do-visitors-tend-to-align-with-a-single-political-party-on-the-quiz-or-are-they-generally-a-blend-of-alignments-what-do-you-think-that-says-about-visitors-about-the-party-system">
                           Do visitors tend to align with a single political party on the quiz, or are they generally a blend of alignments? What do you think that says about visitors? About the party system?
                          </a>
                         </div>
                         <div class="answer fs12">
                          We tend to see a blend of alignments. There is a lot of overlap between parties, especially when you add in smaller, less known parties. However, even if two parties were to have the same stance on every issue, they would most likely differ in how they rank the importance each issue. This is why we strongly recommend that users select which issues are most or least important to them so they can be matched with a party that not only reflects their answers, but also their priority of issues.
                         </div>
                         <li asg-card="">
                          <div class="qa">
                           <div class="question fs14 fw600">
                            <a href="http://www.answers.com/nickboutelier/Q/474952_has-isidewith-been-as-successful-overseas-as-it-has-been-in-the-us">
                             Has iSideWith been as successful overseas as it has been in the US?
                            </a>
                           </div>
                           <div class="answer fs12">
                            Yes, and we would probably say that it has been most successful in the UK. After the 2012 Presidential Election, we focused on building internal tools that would allow us to scale our model internationally. Our system allows us to quickly find which issues are trending in a country as well as finding all the varying stances on those issues. The refinement of that process combined with the UK's passion for politics would explain the success of iSideWith overseas. Last year, we overcame the challenge of scaling internationally.Earlier this year, we feel we finally cracked the barriers to scaling this on a hyper-local city politics level and will be using a brand new set of features to help voters choose their local candidates and proposed legislation in the 2014 US Midterm and the 2015 Canadian and UK General elections.
                           </div>
                           <li asg-card="">
                            <div class="qa">
                             <div class="question fs14 fw600">
                              <a href="http://www.answers.com/nickboutelier/Q/474947_after-users-finish-the-quiz-what-percentage-of-them-say-the-results-are-very-accurate-or-other-responses">
                               After users finish the quiz, what percentage of them say the results are "Very Accurate" or other responses?
                              </a>
                             </div>
                             <div class="answer fs12">
                              The percentage is usually around 85% and we have noticed that the remaining 15% is where voter ideology diverges from policy. That's an important metric we track on a daily basis for quality control and to pinpoint where voters are diverging from leaders within their party by issue.
                             </div>
                             <li asg-card="">
                              <div class="qa">
                               <div class="question fs14 fw600">
                                <a href="http://www.answers.com/nickboutelier/Q/474942_how-do-the-majority-of-your-users-hear-about-your-site">
                                 How do the majority of your users hear about your site?
                                </a>
                               </div>
                               <div class="answer fs12">
                                Through friends comparing answers with each other, and sharing with their friends and family. Once users see how easy it is to compare their beliefs with the candidates', they find it especially fascinating to then see which of their friends most closely aligns with their beliefs. It's very interesting to see that, for example, you side 95% with your best friend, 80% with your dad, 30% with your girlfriend, etc. That's probably one of the most addictive features of the site that keeps it growing outside of the election season.
                               </div>
                               <li asg-card="">
                                <div class="qa">
                                 <div class="question fs14 fw600">
                                  <a href="http://www.answers.com/nickboutelier/Q/474937_you-describe-yourself-on-the-site-as-a-lifelong-undecided-voter-how-has-isidewith-helped-you-personally-stay-updated-on-important-issues-would-you-say-youre-still-undecided">
                                   You describe yourself on the site as a "lifelong undecided voter." How has iSideWith helped you personally stay updated on important issues? Would you say you're still undecided?
                                  </a>
                                 </div>
                                 <div class="answer fs12">
                                  I would say that I am now definitely decided. While I'm still not as passionate about politics like Taylor is, I use iSideWith to keep me informed and educated. I check the personalized news section every morning to filter the top stories down to the issues that I care most about. I compare answers with friends a lot. I'm also really excited about our midterm election voter guide and can't imagine how hard it would be to vote without it. Taylor is really into checking out our polling data pages and watching trends change over time or by region. He is also working on a mock election concept which sounds really interesting. The combination of Taylor and I working on this is what makes the site useful and interesting for users of all levels of political interest and provides an educational, engaging foundation for voters to really understand what (and who) they are voting for.
                                 </div>
