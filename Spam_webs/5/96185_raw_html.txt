             <h2 class="recipe-title">
              Gluten Free Coffee Cake Recipe
             </h2>
             <div class="time-servings">
              <div class="recipe-time" itemprop="totalTime">
               <strong>
                Total Time:
               </strong>
               35 minutes
              </div>
              <div class="recipe-yield" itemprop="recipeYield">
               <strong>
                Serves:
               </strong>
               9
              </div>
             </div>
             <h3>
              Ingredients:
             </h3>
             <ul class="list-ingredients">
              <li itemprop="ingredients">
               1/4 c.
               <a href="https://secure.ttpurchase.com/930D49B9-D357-8845-1B016AEC243D055A" target="_blank">
                coconut oil
               </a>
              </li>
              <li itemprop="ingredients">
               1/4 c.
               <a href="http://amzn.to/1vS8zKi" target="_blank">
                honey
               </a>
              </li>
              <li itemprop="ingredients">
               1 egg
              </li>
              <li itemprop="ingredients">
               1 tsp.
               <a href="http://amzn.to/1ocNUAT" target="_blank">
                Vanilla Extract
               </a>
              </li>
              <li itemprop="ingredients">
               1/2 c.
               <a href="http://amzn.to/1mtkIUN" target="_blank">
                coconut milk
               </a>
              </li>
              <li itemprop="ingredients">
               1 1/2 c.
               <a href="http://amzn.to/1n0DPl1" target="_blank">
                almond flour
               </a>
              </li>
              <li itemprop="ingredients">
               1 tsp.
               <a href="http://amzn.to/1nbkJdm" target="_blank">
                Cinnamon
               </a>
              </li>
              <li itemprop="ingredients">
               2 tsp.
               <a href="http://amzn.to/1zTRgfN" target="_blank">
                Baking Powder
               </a>
              </li>
              <li itemprop="ingredients">
               1/2 tsp.
               <a href="http://amzn.to/1v811bs" target="_blank">
                Sea Salt
               </a>
              </li>
              <li itemprop="ingredients">
               1/2  3/4 c.
               <a href="http://amzn.to/1n0DPl1" target="_blank">
                almond flour
               </a>
              </li>
              <li itemprop="ingredients">
               1 1/2 tbsp.
               <a href="https://secure.ttpurchase.com/930D49B9-D357-8845-1B016AEC243D055A" target="_blank">
                coconut oil
               </a>
              </li>
              <li itemprop="ingredients">
               1 tbsp.
               <a href="http://amzn.to/1vS8zKi" target="_blank">
                honey
               </a>
              </li>
              <li itemprop="ingredients">
               2 tsp.
               <a href="http://amzn.to/1npUBue" target="_blank">
                Cinnamon
               </a>
              </li>
             </ul>
             <h3>
              Directions:
             </h3>
             <ol class="list-directions">
              <li itemprop="recipeInstructions">
               Preheat oven to 350.
              </li>
              <li itemprop="recipeInstructions">
               Stir coconut oil, egg, vanilla and honey together in a bowl.
              </li>
              <li itemprop="recipeInstructions">
               In a small bowl, combine almond flour, cinnamon, baking powder, and sea salt
              </li>
              <li itemprop="recipeInstructions">
               Add dry ingredients to wet along with milk, and stir until smooth.
              </li>
              <li itemprop="recipeInstructions">
               Pour into an 88 baking pan.
              </li>
              <li itemprop="recipeInstructions">
               In a small bowl mix the almond flour, coconut oil, honey, and cinnamon for topping.
              </li>
              <li itemprop="recipeInstructions">
               Spread the topping over the cake.
              </li>
              <li itemprop="recipeInstructions">
               Bake for 20  25 minutes,
              </li>
             </ol>
