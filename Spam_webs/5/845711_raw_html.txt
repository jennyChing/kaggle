function ajaxGetMarkers(ajax,geoloc,rewriteCategory,rewriteLocation,reset) {

	showLoading();

	radius = new Array();

	var category = 0;
	var location = 0;
	var search = '';

	if(ajax){
		if(rewriteCategory){
			category = rewriteCategory;
		} else {
			category = $('#dir-searchinput-category-id').val();
		}
		if(rewriteLocation){
			location = rewriteLocation;
		} else {
			location = $('#dir-searchinput-location-id').val();
		}
		search = $('#dir-searchinput-text').val();

		var ajaxGeo = $('#dir-searchinput-geo').attr("checked");

		if(ajaxGeo &amp;&amp; !reset){
			var inputRadius = geoInputRadius.val();
			if(!isNaN(inputRadius)){
				radius.push(parseInt(inputRadius));
			} else {
				geoInputRadius.val(geoInputRadius.data('default-value'));
				radius.push(parseInt(geoInputRadius.data('default-value')));
			}
			radius.push(parseFloat(geoInputLat.val()));
			radius.push(parseFloat(geoInputLng.val()));
		}
	} else {
		if(reset){
			category = parseInt(mapDiv.data('category'));
			location = parseInt(mapDiv.data('location'));
			search = mapDiv.data('search');
		} else {
			if(geoloc){
				radius.push(parseInt(5));
				radius.push(geoloc.lat());
				radius.push(geoloc.lng());

				category = parseInt(mapDiv.data('category'));
				location = parseInt(mapDiv.data('location'));
				search = mapDiv.data('search');
			}
		}
	}
	// get items from ajax
	$.post(
		// MyAjax defined in functions.php
		MyAjax.ajaxurl,
	    {
	        action : 'get_items',
	        category : category,
	        location : location,
	        search : search,
	        radius : radius
	    },
	    function( response ) {
	    	// show reset ajax button
	    	if((!reset) &amp;&amp; (!geoloc) &amp;&amp; ($('#directory-search').data('interactive') == 'yes')){
	    		$('#directory-search .reset-ajax').show();
	    	}
	    	if((!reset) &amp;&amp; (!geoloc)){
	    		generateContent(response);
	    	}
	    	if(ajaxGeo &amp;&amp; !reset){
	    		var ajaxGeoObj = new google.maps.LatLng(parseFloat(geoInputLat.val()),parseFloat(geoInputLng.val()));
	    		generateMarkers(response,ajaxGeoObj,true);
	    	} else {
	    		generateMarkers(response,geoloc);
	    	}
	    }
	)
	.fail(function(e) { console.log("AJAX ERROR", e); });
}

function generateContent(data) {

	var length = $.map(data, function(n, i) { return i; }).length;

	contentDiv.find('.ajax-content').remove();
	var title;
	if(length == 0){
		title = $('&lt;header class="entry-header"&gt;&lt;h1 class="entry-title"&gt;&lt;span&gt;No result found&lt;/span&gt;&lt;/h1&gt;&lt;/header&gt;');
	} else {
		title = $('&lt;header class="entry-header"&gt;&lt;h1 class="entry-title"&gt;&lt;span&gt;Search result&lt;/span&gt;&lt;/h1&gt;&lt;/header&gt;');
	}
	
	var html;
	if(length &gt; 0){
		html = $('&lt;ul class="items"&gt;&lt;/ul&gt;');
	}
	var limit = 10;
	if(limit &gt; length) {
		limit = length;
	}
	var i = 0;
	for(var key in data){
		var thumbnailHtml;
		if(data[key].timthumbUrlContent){
			var thumbnailHtml = '&lt;div class="thumbnail"&gt;&lt;img src="'+data[key].timthumbUrlContent+'" alt="Item thumbnail" /&gt;&lt;div class="comment-count"&gt;'+data[key].comment_count+'&lt;/div&gt;&lt;/div&gt;';
		} else {
			thumbnailHtml = '';
		}
		var rating = '';
		if(data[key].rating){
			rating += '&lt;span class="rating"&gt;';
			for (var i = 1; i &lt;= data[key].rating['max']; i++) {
				rating += '&lt;span class="star';
				if(i &lt;= data[key].rating['val']) {
					rating += ' active';
				}
				rating += '"&gt;&lt;/span&gt;';
			}
			rating += '&lt;/span&gt;';
		}
		var descriptionHtml = '&lt;div class="description"&gt;&lt;h3&gt;&lt;a href="'+data[key].link+'"&gt;'+data[key].post_title+'&lt;/a&gt;'+rating+'&lt;/h3&gt;'+data[key].excerptDir+'&lt;/div&gt;';
		html.append('&lt;li class="item clear"&gt;'+thumbnailHtml+descriptionHtml+'&lt;/li&gt;');
		if(i &lt;= limit){
			i++;
		} else {
			break;
		}
	};
	ajaxContent = $('&lt;div class="ajax-content"&gt;&lt;/div&gt;').html(title).append(html);
	contentDiv.find('&gt;').hide();
	contentDiv.append(ajaxContent);
}
