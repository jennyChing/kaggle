  <script>
   /*
 *  Validation form with jQuery and animation.css
 *  
 *  Check for every input on onBlur event
 *  
 *  1. Check if input is empty
 *  2. Check if e-mail is invalid
 *  3. Check match of passwords
 *  
 *  Feel free to suggest improvements! _o/
 */

    $(document).ready(function () { 
        $('form').on('blur', 'input', checkForValidations);
        $('form').on('focus', 'input', removeValidationClass);
    });

    function checkForValidations(evt) {
        var $obj = $(this);
        if ($obj.hasClass('required')) {
            if (emptyField($obj)) {
                shake($obj);
                $obj.siblings(".error").text("Please enter a value for this field.");
            } else {
                $obj.removeClass().addClass('required').addClass('success');
                $obj.siblings(".error").text("");
            }
        }
        if ($obj.hasClass('emailcheck')) {
            if (!emptyField($obj)) {
                if (validaEmail($obj) == false) {
                    $obj.siblings(".error").text("Your email address you've entered does not seem valid. Please re-type it in carefully.")
                } else {
                    $obj.siblings(".error").text("");
                };

            } else { $obj.siblings(".error").text("Please enter a value for this field."); };
        }
        if ($obj.hasClass('passcheck')) {
            if (!emptyField($obj)) {
                if (matchFields($obj) == false) {
                    $obj.siblings(".error").text("Your passwords don't match. Please re-type the password again.")
                } else {
                    $obj.siblings(".error").text("");
                };

            } else { $obj.siblings(".error").text("Please enter a value for this field."); };
        }
        if ($obj.hasClass('mobilecheck')) {
            if (!emptyField($obj)) {
                if (validaMobile($obj) == false) {
                    $obj.siblings(".error").text("Please enter a 10-digit South African mobile number.")
                } else {
                    $obj.siblings(".error").text("");
                };

            } else { $obj.siblings(".error").text("Please enter a value for this field."); };
        }
        if ($obj.hasClass('RSAIDcheck')) {
            
            if (!emptyField($obj)) {
                if (validaRSAID($obj) == false) {
                    $obj.siblings(".error").text("Please enter a valid South African ID Number.")
                } else if (validaRSAID($obj) == true) {
                    $obj.siblings(".error").text("");
                };
            } else { $obj.siblings(".error").text("Please enter a value for this field."); };
        }
        if ($obj.hasClass('strongpass')) {
            if (!emptyField($obj)) {
                if (strongPassword($obj) == false) {
                    $obj.siblings(".error").text("The password you've entered isn't strong enough. Please try again.")
                } else {
                    $obj.siblings(".error").text("");
                };

            }
        }
    }

    function removeValidationClass(evt) {
        $(this).removeClass('error animated shake');
    }

    function emptyField($objeto) {
        return (($objeto.val() == '') ? true : false);
    }

    function shake($objeto) {
        /*$objeto.effect('shake', { // needs jQuery UI
            distance: 10,
            times: 2
        });*/
        $objeto.removeClass('success').addClass('error animated shake');
    }

    function validaEmail($objeto) {
        var er = RegExp(/^[A-Za-z0-9_\-\.\+]+@[A-Za-z0-9_\-\.\+]{2,}\.[A-Za-z0-9\+]{2,}(\.[A-Za-z0-9]\+)?/);
        if (er.test($objeto.val()) == false) {
            shake($objeto);
            return false;
        } else {
            $objeto.removeClass().addClass('emailcheck').addClass('success');
            return true;
        }
    }

    function validaMobile($objeto) {
        var er = RegExp(/^\d{10}$/)
        if (er.test($objeto.val()) == false) {
            shake($objeto);
            return false;
        } else {
            $objeto.removeClass().addClass('mobilecheck').addClass('success');
            return true;
        }
    }

    function validaRSAID($objeto) {
        var selected = $("input[type='radio'][name='lockerIdOption']:checked");

        if (selected[0].value == 1 &amp;&amp; validateID($objeto.val()) == false) {
            shake($objeto);
            return false;
        } else if (selected[0].value == 1 &amp;&amp; validateID($objeto.val()) == true) {
            $objeto.removeClass().addClass('RSAIDcheck').addClass('success');
            return true;
        }
        else {
            $objeto.removeClass().addClass('RSAIDcheck').addClass('success');
            return true;
        }
    }

    function matchFields($objeto) {
        var parentID = '#' + $objeto.attr('ID').replace('2', '');
        if ($(parentID).val() != $objeto.val()) {
            shake($objeto);
            return false;
        } else {
            $objeto.removeClass().addClass('passcheck').addClass('success');
            return true;
        }
    }

    function strongPassword($objeto) {
        var p = $objeto[0].value;
        var label1 = 0;
        var label2 = 0;
        var label3 = 0;
        var label4 = 0;
        var label5 = 0;
        if (p.length &gt; 7) {//min length
            label1 = 1;
        }
        //if (/[!@#$%^&amp;*?_~]{2,}/.test(p)) {//min 2 special characters
        //    label2 = 1;
        //}
        if (/[a-zA-Z]{1,}/.test(p)) {//min 2 a-z
            label3 = 1;
        }
        //if (/[A-Z]{2,}/.test(p)) {//min 2 A-Z
        //    label4 = 1;
        //}
        if (/[0-9]{1,}/.test(p)) {//min 2 0-9
            label5 = 1;
        }

        var strong_password = label1 + label2 + label3 + label4 + label5;

        if (strong_password &lt; 3) {
            shake($objeto);
            return false;
            } else {
            $objeto.removeClass().addClass('strongpass').addClass('success');
            return true;
            }
    }
