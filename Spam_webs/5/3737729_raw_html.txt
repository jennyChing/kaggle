		var images_loaded = false;

			if (!images_loaded) {

			}
			if (window.isTouch || menu_button.filter(':visible').length) return;

			$(".menu-item.expanded", nav).removeClass('expanded');

			// If an input field is focused, don't close
			if ($("input:focus", $(this)).length) {
				$(this).addClass('expanded');
			}
			// If an input field in another tab is focused, unfocus it
			else if ($(".menu-item input:focus", nav).length) {
				if (document.activeElement) document.activeElement.blur();
			}

				nav.removeClass('expanded');
				$(".menu-item", nav).removeClass('expanded');
			}
			else {
				nav.addClass('expanded');
				search.removeClass('expanded');
				settings.removeClass('expanded');
				if (document.activeElement) document.activeElement.blur();
			}
			return false;

			var li = $(this).parent();
			if (!$(".dropdown-content", li).length) return;

			var is_expanded = li.hasClass('expanded');

			$(".menu-item.expanded", nav).removeClass('expanded');
			$(".menu-item .nav-link .caret", nav).removeClass('icon-angle-down').addClass('icon-angle-right');
			if (!is_expanded) {
				li.addClass('expanded');
				$('.nav-link .caret', li).removeClass('icon-angle-right').addClass('icon-angle-down');
				nav.addClass('expanded');
			}

			return false;
