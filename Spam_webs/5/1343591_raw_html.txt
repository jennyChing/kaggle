	function updateQueryStringParameter(uri, key, value) {
  var re = new RegExp("([?&amp;])" + key + "=.*?(&amp;|$)", "i");
  var separator = uri.indexOf('?') !== -1 ? "&amp;" : "?";
  if (uri.match(re)) {

  }
  else {

  }
