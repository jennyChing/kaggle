$('.astar').mouseover(function(){
    var rate_number = $(this).index() + 1;
    render_rank(rate_number);
});

$('.astar').click(function(){
    var rate_number = $(this).index() + 1;
    save_rank(rate_number);
});

function render_avg_rank(){ // rank is between 0 to 5
    $('.has_star2').removeClass('has_star2');
    $('.astar2').each(function(i){
        if(i&lt;avg_rank)
            $(this).addClass('has_star2');
    });
}

function render_rank(rank){ // rank is between 0 to 5
    $('.has_star').removeClass('has_star');
    $('.astar').each(function(i){
        if(i&lt;rank)
            $(this).addClass('has_star');
    });
}

function reset_rank(){
    render_rank(current_rank);
}

function save_rank(rank){
    current_rank = rank;
    
    // push to server
    var data = {rank:rank, csrfmiddlewaretoken:$('[name=csrfmiddlewaretoken]').val()};
    $.post('/rank_quote/'+slug+'/', data, function(data) {
      //console.log(data);
    });
}

var current_rank = 0;
var avg_rank = 3.3125;
var slug = 'some-men-storm-imaginary-alps';

function display_quote(){
    $('#main')[0].visibility = 'hidden';
    reset_paddings();
    $('#main')[0].visibility = 'visible';
    $('#main').animate({'zoom':'120%', opacity:0},1);
    $('#main').animate({'zoom':'100%', opacity:1}, 550);
    if($('#main').height() &gt; 55){
        $('#cbracket').css('marginTop', $('#main').height()-30);
    }
    
    $('h1').css('marginLeft', ($('#main').width() - $('h1').width())/2);

}


$(document).ready(function(){
    setTimeout("display_quote();", 00);
    render_avg_rank();
    reset_rank();
    
    // enable search functionality
    $('#homepsearchi')[0].myval = '';
    
    $('#homepsearchi').keyup(function(){
        if(!this.value || this.value == this.myval)
            return;
        
        this.myval = this.value;
        
        $.ajax({
            url: "/suggest?q="+this.value,
            success: function(t){
                $('#sugges').html(t);
                $('#sugges').css('marginTop', -1 * $('#sugges').height());
            }
        });
    });

});

});

function let_server_know_about_this_like(){
    // facebook event on like-press
    FB.Event.subscribe('edge.create',
      function(response) {
          $.ajax({
            url: "/manage/record_fb_like/",
            data: { url : response, slug:slug },
          });
      }
    );
}
