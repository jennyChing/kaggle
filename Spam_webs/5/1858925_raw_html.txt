        function formValidation(formId, emailId, emailId2) {
          var isValid = true;
          var reg = /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/;
          var address = $('#'+emailId).val();
          var address2 = $('#'+emailId2).val();
          $('.err').hide();
          $('#'+formId+' .required').each(function(){
            checkVal = $(this).val();
            if (checkVal.length&lt;2) {
              isValid = false;
              $('#'+$(this).attr('id')+'Err').focus();
				      $('#'+$(this).attr('id')+'Err').show();
              return false;
            }
          });

          if ( isValid ) {
            if ( reg.test(address) == false ) {
              isValid = false;
              $('#'+emailId).focus();
              $('#'+emailId+'Err').html('Invalid email address!');
				      $('#'+emailId+'Err').show();
            }
            
            if ( reg.test(address2) == false ) {
              isValid = false;
              $('#'+emailId2).focus();
              $('#'+emailId2+'Err').html('Invalid email address!');
				      $('#'+emailId2+'Err').show();
            }
          }
          return isValid;
        }
