      $(function() {

					$times = $('.times .time');
					if ($times.length) {

						function updateClocks ( ) {

							var str, colonIndex, hours, minutes, ampm, newStr;
							
							$times.each(function(i) {

								str = $(this).text();
								colonIndex = str.indexOf(':');
								minutes = Number(str.substring(colonIndex + 1));
								hours = Number(str.substring(0, colonIndex));

								minutes += 1;

								if (minutes &lt; 10) {
									
									minutes = '0' + minutes;

								} else if (minutes == 60) {
									
									minutes = '00';
									
									hours += 1;

									if (hours == 12) {

										ampm = $(this).next().text().toLowerCase();
										ampm = (ampm == 'am') ? 'pm' : 'am';
										$(this).next().text(ampm);

									} else if (hours &gt; 12) {
										
										hours = 1;
										
									}
								}
								
								newStr = hours + ':' + minutes;

								$(this).text(newStr);

								// console.log(colonIndex, str, newStr);
							});

							// update the clocks every 60 seconds
							setTimeout(updateClocks, 60 * 1000);
						}

					}

					// start updating clocks at right time
					setTimeout(updateClocks, 12 * 1000);

				});
