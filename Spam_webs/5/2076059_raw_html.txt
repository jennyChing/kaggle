            function UpVoteContent(appId, pictureId) {
                $.ajax({
                    url: '/api/v1/ExternalContent/Like',
                    type: "POST",
                    data: {
                        appid: appId,
                        externalid: pictureId
                    },
                    success: function (data) {
                        if (data.error != null) {
                            toastr.error(data.message);
                        }
                        else {
                            toastr.info("Image rating submitted.");

                            $.post('/ExternalAjax/RefreshContentStats',
                                {
                                    appid: appId,
                                    picid: pictureId,
                                },
                            function (data) {
                                $('#contentStats').empty();
                                $('#contentStats').append(data);
                            });
                        }
                    },
                    error: function (jqXHR, exception) {
                        if (jqXHR.status == 401) {
                            toastr.warning("Must be logged in to rate content");
                        }
                    }
                });
            }

                                    sort: "top"
                                },
                            function (data) {

                    }
                });
