        Stylish Simplicity - Kisui Wedding Dresses
       </h1>
       <p>
        Looking for a slick and stylish wedding dress? The 2014 Kisui collection could have the one for you...
       </p>
       <img alt="AshleyLudaescherPhotography Feature Image Kisui" class="mob featured-image" src="http://images.onefabday.com/2014/03/AshleyLudaescherPhotography-Feature-Image-Kisui.jpg" width="620"/>

        <a href="the-style/">
         <span class="tag">
          The Style
         </span>
        </a>

         </a>
        </span>
       </div>
       <div class="text-block">
        <p>
         Slick and simple with a strong design aesthetic we just love Kisui's latest collection of wedding dresses. A cohesive collection
         <a href="http://kisui-bride.com/" target="_blank" title="Kisui">
          Kisui

                name = name.replace(/[\[]/, "\\[").replace(/[\]]/, "\\]");
                var regex = new RegExp("[\\?&amp;]" + name + "=([^&amp;#]*)"),
                    results = regex.exec(location.search);
                return results === null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));
            }

            $('.countrySelect').change(function() {
                if ((typeof this.value !== 'undefined') &amp;&amp; (this.value)) {

                    var regionData = {};
                    regionData.dropdown = this.value;

                    if (this.value == 'ni') {
                        regionData.region = 'ie';
                    } else if (this.value == 'us' || this.value == 'au') {
                        regionData.region = 'int';
                    } else {
                        regionData.region = this.value;
                    }
                    $.cookie('regiondata', JSON.stringify(regionData), {
                        expires : 30,
                        path    : '/'
                    });
                    window.location.href = _ofd.internationalise.links[regionData.region];
                }
            });

            if ((getParameterByName('redirect') !== '1')
                &amp;&amp; (((_ofd.internationalise.cookie === '') &amp;&amp; (_ofd.internationalise.region !== _ofd.internationalise.country))
                || ((_ofd.internationalise.cookie !== '') &amp;&amp; (_ofd.internationalise.region !== _ofd.internationalise.cookie)))
            ) {
                $('#region-inline').show();
            }

            if (_ofd.internationalise.country !== 'ie' &amp;&amp; getParameterByName('redirect') === '1') {

                //only show on desktop
                if ($('.nav-bar').is(":visible") === true) {

                    //Check if in allowed regions

                    $('#region-overlay').popup({
                        transition: 'all 0.3s',
                        scrolllock: true,
                        autozindex: true,
                        vertical: 'top',
                        onclose: function() {
                            ga('send', 'event', 'popup', 'click', _ofd.internationalise.country + '-close');

                            var regionData = {};
                            regionData.dropdown = _ofd.internationalise.country;
                            regionData.region = _ofd.internationalise.country;

                            $.cookie('regiondata', JSON.stringify(regionData), {
                                expires : 30,
                                path    : '/'
                            });
                        }
                    });
                    $('#region-overlay').popup('show');
                }
            }

            $('.cookie-uk').click(function() {
                ga('send', 'event', 'popup', 'click', $(this).data('reference'));
                _ofd.internationalise.country = 'uk';
                var regionData = {};
                regionData.dropdown = 'uk';
                regionData.region = 'uk';

                $.cookie('regiondata', JSON.stringify(regionData), {
                    expires : 30,
                    path    : '/'
                });
                $('#region-overlay').popup('hide');
                window.location.href = _ofd.internationalise.links[_ofd.internationalise.country];
            });

            $('.cookie-ie').click(function() {
                ga('send', 'event', 'popup', 'click', $(this).data('reference'));
                _ofd.internationalise.country = 'ie';
                var regionData = {};
                regionData.dropdown = 'ie';
                regionData.region = 'ie';

                $.cookie('regiondata', JSON.stringify(regionData), {
                    expires : 30,
                    path    : '/'
                });
                $('#region-overlay').popup('hide');
                window.location.href = _ofd.internationalise.links[_ofd.internationalise.country];
            });

            $('.cookie-int').click(function() {
                ga('send', 'event', 'popup', 'click', $(this).data('reference'));
                _ofd.internationalise.country = 'int';
                var regionData = {};
                regionData.dropdown = 'int';
                regionData.region = 'int';

                $.cookie('regiondata', JSON.stringify(regionData), {
                    expires : 30,
                    path    : '/'
                });
                $('#region-overlay').popup('hide');
                window.location.href = _ofd.internationalise.links[_ofd.internationalise.country];
            });
    });
