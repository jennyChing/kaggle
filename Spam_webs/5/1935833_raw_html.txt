          </p>
         </div>
         <div class="secondPar">
          <p>
           The unsuspecting walker picked up the barnacle-encrusted "stone" for his pet to chase along a beach without realising it was a deadly WW2 explosive.
          </p>
         </div>
         <div class="thirdPar">
          <p>
           A passing bomb specialist spotted the danger and alerted the police and coastguard, who set up a 30-yard cordon around the grenade. It was later taken away by explosive ordnance disposal (EOD) officers.
          </p>
         </div>
         <div class="fourthPar">
          <p>
           The device was discovered washed up on the shoreline at Marine Parade in Harwich, Essex - the fifth in as many weeks.
          </p>
         </div>
         <div class="fifthPar">
          <p>
           Essex Police said a grenade was found yesterday on the beach near West End Lane and The Promenade while three were found near Lower Marine Parade on July 12.
          </p>

                    bindUIActions();

                    var displayed = false;

                    setTimeout(function () {
                        if (!displayed) {
                            displayed == true;
                            readNextWidget.display();
                        }

                    }, 10000);

                    $(window).scroll(function () {
                        if (isScrolledIntoView("#tmg-related-links")) {
                            if (!displayed) {
                                displayed == true;
                                readNextWidget.display();
                            }
                        }

                        if(isScrolledIntoView("#shareBottom") || isScrolledIntoView("#newGigyaShareBottom")) {
                            if(settings.autoOpen) {
                                openArticleContainer();
                            }
                        }

                        if($(window).scrollTop() + $(window).height() == $(document).height()) {
                            if(settings.autoOpen) {
                                openArticleContainer();
                            }
                        }
                    });
                }
            };

            var display = function () {
                $(settings.widgetContainer).fadeIn(2000, function () {
                    //animation complete
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'display_closed', 'version_1']);
                });
            };

            var bindUIActions = function () {
                $(settings.openButton).mouseenter(function (event) {
                    event.preventDefault();
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'hover_closed', 'version_1']);
                    openArticleContainer();
                });
                $(settings.widgetContainer).mouseleave(function (event) {
                    event.preventDefault();
                    closeArticleContainer();
                });
                $(settings.closeButton).click(function (event) {
                    event.preventDefault();
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'click_close', 'version_1']);
                    updateCookie(settings.articleId, settings.articleListToSkip);
                    closeArticleContainer();
                    settings.autoOpen = false;
                });
            };

            var openArticleContainer = function () {
                if (!settings.panelOpen) {
                	$("#readNextArticleContainer").show();//show/hide because height is calculated incorrectly for elements with display:none
                    $("#readNextArticleContainer").hide();
                    $(settings.articleContainer).toggle('slide', {direction: 'left'}, function () {
                        //animation complete
                        _gaq.push(['_trackEvent', 'wayfarer_read_next', 'display_open', 'version_1']);
                    });
                    settings.panelOpen = !settings.panelOpen;
                    settings.autoOpen = false; //set auto open to false once user has viewed this once
                }
                return false;
            };

            var closeArticleContainer = function () {
                if (settings.panelOpen) {
                    $(settings.articleContainer).toggle('slide', {direction: 'left'}, function () {
                        //animation complete
                    });
                    settings.panelOpen = !settings.panelOpen;
                }
                return false;
            };


            var isScrolledIntoView = function(elem) {
                if($(elem).length) { //check if element exists
                    var docViewTop = $(window).scrollTop();
                    var docViewBottom = docViewTop + $(window).height();

                    var elemTop = $(elem).offset().top;
                    var elemBottom = elemTop + $(elem).height();

                    return ((elemBottom &lt;= docViewBottom) &amp;&amp; (elemTop &gt;= docViewTop));
                } else {
                    return false;
                }
            };

            var unique = function(array) {
                return array.filter(function (el, index, arr) {
                    return index == arr.indexOf(el);
                });
            }

            var createCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        articleListToSkip[0] = currentArticleId;
                        $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
                        return true;
                    }
                });
            }

            var updateCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        updateCookie(currentArticleId, articleListToSkip);
                        return true;
                    }
                });
            }

            var updateCookie = function(articleIdToSkip, articleListToSkip) {
                articleListToSkip.push(articleIdToSkip);
                articleListToSkip = unique(articleListToSkip);
                $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
            }

            var chooseNextArticle = function (currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if ((article.id != currentArticleId) &amp;&amp; ($.inArray(article.id, articleListToSkip) === -1)) {

                        //set article
                        settings.articleTitle = article.title;
                        settings.articleUrl = article.url;
                        settings.articleId = article.id;
						settings.articleImage = article.imgUrl;
                        return true;
                    }
                });
            }

            //public API
            return {
                settings: settings,
                init: init,
                display: display,
                openArticleContainer: openArticleContainer,
                closeArticleContainer: closeArticleContainer
            }
        })();

    };
  </script>
