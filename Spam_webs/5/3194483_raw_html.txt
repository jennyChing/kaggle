        function persistNav(e) {
            if (!jQuery.cookie) {
                return;
            }
            var eId = jQuery(e).attr('id');
            var shownItems = jQuery.cookie('shownItems') ? jQuery.cookie('shownItems').split(',') : [$('.aweber_accordion .head:first').attr('id')];
            var arrPos = jQuery.inArray(eId, shownItems);
            if (arrPos == -1) {
                shownItems.push(eId);
            } else {
                shownItems.splice(arrPos, 1);
            }
            jQuery.cookie('shownItems', shownItems, {path: '/empowered-by-ed', expires: 365});
        }

        $("[href='']").each(function() {
            $(this).attr('href', '#');
        });

        $('.aweber_accordion .head').click(function() {
            $(this).next().next().toggle('fast');
            persistNav($(this));
            return false;
        }).next().next().hide();

        if (jQuery.cookie) {
            if (!jQuery.cookie('shownItems')) {
                jQuery.cookie('shownItems', $('.aweber_accordion .head:first').attr('id'), {path: '/empowered-by-ed', expires: 365});
            }
            if (jQuery.cookie('shownItems')) {
                var shownItems = jQuery.cookie('shownItems').split(',');
                for (var i = 0; i &lt; shownItems.length; i++) {
                    jQuery('#' + shownItems[i]).next().next().show();
                }
            }
        } else {
            $('.aweber_accordion .head:first').next().next().show();
