				function randomBG() {

					var coverRandXPositive = Math.ceil(Math.random()*100);
					var coverRandYPositive = Math.ceil(Math.random()*100);
					var coverRandX = Math.ceil(Math.random()*200);
					var coverRandY = Math.ceil(Math.random()*1000);

					var randXPositive = Math.ceil(Math.random()*100);
					var randYPositive = Math.ceil(Math.random()*100);
					var randX = Math.ceil(Math.random()*10);
					var randY = Math.ceil(Math.random()*1000);

					randX = randXPositive &gt; 50 ? -randX : randX;
					randY = randYPositive &gt; 50 ? -randY : randY;

					coverRandX = coverRandXPositive &gt; 50 ? -coverRandX : coverRandX;
					coverRandY = coverRandYPositive &gt; 50 ? -coverRandY : coverRandY;

					var style = "&lt;style&gt;";
					style += "body { background-position:" + randX + "px " + randY + "px; } ";
					style += "#container { background-position:" + coverRandX + "px " + coverRandY + "px; } ";
					style += "&lt;/style&gt;";

					$("head").append(style);
				}
