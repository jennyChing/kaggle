// 9. User Search
function mpSearchHotels(params){
  // determine destination
  var destination = 'All Destinations';
  var destinations = JSON.parse($('.destination-choices').attr('data-destinations'));
  destinations.forEach(function(rec){
    if (rec.id == params.destination) {
      destination = rec.name;
    }
  });

  var properties = {
    Destination: destination,
    'Start Date': params.in,
    'End Date': params.out,
    '# Rooms': params.rooms,
    '# Guests': params.guests,
    'Logged In': params.user_id ? true : false
  };
  mixpanel.track('Search Hotels', properties);
}

function capitalizeFirstLetter(string) {
  return string.charAt(0).toUpperCase() + string.slice(1);
