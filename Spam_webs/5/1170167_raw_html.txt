(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

                 PREVIEW: 'Nixon's Pals' Graphic Novel by Joe Casey and Chris Burnham
                </h3>
                <div class="post-header">
                 <div class="post-header-line-1">
                 </div>
                </div>
                <div class="post-body entry-content" id="post-body-2650636788822680823" itemprop="description articleBody">
                 <div class="separator" style="clear: both; text-align: center;">
                  <a href="http://3.bp.blogspot.com/-vRs1NHs6uNg/VNvww_ZYxAI/AAAAAAAAh8A/eGskPSvXEkU/s1600/unnamed%2B(68).jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                   <img border="0" height="640" src="http://3.bp.blogspot.com/-vRs1NHs6uNg/VNvww_ZYxAI/AAAAAAAAh8A/eGskPSvXEkU/s1600/unnamed%2B(68).jpg" width="402"/>
                  </a>
                 </div>
                 <div class="separator" style="clear: both; text-align: center;">
                  <br/>
                 </div>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="color: black;">
                   <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                    Image Comics and Man of Action Entertainment arepleased to announce a new, full-color hardcover edition of NIXONS PALS, the political superhero send-up graphic novel by OFFICER DOWNE collaborators Joe Casey (SEX, THE BOUNCE) and Chris Burnham (NAMELESS;
                    <em>
                     Batman, Incorporated
                    </em>
                    ).
                   </span>
                  </span>
                 </div>
                 <a name="more">
                 </a>
                 <span style="background-color: #f3f3f3;">
                  <br/>
                 </span>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="color: black;">
                   <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                    Nixon Cooper is parole officer to Los Angeles Countys large population of super-villains, which could be the worlds most thankless job. But somehow, Nixon cares  maybe a little too much. His marriage is falling apart, he got clobbered by an industrial waste mutant, and he has inappropriate feelings for one of his parolees (a superpowered stripper with talking breasts). To top it all off, hes having really weird dreams about space that he just knows mean something  but does he really want to get pulled deeper into the lives of people he just wants to keep out of trouble?
                   </span>
                  </span>
                 </div>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="color: black;">
                   <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                    "This book holds a special place in my heart, because it's where the Casey/Burnham romance first blossomed, said writer Casey. You can feel the love in the work itself... and the new coloring just brings it out that much more. Thankfully, the story still holds up and it's fun to see how Burnham was nailing it right off the bat."
                   </span>
                  </span>
                 </div>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="color: black;">
                   <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                    Almost seven years after it first came out,
                    <em>
                     Nixon's Pals
                    </em>
                    is still one of my all-time favorites, said artist Burnham. It's one of the best things Joe has ever written, and it still feels like me at my most unleashed. It bounces back and forth between every genre from sci-fi to noir to superhero to horror to romance, and is absolutely cram-packed with the craziest stuff that we could think of. I'm super excited for the fans I've made over the last few years to see this hard-to-find gem for the first time, and especially grateful that the amazing colors of Carlos Badilla are here to really make my linework sing.
                   </span>
                  </span>
                 </div>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                   <span style="color: black;">
                    In addition to the dynamic new colors by Badilla, this hardcover edition of NIXONS PALS includes pin-up art by Nathan Fairbairn (
                   </span>
                   <em style="line-height: 20.7999992370605px;">
                    Multiversity
                   </em>
                   <em style="line-height: 20.7999992370605px;">
                    ;
                   </em>
                   <span style="color: black;">
                    <em>
                     Batman, Incorporated
                    </em>
                    ), Jenny Frison (REVIVAL), Rob Guillory (CHEW),
                   </span>
                   <span style="color: black; line-height: 20.7999992370605px;">
                    Matt Horak (
                   </span>
                   <em style="line-height: 20.7999992370605px;">
                    The Covenant
                   </em>
                   <span style="color: black; line-height: 20.7999992370605px;">
                    ,
                   </span>
                   <em style="line-height: 20.7999992370605px;">
                    Amazing Forest
                   </em>
                   <span style="color: black; line-height: 20.7999992370605px;">
                    ),
                   </span>
                   <span style="color: black;">
                    Daniel Warren Johnson (
                    <em>
                     Space-Mullet
                    </em>
                    ), Jason Masters (
                    <em>
                     Guardians of the Galaxy
                    </em>
                    ,
                    <em>
                     Batwoman
                    </em>
                    ), Dean Trippe (
                    <em>
                     Something Terrible
                    </em>
                    ), Buster Moody (
                   </span>
                   <em>
                    <span style="line-height: normal;">
                     Task Force Rad Squad)
                    </span>
                   </em>
                   <span style="color: black;">
                    , Tradd Moore (LUTHER STRODE,
                    <em>
                     Secret Avengers
                    </em>
                    ), Rod Reis (C.O.W.L.),andGreg Smallwood (
                    <em>
                     Moon Knight)
                    </em>
                    .
                   </span>
                  </span>
                 </div>
                 <div dir="ltr" style="line-height: 22.5px; margin-bottom: 1em; margin-top: 1em; padding: 0px;">
                  <span style="color: black;">
                   <span style="background-color: #f3f3f3; font-family: Georgia, Times New Roman, serif; font-size: large;">
                    NIXONS PALS follows Nixon on a perverse journey of discovery, ultra-violence, and new beginnings. It will be in in comic book stores on
                    <span class="aBn" data-term="goog_1437240552" style="border-bottom-color: rgb(204, 204, 204); border-bottom-style: dashed; border-bottom-width: 1px; position: relative; top: -2px; z-index: 0;" tabindex="0">
                     <span class="aQJ" style="position: relative; top: 2px; z-index: -1;">
                      March 18
                     </span>
                    </span>
                    and in bookstores on
                    <span class="aBn" data-term="goog_1437240553" style="border-bottom-color: rgb(204, 204, 204); border-bottom-style: dashed; border-bottom-width: 1px; position: relative; top: -2px; z-index: 0;" tabindex="0">
                     <span class="aQJ" style="position: relative; top: 2px; z-index: -1;">
                      March 31
                     </span>
                    </span>
                    and is available now for pre-order.
                   </span>
                  </span>
                 </div>
                 <br/>
                 <div class="separator" style="clear: both; text-align: center;">
                  <a href="http://2.bp.blogspot.com/-bUo5LtOcowQ/VNvww2g3a-I/AAAAAAAAh7I/jyHQf4fJjx8/s1600/unnamed%2B(69).jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                   <img border="0" height="148" src="http://2.bp.blogspot.com/-bUo5LtOcowQ/VNvww2g3a-I/AAAAAAAAh7I/jyHQf4fJjx8/s1600/unnamed%2B(69).jpg" width="400"/>
