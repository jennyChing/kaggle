    ShiftRight = function (x, y, z, forceReset) {

        var e = document.getElementById(z);
        if ((e.className.indexOf("catNavigationArrowRightDisabled") == -1) || forceReset) {
            var currentWidth = 0;
            var count = 0;

            if (x == "categoryItemSw") {
                RemoveAllCategories(currentSoftwareCategoryIndex, y, x);

                if ((currentSoftwareCategoryIndex &gt;= y) || forceReset) {
                    currentSoftwareCategoryIndex = 0;
                }

                for (var i = currentSoftwareCategoryIndex; (i &lt; y) &amp;&amp; (count &lt; maxCategoryToDisplay) ; i++) {
                    currentWidth = currentWidth + ElementWidth(x + i);
                    if (currentWidth &lt; MAX_WIDTH_CATEGORIES) {
                        FadeInElement(x + i);
                        count++;
                    }
                }
                currentSoftwareCategoryIndex = currentSoftwareCategoryIndex + count;
                SetButtonDisabledStatus(currentSoftwareCategoryIndex, y, "swCat_Btn_Fwd_Container", "swCat_Btn_Back_Container");
            }
            else if (x == "categoryItemHw") {
                RemoveAllCategories(currentHardwareCategoryIndex, y, x);

                if ((currentHardwareCategoryIndex &gt;= y) || forceReset) {
                    currentHardwareCategoryIndex = 0;
                }

                for (var i = currentHardwareCategoryIndex; (i &lt; y) &amp;&amp; (count &lt; maxCategoryToDisplay) ; i++) {

                    currentWidth = currentWidth + ElementWidth(x + i);
                    if (currentWidth &lt; MAX_WIDTH_CATEGORIES) {
                        FadeInElement(x + i);
                        count++;
                    }
                }
                currentHardwareCategoryIndex = currentHardwareCategoryIndex + count;
                SetButtonDisabledStatus(currentHardwareCategoryIndex, y, "hwCat_Btn_Fwd_Container", "hwCat_Btn_Back_Container");
            }
    }
    }

    SetButtonDisabledStatus = function (x, y, z, t) {
        //
        // x - current index (software or hardware)
        // y - max categories
        // z - fwd button
        // t - back button
        //
        if (x &lt;= maxCategoryToDisplay) {
            DisableNavigationBtn(t);
        }
        else {
            EnableNavigationBtn(t);
        }
        if (x &gt;= y) {
            DisableNavigationBtn(z);
        }
        else {
            EnableNavigationBtn(z);
        }
    }

    ShiftLeftIndex = function (x) {
        var modulus = x % maxCategoryToDisplay;
        if (modulus == 0) {
            modulus = maxCategoryToDisplay;
        }
        return Math.max(0, (x - (modulus + maxCategoryToDisplay)));
    }

    function AutoCompleteKbdSelection(y) {
        var dropDown = null;

        if (y == 0) {
            autoCompleteIndex++;
            dropDown = document.getElementById('autoComplete_' + autoCompleteIndex);
            if (dropDown) {
                dropDown.className = 'autoSuggestionsFocus';
                dropDown = document.getElementById('autoComplete_' + (autoCompleteIndex - 1));
                if (dropDown) {
                    dropDown.className = 'autoSuggestionsNoFocus';
                }
            }
            else {
                autoCompleteIndex--;
            }
        }
        else {
            autoCompleteIndex--;
            dropDown = document.getElementById('autoComplete_' + autoCompleteIndex);
            if (dropDown) {
                dropDown.className = 'autoSuggestionsFocus';
                dropDown = document.getElementById('autoComplete_' + (autoCompleteIndex + 1));
                if (dropDown) {
                    dropDown.className = 'autoSuggestionsNoFocus';
                }
            }
            else {
                autoCompleteIndex++;
            }
        }
    }

    function StartTypeAheadSearchWithTimer(x, y) {
        HideWatermark(x);
        if (searchAheadTimer) {
            window.clearTimeout(searchAheadTimer);
        }
        searchAheadTimer = window.setTimeout(RequestAutoComplete, SEARCH_AHEAD_TIMEOUT);
    }

    function RequestAutoComplete() {
        var y = $("#TypeAheadSearchBox").val();
        $.getJSON('/en-us/windows/compatibility/CompatCenter/AutoCompleteTextSearchSelected',
                    { "id": y },
                    function (data) {
                        if ((data != null) &amp;&amp; (data.length &gt; 0)) {
                            $("#autoSuggestionLists").html("&lt;ul id=\"autoCompleteSelection\"&gt;" + "&lt;/ul&gt;");
                            $("#autoCompleteSelection").html(data);
                            autoCompleteIndex = -1;
                            var text = $("#autoSuggestionLists").html();
                            $("#suggestions").show();
                        }
                        else {
                            $("#suggestions").hide();
                        }
                    });
                }

                function hideSuggestions() {
                    $("#suggestions").hide();
                }

                function fill(thisValue) {
                    var textVal = $('#' + thisValue).text();

                    if (null != thisValue) {
                        $("#TypeAheadSearchBox").val(textVal);
                    }
                    $("#suggestions").hide();

        SaveProductData();
    }

    function SaveProductData() {

        var autoCompleteId = document.getElementById('autoComplete_' + autoCompleteIndex);
        var typeAheadBox = document.getElementById("TypeAheadSearchBox");

        if (autoCompleteId) {
            $("#TypeAheadSearchBox").val(autoCompleteId.title);
            $("#suggestions").hide();

            SearchRequestToOmniture(typeAheadBox.value);
            $("#postProductName").attr("value", BuildProductData());

            navigationElement = document.getElementById("Search");
            window.setTimeout(function () {
                navigationElement.submit();
            }, 50);
        } else if (typeAheadBox &amp;&amp; (typeAheadBox.value != "") &amp;&amp; (typeAheadBox.value != typeAheadBox.title)) {
            SearchRequestToOmniture(typeAheadBox.value);
            $("#postProductName").attr("value", BuildProductData());

            navigationElement = document.getElementById("Search");
            window.setTimeout(function () {
                navigationElement.submit();
            }, 50);
        }
    }

    $(document).ready(function () {
        positionFooter();
        var showingLang = false;
        $(document).click(function (e) {
            if (!$(".languageTab").has(e.target).length &amp;&amp; !$(".browseLanguage").has(e.target).length) {
                $(".browseLanguage").hide();
                showingLang = false;
            }
        });

        $("#exploreLink").click(function () {
            $("#SiteMapColumns").toggle(500);
        });

        var $firstNode = $(".browseLanguage ul li:first");
        var $lastNode = $(".browseLanguage ul li:last");
        var $curLang = null;

        $(".languageTab").click(function () {
            showingLang = ShowLanguages(showingLang);
        });
        $(".languageTab").keydown(function (e) {
            if (e.keyCode == 13) {
                // enter key does the same as mouse click on drop down
                showingLang = ShowLanguages(showingLang);
            }
        });

        $(".browseLanguage").keyup(function (e) {
            if (e.keyCode == 38) { // up arrow

                if ($curLang != null &amp;&amp; $curLang.html() == $firstNode.html()) {
                    $curLang.removeClass("choose");
                    $curLang = null;
                }

                if ($curLang != null) {
                    $curLang.removeClass("choose");
                    $curLang = $curLang.prev("li");
                } else {
                    $curLang = $lastNode;
                }
                $curLang.addClass("choose");

            } else if (e.keyCode == 40) { // down arrow
                if ($curLang != null &amp;&amp; $curLang.html() == $lastNode.html()) {
                    $curLang.removeClass("choose");
                    $curLang = null;
                }

                if (null != $curLang) {
                    $curLang.removeClass("choose");
                    $curLang = $curLang.next("li");
                } else {
                    $curLang = $firstNode;
                }

                $curLang.addClass("choose");

            } else if (e.keyCode == 13) { // enter
                if ($curLang != null) {
                    document.location = $curLang.children('a').attr('href');
                }
            } else if (e.keyCode == 27) { // escape
                $(".browseLanguage").hide();
                showingLang = false;
            }

            return false;
        });
        $(window).resize(positionFooter);
    });

    function positionFooter() {

            //
            $("#stickyFooter").css({
                position: "absolute",
                bottom: 0
            });
        } else {
            $("#stickyFooter").attr("style", "");
        }
    }
