      var checkPermissions = function() {
        FB.getLoginStatus(function(response) {
          if (response.status == 'connected') {

            FB.api('/me/permissions', function(response) {
              var permsArray = response.data[0];
              var permsToPrompt = [];

                  permsToPrompt.push(permsNeeded[i]);
                }
              }
              if (permsToPrompt.length &gt; 0) {
                //   alert('Need to re-prompt user for permissions: ' +

                });
              } else {
                //  alert('No need to prompt for any permissions');
                $('.feedback.not_given_permission').fadeOut('slow');

            <h2 itemprop="name">
             Snow Games
            </h2>
           </a>
           <h3 class="date-header">
            <span>
             Monday, 18 February 2013
            </span>
           </h3>
          </div>
          <div class="clear">
          </div>
          <div class="postbody" id="postbody-1400574282963542091" itemprop="articleBody">
           <iframe allowfullscreen="" frameborder="0" height="360" mozallowfullscreen="" src="http://player.vimeo.com/video/59908933?title=0&amp;byline=0&amp;portrait=0&amp;color=fff800" webkitallowfullscreen="" width="640">

           This will make you smile, I hope. There have been a few videos recently of various animals playing in the snow  and here is another one to add to the collection. This one, however, is a little different because the dog playing in the snow has been captured in slow motion. Barolo, who owns filmmaker
           <a href="http://andreyblanco.com/" rel="nofollow" target="_blank">
            Andrey Blanco
           </a>
           , has been filmed at 240 frames per second. So, you get every jump, every leap and  you swear  a dog can smile!
           <br/>
           <br/>
