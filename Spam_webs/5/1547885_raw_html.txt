	;

	var btn_open = '&lt;div class="btn_open_wrap"&gt;&lt;div class="btn_open"&gt;&lt;a href="javascript:void(0);" id="open_button"&gt;Open&lt;/a&gt;&lt;/div&gt;&lt;/div&gt;'

		$("#header_thirdcategory_" + parts[4]).addClass( class_name_3rd );

	// drop down
	var   btn_dropdown = $("#dropdown_categories  .btn_dropdown")
		, dropdown_list = $("#dropdown_categories &gt; ul.dropdown ul")

	$("body").bind("click", function(){
		dropdown_list.hide();
		btn_dropdown.removeClass("current");
		btn_dropdown.attr("current", "");
	});

	btn_dropdown.bind("click", function(e){
		dropdown_list.hide();
		btn_dropdown.removeClass("current");
		if ( $(this).attr("current") == 1 ) {
			$(this).attr("current", "");
			$(this).removeClass("current");
			$(this).nextAll("ul").hide();
		}
		else {
			btn_dropdown.attr("current", "");
			$(this).attr("current", 1);
			$(this).addClass("current");
			$(this).nextAll("ul").show();
		}
		
		e.stopPropagation();
	});

});

function menukHandler(e){
	if ($(this).attr("id") == "open_button") {
		$(".all_categories").fadeIn("fast");
	}
	else if ($(this).attr("id") == "close_button") {
		$(".all_categories").hide();
	}

       <div class="body">
        <div class="text">
         <p>
          A former Japanese soccer star is hoping to bring his scoring touch to an iPhone app that delves into the enchanting world of sake.
         </p>
         <p>
          The "Sakenomy" app, created by Tokyo-based content developer MTI Ltd., recommends specific sake labels from a database of 1,000 brands based on a user's preference and tastes.
         </p>
         <p>
          Hidetoshi Nakata, a former standout on Japan's national soccer team and one of the nation's greatest players, oversaw the development of the app. It was released Oct. 20 on the iOS App Store.
         </p>
         <p>
          A known sake enthusiast, Nakata, who also played in top leagues in Britain and Italy, regularly visits sake breweries across Japan and provided advice to the software developer about the app on such aspects as user-friendliness and graphics.
         </p>
         <p>
          In addition to recommending sake brands, the app allows users to record impressions of sakes they have tried in four categories--full, light, fruity and mild.
         </p>
         <p>
          An English version of the app will be released this year. The developer also plans to release French and Chinese versions. The Japanese version will be released for Android OS users in January 2015.
         </p>
