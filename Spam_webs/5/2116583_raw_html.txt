  // fetches the cookie with the provided name
  function getCookie(c_name)
  {
      if (document.cookie.length&gt;0) {
          c_start=document.cookie.indexOf(c_name + "=")
          if (c_start!=-1)  { 
              c_start=c_start + c_name.length+1 
              c_end=document.cookie.indexOf(";",c_start)
              if (c_end==-1) c_end=document.cookie.length
              return unescape(document.cookie.substring(c_start,c_end))
          }
      }
      return "";
  }

  function featuredProdsLinksClicked(actionFrom, linkName, prdId, prdName)
  {
    // actionFrom - RV -Recently Viewed
    //              YMAL - You may also like
    //              SS - solo subplan pair
    //              FD - Front Door
    // linkName - PD - product detail link
    //            QV - quick view link
    //            BB - buy now button
    var cmvc = new CoremetricsCategory();
    //alert("actionFrom=" + actionFrom + ",linkName=" + linkName);
    if (actionFrom != "")
    {
      if (actionFrom == "RV")
      {
        cmvc.categoryLevel1 = "RECENTLY VIEWED";
        cmvc.categoryLevel2 = currPageType;
      }
      else if (actionFrom == "YMAL")
      {
        cmvc.categoryLevel1 = "FEATURED PRODUCT AREA";
        cmvc.categoryLevel2 = currPageType;
      }
      else if (actionFrom == "SS")
      {
        cmvc.categoryLevel1 = "FEATURED PRODUCT AREA";
        cmvc.categoryLevel2 = currPageType;
        cmvc.categoryLevel3 = "- SOLO/SUBPLAN PAIR";
      }
      else if (actionFrom == "FD")
      {
        cmvc.categoryLevel1 = "FRONT DOOR";
        cmvc.categoryLevel2 = "";
        cmvc.categoryLevel3 = "";
      
      }
      
      
      if (actionFrom != "SS")
      {
        if (linkName == "PD")
        {
          cmvc.categoryLevel4 = "- PRODUCT DETAIL BUY";
        }
        else if (linkName == "QV")
        {
          cmvc.categoryLevel4  = "- QUICK VIEW BUY";
        }
        else if (linkName == "BB")
        {
          cmvc.categoryLevel4 = "- ON PG BUY";
        }
      }
      
      // If this is quick info link, then send productview tag
      // The productview tag from other links will be sent from Product detail page
      if (linkName == "QV" || linkName == "BB")
      {
         cmCreateProductviewTag(prdId, prdName,cmvc.getCategory(), cmvc.getCategory());
      }

      // If the link is quick info, we can not set the category 
      // because if the user close the quick info window without purchasing and then the customer returns
      // to the original screen and the category needs to be stay original as well.      
      if (linkName != "QV")
      {      
        cmCategory = cmvc;
        cmvc.setCategoryCookie();
      }
    }
  }
