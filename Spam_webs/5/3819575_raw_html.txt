  <script type="text/javascript">
   $('#ContactCountry').change(function() {
		if ($('#ContactCountry :selected').val() == 'US') {
			$('#stateDiv').show();
			$('#otherStateDiv').hide();
		} else {
			$('#stateDiv').hide();
			$('#otherStateDiv').show();
		}
	});
	$('#ContactCountry').change();
	
	$('#ContactZip').change(function() {

		// only do the lookup if something's in the Zip field, and only if looking at US city/states
		if ($(this).val() &amp;&amp; ($('#ContactCountry :selected').val() == 'US')) {
		
			$('#ContactCity, #ContactState').addClass('form_field_ajax_spinner');
			
			$.getJSON(
				'/lizabrown/admin/contacts/get_city_state_from_zip/__zip_code__.json'
					.replace('__zip_code__', $('#ContactZip').val()),
				function(response) {
					if (response.city) {
						$('#ContactCity').val(response.city);
						$('#ContactState').val(response.state).attr('selected', 'selected');
					}
					
					$('#ContactCity, #ContactState').removeClass('form_field_ajax_spinner');
				}
			)
		}
	});
  </script>
