            function getRandomItemList(list,current,num)
            {
                                                                                                                                                
                var a = new Array();
                var arr = new Array();
                var result = new Array();                
                num = parseInt(num);                
                if(!(num&gt;0))
                    num = 4;  
                for(var i in list)
                {
                    if(list[i].Id!=current)                           
                        a.push(list[i]);                    
                }
                                                                                                                                                
                if(num &gt;a.length)
                    num = a.length;
                while(arr.length &lt; num){
                    var randomnumber=Math.ceil(Math.random()*a.length)
                    var found=false;
                    for(var i=0;i&lt;arr.length;i++){
                        if(arr[i]==randomnumber){found=true;break}
                    }
                    if(!found)arr[arr.length]=randomnumber;
                }                  
                for (i in arr)
                {
                    result.push(a[arr[i]-1]);               
                }                    
                return result;
            }
