    function getBrowserScrollSize() {
        var $ = ThriveGlobal.$j;
        var css = {
            "border": "none",

            "margin": "0",
            "padding": "0",
            "width": "200px"
        };

        var inner = $("&lt;div&gt;").css($.extend({}, css));
        var outer = $("&lt;div&gt;").css($.extend({

        var scrollSize = {
