    <script type="text/javascript">
     $(document).ready(function () {

        $("a.voterLink").click(function (e) {

            resp = $(this).attr("id").substring(1);

            $.getJSON("/Home/Vote/", { id: resp, c: "" }, function (incomingLetter) {

                $("#b" + resp).html(incomingLetter.letterUp);
                $("#h" + resp).addClass("not_selectable");
                $("#h" + resp).fadeOut(2500);

            });
        });


        $("div.letterEnvelopeContainer").hover(function () {

            fader($(this).attr("id"), "in");

        }, function () {

            fader($(this).attr("id"), "out");

        });


        function fader(id,fadeType) {

            if (typeof id != 'undefined') {
                var start = id.substring(0, 7);
                if (start === "letter-") {
                    var letterId = id.substring(7);

                    if (fadeType == "in") {
                        $("#showOnMouseOver-" + letterId).fadeIn('fast');
                    } else {
                        $("#showOnMouseOver-" + letterId).fadeOut('fast');
                    }
                    
                };
            };

            

        }

    });
    </script>
    <script type="text/javascript">
     $(document).ready(function() {

        var offset = new Date().getTimezoneOffset();
        $.cookie('userTimeZone', offset/60, { expires: 365 });

        var isRobot = $.cookie('robotTest');

        if (isRobot == 'love') {
            $("#robotTest").val("love");
            $("#robotTestCookie").val("love");
            $("#robotTestForm").hide();
        }

        $("#sendComment").click(function () {

            var robotTest = $("#robotTest").val();

            if (robotTest == "love") {
                $.cookie('robotTest', "love", { expires: 365 * 10 });
                $("#robotTestCookie").val("love");
                $("#addCommentForm").submit();
            } else {
                alert('Cannot post this comment. You need to type the word love into the robot test form.');
            }

        });

        $("a.bookmarkLink").click(function(e) {

            resp = $(this).attr("id").substring(2);

            $.getJSON("/Bookmark/" + resp, null, function(result) {

                if (result.Code == '1') {
                    $("#bookmark").html("bookmarked!");
                }
                else {
                    alert('An error occurred. Please report this problem using the "contact" page. (Message: ' + result.Message + ')');
                }

            });

        });

    });
