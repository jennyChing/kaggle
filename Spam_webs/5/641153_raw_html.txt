                    $('#newsletter-signup-form form').removeAttr('disabled');
                                    
                    $("#newsletter-signup-form .required-icon").each(function(i, el) {
                        setTimeout(function() {
                            $(el).fadeIn(200);
                        }, (i * 300));
                    });
                
                    setTimeout(function() {
                        $('#newsletter-signup-form .legend').fadeIn(200);
                    }, ($("#newsletter-signup-form .required-icon").length * 300));
                });
            });
                      
            
            // Check for proper email formatting
            $('#newsletter-email').blur(function() {
                if (!$(this).val().match(/.+\@.+\..+/i) &amp;&amp; $(this).val().length &gt; 0) {
                    if (!$(this).hasClass('error')) {
                        $(this).addClass('error');     
                        var p = $(this).position();           
                        $(this).after('&lt;div class="field-error" style="margin-left: ' + p.left + 'px"&gt;&lt;div class="field-error-arrow"&gt;&lt;/div&gt;&lt;div class="field-error-text"&gt;Not a valid email format. Did you forget something?&lt;/div&gt;&lt;/div&gt;');
                    }
                    $('#newsletter-submit').attr('disabled','disabled');
                    $('#newsletter-signup-form form').attr('disabled','disabled');
                    
                } 
            });
            $('#newsletter-email').keyup(function(){
                if ($(this).val().match(/.+\@.+\..+/i)) {
                    if ($(this).hasClass('error')) {
                        $(this).removeClass('error');
                        if ($(this).next().hasClass('field-error'))
                            $(this).next().remove();
                    }
                    $('#newsletter-submit').removeAttr('disabled');
                    $('#newsletter-signup-form form').removeAttr('disabled');              
                }
            });
            
            // let a person know when the emails don't match
            $('#newsletter-re-type-email').keyup(function(){
                var pattern = new RegExp('^' + $(this).val(), 'i');
            
                if (!$('#newsletter-email').val().match(pattern)) {
                    if (!$(this).hasClass('error')) {
                        $(this).addClass('error');                
                        var p = $(this).position();           
                        $(this).after('&lt;div class="field-error" style="margin-left: ' + p.left + 'px"&gt;&lt;div class="field-error-arrow"&gt;&lt;/div&gt;&lt;div class="field-error-text"&gt;Email doesn\'t match&lt;/div&gt;&lt;/div&gt;');
                    }
                    $('#newsletter-submit').attr('disabled','disabled');
                    $('#newsletter-signup-form form').attr('disabled','disabled');
                } else {
                    if ($(this).hasClass('error')) {
                        $(this).removeClass('error');
                        if ($(this).next().hasClass('field-error'))
                            $(this).next().remove();
                    }
                    $('#newsletter-submit').removeAttr('disabled');
                    $('#newsletter-signup-form form').removeAttr('disabled');
                }
            });
            
            $('#newsletter-signup-form input').keyup(function(){
                if ($(this).hasClass('clear-error-on-keyup')) {
                    $(this).removeClass('clear-error-on-keyup').removeClass('error');   
                 
                    if ($(this).next().hasClass('field-error'))
                        $(this).next().remove();
                }

         Brewer to Congress: 'Stop Obama's manufactured crisis'
        </h1>
        <h2 class="deck">
         Arizona governor Tweets plea to secure border to Boehner, Reid
        </h2>
        <time class="updated" datetime="2015-07-13T19:09:34+00:00" pubdate="">

        <img alt="" class="size-full wp-image-954951" height="300" src="/files/2014/06/children_on_border.jpg" width="600"/>
        <p class="wp-caption-text">
         Thousands of illegal immigrant children have overrun U.S. border security and their processing centers along the U.S.-Mexico border.

       <p>
        WASHINGTON  After an earlier call to the White House to end its dangerous policy of dropping off hundreds of illegal immigrants apprehended in Texas in her own state, Arizona Gov. Jan Brewer Thursday evening tweeted out a plea to congressional leadership to stop what she called a crisis manufactured by President Obama.
       </p>
       <div class="left_prom_160 desktop-only">
        <div class="desktop-only" id="div-gpt-ad-story160LeftSide">
         <script type="text/javascript">
          googletag.cmd.push(function() { googletag.display('div-gpt-ad-story160LeftSide'); });
         </script>
        </div>
       </div>
       <p>
        Our country is in the midst of a crisis manufactured by President Obama and his administrations refusal to enforce our nations immigration laws, she wrote. It has become clear that the administration has encouraged this massive influx and intends to continue ignoring the states calls to end this policy. Furthermore, it is indisputably evident that our border is not secure and is hardly more secure than ever.
       </p>
       <p>
        On Monday she had blasted the federal government for not giving Arizona any notice of the plan to relocate thousands of refugees crossing the border, most of them children.
       </p>
       <p>
        I urge you to end this dangerous and unconscionable policy immediately and instead take actions to fulfill the federal governments fundamental responsibility of protecting our homeland by securing our nations borders,
        <a href="http://www.azgovernor.gov/dms/upload/PR_060214_GovernorBrewerLetterPresidentObama.pdf">
         Brewer wrote in a letter to Obama
        </a>
        .
       </p>
       <p>
        Brewer said that the country and Arizona face significant challenges from the Obama administrations refusal to carry out its responsibility to secure our countrys southern border, which has stretched state and local governments and law enforcement agencies to the breaking point attempting to manage the enormity of these challenges.
       </p>
       <p>
        She said she was deeply concerned that many of the illegal aliens are being abandoned at the bus stations without any food, water or basic necessities.
       </p>
       <p>
        I remind you that the daytime temperatures in Arizona during this time of year are regularly more than 100 degrees, she added. Consequently, this federal operation seems to place expediency over basic humanitarian concerns. The federal government should not shirk its lawful responsibility to care for and properly process these individuals.
       </p>
       <p>
        Immigration officials have reportedly been transporting illegal immigrants from Texas to Arizona for the past eight months, but crisis has intensified in the last few weeks as Border Patrol agents along Texass Rio Grande Valley are apprehending about 1,000 illegal immigrants per day.
       </p>
