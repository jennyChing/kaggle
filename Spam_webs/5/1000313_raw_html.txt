function getVideoAd(data) {
	try {
		// get third party tracker
		if (data.results[0].mediafiles.mediafile[0].thirdPartyTracker &amp;&amp; data.results[0].mediafiles.mediafile[0].thirdPartyTracker != "[impression]") {
			$(document).ready(function() {
				$("body").append('&lt;img src="'+data.results[0].mediafiles.mediafile[0].thirdPartyTracker+'" alt="tracker" /&gt;');
			});
		}

		// get video or audio ad
		switch (data.results[0].mediafiles.mediafile[0].type) {
			case "video/x-flv":
				return {
					type: "video",
					creative: data.results[0].mediafiles.mediafile[0].url,
					audioAdImage: '',
					clickThrough: data.results[0].mediafiles.mediafile[0].clickthrough,
					duration: data.results[0].mediafiles.mediafile[0].duration
				};
				break;
			case "audio":
				return {
					type: "audio",
					creative: data.results[0].mediafiles.mediafile[0].url,
					audioAdImage: data.results[0].companionads.companionad[0].url,
					clickThrough: data.results[0].mediafiles.mediafile[0].clickthrough,
					duration: data.results[0].mediafiles.mediafile[0].duration
				};
				break;
		}
		return null;
	} catch(e) { 
		//if (this.console) console.log("Error loading video ad", e);
		return null;
	}
}

function getOffset(el) {
    if (el.getBoundingClientRect)
        return el.getBoundingClientRect();
    else {
        var x = 0, y = 0;
        do {
            x += el.offsetLeft - el.scrollLeft;
            y += el.offsetTop - el.scrollTop;
        } 
        while (el = el.offsetParent);

        return { "left": x, "top": y }
    }

function getCanvasWidthHeight(image){
	var windowWidth = 303;//parseInt($(window).width());
	var windowHeight = 190;//parseInt($(window).height());
	var height = windowHeight; //parseInt((windowHeight-60) * 0.8);
	var width = windowWidth; //parseInt(windowWidth * 0.92);
	var naturalWidth = image.naturalWidth? image.naturalWidth : image.width;
	var naturalHeight = image.naturalHeight? image.naturalHeight : image.height;
	var imageWidth = naturalWidth;
	var imageHeight = naturalHeight;

	// fit image into container
	if ((width/height) &gt; (imageWidth/imageHeight)) {
		animation = "down";
		imageWidth = width;
		imageHeight = parseInt(naturalHeight * (imageWidth/naturalWidth));
	} else {
		animation = "left";
		imageHeight = height;
		imageWidth = parseInt(naturalWidth * (imageHeight/naturalHeight));
	}
	
	return {
		"width":width,
		"height":height,
		"animation":animation,
		"imageWidth":imageWidth,
		"imageHeight":imageHeight
		};
}

function startUpdateSong(initial) {
	try  {

function updateSong() {
	if (!parent.metaIsWorking &amp;&amp; !playerIsVisible()) {
		startUpdateSong();
		return;
	}
	parent.songIsChanged = false;
	if (playerType == 'program') showProgramInfo();
	else {
		try {
			listenLogData = {
				"cmd":"getLastPlayedSongs",
				"stationID": "29",
				"mode": "radio",
				"playerType": "website",
				"includeNowPlaying": 1,
				"lastPlayedLimit": 6,
				"cacheKill": Math.random()
			};
			
			if (typeof moreClicks !== 'undefined' &amp;&amp; typeof numSongsPerPage !== 'undefined') {
				listenLogData.lastPlayedLimit = (moreClicks*numSongsPerPage)+1;
			}
			
			if (parent.metaIsWorking) {
				listenLogData.artistName = meta_artist;
				listenLogData.songName = meta_song;
			}
					  
			$.ajax({
				type: 'GET',
				dataType: 'json',
				url: '/player/servicePlayer',
				data: listenLogData,
				timeout: LISTEN_LOG_TIMEOUT,
				success: function(data) {
					if (data.lastPlayedSongs !== undefined) {
						var lastPlayedSongs = data.lastPlayedSongs;
						var last_played = '&lt;div class="last-played-item first"&gt;&lt;h2&gt;Last Played&lt;/h2&gt;&lt;/div&gt;';
						if (lastPlayedSongs.lastPlayedTemplate !== undefined &amp;&amp; lastPlayedSongs.songs !== undefined &amp;&amp; lastPlayedSongs.songs.length &gt; 0) {
							for (var i=0; i&lt; lastPlayedSongs.songs.length; i++) {
								last_played += lastPlayedSongs.lastPlayedTemplate.replace(/{SONG_URL}/g,lastPlayedSongs.songs[i].song_url).replace(/{THUMB_IMG}/g,lastPlayedSongs.songs[i].thumb_img).replace(/{SHORT_ARTIST_NAME}/g,lastPlayedSongs.songs[i].short_artist_name).replace(/{SHORT_SONG_NAME}/g,lastPlayedSongs.songs[i].short_song_name).replace(/{LAST_TIME_WORDS}/g,lastPlayedSongs.songs[i].last_played_words);
							}
						} else {
							last_played += '&lt;div class="last-played-item"&gt;No songs found.&lt;/div&gt;';
						}
						
						$('#last-played').html(last_played);
						$('#last-played .last-played-item').last().addClass('last');
					}
					
					if (data.nowPlaying !== undefined) {
						radioSetNowPlaying(data.nowPlaying);
					}
				},
				complete: function(data) {
					startUpdateSong();
				}
			});
		} catch(e) {
			//if(window.console) console.log('UpdateSong Error: ',e);
		}
	}
}

var LISTEN_LOG_INTERVAL = 6000; // ms between each interval
var LISTEN_LOG_TIMEOUT = 3000; // timeout ms
var LISTEN_LOG_ERROR = 5; // num errors until abort
//var listenLogTimer = null; moved to  yii layouts/playerIframe.php
var listenLogData = null;
var listenLogErrors = 0;
var listenLogLastTime = new Date().getTime();
var listenLogLastResponse = null;
var listenLog_diff = 0;
function updateListenLog() {
	return;
	try{
	
		//clearInterval(listenLogTimer);
	
		// determine if interval is correct
		var currTime = new Date().getTime();
		listenLog_diff = currTime-listenLogLastTime;
		if (listenLogLastTime &gt; 0) {
			//if (!isIE()) console.log("updateListenLog: last update was ", listenLog_diff, 'miliseconds ago');
		}
		listenLogLastTime = currTime;

		// check if number of errors exceeded
		if (listenLogErrors &gt;= LISTEN_LOG_ERROR) {
			  //if (!isIE()) console.log("startListenLog: more than ", LISTEN_LOG_ERROR, " errors, aborting");
			  stopListenLog();
		} else {
			  // do update
			  listenLogData = {
					"cmd":"updateListenLog",
					"stationID": "29",
					"mode": "radio",
					"playerType": "website",
					"includeNowPlaying": 1,
					"listenLogTime":listenLog_diff,
					"cacheKill": Math.random()
			  };
						  
			  $.ajax({
					type: 'GET',
					dataType: 'json',
					url: '/player/servicePlayer',
					data: listenLogData,
					timeout: LISTEN_LOG_TIMEOUT,
					success: function(response) {
						if (response.nowPlaying != false){
							var songData = response.nowPlaying;
							previousArtistName1 = previousArtistName2;
							previousSongName1 = previousSongName2;
							previousArtistName2 = previousArtistName3;

							previousSongName3 = songData['song_name'];

								//console.log("previousSongName1="+previousSongName1);
								//console.log("previousSongName2="+previousSongName2);
								//console.log("previousSongName3="+previousSongName3);
							}
							if (previousArtistName3 == previousArtistName1 &amp;&amp; previousSongName3 == previousSongName1 &amp;&amp; previousSongName3 != "undefined"){
								previousSongName2 = previousSongName3;
								previousArtistName2 = previousArtistName3;
							}
							if (previousArtistName3 == previousArtistName2 
									&amp;&amp; previousSongName3 == previousSongName2
									&amp;&amp; (previousArtistName3 != previousArtistName1 
										|| previousSongName3 != previousSongName1)
							){
								if (!metaIsWorking &amp;&amp; playerType == 'songinfo') showSongInfo(response.nowPlaying);
								if (playerType == 'program') showProgramInfo();
							}else{
								//if (!isIE()) console.log("updateListenLog: same song, do nothing");
							}
							// keep update your variables
							listenLogLastResponse = response.nowPlaying;
							listenLogErrors = 0;
						} else if (playerType == 'songinfo') {
							previousArtistName1 = previousArtistName2;
							previousSongName1 = previousSongName2;
							previousArtistName2 = previousArtistName3;
							previousSongName2 = previousSongName3;
							previousArtistName3 = "(CFHK FM) 103.1 Fresh FM";
							previousSongName3 = '';
							if (previousArtistName3 == previousArtistName1 &amp;&amp; previousSongName3 == previousSongName1 &amp;&amp; previousSongName3 != "undefined"){
								previousSongName2 = previousSongName3;
								previousArtistName2 = previousArtistName3;
							}
							if (previousArtistName3 == previousArtistName2 
									&amp;&amp; previousSongName3 == previousSongName2
									&amp;&amp; (previousArtistName3 != previousArtistName1 
										|| previousSongName3 != previousSongName1)
							) {
								$("#artist-gallery").html('');
								artistPictures = null;
								redrawPicture();
								$("#radio-player-artist-name").html("(CFHK FM) 103.1 Fresh FM");
								$("#radio-player-song-name").html("");
					
								// update facebook url
								$('.fb-like').attr('data-href','http://www.1031freshradio.ca');
								$('.fb-like').html('&lt;fb:like href="http://www.1031freshradio.ca" send="false" width="100" show_faces="true" layout="button_count"&gt;&lt;/fb:like&gt;');
								FB.XFBML.parse();
							}
						}
					},
					error: function() {
						  //if (!isIE()) console.log("updateListenLog: ajax error");
						  listenLogErrors++;
					},
					complete: function(){
						//startListenLog();
					}
			  });   
		}
	} catch(e) {
		//if (!isIE()) console.log('Error: updateListenLog',e);
	}
}

function parseMeta(event) {
	if (playerType == 'program') return;
	if(window.top === window.self &amp;&amp; $("iframe#socast").length &gt; 0) {
		var iframe = document.getElementById("socast");
		iframe.contentWindow.parseMeta(event);
		return;
	}
	if (parent.windowIsBlurred) {
		parent.songIsChanged = true;
	}
	//if (window.console) console.log(event.metadata);
	if (event.metadata.title != undefined &amp;&amp; event.metadata.title != '') {
		parent.metaIsWorking = true;
		meta_title = event.metadata.title.split(' - ');
		if (meta_title.length == 2) {
			meta_artist = meta_title[1];
			meta_song = meta_title[0];
			//showSongInfo({});
		} else {
			meta_artist = '(CFHK FM) 103.1 Fresh FM';
			meta_song = '';
		}
		//updateListenLog();
		if (!parent.windowIsBlurred) updateSong();
		setTimeout(clearBuffer,3000);
	} else {
		meta_artist = '(CFHK FM) 103.1 Fresh FM';
		meta_song = '';
	}
	$('#radio-player-artist-name').html(meta_artist);
	$('#radio-player-song-name').html(meta_song);
}
