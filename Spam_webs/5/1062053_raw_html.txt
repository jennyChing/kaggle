                            for(var i = 0; i &lt; nearby.length; i++){
                                allMarkers.push(new google.maps.LatLng(nearby[i].latitude, nearby[i].longitude));
                            }

                            var mapOptions = {
                                center: allMarkers[0],
                                zoom: initialZoom,
                                mapTypeId: google.maps.MapTypeId.ROADMAP
                            };
                            window.gmap = new google.maps.Map(document.getElementById('map-elem'), mapOptions);
                            window.gmap2 = new google.maps.Map(document.getElementById('map-elem2'), mapOptions);

                            // Fit all markers on map
                            //  Make an array of the LatLng's of the markers you want to show
                            //  Create a new viewpoint bound
                            var bounds = new google.maps.LatLngBounds ();
                            //  Go through each...
                            for (var j = 0, LtLgLen = allMarkers.length; j &lt; LtLgLen; j++) {
                              //  And increase the bounds to take this point
                              bounds.extend (allMarkers[j]);
                            }
                            //  Fit these bounds to the map
                            gmap.fitBounds(bounds);
                            gmap2.fitBounds(bounds);

                            var infoWindow = new google.maps.InfoWindow();

                            // marker for the given business
                            var marker = new google.maps.Marker({
                                position: allMarkers[0],
                                map: gmap,
                                icon: "http://www.googlemapsmarkers.com/v1/A/fb8374/000000/ae473c",
                                title: nearby[0].name,
                                iwcontent: '&lt;strong&gt;&lt;a href="'+nearby[0]['url']+'" target="_blank"&gt;"'+nearby[0].name+'&lt;/a&gt;&lt;/strong&gt;'
                            });

                            google.maps.event.addListener(marker, 'click', function (e) {
                                infoWindow.setContent(this.iwcontent);
                                infoWindow.open(gmap, this);
                            });
                            // neighborhood businesses


                            if (nearby !== null) {
                                for (var k = 1; k &lt; nearby.length; k++) {
                                    var marker = new google.maps.Marker({
                                        position: allMarkers[k],
                                        map: null,
                                        title: nearby[k].name,
                                        iwcontent: '&lt;strong&gt;&lt;a href="'+nearby[k]['url']+'" target="_blank"&gt;'+nearby[k]['name']+'&lt;/a&gt;&lt;/strong&gt;'
                                    });
                                    google.maps.event.addListener(marker, 'click', function (e) {
                                        infoWindow.setContent(this.iwcontent);
                                        infoWindow.open(gmap, this);
                                    });
                                    gmapMarkers.push(marker);
                                }

       <p class="author">
        Posted by the Mass Office of Travel and Tourism
       </p>
       <input class="jpibfi" type="hidden"/>
       <p>
        Dont let the name fool you. Theres more to Sandwich than just food.
       </p>
       <p>
        Certainly, the Cape Cod town of about 20,000, which is located towards the top of
        <a href="http://en.wikipedia.org/wiki/Cape_Cod#.22Upper.22_and_.22Lower.22" target="_blank">
         Upper Cape Cod
        </a>
        , has plenty to offer in the way of eats. Glass was what made Sandwich famous, though.
       </p>
       <p>
        Its been nearly two centuries since the Boston &amp; Sandwich Glass Company was the worlds leading glass manufacturer but that history is kept alive and well today inside the
        <a href="http://www.massvacation.com/explore/arts/cultural-districts/#glasstown" target="_blank">
         Glasstown Cultural District
        </a>
        .
       </p>
       <p>
        For those heading to the Cape whod like to break up their beach time with a little history, Sandwich and its cultural district presents the perfect opportunity.
       </p>
       <p>
        Heres where to go and what to see.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Sandwich-Massachusetts-Directions.jpg">
         <img alt="Sandwich Massachusetts Directions" class="aligncenter size-full wp-image-7727" height="1360" src="/blog/wp-content/uploads/2015/05/Sandwich-Massachusetts-Directions.jpg" width="2048"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Old Kings Highway and Cape Cod Rail in Sandwich, MA
        </em>
       </p>
       <p>
        The centerpiece of Sandwichs Glasstown Cultural District is the
        <a href="http://www.massvacation.com/business/3458/" target="_blank">
         Sandwich Glass Museum
        </a>
        , which chronicles the towns rich history of production.
       </p>
       <p>
        Upon entering, guests can browse more than 6,000 antique glass objects on display in 15 different galleries, along with
        <a href="http://www.sandwichglassmuseum.org/" target="_blank">
         a variety
        </a>
        of exhibits.
       </p>
       <p>
        In addition to the permanent and rotating installations, Sandwichs glass museum also serves as the jumping off point for special events and tours. There are often special happenings for the holidays, and during the summer,
        <a href="http://www.sandwichglassmuseum.org/events.php?pgID=26" target="_blank">
         guided walks
        </a>
        leave from the museum, too.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Sandwich-Glass-Museum.jpg">
         <img alt="KONICA MINOLTA DIGITAL CAMERA" class="aligncenter size-full wp-image-7728" height="960" src="/blog/wp-content/uploads/2015/05/Sandwich-Glass-Museum.jpg" width="1280"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Glass on glass at the Sandwich Glass Museum. Photo by Brady Wahl via
        </em>
        <a href="https://www.flickr.com/photos/bradywahl/2549899916/in/photolist-r1ifYt-qHJLd7-qGRx8k-49mYU-qGnDDo-4NnBzh-qLZCSL-q7MwtR-qM1Ye1-qGWiFw-4TviBu-4Tr5RR-49mZu-4TviDj-49n17-4Tk4WW-4Tk4Q1-4TjUvb-4TjUpE-4TfdBF-eZoSp4-qZmJma-q5cUV4-r1ZPMp-qGXbwE-r4r8Xg-r4uGcG-q3fPzw-qMb3Ca-q3tgN8-r1Q6qq-q76txm-4TfEMZ-4TfVhF-q6tZ9s-r3vyEe-qKVHM9-qHRuTf-4Tkaah-4TfEuR-6LUQaW">
         <em>
          Flickr
         </em>
        </a>
       </p>
       <p>
        Just a few steps down the road, travelers can take in another historic Sandwich site in the form of
        <a href="http://www.capecodwindmills.scificincinnati.com/thomas_dexters_grist_mill.htm" target="_blank">
         Dexter Grist Mill
        </a>
        . The mill is one of oldest of its kind in the entire country, containing records that date back to 1640.
       </p>
       <p>
        To this day, Dexter Mill Grist Mill and its grounds are well preserved, making this destination a very popular one for sightseeing and photography.
       </p>
       <p>
        The Hoxie House, a saltbox rumored to be the oldest house on Cape Cod and one that dates back to circa 1675, is right in the area as well.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Dexter-Grist-Mill.jpg">
         <img alt="Dexter Grist Mill" class="aligncenter size-full wp-image-7729" height="2048" src="/blog/wp-content/uploads/2015/05/Dexter-Grist-Mill.jpg" width="1365"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Dexter Grist Mill in Sandwich
        </em>
       </p>
       <p>
        Naturally, given its history, Sandwich does not lack for contemporary art, glassblown or otherwise.
       </p>
       <p>
        The
        <a href="http://www.collectionsgallery.com/default.html" target="_blank">
         Collections Gallery
        </a>
        displays works from more than a dozen different local artists, which run the gamut from paintings to jewelry to porcelain to dolls. To learn about the artists before visiting, just click
        <a href="http://www.collectionsgallery.com/Artists.html">
         here
        </a>
        .
       </p>
       <p>
        Theres even more fine work at
        <a href="https://www.facebook.com/pages/My-Sisters-Gallery/149460125095971" target="_blank">
         My Sisters Gallery
        </a>
        and
        <a href="http://www.sandwich-artisans.com/" target="_blank">
         Sandwich Artisans
        </a>
        . The latter group hosts shows throughout the summer and into fall, which present an excellent venue for souvenir or gift shopping.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Sandwich-Glass.jpg">
         <img alt="Sandwich Glass" class="aligncenter size-full wp-image-7730" height="628" src="/blog/wp-content/uploads/2015/05/Sandwich-Glass.jpg" width="628"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Glass crafted by International Glass Master Michael Maygar
        </em>
       </p>
       <p>
        If a guided tour of town leaves you itching for more time outdoors, there are a couple great places just outside the cultural districts borders.
        <a href="http://dguides.com/capecod/attractions/upper-cape-beaches/town-beach-and-sandwich-boardwalk/" target="_blank">
         The Sandwich Boardwalk
        </a>
        , a breathtaking path that leads to the beach, is worth a trip in and of itself.
       </p>
       <p>
        On the other side of Route 6A, closer to the cultural district, theres
        <a href="http://www.massvacation.com/business/8081/" target="_blank">
         Heritage Museums &amp; Gardens
        </a>
        . The grounds span more than 100 acres of gardens and natural trails, which are tailor-made for walking, exploring and learning.
       </p>
       <p>
        Special exhibitions, folk art, antique cars and a vintage carousel are all also on site at Heritage Gardens, making it one of the best family destinations around.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Heritage-Museums-and-Gardens.jpg">
         <img alt="Heritage Museums and Gardens" class="aligncenter size-full wp-image-7731" height="1254" src="/blog/wp-content/uploads/2015/05/Heritage-Museums-and-Gardens.jpg" width="2048"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Rhododendrons putting the garden in Heritage Museums &amp; Gardens
        </em>
       </p>
       <p>
        And, obviously, what would Sandwich be without a sandwich contest?
       </p>
       <p>
        As summer swings into full force,
        <a href="http://www.sandwichfest.com/" target="_blank">
         Sandwichfest
        </a>
        arrives in town, bringing all kinds of good fun with it. More than 75 artist booths accompany the Best Sandwich in Sandwich competition, with a beer and wine garden providing the beverages; the dog costume parade only makes it better.
       </p>
       <p>
        During other hours of the day, Sandwich has a lively restaurant scene to cover hungry diners.
        <a href="http://www.capecodteashop.com/" target="_blank">
         Beths Bakery &amp; Caf
        </a>
        is a good start to any morning, with tasty lunch options, too.
        <a href="http://www.belfryinn.com/" target="_blank">
         The Belfry Inn &amp; Bistro
        </a>
        and
        <a href="http://www.massvacation.com/business/8083/" target="_blank">
         Caf Chew
        </a>
        are two more tempting restaurants that offer tasty, substantial dishes.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Cafe-Chew-Sandwich-MA.jpg">
         <img alt="Cafe Chew Sandwich MA" class="aligncenter size-full wp-image-7732" height="628" src="/blog/wp-content/uploads/2015/05/Cafe-Chew-Sandwich-MA.jpg" width="628"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Caf Chew may very well have the best sandwich in Sandwich!
        </em>
       </p>
       <p>
        Travelers to the area have a variety of choices for overnights as well, although those do tend to book up quite quickly during the spring and summer.
       </p>
       <p>
        The aforementioned Belfry Inn &amp; Bistro is a TripAdvisor Certificate of Excellent recipient, something it shares with the
        <a href="http://www.massvacation.com/business/38/" target="_blank">
         1750 Inn at Sandwich Center
        </a>
        . The
        <a href="http://www.massvacation.com/business/4638/" target="_blank">
         Danl Webster Inn &amp; Spa
        </a>
        is also a fantastic lodging and you can find more
        <a href="http://www.massvacation.com/stay/" target="_blank">
         here
        </a>
        .
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/05/Daniel-Webster-Inn-and-Spa-Sandwich.jpg">
         <img alt="Daniel Webster Inn and Spa Sandwich" class="aligncenter size-full wp-image-7733" height="628" src="/blog/wp-content/uploads/2015/05/Daniel-Webster-Inn-and-Spa-Sandwich.jpg" width="628"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         The most welcoming Danl Webster Inn &amp; Spa
        </em>
       </p>
       <p>
        <em>
        </em>
        Theres more to Sandwichs Glasstown Cultural District, of course, but the above is a good heading. To learn about everything that there is to see and do thereabouts, click
        <a href="http://www.massvacation.com/explore/arts/cultural-districts/#glasstown" target="_blank">
         here
        </a>
        . For a look at more of Massachusetts cultural districts, just click
        <a href="http://www.massvacation.com/explore/arts/cultural-districts/" target="_blank">
         here
        </a>
        .
       </p>
