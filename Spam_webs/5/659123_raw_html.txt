			var catId = $webkul("#catValue").val();
			clearInterval(timer);  
			timer = setTimeout(function() {
				var keyCode = e.keyCode || e.which;
				if(keyCode==13){
					var title = $webkul("#search_autocomplete ul li.product.over a").attr('href');
					if(title){
						window.location = title;
					}else{
						var title = $webkul("#search_autocomplete ul li.tag.over").attr('title');
						if(title){
							var currTitle = title.replace(/ /g,"+")
							var url ="https://store.webkul.com/catalogsearch/result/"+"?cat="+catId+"&amp;q="+currTitle;
							window.location = url;
						}else{
							var term = $webkul("#search").val();
							term = $webkul.trim(term);
							if(term!=""){
								var currTitle = term.replace(/ /g,"+")
								var url ="https://store.webkul.com/catalogsearch/result/"+"?cat="+catId+"&amp;q="+currTitle;
								window.location = url;
							}
						}
					}	
				}

				if(keyCode!=38 &amp;&amp; keyCode!=40 &amp;&amp; keyCode!=13){

					AjaxData();
				}

				if (keyCode == 40 || keyCode == 38){
					var css = $webkul('#search_autocomplete ul.suggestion-list').css('display');
					if(css){
						if(css=="none"){
							var term = $webkul('#search').val();
							var pterm = $webkul('#prevsearch').val();
							term = $webkul.trim(term)
							if(pterm==term){
								$webkul('#search_autocomplete ul.suggestion-list').css('display','block');
								count=0;
							}

								AjaxData();
							}
						}
					}

				}

				// 	$webkul('#prevsearch').val("");
				// }
			},500);	
		});
		

		$webkul("#search").on('keydown', function(e) {
			var keyCode = e.keyCode || e.which;
			var len = $webkul("#search_autocomplete ul li").length;
			var tagLen = $webkul("#search_autocomplete ul li.tag").length;
			
			if (keyCode == 40){
				count++;
				if(searchSetting==2){
					if(tagLen&gt;0){
						if(count==tagLen+1){
							count=count+1;
						}	
					}
				}	
				if(count&gt;len){
					count=1;
				}
				webkul(count);
			}
			if(len&gt;0){
				if (keyCode == 38){
					count--;
					if(searchSetting==2){
						if(tagLen&gt;0){
							if(count==tagLen+1){
								count=count-1;
							}	
						}
					}	
					if(count&lt;1){
						count=len;
					}
					webkul(count);
				}
			}
		});

		var AjaxData = function(){
			$webkul('#search').addClass('loading');
			var term = $webkul('#search').val();
			var pterm = $webkul('#prevsearch').val();
			term = $webkul.trim(term)
						
			if(term==""){
				$webkul('#search_autocomplete ul.suggestion-list').hide();
			}
			var sug = $webkul('#search_autocomplete ul').html();
			
			if(term!="" &amp;&amp; pterm!=term){
				jQuery.ajax({
					url: "https://store.webkul.com/catalogsearch/ajax/suggest/",

					type:"post",
					datatype :"html",
					success:function(data){
						$webkul('#search_autocomplete').empty();
						$webkul('#search_autocomplete').append(data);
						$webkul('#search').removeClass('loading');
						$webkul('#prevsearch').val(term);
						count=0;
					}
				});
			}else{
				$webkul('#search').removeClass('loading');
			}

		}
