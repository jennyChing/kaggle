       <h4>
        OVERVIEW
       </h4>
       <br/>
       <p>
       </p>
       <div>
        <a href="http://www.sleepsafebed.com/" rel="nofollow" target="_blank">
         SleepSafe Beds
        </a>
        is nationally known for manufacturing adaptable safety beds for those with special needs.
        <a href="http://www.assuredcomfortbed.com" rel="nofollow" target="_blank">
         Assured Comfort Beds
        </a>
        are adjustable beds designed for comfort andwellness with safety options available.
        <br/>
        <br/>
       </div>
       <div>
        <div>
         <b>
          MISSION
          <br/>
          <br/>
         </b>
        </div>
        SleepSafe Beds LLC provides safe, functional products for our customers. With input from caregivers and health care professionals, we continue to develop new products designed to make life easier and safer for individuals with special needs as well as those who care for them.
       </div>
       <div>
        <div>
         <br/>
         <b>
          DESCRIPTION
         </b>
         <br/>
         <br/>
        </div>
        Our patented SleepSafe Beds are designed to address the problemsof entrapment and falls that may occur in traditional hospital beds.
        <br/>
        <br/>
        SleepSafe Beds are custom built to each order in the USA, offering twin or full size; fixed, articulating or HiLo frames; padding; and a wide range offinishes, including brightly colored children's beds.
        <br/>
        <br/>
        In addition to addressing the problemof entrapment found in traditional hospital equipment, our beds' aesthetics and durableconstruction promote an inviting atmosphere for the benefit of not only the resident, where self-esteem may be an issue, but also for family members and caregivers who appreciate a more normalized setting.
        <br/>
        <br/>
        <div>
         <b>
          SAFETY INFORMATION
         </b>
         <br/>
         <br/>
        </div>
        <div>
         SleepSafe Beds are registered and listed with FDA.
         <br/>
         <br/>
         SleepSafe Beds are not only visually appealing, they also address safety issues in ways no other more traditional institutional equipment can match:
         <br/>
         <br/>
          The clear 3/16" windows incorporated in the safety rails affords the opportunit
         <span>
          y for both acknowledgement and need to observe the environment in situations where the need to see and be seen are important.
          <br/>
          <br/>
           Entrapment issues are addressedwith the patented SleepSafe Bed design. The space between the side rails, head and foot boards and the mattress is nearly nonexistent - even with mattress compression.
          <br/>
         </span>
        </div>
       </div>

   $(function(){
    
    var $container = $('#container'),
    $select = $('div#filterGroup select');

    $container.isotope({
      itemSelector : '.color-shape',
      masonry: {
        columnWidth: 80
      }
    });

var $searchcontainer = $('#searchisotope');
$searchcontainer.isotope({
        // options
  itemSelector : '.search-result',
  layoutMode : 'straightDown',
  getSortData : {
    views : function ( $elem ) {
      return parseInt( $elem.find('.views').text(), 10 );
    },
    createdate: function ($elem) {
        var dateStr = $elem.find('.createdate').text(),
            dateArray = dateStr.split('/'),
            year = dateArray[2],
            month = dateArray[0],
            day = dateArray[1];
        return new Date(year, month, day);
    },
    comments : function ( $elem ) {
      return parseInt( $elem.find('.commentcount').text(), 10 );
    },
  }
});

    // filter buttons
    $select.change(function() {
        var $this = $(this);
			filters = {};
        var $optionSet = $this;
        var group = $optionSet.attr('data-filter-group');
    	filters[group] = $this.find('option:selected').attr('data-filter-value');

        var isoFilters = [];
        for (var prop in filters) {
            isoFilters.push(filters[prop])
        }
        var selector = isoFilters.join('');

        $container.isotope({
            filter: selector
        });

        return false;
    });

	$('.filters a').click(function(){
	  var selector = $(this).attr('data-filter');
	      $searchcontainer.isotope({ filter: selector });
	  return false;
	});
	$('.sort-by a').click(function(){
	  // get href attribute, minus the '#'
	  var sortName = $(this).attr('href').slice(1);
	  $searchcontainer.isotope({ sortBy : sortName, sortAscending : false });
	  return false;
	});

  });
  </script>
