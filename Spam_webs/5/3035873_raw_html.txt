            function TxtOverlay(pos, txt, cls, map){

                // Now initialize all properties.
                this.pos = pos;
                this.txt_ = txt;
                this.cls_ = cls;
                this.map_ = map;

                // We define a property to hold the image's
                // div. We'll actually create this div
                // upon receipt of the add() method so we'll
                // leave it null for now.
                this.div_ = null;

                // Explicitly call setMap() on this overlay
                this.setMap(map);
            }

            TxtOverlay.prototype = new google.maps.OverlayView();



            TxtOverlay.prototype.onAdd = function(){

                // Note: an overlay's receipt of onAdd() indicates that
                // the map's panes are now available for attaching
                // the overlay to the map via the DOM.

                // Create the DIV and set some basic attributes.
                var div = document.createElement('DIV');
                div.className = this.cls_;

                div.innerHTML = this.txt_;

                // Set the overlay's div_ property to this DIV
                this.div_ = div;
                var overlayProjection = this.getProjection();
                var position = overlayProjection.fromLatLngToDivPixel(this.pos);
                div.style.left = position.x + 'px';
                div.style.top = position.y + 'px';
                // We add an overlay to a map via one of the map's panes.

                var panes = this.getPanes();
                panes.floatPane.appendChild(div);
            }
            TxtOverlay.prototype.draw = function(){


                var overlayProjection = this.getProjection();

                // Retrieve the southwest and northeast coordinates of this overlay
                // in latlngs and convert them to pixels coordinates.
                // We'll use these coordinates to resize the DIV.
                var position = overlayProjection.fromLatLngToDivPixel(this.pos);


                var div = this.div_;
                div.style.left = position.x + 'px';
                div.style.top = position.y + 'px';



            }
           //Optional: helper methods for removing and toggling the text overlay.  
            TxtOverlay.prototype.onRemove = function(){
                this.div_.parentNode.removeChild(this.div_);
                this.div_ = null;
            }
            TxtOverlay.prototype.hide = function(){
                if (this.div_) {
                    this.div_.style.visibility = "hidden";
                }
            }

            TxtOverlay.prototype.show = function(){
                if (this.div_) {
                    this.div_.style.visibility = "visible";
                }
            }

            TxtOverlay.prototype.toggle = function(){
                if (this.div_) {
                    if (this.div_.style.visibility == "hidden") {
                        this.show();
                    }
                    else {
                        this.hide();
                    }
                }
            }

            TxtOverlay.prototype.toggleDOM = function(){
                if (this.getMap()) {
                    this.setMap(null);
                }
                else {
                    this.setMap(this.map_);
                }
            }

function drawBoundryLayerFromG (district_id,color,d,iw) {
	var bounds = new google.maps.LatLngBounds();
	for (g=0;g&lt;d.length;g++) { // foreach boundary
		triangleCoords = Array();
		for (i=0,iC=d[g].length;i&lt;iC;i++) { // For reach path
			triangleCoords[i] = Array();
			for (ii=0,iiC=d[g][i].length;ii&lt;iiC;ii++) { // For each point
				triangleCoords[i][ii] = Array();
				triangleCoords[i][ii] = new google.maps.LatLng(d[g][i][ii][0],d[g][i][ii][1]);
				bounds.extend(triangleCoords[i][ii]);
			}
		}
		if (typeof window.map_overlays['polygon_'+district_id+'_'+g] !== 'undefined') {
			window.map_overlays['polygon_'+district_id+'_'+g].setMap(null);
		}
		window.map_overlays['polygon_'+district_id+'_'+g] = new google.maps.Polygon({
			paths: triangleCoords,
			strokeColor: color,
			strokeOpacity: (window.subject_type=='election'?.35:.25),
			strokeWeight: 1,
			fillColor: color,
			fillOpacity: .35,
		});
		window.map_overlays['polygon_'+district_id+'_'+g].setMap(window.map);
		// Draw infowindow
		if (iw!='') {
			var contentString = '&lt;div class="iw"&gt;'+iw+'&lt;/div&gt;';

				} else {
					window.infowindow['polygon_'+district_id].open(window.map,window.map_overlays['polygon_'+district_id+'_'+g]);
				}
			});
			google.maps.event.addListener(window.map_overlays['polygon_'+district_id+'_'+g],'mouseover',function(event){
				window.hovering_infowindow = 'polygon_'+district_id;
				if (window.current_infowindow != '' &amp;&amp; window.current_infowindow != 'polygon_'+district_id) {
					window.infowindow[window.current_infowindow].close();
				}
				if ('polygon_'+district_id == window.hovering_infowindow &amp;&amp; (window.current_infowindow=='' || window.current_infowindow != 'polygon_'+district_id)) {
					window.current_infowindow = 'polygon_'+district_id;
					window.infowindow[window.current_infowindow].open(window.map,window.map_overlays['polygon_'+district_id+'_'+g]);
					window.infowindow[window.current_infowindow].setPosition(event.latLng);
				}
			});
			google.maps.event.addListener(window.map_overlays['polygon_'+district_id+'_'+g],'mousemove',function(event){
				if (window.infowindow[window.current_infowindow]) {
					window.infowindow[window.current_infowindow].setPosition(event.latLng);
				}
			});
			google.maps.event.addListener(window.map,'mousemove',function(){
				if (window.infowindow[window.current_infowindow]) {
					window.infowindow[window.current_infowindow].close();
				}
			});
		}
	}
}
