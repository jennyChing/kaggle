	var searchTabIndex = 99;
	function setSearchTab(){
		searchTabIndex = 99;
		$('.searchTab').children('li').each(function(){
			if(!$(this).hasClass('active')){
				$(this).css({'z-index':searchTabIndex--});
			} else {
				$(this).css({'z-index':101});
			}
		})
	}
	setSearchTab();
	$('.search-box-inner').children('div').each(function(){$(this).css({'display':'none'})})
	function openTab(name){
		$('.search-box-inner').children('div').each(function(){
			if($(this).is(':visible')){
				$(this).slideUp(600);
			}
		})
		$('#'+name).slideDown(600);
		return false;
	}
	$('.searchTab').children('li').each(function(){
		$(this).mouseover(function(){
			setSearchTab();
			$(this).css({'z-index':100});
		}).click(function(){
			$('.searchTab').children('li').each(function(){
				$(this).removeClass('active')
			})
			$(this).addClass('active');
			setSearchTab();
			openTab($(this).children('a').attr('href'))
			return false;
		})
	})
	$('.searchTab').children('li:eq(0)').trigger('click')
