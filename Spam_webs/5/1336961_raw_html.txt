 * Allow GA events via `ga-event` attributes on elements, both those
 * static on the page and those added in dynamically.
 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')
 *
 * Elements that have been processed on the page already will have a 'ga-event-added' attribute
 *
 */
$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event
    //  non-interactive (optional) - true for non-interactive events (ie. it shouldn't be used in bounce-rate calcs)
    //
    // var ga_event_handler = function(ev) {
    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';
                
            // Only add a delegated handler once for a given even `type'
            if (!_handlers[type]) {
                // Delegate to handle dynamic elements, track element ref via closure 
                $(document).on(type, selector, (function() {
                    return function(ev) {
                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());
                _handlers[type] = true;
            }

            // mark element as processed
            $(this).attr('ga-event-added', 'true');
        });

        // wait for 100ms and poll again (catch dynmically added elements)
        window.setTimeout(poll_ga_attr_events, 150);
    }


    // let's go, start processing ...
    poll_ga_attr_events();

});

function listen(evnt, elem, func) {
    if (elem.addEventListener) { // W3C DOM
        elem.addEventListener(evnt,func,false);
    }
    else if (elem.attachEvent) { // IE DOM
        var r = elem.attachEvent("on"+evnt, func);
        return r;
    }

            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1175263">

            <span class="hellip">

            <span class="hidden">
             the big state. In the 1986 hit film "Aliens," young Carrie Henn played the role of Rebecca "Newt" Jordan alongside Sigourney Weaver. This role was Carrie's first and last professional acting role. However, she has since participated in a few documentaries about the iconic film. She earned a degree from California State University and became an elementary school teacher.Actress Anna Chlumsky gained significant praise for her role as Vada Margaret Sultenfuss in the 1991 film "My Girl" and the 1994 "My Girl 2." While she did not land any leading film roles after the "My Girl" franchise, she has continued to act into her adulthood. In 2012, she landed a recurring role as Amy Brookheimer on the TV series "Veep."It's likely you remember this face as the child star from "Jumanji." Although he hasn't appeared in much else since then, Pierce voiced the part of Chip in "Beauty and the Beast" and the original Tails in the "Sonic the Hedgehog" series. So, what's Pierce up to these days? He decided to leave Hollywood to play the role of daddy and raise his two children.In the 1984 hit "The Never Ending Story," young actor Barret Oliver played Bastian Bux. The success of this film lead to many coveted roles over the next few years in movies like "Cocoon" and "Frankenweenie." However, by 1989, Barret had left the acting world altogether. Keeping in touch with his creative roots, Barret has become a professional photographer, and has even published a book on photography called "A History of the Woodburytype".Spreading Christmas cheer was Daniel Tay's strong suit in the 2003 holiday classic "Elf." Daniel played younger brother to Buddy the Elf, played by Will Ferrell. Since then, Daniel has done voice work for video games like "Grand Theft Auto IV" and appeared in the 2007 film "Brooklyn Rules." I think Daniel will forever be remembered in our hearts at Christmas time.Ross Malinger is best known for his role as Tom Hank's adorable son in the movie "Sleepless in Seattle." Although his success with the hit movie got him other roles in films like "Sudden Death" and TV series like "Party of Five" and "Seinfeld," his fame was slowly dwindled. He made his last TV appearance in 2006. Since then it was reported Malinger worked in a Malibu car dealership.Ariana Richards began picking up small acting roles in classic TV series such as "Golden Girls" and "Empty Nest." However, it was her role as Lex Murphy in the 1993 hit "Jurassic Park" that caught audience's attention. Arian continues to dabble in acting here and there, but is now a decorated professional painter. One of her most notable paintings is "Lady of the Dahlias."Actor Jeff Cohen is a child actor that seemed to have disappeared after he played Chunk in the 1985 movie, "The Goonies." Cohen continued to act picking up roles in film and TV over the next few years following the success of "The Goonies." While Cohen left his acting days behind, he did not leave the acting world altogether. He is now a successful entertainment lawyer and partner at Cohen Gardner LLP.Jonathan Lipnicki played Ray Boyd, the son of actress Renee Zellweger in the 1996 film "Jerry Maguire." His sincerity and innocence quickly stole the hearts of filmgoers. Shortly after the success of "Jerry Maguire," Jonathan landed a recurring role on the sitcom "The Jeff Foxworthy Show" from 1996-1997. Since then, Jonathan has picked up several small roles in both film and TV.Everybody remembers Ross as the spunky little Nicky on the "Fresh Prince of Bel-Air." After his debut on show, Will Smith got Ross a part in the 1996 hit, "Independence Day."  Since then, Ross has done voice work for "The Wild Thornberrys" and appeared in an episode of "Judging Amy." Although Ross has temporarily fallen off the grid, we should expect a comeback from him in the future. He's currently studying film at California State University, Northridge.After getting her start on "The Young and the Restless" and "Honey, I Shrunk the Kids," O'Neil left Hollywood to pursue less innocent and more revealing roles. Although her success never took off, it seemed the star always had taller dreams. In the 2000s, O'Neill joined a circus act called "Girls On Stilts."Actor Devin Ratray played the mischievous older brother Buzz McCallister in the 1990 cult classic "Home Alone." While Devin has not had any smash hits since "Home Alone," he has continued with his acting career, picking up several smaller roles on TV shows like "Law &amp; Order," along with small parts on shows like "The Good Wife" and "Person of Interest."Mara Wilson played the main character in the 1996 film, "Matilda." Prior to this role, Mara had gained recognition for her roles in 1993 "Mrs. Doubtfire" and the 1994 remake of "The Miracle on 34th Street." Mara pulled away from professional acting a few years after "Matilda," but still acts as a hobby. She has taken an interest in creative writing and young adult fiction.How could we ever forget young Anakin Skywalker from "Star Wars Episode 1: The Phantom Menance"? After his performance in "Star Wars," Lloyd starred alongside Arnold Schwarzenegger in the holiday hit "Jingle All the Way." After a few voicing gigs, Lloyd decided to leave Hollywood for good. He claims his childhood stardom ruined his school life and he would sometimes have to do up to 60 interviews a day.
