   var MF = function() {
	function main(window, domain) {
		var doc = window.document;
		domain = domain || 'mindflash.com';

		function startGAUniversalTracker() {
			window.ga_debug = {trace: true};

			(function(i,s,o,g,r,a,m){i['GoogleAnalyticsObject']=r;i[r]=i[r]||function(){
				(i[r].q=i[r].q||[]).push(arguments)},i[r].l=1*new Date();a=s.createElement(o),
				m=s.getElementsByTagName(o)[0];a.async=1;a.src=g;m.parentNode.insertBefore(a,m)
			})(window,doc,'script','//www.google-analytics.com/analytics.js','ga');

			ga('create',
				'UA-5857047-1',
				'.'+domain,
				 {
					'allowLinker': true,
					'allowHash': false,
					'alwaysSendReferrer': true,
					'legacyCookieDomain': domain
				 });
			ga('require', 'linkid', 'linkid.js');
			ga(function() {
				saveOriginalReferrer();

				function track(err, attribution) {
					trackPageView(attribution);
				}

				analyze(
					function(err, res) {
						if(err) {
							track(err);
							return;
						}

						if(doc.referrer &amp;&amp; weAreNotReferringToOurselves()) {
							bakeCookies(res, track);
						} else {
							track(null, res);
						}
				});
			});
		}

		function weAreNotReferringToOurselves() {
			var referrerAbstract = doc.referrer.substring(0, 25)
			return !contains(referrerAbstract, 'www.' + domain) &amp;&amp; !contains(referrerAbstract, domain);
    	}

		function saveOriginalReferrer() {
			// save original referrer
			if(!docCookies.hasItem('REFERRER')) {
				var stats = getAttributionInfo();
				store('REFERRER', stats.referrer || '');
			}
		}

		// get the referral data depending on which GA they hit originally
 		function analyze(cb) {
			if(cb) {
				if(visitorHasOldGACookie())
					cb(null, getGAReferrerDataClassic());
				else {
					cb(null, getAttributionInfo());
				}
			}
		}

		function bakeCookies(ingredients, cb) {
			store('SOURCE', ingredients.source || '');
			store('MEDIUM', ingredients.medium || '');
			store('CAMPAIGN',ingredients.campaign || '');
			store('KEYWORD',ingredients.keyword || '');
			store('CONTENT',ingredients.content || '');
			store('LANDINGPAGE', ingredients.landingPage || '');
			store('REFERRERURL', ingredients.originalDocReferrer || '');

			cb(null, ingredients);
		}

		function trackPageView(res) {
			if(res &amp;&amp; res.referrer)
				ga('set','dimension2', res.referrer);	// set Original Referrer in GA correctly

			ga('send', 'pageview');
		}

		function getAttributionInfo() {
			var utmParams = docCookies.getItem('CUR') || doc.location.search;
			var gaStats = {
				source: _uGC(utmParams,'utm_source=','&amp;') || '',
				medium: _uGC(utmParams,'utm_medium=','&amp;') || '',
				campaign: _uGC(utmParams,'utm_campaign=','&amp;') || '',
				keyword: _uGC(utmParams,'utm_term=','&amp;') || '',
				content: _uGC(utmParams,'utm_content=','&amp;') || ''
			};

			// can we have utm_parameters alone? without source?
			if(gaStats.source == '') {

				if(doc.referrer.length &gt; 0)
				{
					var uri = parseUri(doc.referrer);

					if(contains(uri.host, "google"))
					{
						// track as google organic
						gaStats.source = 'google';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
					}
					else if(contains(uri.host, "yahoo"))
					{
						// track as yahoo organic
						gaStats.source = 'yahoo';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
					}
					else if(contains(uri.host, "bing"))
					{
						// track as bing organic
						gaStats.source = 'bing';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
						gaStats.keyword = _uGC(uri.query,'q=','&amp;');
					}
					else if(contains(uri.host, "aol")){
						// track as aol organic
						gaStats.source = 'aol';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
						gaStats.keyword = _uGC(uri.query,'query=','&amp;');
						if(!gaStats.keyword) gaStats.keyword = _uGC(uri.query,'encquery=','&amp;');
					}
					else if(contains(uri.host, "ask")){
						// track as ask organic
						gaStats.source = 'ask';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
						gaStats.keyword = _uGC(uri.query,'q=','&amp;');
					}
					else if(contains(uri.host, "startsiden")){
						// track as startsiden organic
						gaStats.source = 'startsiden';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
					}
					else if(contains(uri.host, "yandex")){
						gaStats.source = 'yandex';
						gaStats.medium = 'organic';
						gaStats.campaign = 'organic';
						gaStats.keyword = _uGC(uri.query,'text=','&amp;');
					}
					else
					{
						// track as referral
						gaStats.source = uri.host || '';
						gaStats.medium = 'referral';
						gaStats.campaign = 'referral';
						gaStats.content = uri.path || '';
					}
				}
			}

			// if we still don't know who they are, assume they came in direct
			if(gaStats.source == '') gaStats.source = 'direct';
			if(gaStats.medium == '') gaStats.medium = 'none';
			if(gaStats.campaign == '') gaStats.campaign = 'direct';

			// record landing page
			gaStats.landingPage = doc.location.pathname || '';

			// parse up the original referrer with pathname //source!medium!campaign!term!content
			var referrer = gaStats.source + '!' + gaStats.medium + '!' + gaStats.campaign + '!' + gaStats.keyword + '!' + gaStats.content;
			referrer = referrer.replace('%20','-');
			referrer = referrer.replace(/[^a-zA-Z0-9-~!*_.]/g,'');
			referrer = referrer.substr(0,62);
			gaStats.referrer = referrer || '';

			// url they rode in on
			gaStats.originalDocReferrer = doc.referrer || '';

			return gaStats;

		function getGAReferrerDataClassic() {
			var gaCookie = docCookies.getItem('__utmz');
			var gaStats = {
				source: _uGC(gaCookie,'utmcsr=','|') || '',
				medium: _uGC(gaCookie,'utmcmd=','&amp;') || '',
				campaign: _uGC(gaCookie,'utmccn=','&amp;') || '',
				keyword: _uGC(gaCookie,'utmctr=','&amp;') || '',
				content: _uGC(gaCookie,'utmcct=','&amp;' || '')
			};

			if (_uGC(gaCookie,'utmgclid=','|') != '')
			{
				gaStats.source = 'google';
				gaStats.medium = 'cpc';
			}

			return gaStats;
		}

		function contains(str, expected) {

		function _uGC(l,n,s) {
				if (!l || l=="" || !n || n=="" || !s || s=="") return "";
				var i,i2,i3,c="-";
				i=l.indexOf(n);
				i3=n.indexOf("=")+1;
				if (i &gt; -1) {
						i2=l.indexOf(s,i); if (i2 &lt; 0) { i2=l.length; }
						c=l.substring((i+i3),i2);
				}
				if (c == '-')
						return '';
				else
						return c;
		}

		function parseUri (str) {
			var	o   = parseUri.options,
				m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
				uri = {},
				i   = 14;

			while (i--) uri[o.key[i]] = m[i] || "";

			uri[o.q.name] = {};
			uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
				if ($1) uri[o.q.name][$1] = $2;
			});

			return uri;
		};

		 |*|
		 |*|  Syntaxes:
		 |*|
		 |*|  * docCookies.setItem(name, value[, end[, path[, domain[, secure]]]])
		 |*|  * docCookies.getItem(name)
		 |*|  * docCookies.removeItem(name[, path[, domain]])
		 |*|  * docCookies.hasItem(name)
		 |*|  * docCookies.keys()
		 |*|
		 \*/

		 var docCookies = {
			 getItem: function (sKey) {
				 if (!sKey) { return null; }
				 return decodeURIComponent(doc.cookie.replace(new RegExp("(?:(?:^|.*;)\\s*" + encodeURIComponent(sKey).replace(/[\-\.\+\*]/g, "\\$&amp;") + "\\s*\\=\\s*([^;]*).*$)|^.*$"), "$1")) || null;
			 },
			 setItem: function (sKey, sValue, vEnd, sPath, sDomain, bSecure) {
				 if (!sKey || /^(?:expires|max\-age|path|domain|secure)$/i.test(sKey)) { return false; }
				 var sExpires = "";
				 if (vEnd) {
					 switch (vEnd.constructor) {
						 case Number:
							 sExpires = vEnd === Infinity ? "; expires=Fri, 31 Dec 9999 23:59:59 GMT" : "; max-age=" + vEnd;
							 break;
						 case String:
							 sExpires = "; expires=" + vEnd;
							 break;
						 case Date:
							 sExpires = "; expires=" + vEnd.toUTCString();
							 break;
					 }
				 }
				 //console.log(encodeURIComponent(sKey) + "=" + encodeURIComponent(sValue) + sExpires + (sDomain ? "; domain=" + sDomain : "") + (sPath ? "; path=" + sPath : "") + (bSecure ? "; secure" : ""));
				 doc.cookie = encodeURIComponent(sKey) + "=" + encodeURIComponent(sValue) + sExpires + (sDomain ? "; domain=" + sDomain : "") + (sPath ? "; path=" + sPath : "") + (bSecure ? "; secure" : "");
				 return true;
			 },
			 removeItem: function (sKey, sPath, sDomain) {
				 if (!this.hasItem(sKey)) { return false; }
				 doc.cookie = encodeURIComponent(sKey) + "=; expires=Thu, 01 Jan 1970 00:00:00 GMT" + (sDomain ? "; domain=" + sDomain : "") + (sPath ? "; path=" + sPath : "");
				 return true;
			 },
			 hasItem: function (sKey) {
				 if (!sKey) { return false; }
				 return (new RegExp("(?:^|;\\s*)" + encodeURIComponent(sKey).replace(/[\-\.\+\*]/g, "\\$&amp;") + "\\s*\\=")).test(doc.cookie);
			 },
			 keys: function () {
				 var aKeys = doc.cookie.replace(/((?:^|\s*;)[^\=]+)(?=;|$)|^\s*|\s*(?:\=[^;]*)?(?:\1|$)/g, "").split(/\s*(?:\=[^;]*)?;\s*/);
				 for (var nLen = aKeys.length, nIdx = 0; nIdx &lt; nLen; nIdx++) { aKeys[nIdx] = decodeURIComponent(aKeys[nIdx]); }
				 return aKeys;
			 }
		 };
		 // END cookies.js

	// Fixes moc document to work like document.cookie does
	function setupMocDoc(mocDoc) {
		var cookies = {};
		mocDoc.__defineGetter__('cookie', function () {
				var output = [];
				for (var cookieName in cookies) {
						output.push(cookieName + "=" + cookies[cookieName]);
				}
				return output.join(";");
		});
		mocDoc.__defineSetter__('cookie', function (s) {
				var indexOfSeparator = s.indexOf("=");
				var key = s.substr(0, indexOfSeparator);
				var value = s.substring(indexOfSeparator + 1);
				cookies[key] = value;
				return key + "=" + value;
		});
		mocDoc.clearCookies = function () {
				cookies = {};
		};
		return mocDoc;
	}
