/**
* hoverIntent is similar to jQuery's built-in "hover" function except that
* instead of firing the onMouseOver event immediately, hoverIntent checks
* to see if the user's mouse has slowed down (beneath the sensitivity
* threshold) before firing the onMouseOver event.
*
* hoverIntent r5 // 2007.03.27 // jQuery 1.1.2+
* &lt;http://cherne.net/brian/resources/jquery.hoverIntent.html&gt;
*
* hoverIntent is currently available for use in all personal or commercial
* projects under both MIT and GPL licenses. This means that you can choose
* the license that best suits your project, and use it accordingly.
*
* // basic usage (just like .hover) receives onMouseOver and onMouseOut functions
* $("ul li").hoverIntent( showNav , hideNav );
*
* // advanced usage receives configuration object only
* $("ul li").hoverIntent({
*	sensitivity: 7, // number = sensitivity threshold (must be 1 or higher)
*	interval: 100,   // number = milliseconds of polling interval
*	over: showNav,  // function = onMouseOver callback (required)
*	timeout: 0,   // number = milliseconds delay before onMouseOut function call
*	out: hideNav    // function = onMouseOut callback (required)
* });
*
* @param  f  onMouseOver function || An object with configuration options
* @param  g  onMouseOut function  || Nothing (use configuration options object)
* @author    Brian Cherne &lt;brian@cherne.net&gt;
*/
(function($) {
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 10,
			interval: 200,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
})(jQuery);

               <h1>
                Vaccine meta-analysis of more than 1.2 million children finds no autism link
               </h1>
               <div class="date">
                Posted On:June 11, 2014
               </div>
               <div class="news-contens">
                <style>
                 .social{ margin:10px 0!important; height:20px;}
                </style>
                <div class="Post_heading_Rx">
                 <div class="social">
                  <span class="st_facebook_hcount" displaytext="Facebook">
                  </span>
                  <span class="st_twitter_hcount" displaytext="Tweet">
                  </span>
                  <span class="st_googleplus_hcount" displaytext="Google +">
                  </span>
                  <span class="st_linkedin_hcount" displaytext="LinkedIn">
                  </span>
                  <!-- &lt;span class='st_email_hcount' displayText='Email'&gt;&lt;/span&gt;-->
                 </div>
                 <bdo>
                  <img alt="A new meta-analysis provides further evidence that there is no link between vaccines and autism spectrum disorders." class="attachment-52x70 wp-post-image" height="47" src="http://www.transactrx.com/healthcare-news/wp-content/uploads/a-new-metaanalysis-provides-further-evidence-that-there-is-no-link-betwe_1393_624924_0_14096716_500.jpg" title="A new meta-analysis provides further evidence that there is no link between vaccines and autism spectrum disorders." width="52"/>
                  <p>
                   Although vaccines are among the most important tools in preventative medicine, that fact means nothing if community households refuse to have their children inoculated. Age restrictions, weakened immune systems and allergies are medically legitimate reasons to forgo immunization, but fears about how vaccines may increase the likelihood that a child will develop autism have also discouraged parents. Even though the science behind these fears has been widely debunked, parents' misgivings continue to persist.
                  </p>
                  <p>
                   One team of scientists from the University of Sydney in Australia decided to conduct their own investigation, and concluded that there is no link between vaccines and autism spectrum disorders, as published by the journal Vaccine. Additionally, the patient advocacy group Autism Speaks responded to the study by
                   <a href="http://www.autismspeaks.org/science/science-news/new-meta-analysis-confirms-no-association-between-vaccines-and-autism" target="_blank">
                    reiterating its support for credible research
                   </a>
                   .
                  </p>
                  <p>
                   <strong>
                    Vaccines, components and combination shots are not implicated
                   </strong>
                   <br/>
                   For their meta-analysis, the study authors mined data from five previous cohort studies and five case-control studies, which included more than 1.2 million children and more than 9,900 children, respectively. The prior research was found through searches of MEDLINE, PubMed, EMBASE and Google Scholar, dating up to April 2014.
                  </p>
                  <p>
                   The data revealed that there was no link between autism or autism spectrum disorders and immunization, the vaccine preservative thimerosal or mercury. It also did not matter if a vaccine product was a combination shot, such as the MMR shot.
                  </p>
                  <p>
                   As an acknowledgement of the fears that drive parents to seek non-medical exemptions for vaccines, one of the study authors shared an anecdote about raising two children who both endured febrile seizures following routine inoculations, and how these complications did not dissuade the decision to immunize.
                  </p>
                  <p>
                   In a statement, leaders from Autism Speaks responded by voicing their acceptance of the meta-analysis.
                  </p>
                  <p>
                   "This analysis provides further confirmation for a lack of association between vaccines and autism that the broader healthcare community has understood and embraced for some time," said Rob Ring, Autism Speaks' chief science officer. "Autism Speaks' own policy on vaccines echoes those of other credible healthcare organizations like the American Academy of Pediatrics and the World Health Organization. We strongly encourage parents to work with their physician to ensure their children receive the full benefits immunization offers in protecting their loved ones against a variety of preventable childhood diseases."
                  </p>
                  <p>
                   Michael Rosanoff, Autism Speaks associate director for public health research, added that the meta-analysis approach in this study adds to the credibility of the results.
                  </p>
                  <p>
                   "It assesses the quality of data across multiple studies and combines the highest-quality data to give us a 'higher definition picture' of the relationship between potential risk factors and autism," Rosanoff said.
                  </p>
                  <p>
                   <strong>
                    Tracking local data may help public health workers
                   </strong>
                   <br/>
                   In August 2013, the U.S. Centers for Disease Control and Prevention released a Morbidity and Mortality Weekly Report dedicated to the latest measurements of vaccination rates among schoolchildren in the country during the 2012-2013 school year. Results showed that, among the contiguous48 states and District of Columbia, median immunization rates with MMR, DTaP and varicella shots were 94.5 percent, 95.1 percent and 93.8 percent, respectively. The median for exemptions was 1.8 percent.
                  </p>
                  <p>
                   While the nationwide data may look promising, the MMWR authors pointed out that local health departments may find it more useful to look at data from the surrounding school levels. This will point them in the direction of communities that may be in need of educational campaigns regarding the benefits of vaccines, the risks of not inoculating children and the lack of evidence supporting the claim that immunization causes autism and autism spectrum disorders.
                  </p>
                  <hr/>
                  <p>
                   <strong>
                    It is not uncommon for people to have misgivings about vaccines. Public health
                   </strong>
                   <strong>
                    departments that supplement their
                    <a href="http://www.transactrx.com/health-department-vaccine-billing" target="_blank">
                     medical claim processing software
                    </a>
                    with
                   </strong>
                   <strong>
                    the services of TransactRx can focus more of their attention on correcting the
                   </strong>
                   <strong>
                    vaccination myths that circulate in their community and prevent people from
                   </strong>
                   <strong>
                    being immunized against dangerous infections.
                   </strong>
                  </p>
                 </bdo>
