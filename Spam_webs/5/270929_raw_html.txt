function setMarkers(map, venues, bounds){
  for(var i = 0; i &lt; venues.length; i++){
    var venue = venues[i];
    var myLatLng = new google.maps.LatLng(venue[0], venue[1]);
    var marker = new google.maps.Marker({
        position: myLatLng,
        map: map,
        icon: '//d3m43rlra0pteh.cloudfront.net/assets/map-pin-589916bbfb080f4b3df4314bd46b9a65.png',
        title: venue[2],
        info: venue[3],
        id: i
    });

    var contentString = '&lt;a href="'+ venues[i][5] +'"&gt;' +
                        '&lt;div class="gmaps-card-link extend-card-link"&gt;' +
                          '&lt;div class="gmaps-card-link-image extend-card-link-image"&gt;' +
                            '&lt;img src="'+ venues[i][7] +'" alt="kilo" /&gt;' +
                          '&lt;/div&gt;' +
                          '&lt;div class="extend-card-link-details"&gt;' +
                            '&lt;div class="extend-card-link-title"&gt;' +
                              venues[i][2] +
                            '&lt;/div&gt;' +
                            '&lt;div class="extend-card-link-subtitle"&gt;' +
                              venues[i][6]+ ' Recommendations' +
                            '&lt;/div&gt;' +
                          '&lt;/div&gt;' +
                          '&lt;div class="clearfix"&gt;&lt;/div&gt;' +
                        '&lt;/div&gt;' +
                      '&lt;/a&gt;';

    var infowindow = new google.maps.InfoWindow();

    google.maps.event.addListener(marker, 'click', function() {
      infowindow.setContent(infoWindowContents[this.id])
      infowindow.open(map,markers[this.id]);
    });

    infoWindowContents.push(contentString);
    markers.push(marker);
    bounds.extend(myLatLng);
    }

  google.maps.event.addDomListener(window, 'load', initialize);
  map.fitBounds(bounds);
}
