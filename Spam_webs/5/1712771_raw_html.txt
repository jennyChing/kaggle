(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

            <h3 class="post-title entry-title">
             <a href="http://www.madiganmade.com/2013/03/spring-wreath-diy-lemons.html">
              Spring Wreath DIY: Lemons!
             </a>
            </h3>
            <div class="post-header-line-1">
            </div>
            <div class="postmeta-primary">
             <span class="meta_date">
              5:30 AM
             </span>
             <span class="meta_author">
              Shannon Madigan
             </span>
             <span class="meta_comments">
             </span>
            </div>
            <div class="post-body entry-content">
             <p>
             </p>
             <h3 align="center">
              Madigan made... A simple, faux lemon wreath.
             </h3>
             <p>
              <img alt="Spring Wreath DIY from dollar store lemons" height="453" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/yellow-wreath-lemon-diy-0_zpsc5c11126.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" title="Spring Wreath DIY from dollar store lemons - via madiganmade.com" width="600"/>
             </p>
             <p>
              <br/>
              <strong>
               <em>
                Lemons.
               </em>
              </strong>
              What is it about them that I love so much? The fresh and vibrant smell that hits you when you cut into one? Is it their tart tang yet simple sweetness in desserts? Or that they look so dang pretty in a big bowl?
             </p>
             <p>
              I'll take,
              <em>
               D
              </em>
              , all of the above, I think!
             </p>
             <p>
              When I saw these big lemons in the craft aisle at Dollar Tree... I did a double take. The fruit was pretty life-like and just plain pretty. At a dollar a piece, I couldn't pass them up. I knew I wanted to make a lemon wreath from them. The wreath will be a great spring accent to hang on the front door after Easter is over and I take down my
              <a href="http://www.madiganmade.com/2011/04/simple-large-white-easter-egg-wreath.html" target="_blank">
               big egg wreath
              </a>
              . Easter is so early this year, isn't it? It will be here before we know it!
             </p>
             <p>
              <br/>
              <img height="717" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/spring-wreath-diy-lemons-0_zpsc593a6e1.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" width="600"/>
             </p>
             <h3 align="center">
             </h3>
             <h3 align="center">
              Easy Spring Lemon Wreath
             </h3>
             <p>
             </p>
             <h4>
              Supplies used:
             </h4>
             <ul>
              <li>
               Wire, 18 gauge
              </li>
              <li>
               Faux lemons, about 10-12
              </li>
              <li>
               Wire cutters
              </li>
              <li>
               Pliers
              </li>
              <li>
               Sheer yellow ribbon (or, ahem, yellow dry wall tape)
              </li>
              <li>
               Green ribbon
              </li>
              <li>
               Hot glue gun
              </li>
             </ul>
             <h4>
              Steps taken:
             </h4>
             <p>
              First, I cut a length of wire that was at least 6 inches longer than the desired circumference of my wreath.
             </p>
             <p>
              <img alt="how to make a wreath from faux lemons" height="453" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/spring-wreath-diy-with-lemons-0_zps1a5c3a22.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" title="how to make a wreath from faux lemons" width="600"/>
              <br/>
              Then, I removed the little green plug (nub? You know, that thing that represents the "stem") from each lemon. I poked the wire through that hole and out the other end of the lemon. (the 18 gauge wire should be strong enough to stick through the plastic)
             </p>
             <p>
              <img alt="wreath diy with lemons" height="795" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/summer-wreath-diy-lemons-0_zps96e7662f.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" title="wreath diy with lemons" width="600"/>
             </p>
             <p>
              <br/>
              I continued that process and strung all of the lemons on the wire. Then, I wrapped the wire on itself to hold it shut using pliers.
             </p>
             <p>
             </p>
             <p>
              <img height="453" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/easy-wreath-diy-lemons-0_zpsa06e837a.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" width="600"/>
              <br/>
             </p>
             <p>
              I considered leaving the wreath be at this point, but I felt it needed a little something "else". So, I experimented and ended up really liking the effect yellow dry wall tape gave to the wreath.
             </p>
             <p>
              If you don't have yellow dry wall tape laying around, (what!? You've never made
              <a href="http://www.madiganmade.com/2012/07/diy-neon-cards-with-dry-wall-tape.html" target="_blank">
               handmade cards with neon tape,
              </a>
              I take it? ) you could just use a sheer ribbon in the color of your choice. I wrapped the ribbon/tape around the wreath and secured it down in the back with a little hot glue.
             </p>
             <p>
              All the wreath needed was a green ribbon to hang it... and that was it.
             </p>
             <p>
             </p>
             <p>
              <img alt="easy spring wreath with lemons" height="601" src="http://i981.photobucket.com/albums/ae292/madiganmadeblog/easy-spring-wreath-with-lemons-0_zps39dc7e26.jpg" style="display: block; float: none; margin-left: auto; margin-right: auto" title="easy spring wreath with lemons" width="600"/>
              <br/>
              This spring wreath is simple, sweet, bright and happy, just like lemons on it. I love my Easter dcor but I cant wait to hang this on the door after that holiday is over!
             </p>
