ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.bcpUpdateLink = function(videoId, playlistId) {
  var playlistKey = "bclid";
  var videoKey = "bctid";
  var modSocial = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.modSocial;
  var modVP  = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.modVP;
  var tabBar = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.tabBar;
  // If we are on the /multimedia page, then we use the playlist id, etc. else
  // we just shorten the url that we are on.
  var info = shortUrlAPI = null;
  var newLink = document.URL;

  if (!window.location.origin) {
    window.location.origin = window.location.protocol+"//"+window.location.host;
  }
  var baseURL = window.location.origin + "/multimedia";
  newLink = baseURL + "?" + playlistKey + "=" + playlistId + "&amp;" + videoKey + "=" + videoId;

  // Shorten the URL if we can.
  shortUrlAPI = Drupal.settings.basePath + 'brightcove-player/short-url?url=' + encodeURIComponent(newLink);
  // Fetch the short-url in synchronous manner.
  $.ajax({
    url: shortUrlAPI,
    async: false,
    dataType: "json",
    success: function(result) {
      info = result;
    }
  });
  if (info != null) {
    modSocial.setLink(info.short_url);
  }
  else {
    modSocial.setLink(newLink);
  }
}

    var player = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.player;

    var modAds = player.getModule(APIModules.ADVERTISING);

    modVP.addEventListener(brightcove.api.events.MediaEvent.PLAY, ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.onPlay);
    modVP.addEventListener(brightcove.api.events.MediaEvent.STOP, ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.onStop);

    if (evt.target.experience) {
      return;

    var modExp = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.modExp;
    var tabBar = modExp.getElementByID("playlistTabs");
    ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.tabBar = tabBar;
    // If we have a channel passed in, then set focus to that tab.
    if (typeof multimedia_channel !== 'undefined') {
      var tabs = tabBar.getData();
      for (var i=0, len=tabs.length; i&lt;len; i++) {
        if (multimedia_channel == tabs[i].displayName.toLowerCase().replace(/[^a-z0-9]/g, '')) {
          // Set the tab, and then play the first video in that tab.
          tabBar.setSelectedIndex(i);
          modVP.loadVideo(tabs[i].videoIds[0]);

    }

    ECOmnitureBCLogger(err);
  }
};

ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.onPlay = function(evt) {
  var mediaLength = evt.duration; //Required video duration
  ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.mediaLength = mediaLength; // Needed in another function.
  var mediaOffset = Math.floor(evt.position); //Required video position
  var mediaID = (evt.media.id).toString(); //Required video id
  var mediaFriendly = evt.media.displayName; //Required video title
  var mediaName = mediaID + ":" + mediaFriendly; //Required Format video name
  var mediaRefID = evt.media.referenceId; //Optional reference id
  var mediaPlayerType = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.player.type; //Optional player type
  var mediaTagsArray = evt.media.tags; //Optional tags
  var mediaPlaylist = ""; //Optional Playlist
  var mediaTagsArray2 = new Array();
  var mediaPlayerName = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.mediaPlayerName;
  // HTML5 Players have their tags flattend to a 1-dimensional array already.
  if (mediaPlayerType == 'html') {
    mediaTagsArray2 = mediaTagsArray;
  }
  else {
    for (var i = 0; i &lt; mediaTagsArray.length; i++) {
      mediaTagsArray2.push(mediaTagsArray[i]['name']);
    }
  }
  if (mediaPlayerType == "flash") { //Optional playlist id
    if (evt.media.lineupId != null) {
      mediaPlaylist = (evt.media.lineupId).toString();
    }

  } else {
    if (evt.media.playlistID !== null) {
      mediaPlaylist = (evt.media.playlistID).toString();
    }
  }
  /* Check for start of video */
  if (mediaOffset == 0) {
    /* These data points are optional. If using SC14, change context data variables to hard
    coded variable names and change trackVars above. */
    s.contextData['bc_tags'] = mediaTagsArray2.toString(); //Optional returns list of tags for current video. Flash only.
    s.contextData['bc_refid'] = mediaRefID; //Optional retursn reference id
    s.contextData['bc_player'] = mediaPlayerName; //Optional player name is currently hard coded. Will be dynamic in later releases.
    s.contextData['bc_playertype'] = mediaPlayerType; //Optional returns flash or html
    s.contextData['bc_playlist'] = mediaPlaylist; //Optional returns playlist number for current video.
    s.Media.open(mediaName, mediaLength, mediaPlayerName);
    s.Media.play(mediaName, mediaOffset);
  } else {
    s.Media.play(mediaName, mediaOffset);
  }
}

ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.onStop = function (evt) {

    var mediaLength = ECOmnitureBrightCoveHandlerseb7a4e32601624cb6cf642a2e82514f6.mediaLength;

      s.Media.stop(mediaName, mediaOffset);

    } else {
      s.Media.stop(mediaName, mediaOffset);
    }
  }
  catch(err) {
    ECOmnitureBCLogger(err);
  }
}

      <p>
       <strong>
        Which countries match the GDP and population of Brazil's states?
       </strong>
      </p>
      <p>
       <iframe class="iframed-infographic" frameborder="0" height="760" scrolling="no" src="http://infographics.economist.com/2014/brazil_20140610/?fsrc=rss" width="595">
        Audio and Video content on Economist.com requires a browser that can handle iFrames.
       </iframe>
      </p>
      <p>
       AS FOOTBALLERS and fans descend on Brazil for the World Cup kick-off on June 12th in So Paulo, they will find themselves in not one country but many. As our interactive map shows, Brazilian states' economies, population and GDP per person vary wildly. The biggest state, So Paulo, has the population of Argentina and an economy the size of Turkey's. At the other end of the scale, Roraima has barely more people than Malta and the economic output of Mauritania.
       <br/>
       <br/>
       When Brazil last hosted the World Cup in 1950 half of its states were more destitute than present-day India. The poorest, Piau and Maranho, are now more like Jordan. But that is still a long way from the wealthiest state, around the capital, Braslia, which enjoys an Italian standard of living. For more on Brazilian states' GDP per person,
       <a href="http://www.economist.com/news/americas/21604202-brazils-income-disparities-are-great-so-its-progress-welcome-italordan">
        see article
       </a>
       .
      </p>

        <div class="comment-body">
         <p>
          Excellent info. It should also help take away easy misconceptions and stereotypes about this complex and often confusing country that is Brazil. It is quite clear that Brazil already has some regions that have higher-middle income just like the better parts of Eastern Europe (Poland, Slovakia, Estonia etc.), and that even the poorest states, like Maranho, are not absolutely miserable anymore as once they were.
         </p>
         <p>
          However, one interesting and immediately noticeable thing a Brazilian can figure out from the charts is how extreme income inequality as well as bad public services must really play a huge part in Brazil's woes. After all, if Cear's GDP per person is really similar to Serbia, and Mato Grosso's is really similar to Hungary's, their standard of living should be at least quite similar to those countries', and as far as I know they are not. Cear, where I live, has been one of the states that have improved the most in the last decades in the country, but it still has lots of shameful social issues, such as an iliteracy rate around 15% (especially due to high illiteracy for adults and elders). Cear's HDI (0.682), as measured by Brazilian stats institutes, is comparable to Suriname (105th), Thailand (103th), not Serbia (64th). As for Mato Grosso, definitely the HDI and other social data there don't equal Hungary's, though they are better than in the Northeastern states like Cear.
         </p>
