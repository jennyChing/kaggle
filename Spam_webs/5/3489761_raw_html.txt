	function move_current(direction){
		var $old = $(".current");

					$(".right.light").show();
					$(".right.active").hide();

				if($(".current").prev().hasClass("carage360"))

					$('#r_image').show();
				}
				$(".current").prev(".carage").addClass("current")
				$old.removeClass("current")

				if(!$(".current").prev().hasClass("carage")){
					$(".left.light").show();
					$(".left.active").hide();
				}
			}
		}

	}; // End Move_current
	
	function get_freight(){
		$("#getting_freight").css("display","block");
		var qty = $(".cart_qty").val();
		if($("#freight_postcode").val().length == 4 &amp;&amp; is_int($("#freight_postcode").val())){

			freight_timer = setTimeout(function(){
				$.ajax({
					url: "/ajax/product",
					type: "POST",
					data: {
						action: "get_freight",
						prdid: 55450,
						qty: qty,
						postcode: $("#freight_postcode").val()
						},
					dataType: "json",
					success: function(json){
						if (json.length &lt;= 1){
							$("#multiState").hide();
						}else{
							var count = 0;
							$("#multiState").show();
							$("#multiState").html("");
							$.each(json, function(key, value) {
							     $('#multiState')
							         .append($("&lt;option&gt;&lt;/option&gt;")
							         .attr("value",json[count].exp_freight_cost+"|"+json[count].std_freight_cost)
							         .text(json[count].state)); 
						         count++;
							});
						}
						$("#getting_freight").css("display","none");
						$("#exp_freight_price").html(json[0].exp_freight_cost); //set value from option
						if (json[0].std_freight_cost != false)
							$("#std_freight_price").html(json[0].std_freight_cost); //set value from option
						else
							$("#std_freight_price").html("N/A");
					
					}
				})
			}, 300)
		}
	}; // End Get Freight
