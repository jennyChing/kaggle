      function numberWithCommas(x) {
    return x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}

function setCounter(value, goal) {
    var percentile = parseInt(value/goal*100)
    var widthPercent = value &gt; goal ? 100 : percentile

    $("#goal_caption").html(numberWithCommas(value) + ' actions taken! '+ percentile + '% of our goal of ' + numberWithCommas(goal) + '!')
    $("#goal_percent").html(percentile + '%')
    $("#progress_bar").width(widthPercent+'%')

}

function combineThermometers(pages, onComplete, count) {
    var page;
    count = count || 0;

    if (pages.length) {
        page = pages.shift();
        jQuery.ajax({
            url: 'https://act.democracyforamerica.com/progress/?page=' + page,
            dataType: 'jsonp',
            success: function(progress) { combineThermometers(pages, onComplete, count + progress.total.actions) },
            error: function(msg) { combineThermometers(pages, onComplete, count ) },
      });
    } else {
        onComplete(count, None );
    }
}

function actionkitFormReady() {
   if ( typeof counterList != 'undefined' &amp;&amp; counterList instanceof Array ) {
      combineThermometers(counterList, setCounter);
   }
}
