                function getHiddenProp() {
                    var prefixes = ['webkit', 'moz', 'ms', 'o'];

                    if ('hidden' in document) return 'hidden';

                    for (var i = 0; i &lt; prefixes.length; i++) {
                        if ((prefixes[i] + 'Hidden') in document)

                    var prop = getHiddenProp();

                function notify_user(num) {
                    if (num &gt; 0 &amp;&amp; num != undefined) {
                        // Let's check if the browser supports notifications

                            $zopim.livechat.window.show();
                            return;

                        var options = { icon: "http://mac.remotedesktopmanager.com/favicon.png" };
                        if (Notification.permission === "granted" &amp;&amp; isHidden()) {
                            notification = new Notification("You have " + num + " new chat message", options);
                        }
                        else if (Notification.permission !== 'denied') {
                            Notification.requestPermission(function (permission) {
                                if (!('permission' in Notification)) {
                                    Notification.permission = permission;
                                }

                                if (permission === "granted" &amp;&amp; isHidden()) {
                                    notification = new Notification("You have " + num + " new chat message", options);
                                }
                            });
                        }

                        $zopim.livechat.window.show();
                    }
                }

                function resizeFancyBox() {
                    var iFrameContentHeight = $("#fancybox-frame").contents().find("body").height();
                    var iFrameContentWidth = $("#fancybox-frame").contents().find("body").width();

                    var outer = $('#fancybox-wrap');
                    var inner = $('#fancybox-inner');
                    var paddingTotal = 0;
                    var extra = 20; // some extra space to avoid Scrollbars

                    if (iFrameContentHeight &gt; 0 &amp;&amp; iFrameContentWidth &gt; 0) {
                        outer.css({

                            width: iFrameContentWidth + paddingTotal + extra
                        });

                            height: iFrameContentHeight + extra,
                            width: iFrameContentWidth + extra
                        });

                        $.fancybox.center();
                    }
