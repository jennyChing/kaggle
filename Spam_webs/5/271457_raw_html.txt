	// Sliders
	var sliders = new Array(); // Each slider has settings and values stored in this array
	var thumbs_per_reel = 5; // Hardcoded
	$.each($(".mediaslider"), function(index, value){
		if (!$(this).hasClass('similar_media_slider')) {
			var id = $(this).attr('id'); // Automatic unique idenifier for the slider

			// Set up all settings and options for slider
			sliders[id] = new Array();
			sliders[id]['media_count'] = $("#"+id+" .slider_thumbnail").length; // All medias in reel
			sliders[id]['thumb_width'] = $("#"+id+" .slider_thumbnail:first").outerWidth(true); // Width of a single thumb
			sliders[id]['single_reel_length'] = sliders[id]['thumb_width']*thumbs_per_reel; // Width of single reel with width of
			sliders[id]['total_reel_length'] = sliders[id]['thumb_width']*sliders[id]['media_count']; // Width of all reels combined
			sliders[id]['current_page'] = 1;
			sliders[id]['max_page'] = Math.ceil(sliders[id]['media_count']/thumbs_per_reel);
			sliders[id]['animation_is_running'] = false;

			$(this).find(".mediaslider_inner").css('width', sliders[id]['total_reel_length']+'px');
		}
	});

	$('.mediaslider').not('.similar_media_slider').off('click');
	$(".mediaslider").not('.similar_media_slider').on('click', '.mediaslider_right', function(e){
		var id = $(e.currentTarget).parents('.mediaslider').attr('id');
		if(sliders[id]["animation_is_running"] == false &amp;&amp; sliders[id]["current_page"] != sliders[id]["max_page"]){
			sliders[id]["animation_is_running"] = true;
			sliders[id]["current_page"] = sliders[id]["current_page"]+1;
			$("#"+id+" .mediaslider_current").text(sliders[id]["current_page"]);
			handleNavigation(id);
			$(e.currentTarget).parents('.mediaslider').find('.mediaslider_inner').animate({
				left: '-='+sliders[id]["single_reel_length"]
		    }, 400, function(){
		    	sliders[id]["animation_is_running"] = false;
		    });
		}
	});

	$(".mediaslider").not('.similar_media_slider').on('click', '.mediaslider_left', function(e){
		var id = $(e.currentTarget).parents('.mediaslider').attr('id');
		if(sliders[id]["animation_is_running"] == false &amp;&amp; sliders[id]["current_page"] != 1){
			sliders[id]["animation_is_running"] = true;
			sliders[id]["current_page"] = sliders[id]["current_page"]-1;
			handleNavigation(id);
			$("#"+id+" .mediaslider_current").text(sliders[id]["current_page"]);
			$(e.currentTarget).parents('.mediaslider').find('.mediaslider_inner').animate({
				left: '+='+sliders[id]["single_reel_length"]
		    }, 400, function(){
		    	sliders[id]["animation_is_running"] = false;
		    });
		}
	});
	// Adjust the navigation for the slider, disabling/enabling buttons visually
	function handleNavigation(id){

		if(sliders[id]["max_page"] != 1){
			if(sliders[id]["current_page"] == 1){
				$("#"+id+" .mediaslider_left").addClass('disabled');
			}else{
				$("#"+id+" .mediaslider_left").removeClass('disabled');
			}
			if(sliders[id]["current_page"] == sliders[id]["max_page"]){
				$("#"+id+" .mediaslider_right").addClass('disabled');
			}else{
				$("#"+id+" .mediaslider_right").removeClass('disabled');
			}
		}else{
			$("#"+id+" .mediaslider_left").addClass('disabled');
			$("#"+id+" .mediaslider_right").addClass('disabled');
		}

	}

	// Sliders
	var sliders = new Array(); // Each slider has settings and values stored in this array
	var thumbs_per_reel = 5; // Hardcoded
	$.each($(".mediaslider"), function(index, value){
		if (!$(this).hasClass('similar_media_slider')) {
			var id = $(this).attr('id'); // Automatic unique idenifier for the slider

			// Set up all settings and options for slider
			sliders[id] = new Array();
			sliders[id]['media_count'] = $("#"+id+" .slider_thumbnail").length; // All medias in reel
			sliders[id]['thumb_width'] = $("#"+id+" .slider_thumbnail:first").outerWidth(true); // Width of a single thumb
			sliders[id]['single_reel_length'] = sliders[id]['thumb_width']*thumbs_per_reel; // Width of single reel with width of
			sliders[id]['total_reel_length'] = sliders[id]['thumb_width']*sliders[id]['media_count']; // Width of all reels combined
			sliders[id]['current_page'] = 1;
			sliders[id]['max_page'] = Math.ceil(sliders[id]['media_count']/thumbs_per_reel);
			sliders[id]['animation_is_running'] = false;

			$(this).find(".mediaslider_inner").css('width', sliders[id]['total_reel_length']+'px');
		}
	});

	$('.mediaslider').not('.similar_media_slider').off('click');
	$(".mediaslider").not('.similar_media_slider').on('click', '.mediaslider_right', function(e){
		var id = $(e.currentTarget).parents('.mediaslider').attr('id');
		if(sliders[id]["animation_is_running"] == false &amp;&amp; sliders[id]["current_page"] != sliders[id]["max_page"]){
			sliders[id]["animation_is_running"] = true;
			sliders[id]["current_page"] = sliders[id]["current_page"]+1;
			$("#"+id+" .mediaslider_current").text(sliders[id]["current_page"]);
			handleNavigation(id);
			$(e.currentTarget).parents('.mediaslider').find('.mediaslider_inner').animate({
				left: '-='+sliders[id]["single_reel_length"]
		    }, 400, function(){
		    	sliders[id]["animation_is_running"] = false;
		    });
		}
	});

	$(".mediaslider").not('.similar_media_slider').on('click', '.mediaslider_left', function(e){
		var id = $(e.currentTarget).parents('.mediaslider').attr('id');
		if(sliders[id]["animation_is_running"] == false &amp;&amp; sliders[id]["current_page"] != 1){
			sliders[id]["animation_is_running"] = true;
			sliders[id]["current_page"] = sliders[id]["current_page"]-1;
			handleNavigation(id);
			$("#"+id+" .mediaslider_current").text(sliders[id]["current_page"]);

				left: '+='+sliders[id]["single_reel_length"]
		    }, 400, function(){
		    	sliders[id]["animation_is_running"] = false;
		    });
		}
	});
	// Adjust the navigation for the slider, disabling/enabling buttons visually
	function handleNavigation(id){

		if(sliders[id]["max_page"] != 1){
			if(sliders[id]["current_page"] == 1){
				$("#"+id+" .mediaslider_left").addClass('disabled');
			}else{
				$("#"+id+" .mediaslider_left").removeClass('disabled');
			}
			if(sliders[id]["current_page"] == sliders[id]["max_page"]){
				$("#"+id+" .mediaslider_right").addClass('disabled');
			}else{
				$("#"+id+" .mediaslider_right").removeClass('disabled');
			}
		}else{
			$("#"+id+" .mediaslider_left").addClass('disabled');
			$("#"+id+" .mediaslider_right").addClass('disabled');
		}

	}
