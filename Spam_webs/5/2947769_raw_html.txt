function performBlockUnblock(objCon){		
	$.ajax({
		url: "/frontajax/" + Math.random(),
		type: "POST",
		data: {'requesttype':'ajax', 'action':'followUnfollow', 'subTask':objCon.attr('subTask'), 'id':objCon.find('.follow_btn').attr('id')},
		dataType: "json",
		beforeSend: function(){
			objCon.attr('ajaxRunning','Y');
			objCon.find('.follow_btn img').hide();
			objCon.find('.ajaxLoader').show();
		},
		success: function(data){
			objCon = $('#follow_con_'+data.id);
			
			// check user loggedin
			if(data.is_loggedin==1){
				alert(data.msg);
				showButton(objCon);
				return false;
			} 
						
			if(data.status == 200){
				//show_message(data.msg, 'info');
				objCon.attr('blocked', data.blocked);
				objCon.attr('isFollowing', data.isFollowing);
				if(data.isFollowing == 'N' &amp;&amp; '' == 'followings')
					objCon.remove()
				//Show count - its now only showing on public profile page
				$('.public_proTop .num_rig .number').html(data.cntFollower);
				
				//Showig the button
				showButton(objCon);
			} else {
				show_message(data.msg, 'error');
				showButton(objCon);
			}
			
			objCon.attr('ajaxRunning','N');
		},
		error: function(data) {
			show_message('Error!!! Please try again later', 'error');
			objCon.attr('ajaxRunning','N');
			showButton(objCon);
		}
	});
}

function showButton(objCon){
	objCon.find('.ajaxLoader').hide();
	if(objCon.attr('blocked') == 'Y'){
		objCon.find('.follow_btn img.blockedUser').show();
		objCon.attr('subTask', 'unblock');
	} else if(objCon.attr('isFollowing') == 'Y'){
		objCon.find('.follow_btn img.followingUser').show();
		objCon.attr('subTask', 'unfollow');
	} else {
		objCon.find('.follow_btn img.followUser').show();
		objCon.attr('subTask', 'follow');
	}
}

			conditional : function(value) {if($('#chk_confirm').is(':checked')) 	return true; else return false;}

		usr_password : {
			required : true,
			conditional : function(value) {
				if(value=='') {$('#user_password + p').html('Password is required!');return false;}
				if(value.length&gt;=8 &amp;&amp; value.length&lt;=15 ){return true;}
				if(value.length&lt;8){$('#user_password + p').html('Password must be atleast 8 characters');return false;}
				if(value.length&gt;15){$('#user_password + p').html('Password must be less than 15 characters');return false;}
			},
			pattern : /^[a-zA-Z0-9\!\@\#\$\%\^\&amp;\*()\-\_\+\=\?]{8,15}$/
		},
		usr_email_id:{
			required : true,
			conditional : function(value) {
				if(value=='') {$('#user_email_id + p').html('Email Address is required!');return false;}
				if(checkUniqueEmail(value)==true){$('#user_email_id + p').html('This email address is not valid.');return true;}
				else{$('#user_email_id + p').html('This email is already in use!');return false;}
				return true;	
			},
			pattern : /^\b[A-Z0-9._%-]+@[A-Z0-9.-]+\.[A-Z]{2,4}\b$/
		},
		usr_userid:{
			required : true,
			conditional : function(value) {

				if(checkUniqueUser(value)==false){$('#user_userid + p').html(' This username is already in use!');return false;}
				else{$('#user_userid + p').html('Please enter a proper username!');return true;}
			},
			pattern : /^[a-zA-Z0-9\_]{1,}$/
		}
