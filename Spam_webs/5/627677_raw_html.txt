     <h1>
      Wal-Mart U.S. asking meat suppliers to reduce antibiotic use
     </h1>
     <div>
      <div>
       <span class="partner">
        <img alt="Reuters" src="//img.s-msn.com/tenant/amp/entityid/AA1Q4Ad.img?h=17&amp;w=0&amp;m=6&amp;q=60&amp;u=t&amp;o=t&amp;l=f&amp;f=png"/>

     <div class="richtext" data-aop="articlebody">
      <span class="storyimage fullwidth inlineimage" data-aop="image">
       <span class="image" data-attrib="Robyn Beck/AFP/Getty Images" data-caption="Lean ground beef for sale on the opening day of the new Walmart Neighborhood Market in Panorama City, Calif. on 28, 2012.">
        <img alt="Lean ground beef for sale on the opening day of the new Walmart Neighborhood Market in Panorama City, Calif. on 28, 2012." height="176" src="//img.s-msn.com/tenant/amp/entityid/BBcForN.img?h=176&amp;w=300&amp;m=6&amp;q=60&amp;o=f&amp;l=f" width="300"/>

      <p xmlns="http://www.w3.org/1999/xhtml">
       Wal-Mart Stores Inc said its U.S. units were asking meat suppliers to reduce the use of antibiotics in farm animals and provide annual reports on antibiotics management and animal welfare.
      </p>
      <p xmlns="http://www.w3.org/1999/xhtml">
       Walmart U.S. and Sam's Club U.S. are asking suppliers to limit the use of antibiotics to sick animals and not use antibiotics for boosting animals' growth, the company said on Thursday.
      </p>
      <p xmlns="http://www.w3.org/1999/xhtml">
       "We have listened to our customers, and are asking our suppliers to engage in improved reporting standards and transparency measures regarding the treatment of farm animals," Kathleen McLaughlin, senior vice president of Walmart sustainability, said in a statement.      (Reporting by Sruthi Ramakrishnan in Bengaluru; Editing by Kirti Pandey)
      </p>

                            {
                                var dimensionProperties = isWidth ? ["width", "offsetWidth", "scrollWidth"] : ["height", "offsetHeight", "scrollHeight"];
                                var pixelStyle = isWidth ? "pixelWidth" : "pixelHeight";
                                var dimensionStyle = isWidth ? "width" : "height";
                                var totalProperties = 3;
                                var maxDimension = 0, dimension = 0;

                                for (var i = 0; i &lt; totalProperties; i++)
                                {
                                    if ((dimension = element[dimensionProperties[i]]) &gt; maxDimension) 
                                    {
                                        maxDimension = dimension;
                                        if (maxDimension &gt; threshold) 
                                        {
                                            break;
                                        }
                                    }
                                }

                                var elemStyle = element.style;
                                if (maxDimension &lt;= threshold &amp;&amp; elemStyle)
                                {
                                    if ((dimension = elemStyle[pixelStyle]) &gt; maxDimension)
                                    {
                                        maxDimension = dimension;
                                        if (maxDimension &lt;= threshold &amp;&amp; (dimension = parseInt(elemStyle[dimensionStyle])) &gt; maxDimension)
                                        {
                                            maxDimension = dimension;
                                        }
                                    }
                                }

                                return maxDimension;
                            }

                            function setAdContainerDisplayState(elemId, doShow, adSizeClassname)

                                adHtmlContainer.style.display = doShow ? "" : "none";
                                var className = adHtmlContainer.className;
                                
                                className = addOrRemoveClassname(className, postEvaluationClassname, doShow);
                                className = addOrRemoveClassname(className, postEvaluationAdSmallClassname, adSizeClassname === postEvaluationAdSmallClassname);
                                className = addOrRemoveClassname(className, postEvaluationAdMediumClassname, adSizeClassname === postEvaluationAdMediumClassname);
                                className = addOrRemoveClassname(className, postEvaluationAdLargeClassname, adSizeClassname === postEvaluationAdLargeClassname);
                                className = addOrRemoveClassname(className, postEvaluationNoAdClassname, adSizeClassname === postEvaluationNoAdClassname);

                                adHtmlContainer.className = className;
                            }

                            function addOrRemoveClassname(classNameList, className, add)
                            {
                                var classIndex = classNameList.indexOf(className);
                                if (add)
                                {
                                    if (classIndex === -1)
                                    {
                                        return classNameList + " " + className;
                                    }
                                }
                                else if (classIndex &gt;= 0)
                                {
                                    return classNameList.replace(className, "");
                                }
                                return classNameList;
