           $(function() {
    if (page.ShowView.Videos !== undefined) {
      return;
    }

    page.ShowView.Videos = {};

    var Sort = {
      ASC: 'oldest',
      DESC: 'newest'
    };
    // Model for keeping track of video tab
    var Tab = Backbone.Model.extend({
      initialize: function() {
        this.set('sort', Sort.DESC);
      },

      reverseSort: function() {
        if (this.get('sort') === Sort.DESC) {
          this.set('sort', Sort.ASC);
        }
        else {
          this.set('sort', Sort.DESC);
        }
      }
    });

    // View for each individual season
    var SeasonSectionView = Backbone.View.extend({
      initialize: function() {
        this.model.on("change:sort", this.reverse, this);
      },

      reverse: function() {
        var ul = this.$el.find('ul.portrait-grid');
        this.$el.find('li.group-item').each(function() {
          ul.prepend(this);
        });
      },

      animationDuration: 200,

      expand: function() {
        var handle = this.$el.find('a.season-dropdown');
        handle.addClass('open');
        handle.siblings('ul.portrait-grid').slideDown(this.animationDuration);
        handle.siblings('ul.portrait-grid').find('li.group-item img.landscape').each(function() {
          if ($(this).attr('src') !== undefined) {
            return;
          }
          $(this).attr('src', $(this).attr('data-thumbnailUrl'));
        });
        //ellipsis for newly visible items
        this.$el.find('.short-desc').ellipsis();
      },

      collapse: function() {
        var handle = this.$el.find('a.season-dropdown');
        handle.removeClass('open');
        handle.siblings('ul.portrait-grid').slideUp(this.animationDuration);
      },

      toggleVisibility: function(ev) {
        ev.preventDefault();

        var handle = this.$el.find('a.season-dropdown');
        if (handle.hasClass('open')) {
          this.collapse();
        }
        else {
          this.expand();
        }
      },

      events: {
        "click a.season-dropdown": 'toggleVisibility'
      }
    });

    // View for season list as a whole.
    var SeasonListView = Backbone.View.extend({
      initialize: function() {
        this.model.on("change:sort", this.reverse, this);
      },

      reverse: function() {
        var self = this;
        $(".list-of-seasons").animate({opacity:0}, 150);
        self.$el.find('li.season').each(function() {
          self.$el.prepend(this);
        });
        $(".list-of-seasons").animate({opacity:1}, 150);
      }
    });

    // Subtab view for sorting
    var SortSubtabView = Backbone.View.extend({
      initialize: function(options) {
        this.model.on("change:sort", this.toggleSelected, this);
        if (options.sort !== undefined) {
          this.sort = options.sort;
        }
      },

      toggleSelected: function() {
        this.$el.toggleClass('selected');
      },

      events: {
        "click": function(ev) {
          ev.preventDefault();
          if (this.model.get('sort') !== this.sort) {
            this.model.reverseSort();
          }
        }
      }
    });


    // Create the objects.
    var videosPage = new Tab();
    var seasonList = new SeasonListView({model: videosPage, el: $('ul.list-of-seasons').get(0)});
    var newestSubtab = new SortSubtabView({model: videosPage, el: $('a#subtabs_videos_newest').get(0), sort: Sort.DESC});
    var oldestSubtab = new SortSubtabView({model: videosPage, el: $('a#subtabs_videos_oldest').get(0), sort: Sort.ASC});

    var seasonViews = [];
    $('li.season').each(function() {
      seasonViews.push(new SeasonSectionView({model: videosPage, el: this}));
    });
    videosPage.on('change:sort', function() {
      if (videosPage.get('sort') === Sort.ASC) {
        _.last(seasonViews).expand();
      }
      else {
        _.first(seasonViews).expand();
      }
    });

    // Export variables
    page.ShowView.Videos.videosPage = videosPage;
  });
