    var populateTooltip = function(dataObject, productPrice, bounds, productRate) {
        var diff = dataObject.maximum - dataObject.minimum;

        footerElement.innerHTML = (dataObject.message !== undefined ? dataObject.message : dataObject.globalMessage).replace(/%moreWith%/g, diff);
        
        if (options.type == 'textGraphic') {
            if (productPrice &lt; bounds.absoluteMinimumValue) {
                percentage = 0;
            } else if (productPrice &gt;= bounds.absoluteMaximumValue){
                percentage = 100;
            } else {
                percentage = (((productPrice-bounds.productMinimumValue)/(bounds.productMaximumValue-bounds.productMinimumValue)*bounds.intervalBound))+(bounds.intervalBound*bounds.currentIndex);
            }

            // fill ecredit bar
            percentage = Math.round(percentage * 100) / 100;
            barFill.style.width = percentage + '%';
            barFillArrow.style.marginLeft = percentage + '%';
        }
    };

    var addCetelemBar = function() {
        /* create cetelem bar based on number of input intervals + find absolute minimum/maximum value  */
        for (var i = 0, len = cetelemCampaignData.length; i &lt; len; i++) {
            if (cetelemCampaignData[i].productMaximumValue &gt; absoluteMaximumValue)
                absoluteMaximumValue = cetelemCampaignData[i].productMaximumValue;
            if (cetelemCampaignData[i].productMinimumValue &lt; absoluteMinimumValue)
                absoluteMinimumValue = cetelemCampaignData[i].productMinimumValue;

            var cetelemBarPart = document.createElement('div'),
                    cetelemBarTextValue = document.createElement('div'),
                    cetelemBarVertical = document.createElement('div'),
                    cetelemBarPartText = document.createElement('div');

            cetelemBarPart.className = 'cetelem-bar-part';

            if (i === len - 1) {
                cetelemBarPart.className += ' cetelem-last-child';
            }
            cetelemBarTextValue.className = 'cetelem-bar-text-value';
            cetelemBarVertical.className = 'cetelem-bar-vertical';
            cetelemBarPartText.className = 'cetelem-bar-part-text';

            cetelemBarTextValue.appendChild(document.createTextNode(cetelemCampaignData[i].productMinimumValue.toString() + ' Lei'));
            cetelemBarPartText.appendChild(document.createTextNode(cetelemCampaignData[i].interest.toString() + '%'));

            cetelemBarPart.appendChild(cetelemBarTextValue);
            cetelemBarPart.appendChild(cetelemBarVertical);
            cetelemBarPart.appendChild(cetelemBarPartText);

            tooltipInnerComponent.appendChild(cetelemBarPart);
        }
    }

    CetelemComponent.prototype.refresh = function(productPrice, selectedCreditPeriod, productRate) {
        var thresholdObjects, thresholdObject;
        productPrice = productPrice || productPriceGlobal;
        productRate = productRate || 0;
        for (var i = 0, len = cetelemCampaignData.length; i &lt; len; i++) {
            thresholdObjects = cetelemCampaignData[i].thresholds,
                    bounds = {
                        productMinimumValue: cetelemCampaignData[i].productMinimumValue,
                        productMaximumValue: cetelemCampaignData[i].productMaximumValue,
                        absoluteMaximumValue: absoluteMaximumValue,
                        absoluteMinimumValue: absoluteMinimumValue,
                        currentIndex: i,
                        intervalBound: intervalBound
                    };

            if ((productPrice &gt;= cetelemCampaignData[i].productMinimumValue) &amp;&amp;
                    (productPrice &lt; cetelemCampaignData[i].productMaximumValue)) {
                for (var j = 0, jlen = thresholdObjects.length; j &lt; jlen - 1; j++ ){
                    thresholdObject = thresholdObjects[j];
                    thresholdObject.globalMessage = cetelemCampaignData[i].message;
                    if ((selectedCreditPeriod == thresholdObject.period) &amp;&amp;
                            (productPrice &gt;= thresholdObject.minimum) &amp;&amp;
                            (productPrice &lt; thresholdObject.maximum)) {
                        populateTooltip(thresholdObject, productPrice,bounds, productRate);
                        return;
                    }
                }
                thresholdObject = thresholdObjects[thresholdObjects.length - 1];
                thresholdObject.globalMessage = cetelemCampaignData[i].message;
                populateTooltip(thresholdObjects[thresholdObjects.length - 1], productPrice, bounds, productRate);
                return;
            }
        }

        if (productPrice &lt; absoluteMinimumValue){
            thresholdObject = thresholdObjects[0];
            thresholdObject.globalMessage = cetelemCampaignData[i].message;
            populateTooltip(thresholdObjects[0], productPrice, bounds, productRate);
            return;
        }

        thresholdObject = thresholdObjects[thresholdObjects.length - 1];
        thresholdObject.globalMessage = cetelemCampaignData[i].message;
        populateTooltip(thresholdObjects[thresholdObjects.length - 1], productPrice, bounds, productRate);
    }

    initElements();
    if (options.type == 'textGraphic') {
        addCetelemBar();
    }
    this.refresh(productPriceGlobal, options.period, productRate);
    window.x = this.refresh;

     function repositioning(where){
        var scrollToElement = false;
        if (typeof where !== 'undefined' &amp;&amp; where !== '') {
            var currentElement = where;
        } else {
            var currentElement = window.location.hash.replace('!','');
            scrollToElement = false;
        }

        if(typeof currentElement !== 'undefined' &amp;&amp; currentElement !== '') {
            switch(currentElement) {
                case '#add_review':
                    $('#scrie_review.on').click();
                    if($(window).height() &gt; 700) currentElement = '#new_reviews';
                    scrollToElement = true;
                    break;
                case '#top':
                    $("html, body").stop().animate({scrollTop: 0}, 'slow');
                    break;
                default:
                    break;
            }

            if ($(currentElement).length) {
                var $butonCurrentElement = $('#but_' + currentElement.replace('#', ''));
                var butonCEoffset = 0;
                if ($butonCurrentElement.length) {
                    butonCEoffset = parseInt($butonCurrentElement.offset().top) - $('#bannerMenuReviews').offset().top + 20;
                }

                                var reposition = parseInt($(currentElement).offset().top) - butonCEoffset - 170;
                                if (scrollToElement) {
                    reposition = parseInt($(currentElement).offset().top) - 50;
                }

                $('html,body').animate({scrollTop: reposition}, 'slow');
            }
        }
    }

			function getZoomAreaPosition() {
				var imgX = $("#zoom-image img").width();
				var imgY = $("#zoom-image img").height();
				var viewX = 525;
				var viewY = 350;
				var zoomX = viewX * 100 / imgX;
				var zoomY = viewY * 100 / imgY;
				var areaX = zoomX / 100 * 350;
				var areaY = zoomY / 100 * 350;
				var offsetX = (nx * 100 / imgX ) / 100 * 350;
				var offsetY = (ny * 100 / imgY ) / 100 * 350;

				var x = qst.pageX - (areaX / 2);
				var y = qst.pageY - (areaY / 2);

				if (x &lt; imgPosition.topX) x = imgPosition.topX;
				if (y &lt; imgPosition.topY) y = imgPosition.topY;

				if (x + areaX &gt; imgPosition.bottomX) x = x - (x + areaX - imgPosition.bottomX);
				if (y + areaY &gt; imgPosition.bottomY) y = y - (y + areaY - imgPosition.bottomY);

				var top = y;
				var left = x;

				return {'areaX' : parseInt(areaX), 'areaY' : parseInt(areaY), 'top' : parseInt(top), 'left' : parseInt(left) }
