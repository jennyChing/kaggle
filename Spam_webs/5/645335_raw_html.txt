         <div id="recipe_directions">
          <h2>
           Directions
          </h2>
          <div itemprop="recipeInstructions">
           <p>
            Heat
            <a href="/howto/it-pays-to-know-your-oven-344" title="It pays to know your oven">
             oven
            </a>
            to 375 (190). Spread toast lightly with mustard; place on ungreased cookie sheet.
           </p>
           <p>
            Mix sauerkraut, corned beef, cheese and mayonnaise.
Spread about  cup sauerkraut mixture on each toast slice.
           </p>
           <p>
            Bake about 10 minutes or until sauerkraut mixture is hot and cheese is melted.
           </p>
           <p>
            Cut sandwiches in halves.
           </p>

function wiggle(wigSpeed, wigHt) {
  var $elem = $('#u-r-set'),wigS=wigSpeed, wigH=wigHt;    

    $elem.animate({
        top: wigH
    }, wigS, function() {
        $elem.animate({
            top: 0
        }, wigS, function() {
            wiggle(wigS, wigH);
