OB.Verse =
{
	previewUrl: '/topics/tools/verse',

	Preview: function(myEvent)
	{
		myEvent.preventDefault();
		$.getJSON(OB.Verse.previewUrl, {verse: $('#preview').val()}, OB.Verse.PreviewResponse);
	},

	PreviewResponse: function(response)
	{
		if (response.content &amp;&amp; response.id &amp;&amp; response.reference)
		{
			$('#preview-text').html('&lt;h3&gt;' + response.reference + '&lt;/h3&gt;&lt;p&gt;' + response.content + '&lt;/p&gt;&lt;p&gt;&lt;input type="submit" value="Suggest This Verse" /&gt;&lt;/p&gt;');
			$('#verse').val(response.id);
		}
		else
		{
			$('#preview-text').html('&lt;p class="error"&gt;Sorry, we couldn&rsquo;t find a passage matching the verse you suggest. Please double-check your spelling and try again.&lt;/p&gt;');
		}
	}
};

OB.Vote =
{
	myUrl: '/topics/tools/vote',

	Vote: function(myEvent)
	{
		myEvent.preventDefault();
		$.post(OB.Vote.myUrl, {
			word: document.URL.replace(/\/+$/, '').split(/\//).pop(),
			db: $("#db").val(),
			id: $(this).attr("id")
			});
		$(this).parent().html('Thank you for your vote.');
	},

	Over: function()
	{
		$(this).addClass("visible");
	},

	Out: function()
	{
		$(this).removeClass("visible");
	}
};
