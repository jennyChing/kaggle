            sensitivity: 7,

          // pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
          var cX, cY, pX, pY;
          // A private function for getting mouse position
          var track = function(ev) {
            cX = ev.pageX;
            cY = ev.pageY;
          };
          // A private function for comparing current and previous mouse position
          var compare = function(ev,ob) {
            ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
            // compare mouse positions to see if they've crossed the threshold
            if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
              $(ob).unbind("mousemove",track);
              // set hoverIntent state to true (so mouseOut can be called)

              pY = cY;

              ob.hoverIntent_t = setTimeout( function(){
                compare(ev, ob);
              }

              opts.els[prepend?"unshift":"push"](s);

            opts.slideCount=els.length;

              opts.currSlide++;
              opts.nextSlide++;
            }
            if(!$.support.opacity&amp;&amp;opts.cleartype&amp;&amp;!opts.cleartypeNoBg){
              clearTypeFix($s);
            }
            if(opts.fit&amp;&amp;opts.width){
              $s.width(opts.width);
            }
            if(opts.fit&amp;&amp;opts.height&amp;&amp;opts.height!="auto"){
              $slides.height(opts.height);
            }
            s.cycleH=(opts.fit&amp;&amp;opts.height)?opts.height:$s.height();
            s.cycleW=(opts.fit&amp;&amp;opts.width)?opts.width:$s.width();
            $s.css(opts.cssBefore);
            if(opts.pager||opts.pagerAnchorBuilder){
              $.fn.cycle.createPagerAnchor(els.length-1,s,$(opts.pager),els,opts);
            }
            if($.isFunction(opts.onAddSlide)){
              opts.onAddSlide($s);
            }
            else{
              $s.hide();
            }

            return;

            return;
          }
          if(!manual&amp;&amp;!p.cyclePause&amp;&amp;!opts.bounce&amp;&amp;((opts.autostop&amp;&amp;(--opts.countdown&lt;=0))||(opts.nowrap&amp;&amp;!opts.random&amp;&amp;opts.nextSlide&lt;opts.currSlide))){

              opts.end(opts);

            return;

              opts.currFx=fx;

              fx=opts.oneTimeFx;
              opts.oneTimeFx=null;

            $.fn.cycle.resetState(opts,fx);

              $.each(opts.before,function(i,o){
                if(p.cycleStop!=opts.stopCount){
                  return;
                }
                o.apply(next,[curr,next,opts,fwd]);
              }
                    );
            }
            var after=function(){
              $.each(opts.after,function(i,o){
                if(p.cycleStop!=opts.stopCount){
                  return;
                }
                o.apply(next,[curr,next,opts,fwd]);
              }

            };

            opts.busy=1;
            if(opts.fxFn){
              opts.fxFn(curr,next,opts,after,fwd,manual&amp;&amp;opts.fastOnEvent);
            }
            else{
              if($.isFunction($.fn.cycle[opts.fx])){
                $.fn.cycle[opts.fx](curr,next,opts,after,fwd,manual&amp;&amp;opts.fastOnEvent);
              }
              else{
                $.fn.cycle.custom(curr,next,opts,after,fwd,manual&amp;&amp;opts.fastOnEvent);
              }
            }
          }
          if(changed||opts.nextSlide==opts.currSlide){
            opts.lastSlide=opts.currSlide;
            if(opts.random){
              opts.currSlide=opts.nextSlide;
              if(++opts.randomIndex==els.length){
                opts.randomIndex=0;
              }
              opts.nextSlide=opts.randomMap[opts.randomIndex];
              if(opts.nextSlide==opts.currSlide){
                opts.nextSlide=(opts.currSlide==opts.slideCount-1)?0:opts.currSlide+1;
              }
            }
            else{
              if(opts.backwards){
                var roll=(opts.nextSlide-1)&lt;0;
                if(roll&amp;&amp;opts.bounce){
                  opts.backwards=!opts.backwards;
                  opts.nextSlide=1;
                  opts.currSlide=0;
                }
                else{
                  opts.nextSlide=roll?(els.length-1):opts.nextSlide-1;
                  opts.currSlide=roll?0:opts.nextSlide+1;
                }
              }
              else{
                var roll=(opts.nextSlide+1)==els.length;
                if(roll&amp;&amp;opts.bounce){
                  opts.backwards=!opts.backwards;
                  opts.nextSlide=els.length-2;
                  opts.currSlide=els.length-1;
                }
                else{
                  opts.nextSlide=roll?0:opts.nextSlide+1;
                  opts.currSlide=roll?els.length-1:opts.nextSlide-1;
                }
              }
            }
          }
          if(changed&amp;&amp;opts.pager){
            opts.updateActivePagerLink(opts.pager,opts.currSlide,opts.activePagerClass);
          }
          var ms=0;
          if(opts.timeout&amp;&amp;!opts.continuous){
            ms=getTimeout(els[opts.currSlide],els[opts.nextSlide],opts,fwd);
          }
          else{
            if(opts.continuous&amp;&amp;p.cyclePause){
              ms=10;
            }
          }
          if(ms&gt;0){
            p.cycleTimeout=setTimeout(function(){
              go(els,opts,0,(!opts.rev&amp;&amp;!opts.backwards));
            }
                                      ,ms);
          }
        }

          if(opts.timeoutFn){

            }
          }
          return opts.timeout;
        }

        function advance(opts,val){
          var els=opts.elements;
          var p=opts.$cont[0],timeout=p.cycleTimeout;
          if(timeout){
            clearTimeout(timeout);
            p.cycleTimeout=0;
          }
          if(opts.random&amp;&amp;val&lt;0){
            opts.randomIndex--;
            if(--opts.randomIndex==-2){
              opts.randomIndex=els.length-2;
            }
            else{
              if(opts.randomIndex==-1){
                opts.randomIndex=els.length-1;
              }
            }
            opts.nextSlide=opts.randomMap[opts.randomIndex];
          }
          else{
            if(opts.random){
              opts.nextSlide=opts.randomMap[opts.randomIndex];
            }
            else{
              opts.nextSlide=opts.currSlide+val;
              if(opts.nextSlide&lt;0){
                if(opts.nowrap){
                  return false;
                }
                opts.nextSlide=els.length-1;
              }
              else{
                if(opts.nextSlide&gt;=els.length){
                  if(opts.nowrap){
                    return false;
                  }
                  opts.nextSlide=0;
                }
              }
            }
          }
          var cb=opts.onPrevNextEvent||opts.prevNextClick;
          if($.isFunction(cb)){
            cb(val&gt;0,opts.nextSlide,els[opts.nextSlide]);
          }
          go(els,opts,1,val&gt;=0);
          return false;
        }

              Aneka Tambang Persero - Recruitment Jobs S1 Experience Hire Program ANTAM December 2014
             </a>
            </h3>

            <div class="post-body entry-content">
             <p>
              Aneka Tambang Persero (ANTAM) is one of the largest mining company in Indonesia with a majority stake owned by the Indonesian Government. Antam is a diversified mining company and vertically integrated, export-oriented. Through the area of operation spread throughout Indonesia rich in minerals, Antam activities include exploration, mining, processing and marketing of nickel ore, feronike, gold, silver, bauxite and coal.
              <br/>
              <br/>
              <span style="font-family: Arial, Helvetica, sans-serif; font-size: x-large;">
               <a href="http://www.lowongankerjaindonesiaterbaru.com/2014/08/aneka-tambang-persero.html">
                Aneka Tambang Persero
               </a>
              </span>
              <br/>
              <span style="font-family: Verdana, sans-serif;">
               Currently looking for:
              </span>
              <br/>
              <span style="color: #c90a0a; font-family: Verdana, sans-serif; font-size: x-large;">
               Vice President, Accounting and Budgeting
              </span>
              <br/>
              <span style="color: #c90a0a; font-family: Verdana, sans-serif; font-size: x-large;">
               CSR Junior Specialist
              </span>
              <br/>
              <br/>
              Aneka Tambang Persero, current career opportunities, for those of you who are young, aggressive, tenacious, communicative and honest, welcome to send your application to join us. Career offered, for a minimum graduate Bachelor Degree (S1) majoring in Accounting or relevant to the position.
              <br/>
              <br/>
             </p>
             <div class="separator" style="clear: both; text-align: center;">
              <a href="http://www.lowongankerjaindonesiaterbaru.com/2014/08/aneka-tambang-persero.html" style="margin-left: 1em; margin-right: 1em;" title="ANTAM">
               <img alt="Aneka Tambang Persero - Recruitment Jobs S1 Experience Hire Program ANTAM December 2014" border="0" height="250" src="http://4.bp.blogspot.com/-FS9LrA4Rrpw/VHyAp_-WIEI/AAAAAAAAu9U/bzYiJXnV6RE/s320/Aneka%2BTambang%2BPersero.jpg" title="Aneka Tambang Persero" width="300"/>
              </a>
             </div>
             <div style="text-align: center;">
              <i>
               Credit Image : lowongankerjaindonesiaterbaru.com
              </i>
             </div>
             <br/>
             Applicants who register will be required to have a minimum GPA of 3.00 on a scale of 4.00, have the experience in the accounting field is preferred and the field of Corporate Social Responsibility (CSR), especially in Kalimantan, preferably at least 2-10 years, It has certificate of Financial Management, Accounting Finance, and Advanced English Course, Fluent in English both oral and written, willing to work in a tight schedule, and applicants must be willing to be assigned to and moved to one of Antam Unit / Business Unit / Subsidiary.
             <br/>
             <br/>
             <span style="color: #c90a0a; font-family: Verdana, sans-serif; font-size: x-large;">
              How to Apply?
             </span>
             <br/>
             If you match the profile and are interested in this exiting career opportunity, please send application forms ( can be downloaded at www.antam.com) and requirement document through following link below or by email to
             <span style="color: #c90a0a;">
              recruitment@antam.com
             </span>
             with the subject line VP, Accounting and Budgeting or CSR Junior Specialist. Closing date
             <span style="color: #c90a0a;">
              9 December 2014
             </span>
             . Only qualified candidates that meet with the above requirements would be processed in this recruitment.
