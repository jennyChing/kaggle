            function update_form_fields(no_inst_check) {

                var i;
                var f  = gid('f_inst');
                var b  = gid('f_branch');
                var si = gid('f_sub_inst');
                var s  = gid('search');

                var sd = new Array();
                var disp = '';
                var uniq_sites    = {};
                var uniq_branches = {};
                var uniq_inst     = {};   // for the dropdown

                var orig_f = f.value;
                var orig_b = b.value;
                var orig_s = s.value;
                var re     = '';
                    
                var oo  = false;
                var tmp = gid('f_oconus_only');
                if (tmp) { oo = tmp.checked; }

                var d = gid('sub_inst_div');
                if (d) { d.style.display = 'none'; }

                if (!f || !b || !s || !si) { return false; }

                // trim leading and trailing white space from the user entered value
                if (s.value) {
                    
                    var tmp = s.value;
                    
                    tmp = tmp.replace("(", '\\(');
                    tmp = tmp.replace(")", '\\)');
                    re = new RegExp('^' + tmp.replace(/^\s+|\s+$/g, '') + '$', 'i'); 

                }
                for(i = f.options.length - 1; i &gt;= 1; i--) { f.remove(i); }
                for(i = b.options.length - 1; i &gt;= 1; i--) { b.remove(i); }
                for(i = si.options.length - 1; i &gt;= 1; i--) { si.remove(i); }

                // add back in what we got back but only if we selected a branch and it matches
                var opt_index = 1;  // zero is please choose!

                for (i = 0; i &lt; all_sites.length; i++) {
                    
                    if (oo) { if (all_sites[i].oconus == 0) { continue; } }
                    
                    // this is for the auto complete
                    if (!ac_done) { 
                        if (typeof(uniq_sites[all_sites[i].basename]) == 'undefined') {
                            sd[sd.length] = all_sites[i].basename; 
                            uniq_sites[all_sites[i].basename] = true;
                        }
                    }
                    
                    disp = all_sites[i].country + ' - ' + all_sites[i].basename;
                    if (orig_b == '' || orig_b == 'Please Choose') { disp = all_sites[i].mha_desc + ' - ' + disp; } // if nothing is selected, draw branch

                    // if we have something in the search box, do this here
                    if (re) {
                

                        if (re.exec(all_sites[i].basename)) {   // we found a match, add the branch and the base
                        
                            if (typeof(uniq_branches[all_sites[i].branch]) == 'undefined') {
                                uniq_branches[all_sites[i].branch] = true;
                                b.options[b.options.length] = new Option(all_sites[i].mha_desc, all_sites[i].branch);
                                if (orig_b == all_sites[i].branch) { b.selectedIndex = b.options.length - 1; }
                            }
                                
                            if (typeof(uniq_inst[all_sites[i].disp]) == 'undefined') {
                                f.options[f.options.length] = new Option(disp, all_sites[i].siteid);
                                uniq_inst[all_sites[i].disp] = true;
                            }

                        }

                    } else {

                        // add them all back in uniquely
                        if (typeof(uniq_branches[all_sites[i].branch]) == 'undefined') {
                            uniq_branches[all_sites[i].branch] = true;
                            b.options[b.options.length] = new Option(all_sites[i].mha_desc, all_sites[i].branch);
                            if (orig_b != '' &amp;&amp; orig_b == all_sites[i].branch) { b.selectedIndex = b.options.length - 1; }
                        }

                        // now the installations
                        if (orig_b != 'Please Choose' &amp;&amp; orig_b != '') {
                            // if we selected a branch and the current branch != the one we selected, don't add to the drop down list
                            if (orig_b != all_sites[i].branch) { continue; }
                        }

                        if (typeof(uniq_inst[disp]) == 'undefined') {
                            f.options[f.options.length] = new Option(disp, all_sites[i].siteid);
                            uniq_inst[disp] = true;
                        }
                    
                    }


                }

                // if there's only one, choose it
                if (b.options.length == 2) { b.selectedIndex = 1; }
                if (f.options.length == 2) { f.selectedIndex = 1; }

                // now, if the final installation is selected and we actually changed it, run our sanity checks
                if (f.selectedIndex != 0 &amp;&amp; f.options[f.selectedIndex].value != orig_f) { 
                    if (no_inst_check != true) {
                        remote_registration.inst_merge(f.options[f.selectedIndex].value, { new_site: f.options[f.selectedIndex].value, old_site: orig_f }); 
                    } else {
                        remote_registration.check_joint_base(f.options[f.selectedIndex].value, orig_f);
                    }
                }

                if (!ac_done) {
                    AutoComplete_Create('search', sd, 20, function(o) { var f = gid('search'); if (f) { f.blur(); }}); 
                    ac_done = true;
                }

            }
