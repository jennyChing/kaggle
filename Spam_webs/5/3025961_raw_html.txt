                    loading();
                   
                    if (navigator.geolocation) {
                        navigator.geolocation.getCurrentPosition(locationSuccess, locationError);
                    }
                    else{
                        $j("#loading p").text("Your browser does not support Geolocation!");
                        $j("#loading img").hide();
                    }

//////////////////////////////////////////////////////   
/*                 geo success function             */
/////////////////////////////////////////////////////       
                function locationSuccess(position) {
                    var lat_g = position.coords.latitude;
                    var lon_g = position.coords.longitude;
                    
                    
                    geoAPI ="http://maps.googleapis.com/maps/api/geocode/json?latlng="+lat_g+","+lon_g+"&amp;sensor=false";
                    $j.getJSON(geoAPI, function(r){
               
                        if(r.status == "OK"){
                        
                            var city_find = r.results[0].address_components[1].short_name;
                            
                            data(lat_g+","+lon_g, city_find);
                            refresh_geo = setInterval( function() 
                            {
                                data(lat_g+","+lon_g,city_find);

                            }, refresh);//end setinterval                   
                        } // end if status
                        else{
                            $j("#loading p").text(r.status);
                            $j("#loading img").hide();
                        }
                    }).fail(function() { 
                        $j("#loading p").text("Failed to connect to server");
                        $j("#loading img").hide();
                     });//end json
                }//end location success 

//////////////////////////////////////////////////////   
/*             geo Error function                  */
/////////////////////////////////////////////////////       
            function locationError(error){
                
                $j("#loading img").hide();
                switch(error.code) {
                    case error.TIMEOUT:
                         $j("#loading p").text("A timeout occured! Please try again!");
                        break;
                    case error.POSITION_UNAVAILABLE:
                        $j("#loading p").text("We can\"t detect your location. Sorry!");
                        break;
                    case error.PERMISSION_DENIED:
                        $j("#loading p").text("Please allow geolocation access for this to work.");
                        break;
                    case error.UNKNOWN_ERROR:
                        $j("#loading p").text("An unknown error occured!");
                        break;
                }
            }// end locationError
        
        }//end geo function
    });//end ready

         <h1 class="name post-title entry-title" itemprop="itemReviewed" itemscope="" itemtype="http://schema.org/Thing">
          <span itemprop="name">
           29 killed in Baghdad car bombings
          </span>
         </h1>
         <p class="post-meta">
          <span class="post-views">
           8 Views
          </span>
         </p>
         <div class="clear">
         </div>
         <div class="entry">
          <p>
           <strong>
            BAGHDAD: The Iraqi Capital Baghdad was rocked by nine bomb blasts on Monday morning, killing at least 29 people and injuring more than 120, emergency officials said.
           </strong>
          </p>
          <p>
           <a href="http://i1.wp.com/thekooza.com/wp-content/uploads/2013/07/Baghdad-car-bomb.kooza_.jpg">
            <img alt="Baghdad-car-bomb.kooza" class="size-medium wp-image-4555 alignright" data-recalc-dims="1" src="http://i1.wp.com/thekooza.com/wp-content/uploads/2013/07/Baghdad-car-bomb.kooza_.jpg?resize=300%2C225"/>
           </a>
          </p>
          <p>
           One of the blasts occurred in Mahmudiya area, south of Baghdad, while five of them occurred in Shiite majority areas.
          </p>
          <p>
           Two car bombs exploded in Kut, killing at least five people and injuring 35.
          </p>
          <p>
           The recent car bombings have been connected with the recent discontent of Sunni minority, amid fears of a sectarian unrest, which could trigger if the government failed to address their problems.
          </p>
          <p>
          </p>
          <div class="sharedaddy sd-sharing-enabled">
