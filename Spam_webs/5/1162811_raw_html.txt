  <script>
   $(function(){
        function RefreshCaptcha(captchaImg){
            $.ajax({
                url: "/api/captchaapi/getimage",
                success: function (data) {
                    captchaImg.fadeOut(200,function(){
                        var d = new Date();
                        captchaImg.attr("src", "/api/captchaapi/getimage?"+d.getTime());
                        captchaImg.fadeIn(200);
                    });
                }
            });
        }

        $(document).on("click",".refreshCaptchaImage",function(){
            var captchaImg = $(this).siblings(".captchaImage");
            RefreshCaptcha(captchaImg);
        });

        $(document).on("focusin",".captchaImageInput",function(){
            var captchaImg = $(this).closest(".captchaField").find(".captchaImage");
            RefreshCaptcha(captchaImg);
        });

        $(document).on("click",".submitForm",function(){

            var $form = $(this).closest(".form");

            var formMessage = "CONTACT FORM:\n\n";
            var captchaValue = "";
            $form.find("input[type='text'],textarea,select").each(function(){
                var value = $(this).val();
                var name = $(this).attr("name");

                if(name=="captcha"){
                    captchaValue = value;
                }
                else {
                    formMessage += name +": "+ value +"\n\n";
                }
            });

            $form.find("input[type='checkbox']").each(function(){
                var name = $(this).attr("name");
                var value = "Not checked";
                if($(this).is(':checked')){
                    value = "CHECKED";
                }

                formMessage += name +": "+ value +"\n\n";
            });

            $form.find("input[type='radio']:checked").each(function(){
                var value = $(this).val();
                var name = $(this).attr("name");
                formMessage += name +": "+ value +"\n\n";
            });

            var $captcha = $form.find(".captchaImageInput");
            if(captchaValue==""){
                $captcha.addTemporaryClass("error",4000);
                return false;
            }
            else {
                $.ajax({
                    url: "/api/contactmessageapi/newblockmessage",
                    type: "POST",
                    dataType: "json",
                    data: {
                        message : formMessage,
                        captcha : captchaValue
                    },
                    traditional: true,
                    async: true,
                    success: function (data) {
                        if (data.IsSuccess) {
                            alert(data.SuccessMessage);

                            $form.find("input[type='text'],textarea,select").each(function(){
                                $(this).val("");
                            });
                        }
                        else {
                            if(data.Data != undefined &amp;&amp; data.Data.CaptchaWrong != undefined &amp;&amp; data.Data.CaptchaWrong == 1){
                                $captcha.addTemporaryClass("error",2000);
                            }
                            else {
                                alert("Something went wrong. Please use another way to contact this person.");
                            }
                        }
                    },
                    error: function (jqXHR, textStatus, errorThrown) {
                        alert("Something went wrong. Please use another way to contact this person.");
                    }
                });

                return false;
            }



        });
    });
  </script>

        $("#commentRefreshCaptcha").click(function(){
            var captchaImg = $(this).siblings(".captchaImg");
            $.ajax({
                url: "/api/captchaapi/getnewimage",
                success: function (data) {
                    captchaImg.fadeOut(200,function(){
                        var d = new Date();
                        captchaImg.attr("src", "/api/captchaapi/getnewimage?"+d.getTime());
                        captchaImg.fadeIn(200);
                    });

                }
            });
        });

        $("#postComment").submit(function(){
            var contentObj = $('#comment');
            var nameObj = $('#commentName');
            var captchaObj = $('#commentCaptcha');

            var postData = {
                content : contentObj.val() || "",
                personName : nameObj.val() || "",
                captcha : captchaObj.val() || "",
                pageGuid : '20366614213N2bM'
            };
            if(postData.content=="") {
                contentObj.addTemporaryClass("error",4000);
                return false;
            }
            else if(postData.personName=="") {
                nameObj.addTemporaryClass("error",4000);
                return false;
            }
            else if(postData.captcha==""){
                captchaObj.addTemporaryClass("error",4000);
                return false;
            }
            else
            {
                var ajHelper = new AjaxFrontHelper();
                ajHelper.PostUrl = "/api/other/newcomment";
                ajHelper.PostData = postData;
                ajHelper.OnSuccessReturn = function(){
                    $("#postComment").fadeOut(200,function(){
                        $("#commentAdded").html(ajHelper.ReturnData.SuccessMessage);
                        $("#commentAdded").fadeIn();
                    })
                };
                ajHelper.OnErrorReturn = function(){
                    if(ajHelper.ReturnData.Data != undefined &amp;&amp; ajHelper.ReturnData.Data.CaptchaWrong != undefined &amp;&amp; ajHelper.ReturnData.Data.CaptchaWrong == 1){
                        captchaObj.addTemporaryClass("error",2000);
                    }
                };
                ajHelper.DoPost();
                return false;
            }
        });
    });
  </script>
