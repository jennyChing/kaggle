                                allMarkers.push(new google.maps.LatLng(nearby[i].latitude, nearby[i].longitude));
                            }

                            var mapOptions = {
                                center: allMarkers[0],
                                zoom: initialZoom,
                                mapTypeId: google.maps.MapTypeId.ROADMAP
                            };
                            window.gmap = new google.maps.Map(document.getElementById('map-elem'), mapOptions);
                            window.gmap2 = new google.maps.Map(document.getElementById('map-elem2'), mapOptions);

                            // Fit all markers on map
                            //  Make an array of the LatLng's of the markers you want to show
                            //  Create a new viewpoint bound
                            var bounds = new google.maps.LatLngBounds ();
                            //  Go through each...
                            for (var j = 0, LtLgLen = allMarkers.length; j &lt; LtLgLen; j++) {
                              //  And increase the bounds to take this point
                              bounds.extend (allMarkers[j]);
                            }
                            //  Fit these bounds to the map
                            gmap.fitBounds(bounds);
                            gmap2.fitBounds(bounds);

                            var infoWindow = new google.maps.InfoWindow();

                            // marker for the given business
                            var marker = new google.maps.Marker({
                                position: allMarkers[0],
                                map: gmap,
                                icon: "http://www.googlemapsmarkers.com/v1/A/fb8374/000000/ae473c",
                                title: nearby[0].name,
                                iwcontent: '&lt;strong&gt;&lt;a href="'+nearby[0]['url']+'" target="_blank"&gt;"'+nearby[0].name+'&lt;/a&gt;&lt;/strong&gt;'
                            });

                            google.maps.event.addListener(marker, 'click', function (e) {
                                infoWindow.setContent(this.iwcontent);
                                infoWindow.open(gmap, this);
                            });
                            // neighborhood businesses


                            if (nearby !== null) {
                                for (var k = 1; k &lt; nearby.length; k++) {
                                    var marker = new google.maps.Marker({
                                        position: allMarkers[k],
                                        map: null,
                                        title: nearby[k].name,
                                        iwcontent: '&lt;strong&gt;&lt;a href="'+nearby[k]['url']+'" target="_blank"&gt;'+nearby[k]['name']+'&lt;/a&gt;&lt;/strong&gt;'
                                    });
                                    google.maps.event.addListener(marker, 'click', function (e) {
                                        infoWindow.setContent(this.iwcontent);
                                        infoWindow.open(gmap, this);
                                    });
                                    gmapMarkers.push(marker);
                                }
                            }


                        }

       <p class="author">
        Posted by the Mass Office of Travel and Tourism
       </p>
       <input class="jpibfi" type="hidden"/>
       <p>
        Fresh air is the best kind of cure for too much time inside.
       </p>
       <p>
        For hiking, biking and all other manner of outdoorsy pursuits, there are few better places to go in Massachusetts than the western part of the Commonwealth. Theres all kinds of room to get out and about, and the bucolic landscapes only make it more enjoyable while you do.
       </p>
       <p>
        Whether its during spring, summer or fall, here are five of the best places to go when you go to western Mass.
       </p>
       <p>
        <strong>
         Monument Mountain in Great Barrington, MA
        </strong>
       </p>
       <p>
        The Berkshires are known for stunning scenery and climbing
        <a href="http://www.massvacation.com/business/2701/" target="_blank">
         Monument Mountain
        </a>
        is one of the absolute best ways to see it. The ascent takes a couple of hours, and is moderately strenuous. So long as the weather cooperates and there hasnt been recent precipitation, though, the trip isnt troublingly steep or dangerous.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/04/Monument-Mountain.jpg">
         <img alt="Monument Mountain" class="aligncenter size-full wp-image-7550" height="628" src="/blog/wp-content/uploads/2015/04/Monument-Mountain.jpg" width="628"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         What awaits at the top of Monument Mountain
        </em>
       </p>
       <p>
        At the summit, hikers are rewarded with a magnificent panorama of the Housatonic River Valley; Squaw Peak, Monument Mountains highest point and one that stands 1,600+ feet above sea level, provides views as far as
        <a href="http://www.mass.gov/eea/agencies/dcr/massparks/region-west/mt-greylock-state-reservation-generic.html" target="_blank">
         Mount Greylock
        </a>
        and, in the distance, New Yorks Catskill Mountains. The route is open year round, and for those whod like to stay nearby, the
        <a href="http://www.massvacation.com/business/2702/" target="_blank">
         Monument Mountain Hotel
        </a>
        is the place to go.
       </p>
       <p>
        <strong>
         Gunn Brook Falls in Sunderland, MA
        </strong>
       </p>
       <p>
        Sunderland Bridge, which connects the town of the same name to Deerfield, is one of the best foliage peeping spots in Massachusetts during autumn. A little bit off the beaten trail, visitors can also find a lesser-known hidden gem thats native to the area: Gunn Brook Falls.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/04/Gun-Brook-Falls.jpg">
         <img alt="Gun Brook Falls" class="aligncenter size-full wp-image-7551" height="598" src="/blog/wp-content/uploads/2015/04/Gun-Brook-Falls.jpg" width="900"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Gunn Brook Falls in spring by Evan Gregg
        </em>
       </p>
       <p>
        Located on the lower west side of
        <a href="http://en.wikipedia.org/wiki/Mount_Toby" target="_blank">
         Mount Toby
        </a>
        , the waterfall cascades down 35 feet through a verdant, forested area. The site is accessible through a path thats connected to a public road, giving guests the option to either mostly drive and walk a bit or walk the whole way. For directions, click
        <a href="http://www.berkshirephotos.com/AABerkshireWaterfall360/AACenteral/GunnBrookWeb360/GunnBrook2Dec03PCD.htm">
         here
        </a>
        . For overnights in the area, the
        <a href="http://www.massvacation.com/business/7037/" target="_blank">
         Red Roof Inn
        </a>
        in South Deerfield is convenient.
       </p>
       <p>
        <strong>
         Naumkeag in Stockbridge, MA
        </strong>
       </p>
       <p>
        Vestiges of the
        <a href="http://en.wikipedia.org/wiki/Gilded_Age" target="_blank">
         Gilded Age
        </a>
        can still be found in many corners of western Massachusetts, where the upper crust of centuries past constructed opulent mansions for their summer retreats. If your trip only allows enough time to see one of the sites, no one could fault you for choosing
        <a href="http://www.massvacation.com/business/2800/" target="_blank">
         Naumkeag
        </a>
        .
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/04/Naumkeag-Gardens.jpg">
         <img alt="Naumkeag Gardens" class="aligncenter size-full wp-image-7552" height="1079" src="/blog/wp-content/uploads/2015/04/Naumkeag-Gardens.jpg" width="1600"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Blue Steps at Naumkeag Gardens
        </em>
       </p>
       <p>
        Built for Ambassador Joseph Choate once upon a time, the property is nowadays administered by the
        <a href="http://www.thetrustees.org/places-to-visit/berkshires/naumkeag.html" target="_blank">
         Trustees of Reservations
        </a>
        , who make it open for visitation from May through mid-October. The shingled house is majestic, and the grounds are maybe even more so, with the Blue Steps (pictured above) being especially gorgeous. To sleep over, the
        <a href="http://www.massvacation.com/business/4120/" target="_blank">
         Red Lion Inn
        </a>
        , which also caters Naumkeags Oak Caf, is a great pick.
       </p>
       <p>
        <strong>
         Bartholomews Cobble in Sheffield, MA
        </strong>
       </p>
       <p>
        Naumkeag is to botanical wonder as
        <a href="http://www.massvacation.com/business/314/" target="_blank">
         Bartholomews Cobble
        </a>
        is to geological wonder. The cobble, which consists of twin rocky knolls, has been around quite a bit longer than anything man-made in western Mass, as its shape began to form from coral, shells and sand more than 500 million years ago.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/04/Bartholomews-Cobble.jpg">
         <img alt="Bartholomew's Cobble" class="aligncenter size-full wp-image-7553" height="1362" src="/blog/wp-content/uploads/2015/04/Bartholomews-Cobble.jpg" width="2048"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Hiking towards Hurlburts Hill at Bartholomews Cobble
        </em>
       </p>
       <p>
        If intellectual interest in geology isnt quite enough to draw you to Bartholomews Cobble, there is some terrific hiking and wildlife, too.
        <a href="http://www.thetrustees.org/places-to-visit/berkshires/bartholomews-cobble.html" target="_blank">
         Five miles of trails
        </a>
        offer ample space for leg stretching, with Hurlburts Hill, the high point, rising more than 1,000 feet to present a superb view of the surrounding Housatonic River Valley; the area is also home to hundreds of bird species. Those coming to Sheffield can put their feet up nearby at the
        <a href="http://www.massvacation.com/business/3691/" target="_blank">
         Staveleigh House B&amp;B
        </a>
        .
       </p>
       <p>
        <strong>
         Mount Tom State Reservation in Holyoke, MA
        </strong>
       </p>
       <p>
        Rumor has it that
        <a href="http://www.mass.gov/eea/agencies/dcr/massparks/region-west/mount-tom-state-reservation.html" target="_blank">
         Mount Tom
        </a>
        was the inspiration for literary legend and Springfield native Dr. Seusss
        <a href="http://en.wikipedia.org/wiki/How_the_Grinch_Stole_Christmas!" target="_blank">
         Mount Crumpit
        </a>
        . The reservation is not, it must be said, home to any grinches. It is home to more than 2,100 acres of natural beauty, however, which is better for everyone anyway.
       </p>
       <p>
        <a href="/blog/wp-content/uploads/2015/04/Mount-Tom.jpg">
         <img alt="Mount Tom" class="aligncenter size-full wp-image-7554" height="1365" src="/blog/wp-content/uploads/2015/04/Mount-Tom.jpg" width="2048"/>
        </a>
       </p>
       <p style="text-align: center;">
        <em>
         Looking out from the top of Mount Tom during autumn by Lynne Graves
         <br/>
        </em>
       </p>
       <p>
        How can you make the most of that natural beauty? Well, for starters, theres a whole lot of room for walking and hiking, with 22 miles of trails. Lake Bray has fishing, while the childrens play area is suitable for young ones. Hawks and eagles also make it a habit to pass over the mountain, with countless birds passing through each year. Parking near Mount Tom runs $5 or $6, and
        <a href="http://www.massvacation.com/business/8302/" target="_blank">
         D. Hotel &amp; Suites
        </a>
        in town is just a few minutes drive away.
       </p>
       <p>
        With all the green space there is in western Mass, these five places offer only a sampling of the great outdoors. You can find much more
        <a href="http://www.massvacation.com/explore/outdoors/?stype=1&amp;cid=1&amp;tid=6&amp;gid=0&amp;sid=6&amp;rid=0&amp;attr1=0&amp;attr2=0&amp;so=0&amp;map=0&amp;pnum=1&amp;dnum=20&amp;mid=0" target="_blank">
         here
        </a>
        . And for a look at outside activities throughout all of Massachusetts, just click
        <a href="http://www.massvacation.com/explore/outdoors/" target="_blank">
         here
        </a>
        .
       </p>
       <p>
        Do you have a favorite scenic spot in western Massachusetts? Let us know below in the comments!
