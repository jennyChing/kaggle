             // experimenting with $.getUrlVar(s)
jQuery.extend({
    getUrlVars: function(){
        var vars = [], hash;
        var hashes = window.location.href.slice(window.location.href.indexOf('?') + 1).split('&amp;');
        for(var i = 0; i &lt; hashes.length; i++) {
            hash = hashes[i].split('=');
            vars.push(hash[0]);
            vars[hash[0]] = hash[1];
        }
        return vars;
    },
    getUrlVar: function(name){
        return jQuery.getUrlVars()[name];
    }
});


//A global counter used for the individual fact div ids
var counter = 0;
var firstLoad = true;
var fact = new Array();
var hashCount = 0;

//Perform the actual animation
function slide() {
    _gaq.push(["_trackEvent", "Generators", "Amazing Fact Page Click", "Hit Me With Another Fact"]);
    //Animate the main content panel with queuing, and then remove the now invisible af-slider-cell panel
    jQuery(".af-slider").animate({"top": "-=246px"}, {duration:300, queue:true, complete:function() {
        jQuery(".af-slider-cell:lt(1)").empty();
    }});
    //Animate the left and right edges with queuing, and then reset the repeating divs back to their original position
    jQuery(".af-left").animate({"top": "-=31px"}, {duration:300, queue:true, complete:function() {
    	jQuery(".af-left").css("top", "0px");
    }});
    jQuery(".af-right").animate({"top": "-=31px"}, {duration:300, queue:true, complete:function() {
    	jQuery(".af-right").css("top", "0px");
    }});

    bumpHash();

    //Load the next fact (there should always be a queue of 5)
    loadFact(1);

}

function bumpHash() {
    //console.log("Bumphash for count: "+hashCount);
    //console.log(fact[hashCount]);
    location.hash = 'f' + fact[hashCount].nid;
    updateSocialButtons(fact[hashCount]);
    hashCount++;
}

//Load (or Preload) the facts 
function loadFact(num, cycleAds, url) {
    if(undefined == num) num = 1;
    if(undefined == cycleAds) cycleAds = true;
    if(undefined == url) url = "/api/1.0/views/amazing_facts.json";
    var id = new Array(num);
    //For each new fact requested. Pick a random number for the id and create new divs for them
    for(i=0; i&lt;num; i++) {
        id[i] = "f_"+counter++;
        jQuery(".af-slider").append("&lt;div id=\""+id[i]+"\" class=\"af-slider-cell\"&gt;&lt;/div&gt;");
    }

    //Make a JSON call to get the proper number of facts and set the html of the 
    jQuery.ajax({
      url: url,
      jsonp: false,
      data: { limit : num , display_id: 'xhr', cb: Math.random() }, 
      success: function(data) {
        jQuery.each(data, function(key, val) {
            fact.push(val); 
        	//Create the fact slider panel contents, either with or without the submitted by field
        	sliderCellContents = '';
        	if (val['submitted_by'].length &gt; 0) {
        		sliderCellContents = sliderCellContents + '&lt;div class="af-submitted-by"&gt;&lt;p&gt;';
        		sliderCellContents = sliderCellContents + 'Submitted by &lt;span class="af-submitter"&gt;';
        		sliderCellContents = sliderCellContents + val['submitted_by'];
        		sliderCellContents = sliderCellContents + '&lt;/span&gt;&lt;/p&gt;&lt;/div&gt;';
        		sliderCellContents = sliderCellContents + '&lt;div class="af-fact-with-submitter"&gt;&lt;p&gt;' + val.fact_body + '&lt;/p&gt;&lt;/div&gt;';
        	} else {
        		sliderCellContents = sliderCellContents + '&lt;div class="af-fact"&gt;&lt;p&gt;' + val.fact_body + '&lt;/p&gt;&lt;/div&gt;';
        	}
        
            //Adjust the font size based on the character count. Font size will end up
            //being between 20 &amp; 40
            jQuery("#"+id[key]).html(sliderCellContents);
            factLength = jQuery("#"+id[key]).text().length;
            if (factLength &gt;= 200) {
            	newFontSize = 20;
            } else if (factLength &lt;= 40) {
        		newFontSize = 40;
        	} else {
        		newFontSize = Math.floor(40 - ((factLength - 40) / 8));
        	}
            jQuery("#"+id[key]).css("font-size", newFontSize + "px");
          }) // end jquery.each()
        
        //Load a new ad every 2 requests
        //if(counter % 2 == 0) refreshAds(); 
        if(!firstLoad) refreshAds(); // don't refresh on the first page load
        
        if(hashCount == 0) bumpHash();
		firstLoad = false;

      } // end success callback
    });
}

function updateSocialButtons(fact) {
    //console.log("updating social buttons for fact:");
    //console.log(fact);

    var toolbox_arr = jQuery('.addthis_toolbox').get();
    jQuery.each(toolbox_arr, function(idx, toolbox_obj) {
        var share_config = {
            title: "Amazing Fact #"+fact.nid+" "+fact.fact_body,
            description: fact.fact_body,
            html: fact.fact_body,
            url: "http://mentalfloss.com/amazing-facts?f="+fact.nid,
        };
        addthis.toolbox(toolbox_obj, addthis_config, share_config);
        
    });
}

	jQuery(".page-amazing-facts h1.title &gt; span").html("AMAZING FACT GENERATOR").show();
    // Check to see if a specific fact is being requested
    var legacy_id='';
    var nid = '';

    // first check the query string param 'p' for a legacy id
    var qsp = jQuery.getUrlVar('p');
    if(undefined != qsp) {
        legacy_id = qsp;
    }
    // check the query string param 'f' for a node id
    var qsf = jQuery.getUrlVar('f');
    if(undefined != qsf) {
        nid = qsf;
    }

    // then a hash argument (takes precedence)
    hasharg = window.location.hash.replace('#','');
    if('' != hasharg) {
        // if the hash begins with f it's a new style hash
        if(undefined != hasharg.match(/^f(.*)/g)) {
            nid = hasharg.replace('f', '');
        }
        // otherwise it's a legacy hash
        else {
            legacy_id = hasharg;
        }
    }
    if('' != nid) {
        loadFact(1, false, "/api/1.0/views/amazing_facts.json?id="+nid);
    } 
    else if('' != legacy_id) {
        loadFact(1, false, "/api/1.0/views/amazing_facts.json?legacy_id="+legacy_id);
    }
    
    //Preload 5 facts
    loadFact(5, false);
});
