        custom_facets_types["slider"] = function (helper, content, facet) {
            if (content.getFacetByName(facet.attribute) != undefined)
            {
                var min = content.getFacetByName(facet.attribute).stats.min;
                var max = content.getFacetByName(facet.attribute).stats.max;

                var current_min = helper.state.getNumericRefinement(facet.attribute, "&gt;=");
                var current_max = helper.state.getNumericRefinement(facet.attribute, "&lt;=");

                if (current_min == undefined)
                    current_min = min;

                if (current_max == undefined)
                    current_max = max;

                var params = {
                    type: {},
                    current_min: Math.floor(current_min),
                    current_max: Math.ceil(current_max),
                    count: min == max ? 0 : 1,
                    min: Math.floor(min),
                    max: Math.ceil(max)
                };

                params.type[facet.type] = true;

                return [params];
            }

            return [];

        var updateUrl = function (push_state) {

            var refinements = [];

            /** Get refinements for conjunctive facets **/
            for (var refine in helper.state.facetsRefinements)
            {
                if (helper.state.facetsRefinements[refine])
                {
                    var r = {};

                    r[refine] = helper.state.facetsRefinements[refine];

                    refinements.push(r);
                }
            }

            /** Get refinements for disjunctive facets **/
            for (var refine in helper.state.disjunctiveFacetsRefinements)
            {
                for (var i = 0; i &lt; helper.state.disjunctiveFacetsRefinements[refine].length; i++)
                {
                    var r = {};

                    r[refine] = helper.state.disjunctiveFacetsRefinements[refine][i];

                    refinements.push(r);
                }
            }

            var url = '#q=' + encodeURIComponent(helper.state.query) +
                '&amp;page=' + helper.getCurrentPage() +
                '&amp;refinements=' + encodeURIComponent(JSON.stringify(refinements)) +
                '&amp;numerics_refinements=' + encodeURIComponent(JSON.stringify(helper.state.numericRefinements)) +
                '&amp;index_name=' + encodeURIComponent(JSON.stringify(helper.getIndex()));

            /** If push_state is false wait for one second to push the state in history **/
            if (push_state)
                history.pushState(url, null, url);
            else
            {
                clearTimeout(history_timeout);
                history_timeout = setTimeout(function () {
                    history.pushState(url, null, url);
                }, 1000);
            }
        };

        var getRefinementsFromUrl = function() {

            if (location.hash &amp;&amp; location.hash.indexOf('#q=') === 0)
            {
                var params                          = location.hash.substring(3);
                var pageParamOffset                 = params.indexOf('&amp;page=');
                var refinementsParamOffset          = params.indexOf('&amp;refinements=');
                var numericsRefinementsParamOffset  = params.indexOf('&amp;numerics_refinements=');
                var indexNameOffset                 = params.indexOf('&amp;index_name=');

                var q                               = decodeURIComponent(params.substring(0, pageParamOffset));
                var page                            = parseInt(params.substring(pageParamOffset + '&amp;page='.length, refinementsParamOffset));
                var refinements                     = JSON.parse(decodeURIComponent(params.substring(refinementsParamOffset + '&amp;refinements='.length, numericsRefinementsParamOffset)));
                var numericsRefinements             = JSON.parse(decodeURIComponent(params.substring(numericsRefinementsParamOffset + '&amp;numerics_refinements='.length, indexNameOffset)));
                var indexName                       = JSON.parse(decodeURIComponent(params.substring(indexNameOffset + '&amp;index_name='.length)));

                helper.setQuery(q);

                helper.clearRefinements();

                /** Set refinements from url data **/
                for (var i = 0; i &lt; refinements.length; ++i) {
                    for (var refine in refinements[i]) {
                        helper.toggleRefine(refine, refinements[i][refine]);
                    }
                }

                for (var key in numericsRefinements)
                    for (var operator in numericsRefinements[key])
                        helper.addNumericRefinement(key, operator, numericsRefinements[key][operator]);

                helper.setCurrentPage(page);
                helper.setIndex(indexName);

                $(algoliaConfig.autocomplete.selector);
                $(algoliaConfig.autocomplete.selector).val(helper.state.query);

                helper.search();

            }

        var getFacets = function (content) {

            var facets = [];

            for (var i = 0; i &lt; algoliaConfig.facets.length; i++)
            {
                var sub_facets = [];

                if (custom_facets_types[algoliaConfig.facets[i].type] != undefined)
                {
                    try
                    {
                        var params = custom_facets_types[algoliaConfig.facets[i].type](helper, content, algoliaConfig.facets[i]);

                        if (params)
                            for (var k = 0; k &lt; params.length; k++)
                                sub_facets.push(params[k]);
                    }
                    catch(error)
                    {
                        throw("Bad facet function for '" + algoliaConfig.facets[i].type + "'");
                    }
                }
                else
                {
                    var content_facet = content.getFacetByName(algoliaConfig.facets[i].attribute);

                    if (content_facet == undefined)
                        continue;

                    for (var key in content_facet.data)
                    {
                        var checked = helper.isRefined(algoliaConfig.facets[i].attribute, key);

                        var nameattr = window.facetsLabels &amp;&amp; window.facetsLabels[key] != undefined ? window.facetsLabels[key] : key;
                        var explode = nameattr.split(' /// ');
                        var name = explode[explode.length - 1];

                        var params = {
                            type: {},
                            checked: checked,
                            nameattr: nameattr,
                            name: name,
                            count: content_facet.data[key]
                        };

                        params.type[algoliaConfig.facets[i].type] = true;

                        sub_facets.push(params);
                    }

                    sub_facets.sort(function sortByCheckedThenCount(f1, f2) {
                      if (f1.checked !== f2.checked)
                          return f1.checked ? -1 : 1;

                      if (f2.count !== f1.count)
                          return f2.count - f1.count;

                      return f1.name.localeCompare(f2.name);
                    });

                }
                facets.push({count: sub_facets.length, attribute: algoliaConfig.facets[i].attribute, facet_categorie_name: algoliaConfig.facets[i].label, sub_facets: sub_facets });
            }

            return facets;
        };

        var getPages = function (content) {
            var pages = [];
            if (content.page &gt; 5)
            {
                pages.push({ current: false, number: 1 });
                pages.push({ current: false, number: '...', disabled: true });
            }

            for (var p = content.page - 5; p &lt; content.page + 5; ++p)
            {
                if (p &lt; 0 || p &gt;= content.nbPages)
                    continue;

                pages.push({ current: content.page == p, number: (p + 1) });
            }
            if (content.page + 5 &lt; content.nbPages)
            {
                pages.push({ current: false, number: '...', disabled: true });
                pages.push({ current: false, number: content.nbPages });
            }

            return pages;
        };

                var index_name = render(val);

        var getDate = function () {
            return function (val) {
                var template = algoliaBundle.Hogan.compile(val);

                var renderer = function(context) {
                    return function(text) {
                        return template.c.compile(text, template.options).render(context);
                    };
                };

                var render = renderer(this);

                var timestamp = render(val);


                var date = new Date(timestamp * 1000);

                var days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
                var months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];

                var day = date.getDate();

                if (day == 1)
                    day += "st";
                else if (day == 2)
                    day += "nd";
                else if (day == 3)
                    day += "rd";
                else
                    day += "th";

                return days[date.getDay()] + ", " + months[date.getMonth()] + " " + day + ", " + date.getFullYear();
            }
        };

            function activateAutocomplete()
            {
                $(algoliaConfig.autocomplete.selector).each(function (i) {
                    $(this).typeahead({hint: false}, hogan_objs);

                    var typeaheadInputElement = $(this);
                    var typeaheadInstance = typeaheadInputElement.data('ttTypeahead');

                    var cursorMovedFn = typeaheadInstance.dropdown._callbacks.cursorMoved.sync[0];
                    var selectedFn = typeaheadInstance.dropdown._callbacks.suggestionClicked.sync[0];
                    var enterKeyedFn = typeaheadInstance.input._callbacks.enterKeyed.sync[0];

                    var dropdownContext = typeaheadInstance.dropdown;
                    var inputContext = typeaheadInstance.input;

                    var query = '';

                    // Proxy to keep the value when cursor changes
                    var proxy = function () {
                        query = typeaheadInputElement.val();
                        // Call original method onCursorMoved()
                        cursorMovedFn.apply(dropdownContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    var proxy2 = function () {
                        query = typeaheadInstance.input.query;
                        // Call original method onSuggestionClicked()
                        selectedFn.apply(dropdownContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    var proxy3 = function () {
                        query = typeaheadInputElement.val();
                        // Call original method onEnterKeyed()
                        enterKeyedFn.apply(inputContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    // Replacing original method with our proxy
                    typeaheadInstance.dropdown._callbacks.cursorMoved.sync[0] = proxy;
                    typeaheadInstance.dropdown._callbacks.suggestionClicked.sync[0] = proxy2;
                    typeaheadInstance.input._callbacks.enterKeyed.sync[0] = proxy3;

                    $(this).on('typeahead:selected', function (e, item) {
                        autocomplete = false;
                        instant = false;

                        $(this).val(query);

                        if (typeof item.url !== 'undefined') {
                            window.location.href = item.url;
                        } else {
                            if ($(this).val() != '') {
                                $(this).parents('form:first').submit();
                            }
                        }
                    });
                });
            }

            custom_facets_types["menu"] = function (helper, content, facet) {

                var data = [];

                var all_count = 0;
                var all_unchecked = true;

                var content_facet = content.getFacetByName(facet.attribute);

                if (content_facet == undefined)
                    return data;

                for (var key in content_facet.data)
                {
                    var checked = helper.isRefined(facet.attribute, key);

                    all_unchecked = all_unchecked &amp;&amp; !checked;

                    var name = window.facetsLabels &amp;&amp; window.facetsLabels[key] != undefined ? window.facetsLabels[key] : key;
                    var explode = name.split(' /// ');
                    var nameattr = explode[explode.length - 1];

                    var params = {
                        type: {},
                        checked: checked,
                        nameattr: nameattr,
                        name: name,
                        print_count: true,
                        count: content_facet.data[key]
                    };

                    all_count += content_facet.data[key];

                    params.type[facet.type] = true;

                    data.push(params);
                }

                var params = {
                    type: {},
                    checked: all_unchecked,
                    nameattr: 'all',
                    name: 'All',
                    print_count: false,
                    count: all_count
                };

                params.type[facet.type] = true;

                data.unshift(params);

                return data;
            };

            function updateSliderValues()
            {
                $(".algolia-slider-true").each(function (i) {
                    var min = $(this).attr("data-min");
                    var max = $(this).attr("data-max");

                    var new_min = helper.state.getNumericRefinement($(this).attr("data-attribute"), "&gt;=");
                    var new_max = helper.state.getNumericRefinement($(this).attr("data-attribute"), "&lt;=");

            {
                $(algoliaConfig.autocomplete.selector).each(function (i) {

                    var typeaheadInstance = typeaheadInputElement.data('ttTypeahead');

                    var cursorMovedFn = typeaheadInstance.dropdown._callbacks.cursorMoved.sync[0];
                    var selectedFn = typeaheadInstance.dropdown._callbacks.suggestionClicked.sync[0];
                    var enterKeyedFn = typeaheadInstance.input._callbacks.enterKeyed.sync[0];

                    var dropdownContext = typeaheadInstance.dropdown;
                    var inputContext = typeaheadInstance.input;

                    var query = '';

                    // Proxy to keep the value when cursor changes
                    var proxy = function () {
                        query = typeaheadInputElement.val();
                        // Call original method onCursorMoved()
                        cursorMovedFn.apply(dropdownContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    var proxy2 = function () {
                        query = typeaheadInstance.input.query;
                        // Call original method onSuggestionClicked()
                        selectedFn.apply(dropdownContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    var proxy3 = function () {
                        query = typeaheadInputElement.val();
                        // Call original method onEnterKeyed()
                        enterKeyedFn.apply(inputContext, arguments);
                        typeaheadInputElement.val(query);
                    };

                    // Replacing original method with our proxy
                    typeaheadInstance.dropdown._callbacks.cursorMoved.sync[0] = proxy;
                    typeaheadInstance.dropdown._callbacks.suggestionClicked.sync[0] = proxy2;
                    typeaheadInstance.input._callbacks.enterKeyed.sync[0] = proxy3;

                        instant = false;

                        } else {
                            if ($(this).val() != '') {
                                $(this).parents('form:first').submit();
                            }
                        }
                    });
                });

            custom_facets_types["menu"] = function (helper, content, facet) {

                var data = [];

                var all_count = 0;
                var all_unchecked = true;

                var content_facet = content.getFacetByName(facet.attribute);

                if (content_facet == undefined)
                    return data;

                for (var key in content_facet.data)
                {
                    var checked = helper.isRefined(facet.attribute, key);

                    all_unchecked = all_unchecked &amp;&amp; !checked;

                    var name = window.facetsLabels &amp;&amp; window.facetsLabels[key] != undefined ? window.facetsLabels[key] : key;
                    var explode = name.split(' /// ');
                    var nameattr = explode[explode.length - 1];

                    var params = {
                        type: {},
                        checked: checked,
                        nameattr: nameattr,
                        name: name,
                        print_count: true,
                        count: content_facet.data[key]
                    };

                    all_count += content_facet.data[key];

                    params.type[facet.type] = true;

                    data.push(params);
                }

                var params = {
                    type: {},
                    checked: all_unchecked,
                    nameattr: 'all',
                    name: 'All',
                    print_count: false,
                    count: all_count
                };

                params.type[facet.type] = true;

                data.unshift(params);

                return data;
            };
