       function NumbersOnly(MyField, e, dec)
			{ 			var key;
			var keychar;
			if (window.event)
			   key = window.event.keyCode;
			else if (e)
			   key = e.which;
			else
			   return true;
			keychar = String.fromCharCode(key);
			if ((key==null) || (key==0) || (key==8) ||
				(key==9) || (key==13) || (key==27) )
			   return true;
			else if ((("0123456789").indexOf(keychar) &gt; -1))
			   return true;
			else if (dec &amp;&amp; (keychar == "."))
			   {
			   MyField.form.elements[dec].focus();
			   return false;
			   }
			else
			   return false;
			}

			if(document.formElem.txtname.value == ""){
				alert("Please enter your Name");
				document.formElem.txtname.focus();
				e.preventDefault();
				return false;
			}else{
				if (!document.formElem.txtname.value.match(/^[a-z  A-Z]+$/) &amp;&amp; document.formElem.txtname.value !="")
               {
                    document.formElem.txtname.value="";
                    document.formElem.txtname.focus(); 
                    alert("Please Enter only alphabets");
					return false;
               }	}
			if(document.formElem.txtemail.value == ""){
				alert("Please enter your email address");
				document.formElem.txtemail.focus();
				e.preventDefault();
				return false;	
			} else {
				var pattern = /^([a-zA-Z0-9_.-])+@([a-zA-Z0-9_.-])+\.([a-zA-Z])+([a-zA-Z])+/;				
				if(!pattern.test(document.formElem.txtemail.value)){
					alert("Please enter your valid email address");
					e.preventDefault();
					return false;	
				}
			}			
			if(document.formElem.txtphone.value == "" ){
				alert("Please enter your phone number");
				document.formElem.txtphone.focus();
				e.preventDefault();
				return false;	
			}else{
				var pattern_mobile = /(\+91-?|0)?\d{10}/;
				if(!pattern_mobile.test(document.formElem.txtphone.value)){
					alert("Please enter 10 digits mobile number");
					document.formElem.txtphone.focus();
					e.preventDefault();
					return false;	
				}
			}

      <h1 class="pagename" style="font-size:18px;font-weight:bold;line-height:normal;margin-top:15px;">
       Australia all Set to Launch Coming to Australia Video to Help Visa Applicants
      </h1>
      <div class="mv_text_buttons">
       <div style="width:auto;float:left; font-size:12px; font-weight:bold; text-transform:capitalize;">
        December 2014, 16
       </div>
       <div style="width:auto; float:right; font-size:12px;">
        <a class="print_class" href="#" onclick="EmailPopup();">
         EMAIL
        </a>
        <a class="print_class" href="#" onclick="PrintElem('#aboutusimmi')">
         Print
        </a>
        <span class="mv_textincrease textincrease">
         A+
        </span>
       </div>
      </div>
      <div class="inner_right_col mv_aboutus" id="aboutusimmi">
       <div align="justify" style="width:670px">
        <div>
         <div style="float: left; margin-right: 10px;">
          <img alt="Australia Plans to Launch A Video " australia="" coming="" src="http://www.morevisas.com/media-library/Coming-to-Australia-video-gives-visa-applicants-vital-info.jpg" title="Australia Plans to Launch A Video " to=""/>
         </div>
         <div style="text-align: justify;">
          In a major decision taken to help people who are wishing to move to Australia in 2015, the government has launched a new video termed as Coming to Australia, which provides information with regards to visa application procedure, application requirements and the importance of abiding by the visa conditions.
         </div>
        </div>
        <div style="text-align: justify;">
         <br/>
        </div>
        <div style="text-align: justify;">
         <div>
          According to the Department of Immigration and Border Protection (DIBP), the new video is available on the DIBP website and has information about
          <a href="http://www.morevisas.com/view-services/working-holiday-visa-aus/" target="_blank" title="Australia Working Holiday Visa">
           workingholiday visa
          </a>
          ,
          <a href="http://www.morevisas.com/view-services/student-visa-aus/" target="_blank" title="Australia Student Visa">
           study visa
          </a>
          and
          <a href="http://www.morevisas.com/australia-immigration/business-visa-for-australia/" target="_blank" title="Australia Business Visa">
           business visa
          </a>
          .
         </div>
         <div>
          <br/>
         </div>
        </div>
        <div style="text-align: justify;">
         Sources stated that applicants can now create an ImmiAccount and save time by lodging application for visa online. If applicants are not sure about the right type of visa, then find a Visa Tool present on the website can assist them. Also, prospective applicants can also check the details of the visa at around the time online through Visa Entitlement Verification Online (VEVO), which is a free service.
        </div>
        <div style="text-align: justify;">
         <br/>
        </div>
        <div style="text-align: justify;">
         A spokesman of DIBP stated that applicants should contact an immigration authority in case of any emergency situations such as staying for longer period, to know the details about visa expiry date and visa has been expired.
        </div>
        <div style="text-align: justify;">
         <br/>
        </div>
        <div style="text-align: justify;">
         Besides this, he requests people to seek immigration assistance from a Registered Migration Agent to minimize the red tape and pave the path for successful visa application process.
        </div>
        <div style="text-align: justify;">
         <br/>

function ValidateEmailForm(){
	if(trim(document.emailform.your_name.value) == ""){
		alert("Please enter your name");
		document.emailform.your_name.focus();
		return false;
	}else{
		if (!document.emailform.your_name.value.match(/^[a-z A-Z]+$/) &amp;&amp; document.emailform.your_name.value !="")
	   {
			document.emailform.your_name.value="";
			document.emailform.your_name.focus(); 
			alert("Please Enter only alphabets");
			return false;
	   }	
	}	
	if( trim(document.emailform.your_email.value) == ""){
		alert("Please enter your EmailID");
		document.emailform.your_email.focus();
		return false;
	}else{
		var pattern=/^([a-zA-Z0-9_.-])+@([a-zA-Z0-9_.-])+\.([a-zA-Z])+([a-zA-Z])+/;
		if(!pattern.test(document.emailform.your_email.value)){
			alert("Please enter your Correct EmailID");
			document.emailform.your_email.focus();
			return false;
		}	
	}	
	if( trim(document.emailform.your_friendemail.value) == ""){
		alert("Please enter your Friend EmailID");
		document.emailform.your_friendemail.focus();
		return false;
	}else{
		var pattern=/^([a-zA-Z0-9_.-])+@([a-zA-Z0-9_.-])+\.([a-zA-Z])+([a-zA-Z])+/;
		if(!pattern.test(document.emailform.your_friendemail.value)){
			alert("Please enter your Friend Correct EmailID");
			document.emailform.your_friendemail.focus();
			return false;
		}	
	}	
	if( trim(document.emailform.short_message.value) == ""){
		alert("Please enter your Message");
		document.emailform.short_message.focus();
		return false;
	}
}
