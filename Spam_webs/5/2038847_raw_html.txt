   function arc_submitajax(e){
     				e.preventDefault();
     				var form = this;

				if( typeof( arc_remove_editor_423 ) == 'function' ) {
					arc_remove_editor_423($('form.arc_ajax .editor').attr('id'));
				}
				$(".CommentForm").hide();
				arc_filterform(form);

				$('&lt;input&gt;').attr({
					    type: 'hidden',
					    name: 'arc_token',
					    value: window.arc_token
				}).appendTo(form);

				$.ajax({
					type: 'POST',
					url: form.action,
					data: $(form).serialize(),
					success: function(data, status, xhr){
						var matches = /url=(.*)/.exec(xhr.getResponseHeader("refresh"));
						if( matches != null ) {
							location.replace(matches[1]);
							return;
						}
						$(form).replaceWith(data);
						$(".CommentForm").slideDown();
						if( typeof( arc_editor_423 ) == 'function' ) {
							arc_editor_423($('form.arc_ajax .editor').attr('id'));
						}
					}
				});
			}

		function arc_filterform(form){
			if(typeof( arc_editor_423 ) == 'function' &amp;&amp; $(form).find('.editor').length &gt; 0){
				$(form).find('.editor').each(function(editor_key, editor_value){
					var start = '';
					var end = '';
					var body_html = $('&lt;div&gt;'+$(editor_value).val()+'&lt;/div&gt;');
					var ptags = body_html.find('p').length;
					if(ptags &gt; 0){
						for(var i=0; i&lt;ptags; i++){
							var ptag_html = $.trim(body_html.find('p').eq(i).html());
							if(ptag_html != '' &amp;&amp; ptag_html != '&nbsp;')
							{
								if(start === '')
								{
									start = i; //Get the start index of p tag that is not blank
								}
								end = i; //Get the end index of p tag that is not blank
							}
						}

						$(body_html).find('p').each(function(key, value){
							if(key &lt; start || key &gt; end)
							{
								$(value).remove();
							}
						});
					}
                    arc_remove_editor_423($(editor_value).attr('id'));
					$(editor_value).val($(body_html).html());
                    if($(editor_value).attr('name') != 'chunk_html[]') {
                        arc_editor_423($(editor_value).attr('id'));
                    }
                    else {
                        arc_editor_feature_423($(editor_value).attr('id'));
                    }
				});
			}
			return true;
		}

		$(el).find('a').each(function(){

			$(this).removeAttr('target');

			if(

				&amp;&amp; ( $(this).attr('href').indexOf('http://') == 0 || $(this).attr('href').indexOf('https://') == 0 )

        <a class="floatbox" href="http://cdn.archinect.net/images/1200x/75/7546ahcausk9izt7.jpg" title="">
         <img alt="" border="0" class="Cover" height="303" src="http://cdn.archinect.net/images/514x/75/7546ahcausk9izt7.jpg" title="" width="514"/>

        <div class="Intro">
         <p>
          <i>
           If trees could scream, would we be so cavalier about cutting them down? We might, if they screamed all the time, for no good reason. -- Jack Handey
          </i>
         </p>
        </div>
       </div>
       <p>
       </p>
       <p>
        They're age-old questions: Who will speak for the trees? Can plants feel pain? And what do weeds really want? We can imagine that casting gusts of wind at dandelions might be your wish, but it's probably not theirs. And while you might feel charmed, that four-leaf clover you plucked from the ground is now out of luck. And if youre pulling off petals in a litmus test for your love-life, maybe that daisy loves you not.
        <br/>
        <br/>
        But now, plants have a voice -- at least, they do in Switzerland. In the 1990s, the Swiss Government enshrined the rights of living creatures into their constitution, and in 2008,
        <a href="http://www.ekah.admin.ch/en/">
         The Swiss Federal Ethics Committee
        </a>
        issued guidelines in their report,
        <i>
         The dignity of living beings with regard to plants - Moral consideration of plants for their own sake.
        </i>
        This earned the committee an Ig Noble Prize, from the
        <a href="http://improbable.com/ig/">
         Annals of Improbable Research
        </a>
        , for research that first makes you laugh, but then makes you think.
        <br/>
        <br/>
        Despite their differences, the ethics committee unanimously agreed that arbitrary harm caused to plants is morally impermissible. In other words: it might be fine for farmers to sow crops, but its not so fine if they start casually lopping off the heads of wildflowers on their walk home from the fields. More significantly, particularly for any grant-funded research, is that a majority of the reports contributors also recommended that limits should be placed on the genetic modification of plants.
        <br/>
        <br/>
        Unusually, the religious and scientific communities find themselves on the same side as they protest against this policy. "Dignity is human trait," the devout decry, and humans, they claim, are closer to angels than animals, much less plants. Which seems odd: dont all good plants go to heaven? Similarly, Swiss scientists, now tasked with justifying their genetic research to yet another ethics committee, find the so-called "dignity of plants" to be a humiliating barrier to gaining government approval for their field studies and clinical trials.
        <br/>
        <br/>
        Its easy to empathize with the scientists and smile at the religious, but its hard not be struck by the hubris which unites both their complaints. Evidence from the field of genetic research has proven that even the best intentions can have unintended consequences: genetically modified plants have cross-pollinated with non-GM crops, contaminating species with an all-new form of genetic pollution.
        <br/>
        <br/>
        More widely, humans have amply demonstrated their capacity to catastrophically disrupt ecological flows and destroy ecosystems, often by altering what seems like the most insignificant of variables. When these disasters happen, hands are wrung, brows are beaten, but if the dignity of plants was properly considered in the first place then maybe the world wouldnt suffer so much in the long run.
        <br/>
        <br/>
        Switzerland isn't the only government making a move. Last month, Ecuador
        <a href="http://www.opednews.com/articles/Ecuador-Approves-New-Const-by-Mari-Margil-The-C-090318-265.html">
         amended their constitution
        </a>
        to acknowledge the rights of nature, giving the country's rivers and forests recourse in the court of law. Specifically, nature now has "the right to exist, persist and...regenerate." No doubt, plant migration patterns may shift towards Switzerland, and the world's natural wonders may emigrate to Ecuador.
        <br/>
        <br/>
        Either way, the whole issue of plant-dignity is murky. Fortunately, Klaus Peter Rippe, Chair of the Swiss Ethics Committee, was kind enough to take our call and answer a few of our questions.
        <br/>
        <br/>
        -- Heather Ring, on behalf of
        <a href="http://waywardplants.org">
         The Wayward Plant Registry
        </a>
        , which sets up halfway homes for unwanted plants.
        <br/>
        <br/>
        <b>
         How did the policy come about?
        </b>
        <br/>
        <br/>
        In 1992, there was a very controversial discussion about gene technology and in this debate, the liberal side suggested the constitutional term, "the dignity of living beings." There was a public vote, and two-thirds of the public agreed with this concept. But afterwards a new debate emerged to understand what this term could mean. The constitution made clear that "living beings" included not just humans and animals, but also plants and even so called other organisms.
        <br/>
        <br/>
        <a href="http://files.archinect.com/uploads/ai/aiu_032009_165717.jpg" title="The Decision Tree">
         <img alt="image" border="0" height="388" name="image" src="http://files.archinect.com/uploads/ai/aiu_plantdiagram4.jpg" width="436"/>
        </a>
        <br/>
        <b>
          Click image to enlarge
        </b>
        <br/>
        The Decision Tree, from
        <a href="http://www.ekah.admin.ch/uploads/media/e-Broschure-Wurde-Pflanze-2008.pdf%20">
         <i>
          The Dignity of Living Beings with Regard to Plants
         </i>
         (pdf)
        </a>
        .
        <br/>
        <br/>
        <b>
         The word dignity is very expressive...
        </b>
        <br/>
        <br/>
        Dignity is a very expressive term - especially in the German language [
        <i>
         Wrde der Kreatur
        </i>
        ]. After the Second World War, theologians suggested dignity of creatures as the term to recognize that all living beings created by God have moral standing. Switzerlands concept has these theological roots, but in the 21st century, the debate has to be secular.
        <br/>
        <br/>
        It has been very surprising that today some religious groups have difficulty with the concept of animal and plant dignity. For even if human beings have a special moral standing in the Judeo-Christian tradition, other living beings have a standing as well. This debate is one that occurs more within the religious groups than in the legal and ethical debates of Switzerland. The Swiss Law takes a
        <a href="http://en.wikipedia.org/wiki/Biocentrism">
         biocentric position;
        </a>
        [
        <i>
         the belief that all forms of life are equally valuable and that humanity is not the center of existence
        </i>
        ].
        <br/>
        <br/>
        <img alt="image" height="176" name="image" src="http://files.archinect.com/uploads/ai/aiu_pumpkin.jpg" width="436"/>
        <br/>
        Mickey-shaped pumpkins at EPCOT Centre, via
        <a href="http://www.inhabitat.com/2007/07/26/mickey-mouse-sustainable-farming/">
         Inhabitat
        </a>
        .
        <br/>
        <br/>
        <b>
         Even before this policy came about there have been
         <a href="http://www.youtube.com/watch?v=KmK0bZl4ILM">
          ironic organizations for the rights of plants
         </a>
          as a critique of the rights of animals. As though, once we advocate plants rights, weve crossed the line to absurdity.
        </b>
        <br/>
        <br/>
        In the 19th century, the reaction was: if there are rights for women, there should be rights for animals. But this kind of mockery opened the door for a dialogue. If we base the recognition of rights on certain characteristics, we are unable to draw a strict line between human beings and animals. There are humans who do not or will no longer or never fulfil the criteria and there are animals that do. Equally, its also quite difficult to draw the line between animals and plants - even if you dont use the concept of dignity. But I would like to stress that we dont say that plants are sacred, we only say that we have duties regarding them, and that we have to justify our actions.
        <br/>
        <br/>
        <b>
         Animals are subject to a great deal of anthropomorphism in our culture  but we dont see this happen often with plants. Might this suggest an inability of people to empathize with plants?
        </b>
        <br/>
        <br/>
        There are few exceptions: the movies
        <a href="http://www.youtube.com/watch?v=BGRN39oifsE">
         Little Shop of Horrors
        </a>
        or Attack of the Killer Tomatoes 
        <br/>
        <br/>
        <b>
         They only come alive when they want to eat us! But there's also the classic BBC children's show,
         <a href="http://www.youtube.com/watch?v=AN9GW_KO-XA">
          The Herbs
         </a>
         , where they merged plants with animals in order to animate them.
        </b>
        <br/>
        <br/>
        Even if they cannot speak, they are still living beings. And if living beings have a moral standing, we have to ask, "Can we harm them?" That's the question.
        <br/>
        <br/>
        <img alt="image" height="201" name="image" src="http://files.archinect.com/uploads/ai/aiu_diznity.jpg" width="436"/>
        <br/>
        LOL plant,
        <a href="http://waywardplants.org">
         The Wayward Plant Registry
        </a>
        . LOL cats + plants can be found
        <a href="http://icanhascheezburger.com/?s=plant">
         here
        </a>
        .
        <br/>
        <br/>
        <b>
         We are very capable of separating one person from the human species, but its hard to think about individual plants as separate from a larger ecosystem. How does this policy work within a larger set of environmental strategies?
        </b>
        <br/>
        <br/>
        You are completely correct, because one of the questions is: What is the moral objective - the individual plant or the ecosystem? First we look at the moral objective towards living beings, and second we consider the role of plants within ecosystems and landscapes. The first is the intrinsic value of a plant, the second the instrumental value, an indirect moral duty. Its not a contradiction  there are two sides we have to consider.
        <br/>
        <br/>
        <b>
         Does this provide an ethical conflict -- or at least, a logistical problem -- for vegetarians?
        </b>
        <br/>
        <br/>
        There have been a number of discussions in the vegetarian community  its a very sensible field. Actually, vegetarians are used to omnivores challenging them - "If you don't eat animals, you cannot eat plants either - because you don't know whether or not they will suffer." But what is there left to eat? This discussion extends to questions such as  under what condition should tomatoes grow? This has become not just a question of eating - but a question of the production of foods, especially genetically modified food and the food industry as a whole. The difference between animals and plants is that we know animals can suffer, and with plants it's only a speculation. And so if we have to choose - we choose eating plants, not animals. (Or not too many animals, if you're not a vegetarian.)
        <br/>
        <br/>
        <img alt="image" height="228" name="image" src="http://files.archinect.com/uploads/ai/aiu_Gmsleeper.jpg" width="436"/>
        <br/>
        GM vegetables ...
        <a href="http://www.youtube.com/watch?v=VfrShu_Lp2A">
         more here.
        </a>
        <br/>
        <br/>
        <b>
         What would constitute the humiliation of plants? Maybe topiaries pruned to look like characters? That must be humiliating for the shrub.
        </b>
        <br/>
        <br/>
        We had a letter from the States regarding
        <a href="http://waywardplants.org/records/0001/2/">
         Christmas trees
        </a>
        . But this is humiliation in more of a metaphorical sense.
        <br/>
        <br/>
        <b>
         Could you give some examples of how this policy might affect our behavior?
        </b>
        <br/>
        <br/>
        This is the most difficult question - the concrete cases. If you look at all the discussions in the ethics community, in three years we discussed many examples and cases and theres not very much agreement. The constitutional concept of dignity of living beings originally roots in the debate on genetic modification of animals and plants. One very controversial debate was if it is morally justifiable to insert a so called
        <a href="http://en.wikipedia.org/wiki/Terminator_technology">
         terminator technology
        </a>
        into plants. [
        <i>
         The genetic modification of plants, restricting their ability to reproduce by rendering them infertile -- often in the interest of plant breeding copyrights and corporate patents.
        </i>
        ] We have a responsibility through education and research to define a different moral role for living beings  as opposed to one that perceives non-humans as purely instruments.
        <br/>
        <br/>
        <b>
         Are there any implications for the fields of architecture and landscape architecture?
        </b>
        <br/>
        <br/>
        There is a certain responsibility that comes with using a living being for your projects. The concept - the dignity of living beings - is there to remind people that living beings are not "things  and one must handle them as materials in a way that's different from non-living things, like stone. For instance, it would be morally questionable to use a plant in such a way that it would die shortly after it was planted.
        <br/>
        <br/>
        <b>
         Are there any architects on the ethics committee?
        </b>
        <br/>
        <br/>
        We have philosophers and biologists and so on. But we don't have an architect - because the mandate of the ethics committee is focused on the ethical evaluation of biotechnology, and I assume very few architects use biotechnology in their work.
        <br/>
        <br/>
        <img alt="image" height="248" name="image" src="http://files.archinect.com/uploads/ai/aiu_bio-zoo.jpg" width="436"/>
        <br/>
        Transgenic Zoo by
        <a href="http://www.yeadon.net/yeadon/support/projects/projects.html">
         Peter Yeadon
        </a>
        .
        <br/>
        <br/>
        <img alt="image" height="195" name="image" src="http://files.archinect.com/uploads/ai/aiu_NatalieJeremijenko2.jpg" width="436"/>
        <br/>
        Natalie Jeremijenko's
        <a href="http://www.nyu.edu/projects/xdesign/onetrees/">
         One Trees
        </a>
        project, which cloned 1000 trees and planted them in the Bay Area to monitor their interactions with their environments.
        <br/>
        <br/>
        <img alt="image" height="243" name="image" src="http://files.archinect.com/uploads/ai/aiu_biopresence_image.jpg" width="436"/>
        <a href="http://www.biopresence.com/">
         <br/>
         Biopresence
        </a>
        creates Human DNA trees by transcoding the essence of a human being within the DNA of a tree in order to create "Living Memorials" or "Transgenic Tombstones."
        <br/>
        <br/>
        <img alt="image" height="174" name="image" src="http://files.archinect.com/uploads/ai/aiu_Fabtimedignity.jpg" width="436"/>
        <br/>
        <a href="http://www.archinode.com/bienal.html">
         Fab Tree Hab
        </a>
        from H.E.D. [Human Ecology Design].
        <br/>
        <br/>
        <b>
         Well  there have been proposals to grow our own homes 
        </b>
        <br/>
        <br/>
        That's true - I'm a great fan of science fiction journals - and there are stories which use the concept of living plants as buildings. The question would be -- is this symbiosis, or are we a parasite of the plant? If we accept that plants have a moral standing because they are living beings we have to evaluate carefully if such projects affect them in a way that could not be justified. However, justifiable projects are not unthinkable.
        <br/>
        <br/>
        The Wayward Plant Registry is currently part of
        <a href="http://cca-actions.org/">
         Actions: What You Can Do With The City
        </a>
        at the
        <a href="http://cca.qc.ca/">
         Canadian Centre for Architecture, Montreal
        </a>
        . There will be a Halfway Home for Wayward Plants at the Barbican this summer in conjunction with
        <a href="http://www.barbican.org.uk/artgallery/event-detail.asp?ID=8908">
         Radical Nature, Art and Architecture for a Changing Planet 19692009.
        </a>

    <script type="text/javascript">
     $(document).ready(function(){

		$('#SectionSelect').hide();
		var SectionSelectOpen = false;

		$('#SectionSelectToggle').click(function(){
			if (SectionSelectOpen == false) {
				$('#SectionSelect').slideDown('fast');
				$('#SectionSelect').fadeTo('fast', 1);
				SectionSelectOpen = true;
			}
		});

		$('#SectionSelectClose').click(function(){
			$('#SectionSelect').slideUp('fast');
			$('#SectionSelect').fadeTo('fast', 0);
			SectionSelectOpen = false;
		});

		$('#SectionSelect .Checkbox.Primary').click(function () {
			$(this).parents('fieldset:eq(0)').find(':checkbox').attr('checked', this.checked);
		});

		$('#SectionSelect .Checkbox.Secondary').click(function () {
			$(this).parents('fieldset:eq(0)').find('.Checkbox.Primary').removeAttr('checked');
			selectAllCB();
		});

		selectAllCB();

	});

	function selectAllCB(){
		if($('#SectionSelect .Checkbox.Secondary').length == $('#SectionSelect .Checkbox.Secondary:checked').length)
			$('#SectionSelect .Checkbox.Primary').attr('checked','checked');
		if($('#SectionSelect .Checkbox.Primary').is(':checked'))
			$('#SectionSelect .Checkbox.Secondary').attr('checked','checked');
	}
    </script>
