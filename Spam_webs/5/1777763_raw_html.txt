    <td align="center" colspan="3" valign="top">
     <script type="text/javascript">
      $(document).ready(startRatings)
        var scoreSum = 5
        var totalRatings = 1
        function startRatings () {
          doStars()
        }
        function doStars () {
          var avg = (totalRatings == 0) ? 0 : scoreSum / totalRatings
          avgRating = avg.toFixed(1)
          $('#score').html(avgRating)
          $('#totalRatings').html(totalRatings)
          var stars = Math.round(avgRating)
          for (s = 0;s &lt; 5;s++)
            if (s &lt; stars)
              $('#ratings-star'+s).css('background', 'url(http://popstrip.com/i/star1.png)')
            else

          $.getJSON("http://popstrip.com/ratings.php", {
                  "s": "second-contact", 
                  "score": score,
                  "rate": 1
              }, rateStripCB)
        }
        function rateStripCB (r) {
          // if (typeof r.success == 'undefined') {
          if (r.error) {
            alert(r.msg)
            return
          }

            $('#rating-msg').fadeOut(2000, removeRatingsMsg)
          }
          scoreSum = r.scoreSum
          totalRatings = r.totalRatings
          doStars()
        }

          $('#rating-msg').remove()
