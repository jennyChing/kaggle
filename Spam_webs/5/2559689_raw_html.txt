    <script type="text/javascript">
     $(function () {
      var duration = 0;
      var selectorImagePath = "/images/global/select-down-black.png";
      var divPreferencesPrefix = "div-customer-preferences-";
      var listPreferencesPrefix = "list-customer-preferences-";
      var inputPreferencesPrefix = "input-customer-preferences-";

      // Affichage/masquage des prfrences
      $("#customer-preferences").click(function () {
        var elem = $("#container-customer-preferences");
        elem.css("width", $("#basket_zone").width()).css({
          "left": $("#content").width() - elem.width(),
          "display": elem.css("display") === "none" ? "block" : "none"
        });
      });

      // Affichage/masquage des listes droulantes
      $("div[id^='" + divPreferencesPrefix + "']").click(function () {
        var type = $(this).attr("id").replace(divPreferencesPrefix, "");
        var list = $("#" + listPreferencesPrefix + type);
        if (list.is(':visible')) {
          closeAllLists();
          return false;
        }
        closeAllLists();
        list.slideDown(duration);
      });

      // Changement des infos contenues dans les div de prfrence
      $("ul[id^='" + listPreferencesPrefix + "'] &gt; li").click(function () {
        var type = $(this).parent().attr("id").replace(listPreferencesPrefix, "");
        var divContainer = $("#" + divPreferencesPrefix + type);
        $("#" + inputPreferencesPrefix + type).val($(this).attr("id_" + type));
        divContainer.css("background-image", (type === "country" ? ($(this).css("background-image") + ",") : "") + "url('" + selectorImagePath + "')");
        divContainer.text($(this).text());
        $("#" + listPreferencesPrefix + type).toggle(duration);
      });

      // Chargement dynamic des libells lors du potentiel changement de langue
      $("#" + listPreferencesPrefix + "language").click(function () {
        var arrayTraduction = {"Delivery":{"de":"Lieferung","en":"Delivery","es":"Entrega","fr":"Livraison","it":"Consegna","pt":"Entrega"},"Currency":{"de":"W\u00e4hrung","en":"Currency","es":"Divisa","fr":"Devise","it":"Valuta","pt":"Moeda"},"Language":{"de":"Sprache","en":"Language","es":"Lengua","fr":"Langue","it":"Lingua","pt":"L\u00edngua"},"Change":{"de":"Aktualisieren","en":"Change","es":"Actualizar","fr":"Mettre \u00e0 jour","it":"Cambiare","pt":"Atualizar"}};
        var codeLangue;

        switch ($("#" + inputPreferencesPrefix + "language").val()) {
          case "4" :
            codeLangue = "fr";
            break;
          case "20" :
            codeLangue = "en";
            break;
          case "21" :
            codeLangue = "pt";
            break;
          case "22" :
            codeLangue = "es";
            break;
          case "23" :
            codeLangue = "it";
            break;
          case "24" :
            codeLangue = "de";
            break;
          default :
            codeLangue = "fr";
        }

        var spanElements = $("#container-customer-preferences &gt; div &gt; span");
        $(spanElements[0]).html(arrayTraduction.Delivery[codeLangue] + " : ");
        $(spanElements[1]).html(arrayTraduction.Currency[codeLangue] + " : ");
        $(spanElements[2]).html(arrayTraduction.Language[codeLangue] + " : ");
        $(spanElements[3]).children().val(arrayTraduction.Change[codeLangue]);
      });

      function closeAllLists() {
        $("ul[id^='" + listPreferencesPrefix + "']").slideUp(0);
      }
    });
    </script>

    date.setTime(date.getTime() + (expire * 24 * 60 * 60 * 1000));

    document.cookie = name + "=" + value + expires + "; path=/";

    if (document.cookie.length &gt; 0) {
      var start = document.cookie.indexOf(name + "=");

        var end = document.cookie.indexOf(";", start);
