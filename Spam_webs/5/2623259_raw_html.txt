    } else {
        var index = -1;

        try {
            var parts = resp.msg.split(' - ',2);
            if (parts[1]==undefined){
                msg = resp.msg;
            } else {
                i = parseInt(parts[0]);
                if (i.toString() == parts[0]){
                    index = parts[0];
                    msg = parts[1];
                } else {
                    index = -1;
                    msg = resp.msg;
                }
            }
        } catch(e){
            index = -1;
            msg = resp.msg;

            if (index== -1){
                $('#mce-'+resp.result+'-response').show();
                $('#mce-'+resp.result+'-response').html(msg);

                var input_id = '#mc_embed_signup';

                if (ftypes[index]=='address'){

                    f = $(input_id).parent().parent().get(0);

                    input_id = '#mce-'+fnames[index]+'-month';
                    f = $(input_id).parent().parent().get(0);
                } else {
                    input_id = '#mce-'+fnames[index];
                    f = $().parent(input_id).get(0);

                    $(input_id).focus();
                } else {
                    $('#mce-'+resp.result+'-response').show();
                    $('#mce-'+resp.result+'-response').html(msg);
                }
            }

            $('#mce-'+resp.result+'-response').show();

       <p>
        <strong>
         How to make a DIY Mothers Day Card using a Polka Dot Stencil:
        </strong>
       </p>
       <p>
        Using painters tape, tape off the word MOM. I made my Ms about 4 polka dot circles tall. Its best to start in the center with the O and then work left and right.
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_2.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Cut a piece of cardstock to fit your envelope to be your card. Check out my tips on
        <a href="http://www.merrimentdesign.com/chicago-skyline-notecards-and-lined-envelopes.php">
         how to score and fold paper cards
        </a>
        .
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_3.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Peel the stencil off of the backer sheet very carefully from the backer and all surfaces so you dont tear it. Smooth thoroughly onto the surface. Be sure to save the backing for later. Learn more about
        <a href="http://www.plaidonline.com/blog/post/2014/01/23/learning-how-to-use-mod-podge-peel-stick-stencils">
         working with adhesive stencils
        </a>
        .
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_4.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Paint the polka dot stencil.
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_5.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Peel off the painters tape.
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_6.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Paint the M-O-M polka dots.
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_7.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        Peel up the stencils.
       </p>
       <p>
        <img alt="DIY Mother's Day Card using a Polka Dot Stencil" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_8.jpg" title="DIY Mother's Day Card"/>
       </p>
       <p>
        There are six basic patterns to choose from in the Mod Podge Rocks adhesive stencils line:
       </p>
       <p>
        <img alt="Mod Podge Rocks stencils" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_1.jpg" title="Modge Podge Glitter Patterned Mother's Day Card"/>
       </p>
       <p>
        I used paint for these stencils but Amy also figured out a fresh and easy way to add glitter and sand patterns to projects using Mod Podge and these stencils so fans of glitter and sand, be sure to check out her tutorial on
        <a href="http://www.plaidonline.com/blog/post/2014/01/23/learning-how-to-use-mod-podge-peel-stick-stencils">
         learning how to use Mod Podge Peel and Stick stencils with Mod Podge and glitter
        </a>
        .
       </p>
       <p>
        <img alt="Mod Podge Rocks stencils" class="imgpostwidth" src="http://cdn.merrimentdesign.com/images/diy-mothers-day-card-using-a-polka-dot-stencil-and-giveaway_15.jpg" title="Modge Podge Glitter Patterned Mother's Day Card"/>

                                        fields[i] = this;
                                        i++;
                                    });
                                $(':hidden', this).each(
                                    function(){
                                        var bday = false;
                                        if (fields.length == 2){
                                            bday = true;
                                            fields[2] = {'value':1970};//trick birthdays into having years
                                        }
                                    	if ( fields[0].value=='MM' &amp;&amp; fields[1].value=='DD' &amp;&amp; (fields[2].value=='YYYY' || (bday &amp;&amp; fields[2].value==1970) ) ){
                                    		this.value = '';
									    } else if ( fields[0].value=='' &amp;&amp; fields[1].value=='' &amp;&amp; (fields[2].value=='' || (bday &amp;&amp; fields[2].value==1970) ) ){
                                    		this.value = '';
									    } else {
	                                        this.value = fields[0].value+'/'+fields[1].value+'/'+fields[2].value;
	                                    }
                                    });
                            });
                        return mce_validator.form();
                    }, 
                    success: mce_success_cb
                };
      $('#mc-embedded-subscribe-form').ajaxForm(options);
      
      
    });
}
function mce_success_cb(resp){
    $('#mce-success-response').hide();
    $('#mce-error-response').hide();
    if (resp.result=="success"){
        $('#mce-'+resp.result+'-response').show();
        $('#mce-'+resp.result+'-response').html(resp.msg);
        $('#mc-embedded-subscribe-form').each(function(){
            this.reset();
    	});
    } else {
        var index = -1;
        var msg;
        try {
            var parts = resp.msg.split(' - ',2);
            if (parts[1]==undefined){
                msg = resp.msg;
            } else {
                i = parseInt(parts[0]);
                if (i.toString() == parts[0]){
                    index = parts[0];
                    msg = parts[1];
                } else {
                    index = -1;
                    msg = resp.msg;
                }
            }
        } catch(e){
            index = -1;
            msg = resp.msg;
        }
        try{
            if (index== -1){
                $('#mce-'+resp.result+'-response').show();
                $('#mce-'+resp.result+'-response').html(msg);            
            } else {
                err_id = 'mce_tmp_error_msg';
                html = '&lt;div id="'+err_id+'" style="'+err_style+'"&gt; '+msg+'&lt;/div&gt;';
                
                var input_id = '#mc_embed_signup';
                var f = $(input_id);
                if (ftypes[index]=='address'){
                    input_id = '#mce-'+fnames[index]+'-addr1';
                    f = $(input_id).parent().parent().get(0);
                } else if (ftypes[index]=='date'){
                    input_id = '#mce-'+fnames[index]+'-month';
                    f = $(input_id).parent().parent().get(0);
                } else {
                    input_id = '#mce-'+fnames[index];
                    f = $().parent(input_id).get(0);
                }
                if (f){
                    $(f).append(html);
                    $(input_id).focus();
                } else {
                    $('#mce-'+resp.result+'-response').show();
                    $('#mce-'+resp.result+'-response').html(msg);
                }
            }
        } catch(e){
            $('#mce-'+resp.result+'-response').show();
            $('#mce-'+resp.result+'-response').html(msg);
        }
    }
}
