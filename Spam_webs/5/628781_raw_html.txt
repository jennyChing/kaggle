                        $(this).css({
                            'background-color' : '#FFC8C8'
                        });

                            'background-color' : '#FFC8C8'
                        });

                        return false;

                        $(this).css({
                            'background-color' : '#ffffff'

                    };
                });

                if (return_false) {
                    return false;
                } else {
                    return true;
                }
            }
