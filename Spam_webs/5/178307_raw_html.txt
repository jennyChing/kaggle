		$('input[name=email]').focus(function() {
    		if($(this).val() == 'Enter E-Mail Address') $(this).val('');
		}).blur(function() {
    		if( $(this).val() == '') $(this).val('Enter E-Mail Address');
		});

	    $('#enews_signup_btn').click(function() { 
			$('#enews_signup_error').hide();
			$('#enews_signup_thanks').hide();
			 
			var email = $("input#email").val();
			var emailReg = /^([\w-\.]+@([\w-]+\.)+[\w-]{2,4})?$/;
			
			if(!emailReg.test(email)){
				$('#enews_signup_error').show();
				$('#enews_signup_error').html('&lt;span class="error"&gt;&lt;strong&gt;Please enter a valid email address&lt;/strong&gt;&lt;/span&gt;');
			
				$('input#email').focus();

				return false;
			}
			
			//otherwise, submit the data
			var dataString = 'email='+ email;	
		
			$.ajax({
				type: "POST",
				url: "/processEmailSignup.asp",
				data: dataString,
				success: function(data) {
					//show thank you; hide other divs
					$('#enews_signup_main').hide();
					$('#enews_signup_error').hide();
					$('#enews_signup_thanks').show();
					
					//unbind click event
				}
			  });		
				
			return false;
			});
				
  });
