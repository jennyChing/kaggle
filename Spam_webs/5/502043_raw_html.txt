     $(document).ready(function() {
        var url = "/mybookshelf/songwritermusic/";

        createDropDown();

        $(".dropdown dt a").click(function() {
            $(".dropdown dd ul").toggle();
            return false;
        });

        $(document).bind('click', function(e) {
            var $clicked = $(e.target);
            if (!$clicked.parents().hasClass("dropdown"))
                $(".dropdown dd ul").hide();
        });

        $(".dropdown dd ul li a").click(function() {
            var text = $(this).html();
            $(".dropdown dt a").html(text);
            $(".dropdown dd ul").hide();

            var source = $("#source");
            source.val($(this).find("span.value").html())
            window.location.replace("/language?lan=" + source.val() + "&amp;red=" + url);
            return false;
        });

        function createDropDown() {
            var source = $("#source");
            var selected = source.find("option:selected");
            var options = $("option", source);

            $("#drop-down").append('&lt;dl id="target" class="dropdown"&gt;&lt;/dl&gt;')
            $("#target").append('&lt;dt&gt;&lt;a href="#"&gt;' + selected.text() +
                '&lt;span class="value"&gt;' + selected.val() +
                '&lt;/span&gt;&lt;/a&gt;&lt;/dt&gt;')
            $("#target").append('&lt;dd&gt;&lt;ul&gt;&lt;/ul&gt;&lt;/dd&gt;')

            options.each(function() {
                $("#target dd ul").append('&lt;li&gt;&lt;a href="#"&gt;' +
                    $(this).text() + '&lt;span class="value"&gt;' +
                    $(this).val() + '&lt;/span&gt;&lt;/a&gt;&lt;/li&gt;');
            });
        }
    });
