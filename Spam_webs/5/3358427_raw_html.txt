    function changeTab(day, language)
	{
		
		//var maxDays = document.getElementById('maxDay').innerHTML;
		var maxDays = 14;
		
		// hide all divs
		for (var i=1; i&lt;=maxDays; i++)
		{
			
			if (document.getElementById('webcast_files_'+language+'_'+i))
			{
				document.getElementById('webcast_files_'+language+'_'+i).style.display = 'none';
			}
			if (document.getElementById('day_tab_'+language+'_'+i))
			{
				document.getElementById('day_tab_'+language+'_'+i).className = 'webcast-file-tab';
			}
			
		}
		
		if (document.getElementById('webcast_files_'+language+'_'+day))
		{
			document.getElementById('webcast_files_'+language+'_'+day).style.display = 'block';
		}
		if (document.getElementById('day_tab_'+language+'_'+day))
		{
			document.getElementById('day_tab_'+language+'_'+day).className = 'webcast-file-tab-selected';
		}
		
	}
	
	function changeLanguage(language)
	{
		
		var languages = document.getElementById('languages').innerHTML.split("|");
		
		var selectedLangIndex = 0;
		for (var i=0; i&lt;languages.length; i++)
		{
			if (language==languages[i]) { selectedLangIndex = i; }
			if (document.getElementById('language_tab_'+languages[i])) 
			{
				document.getElementById('language_tab_'+languages[i]).style.display = 'none';				
			}
			
		}
		
		if (document.getElementById('language_tab_'+language)) 
		{
			// reset language select boxes
			document.getElementById('language_select_'+language).selectedIndex = selectedLangIndex;
			//
			document.getElementById('language_tab_'+language).style.display = 'block';
		}
	
	}
	
	function postBack(file_id)
	{
		document.getElementById('file'+file_id).submit();
	}
