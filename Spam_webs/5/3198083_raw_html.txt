	function utf8_decode ( str_data ) {
	    // Converts a UTF-8 encoded string to ISO-8859-1  
	    // 
	    // version: 1008.1718
	    // discuss at: http://phpjs.org/functions/utf8_decode    // +   original by: Webtoolkit.info (http://www.webtoolkit.info/)
	    // +      input by: Aman Gupta
	    // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
	    // +   improved by: Norman "zEh" Fuchs
	    // +   bugfixed by: hitwork    // +   bugfixed by: Onno Marsman
	    // +      input by: Brett Zamir (http://brett-zamir.me)
	    // +   bugfixed by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
	    // *     example 1: utf8_decode('Kevin van Zonneveld');
	    // *     returns 1: 'Kevin van Zonneveld'    
		var tmp_arr = [], i = 0, ac = 0, c1 = 0, c2 = 0, c3 = 0;
	    
	    str_data += '';
	    
	    while ( i &lt; str_data.length ) {        c1 = str_data.charCodeAt(i);
	        if (c1 &lt; 128) {
	            tmp_arr[ac++] = String.fromCharCode(c1);
	            i++;
	        } else if ((c1 &gt; 191) &amp;&amp; (c1 &lt; 224)) {            c2 = str_data.charCodeAt(i+1);
	            tmp_arr[ac++] = String.fromCharCode(((c1 &amp; 31) &lt;&lt; 6) | (c2 &amp; 63));
	            i += 2;
	        } else {
	            c2 = str_data.charCodeAt(i+1);            c3 = str_data.charCodeAt(i+2);
	            tmp_arr[ac++] = String.fromCharCode(((c1 &amp; 15) &lt;&lt; 12) | ((c2 &amp; 63) &lt;&lt; 6) | (c3 &amp; 63));
	            i += 3;
	        }
	    } 
	    return tmp_arr.join('');
	}

        <h6>
         Product Profile
        </h6>
        <div class="content">
         ACI Engineers offers Scroll Compressors manufactured by Anest Iwata, Japan and marketed in India under the joint venture between Anest Iwata and Sumi Motherson Group. The venture is popularly recognised as Anest-Iwata Motherson Ltd. These oil free scroll compressors are extensively used for industrial air applications.
         <br/>
         <br/>
         ACI Engineers is an established company having an experience of 12 years. It is the authorised stockist and distributor for Festo pneumatic products and Anest-Iwata air compressors like lubricated air compressors, oil free air compressors, scroll compressors. The Festo products include cylinders, valves, interface devices, converters, transducers, control units, accessories and service units.
