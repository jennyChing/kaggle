		function shareToFeed(){
			if( !FB ){
				fbshare_w( );
				return;
			}
			
			var doPostFeed = function(){
				try{
					var link = document.location.href;
					if( !link || link.length == 0 ){
						fbshare_w( );
						return;
					}
					
					var title = document.title;
					if( !title || title.length == 0 ){
						fbshare_w( );
						return;
					}
					
					var description = "";
					var metas = document.getElementsByName("description");			
					if (metas.length &gt; 0) {
						description = metas[0].getAttribute("content");
					}
					
					// calling the API ...
					var ref = document.location.host.replace(/\./gi,"");
					var obj = {
						method: 'feed',
						link:  link,
						name: title.replace(/[^\u0020-\u007e]/gi,""),
						description: description.replace(/[^\u0020-\u007e]/gi,""),
						ref: ref
					};

					var callback = function(response){}
					
					FB.ui(obj, callback);
				
				}catch(e){
					fbshare_w();
				}
			}
			try{
				FB.getLoginStatus(
					function(response) {
						if (response.status === 'connected') {
							doPostFeed();
						}
						else{
							fbshare_w();
						}
					}
				);
			}catch(err){}
			
			return false;
		}
