      function isEDT(dt) {
         dt = dt || new Date(); // default to current date
         var month = dt.getUTCMonth(); // get month in UTC
         if (month &gt; 2 &amp;&amp; month &lt; 10) return true; // in EDT if Apr to Oct
         if (month &lt; 2 || month &gt; 10) return false; // not in EDT if Jan,Feb,Dec

         var day = dt.getUTCDay(); // get day of week in UTC

          <!-- T24708711 for MLB_20140913_CLE@DET -->
          <p>
           DETROIT (AP)
           <a class="playerLink" href="/mlb/players/playerpage/1660422/alex-avila">
            Alex Avila
           </a>
           has had a tough season at the plate and
           <a class="playerLink" href="/mlb/players/playerpage/10813/torii-hunter">
            Torii Hunter
           </a>
           has had a tough season in the field.
          </p>
          <p>
           However, Avila hit a two-out, two-run homer in the eighth and Hunter made an outstanding defensive play to help the
           <a href="/mlb/teams/page/DET/detroit-tigers">
            Detroit Tigers
           </a>
           beat the
           <a href="/mlb/teams/page/CLE/cleveland-indians">
            Cleveland Indians
           </a>
           5-4 Saturday night to retain their slim lead in the AL Central.
          </p>
          <p>
           The Tigers remained a half-game ahead of Kansas City, which topped Boston 7-1.
          </p>
          <p>
           Avila sent a full-count pitch from reliever
           <a class="playerLink" href="/mlb/players/playerpage/1734606/bryan-shaw">
            Bryan Shaw
           </a>
           (5-4) into the right-field stands for his 11th home run.
          </p>
          <p>
           ''It was up enough to get a good swing on it,'' said Avila, who is hitting .222 with 11 home runs and 47 RBIs.
          </p>
          <p>
           He hit a slider that got too much of the plate.
          </p>
          <p>
           ''Just a bad pitch . bad location,'' said Shaw, who leads the American League with 73 appearances this season.
          </p>
          <p>
           Hunter, Detroit's right fielder, saved two runs in the top of eighth with a two-out, diving catch of
           <a class="playerLink" href="/mlb/players/playerpage/1945612/yan-gomes">
            Yan Gomes
           </a>
           ' short fly ball with runners on second and third.
          </p>
          <p>
           ''Just a little bleeder off the end of the bat, just tried to do everything I could right there to get under that ball,'' Hunter said. ''It was sinking pretty fast but I just said, `Feet don't fail me now.'''
          </p>
          <p>
           Hunter was a perennial gold glove center fielder in his younger days, but this season - at age 39 - ranks at the bottom of AL right fielders in several defensive statistical rankings.
          </p>
          <p>
           ''Torii Hunter catches that ball, saves two runs. A lot more goes into it than just one pitch,'' Cleveland manager Terry Francona said.
          </p>
          <p>
           <a class="playerLink" href="/mlb/players/playerpage/367942/victor-martinez">
            Victor Martinez
           </a>
           hit his career-best 31st home run for Detroit.
          </p>
          <p>
           <a class="playerLink" href="/mlb/players/playerpage/1232125/joba-chamberlain">
            Joba Chamberlain
           </a>
           (2-5) pitched an inning of scoreless relief and
           <a class="playerLink" href="/mlb/players/playerpage/21655/joe-nathan">
            Joe Nathan
           </a>
           worked the ninth for his 31st save.
          </p>
          <p>
           <a class="playerLink" href="/mlb/players/playerpage/1103277/michael-brantley">
            Michael Brantley
           </a>
           homered for Cleveland.
          </p>
          <p>
           <a class="playerLink" href="/mlb/players/playerpage/490430/mike-aviles">
            Mike Aviles
           </a>
           ' two-out, two-run double into the right-field corner in the fifth gave Cleveland a 4-3 lead.
          </p>
          <p>
           Brantley it a two-run homer in the first against Kyle Lobstein, connecting on the 10th pitch of the at-bats. Brantley has 19 home runs overall this season, and is batting .400 (30 for 75) against Detroit this season with three home runs and 15 RBIs.
          </p>
          <p>
           The Tigers tied it at 2 in the third on
           <a class="playerLink" href="/mlb/players/playerpage/489854/ian-kinsler">
            Ian Kinsler
           </a>
           's RBI single and a sacrifice fly by Hunter.
          </p>
          <p>
           Martinez hit a solo homer in the fourth off Danny Salazar.
          </p>
          <p>
           INSTANT REPLAY REVERSAL
          </p>
          <p>
           Ausmus won an instant replay challenge in the top of the eighth.
           <a class="playerLink" href="/mlb/players/playerpage/1741006/jb-shuck">
            J.B. Shuck
           </a>
           was originally ruled safe at first on an infield single to shortstop
           <a class="playerLink" href="/mlb/players/playerpage/1599178/andrew-romine">
            Andrew Romine
           </a>
           . But after a 39-second review, the ruling on the field was overturned.
          </p>
          <p>
           VICTOR MARTINEZ HITS INDIANS
          </p>
          <p>
           Victor Martinez is hitting .343 (23 for 67) this season with seven home runs and 15 RBIs against Cleveland, one of his former teams, and .362 (85 for 236) with 12 homers and 54 RBIs against the Indians in his career.
          </p>
          <p>
           TRAINER'S ROOM
          </p>
          <p>
           Indians:
           <a class="playerLink" href="/mlb/players/playerpage/448416/ryan-raburn">
            Ryan Raburn
           </a>
           started in right field but left after the third because of a hyperextended left knee. Francona said he would see how Raburn feels on Sunday morning.
          </p>
          <p>
           Tigers: RHP
           <a class="playerLink" href="/mlb/players/playerpage/533212/anibal-sanchez">
            Anibal Sanchez
           </a>
           , on the disabled list with a strained right pectoral, played long toss from a distance of 120 feet before the game and said he felt ''very good''. Manager Brad Ausmus said if Sanchez doesn't have a setback, he will long toss again on Sunday and throw off the mound in the bullpen on Monday.
          </p>
          <p>
           UP NEXT
          </p>
          <p>
           The wrapup of the three-game weekend series Sunday matches Tigers RHP
           <a class="playerLink" href="/mlb/players/playerpage/530362/justin-verlander">
            Justin Verlander
           </a>
           (13-12, 4.82) against
           <a class="playerLink" href="/mlb/players/playerpage/1852623/trevor-bauer">
            Trevor Bauer
           </a>
           (5-8, 4.12). Verlander is 3-1 in his last four starts despite a 5.11 ERA.
          </p>

    function updateOneLineComments(){
        
        //console.log("Checking Comments");
        $(".fyre-comment-wrapper").each(function() {
            if ($(this).find(".fyre-comment").css('display') != 'none'){
                var commentText = $(this).find(".fyre-comment").html();
                if (commentText != null){
                    var loopCtr = 0;
                    // for IE8
                    while ( (commentText.indexOf("&lt;P&gt;") !== -1) &amp;&amp; (loopCtr &lt; 100) ){
                        commentText = commentText.replace("&lt;P&gt;",' ');
                        commentText = commentText.replace("&lt;/P&gt;",' ');
                        loopCtr++;
                    }
                    //for other browsers
                    while ( (commentText.indexOf("&lt;p&gt;") !== -1) &amp;&amp; (loopCtr &lt; 100) ){
                        commentText = commentText.replace("&lt;p&gt;",' ');
                        commentText = commentText.replace("&lt;/p&gt;",' ');
                        loopCtr++;
                    }
                    //console.log(commentText);
                    $(this).find(".fyre-comment").css('display','none');
                    $(this).find(".fyre-comment-head").append(commentText);
                } // end check for blank text
            }
            //console.log("Checking Height");
            cntHeight = $(this).find(".fyre-comment-head").height();
            if (cntHeight &gt; 0){
                //console.log("Setting Height");
                cntHeight = -1 * ((cntHeight / 2) + 12);
                $(this).find(".fyre-flag-link").css('top',cntHeight);
            }
        });
        
   }
