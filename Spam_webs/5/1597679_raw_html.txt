	if(document.getElementById("cartAddition") !=null) {
		var ele = $('cartAddition');
		if(ele){
			if(state == 'hide'){
				$('cartAddition').addClassName('disabled');
			}
			else{
				$('cartAddition').removeClassName('disabled');
			}
		}
	}else{
		if(state == 'hide'){
			document.getElementById('toShow').style.display='none';
			}else{
				document.getElementById('toShow').style.display='block';
			} 
	}
	flag1 = true;
}
