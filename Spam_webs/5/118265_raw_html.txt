function addTags(){
    var tags = $("#form_tags").val();
    var options = {
            url: "/presentation/5991545/addtags",
            request: {tags: tags},
            callback: function(xhr) {
                $("#addtags").fadeOut();
                $("#form_tags").val("");
                if (true === xhr.result) {
                    $.blink("Tag(s) added", 3);
                    var tags = xhr.tags;
                    var htmlTags = new Array();
                    $.each(tags, function(index, value){
                        var htmlValue = '&lt;a href="/search/kw/presentations/' + value + '"&gt;' + value + '&lt;/a&gt;';
                        htmlTags.push(htmlValue);
                    });
                    $('#tags').html(htmlTags.join(' | '));
                }else{
                    if (xhr.hasOwnProperty("error_message")) {
                        $.blink(xhr.error_message, 5);
                    } else {
                        $.blink("Tag(s) not added", 3);
                    }
                }
            }
    };
    AppAjax.get("#tags", options);
}
