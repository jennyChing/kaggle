        function success(position) {
            jQuery('#m-wait-message').html("Locating local store, please wait");

            var params = {};
            params.accept = 1;
            params.lat = position.coords.latitude;
            params.lng = position.coords.longitude;
            params.ajax = 1;

            postGeoResult(params, geoSuccessCallbackJson);
        }

        function notEnabled() {
            var params = {};
            params.accept = 0;
            params.ajax = 1;

            postGeoResult(params, geoNotEnabledCallbackJson);
        }

        function postGeoResult(params, callback) {
            new Ajax.Request('http://www.superamart.com.au/store-locator/set/geo/', {method:'post', parameters:params, onSuccess:callback});
