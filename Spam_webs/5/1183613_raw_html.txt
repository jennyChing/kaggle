	function selectSiteOnMap(siteId, mapId) {
		if ($j("#selectedSiteRb").length &gt; 0) {
			$j("#selectedSiteRb").val(siteId);
		}
		if ($j("#" + mapId).length == 0) {
			var campareaObj = $j("#camparea");
			if (campareaObj.length &gt; 0) {
				campareaObj.val(mapId);
				$j("#viewfull").submit();
			}
		} else {
			changeSelectedSite(siteId);
		}
	}

	function changeSelectedSite(siteId) {
		var o_sites = document.getElementById('viewmap').childNodes;
		for (var i = 0; i &lt; o_sites.length; i++) {
			if (o_sites[i].className) {
				o_sites[i].className = o_sites[i].className.replace('slct', '');
			}
			if (o_sites[i].id == siteId) {
				o_sites[i].className = o_sites[i].className + 'slct';
			}
		}
	}

          </p>

           Wells State Park is a popular 1,400-acre woodland park close to the cultural attractions of central Massachusetts including the city of Worcester and Old Sturbridge Village.  The campground has 60 attractive sites, including 1 small group site, 2 Yurts and some sites perched on a hillside adjacent to Walker Pond.
