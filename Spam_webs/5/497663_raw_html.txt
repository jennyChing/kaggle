function saveFeedback()
{
    
        document.getElementById("divLoading").style.display = "block";

        // collect the data to a Json object.
        var strIsNLSubscribed = "";
        if (document.getElementById("chkOffersnPromo").checked)
        {
            strIsNLSubscribed = "true";
       }
        else
        {
            strIsNLSubscribed = "false";
        }


        var dataFeedback = {
            firstName: removeHTML("txtFeedbackFname"),
            lastName: "",
            emailId: removeHTML("txtFeedbackEmail"),
            IsNLSubscribed: strIsNLSubscribed,
            feedback: removeHTML("txtfeedback")
        };


        $.ajax({
            cache: false,
            type: "POST",
            url: "/Common/SaveFeedback",
            data: dataFeedback,
            dataType: "json",
            success: function (dataFeedback) {
                document.getElementById("divLoading").style.display = "none";
                // There is no problem with the validation
                if (dataFeedback.Valid) {
                    if (dataFeedback.ContactStatus == 2) {
                        document.getElementById('8_lblFeedbackMsg').innerHTML = document.getElementById('8_hdnMsgCreated').value.replace('&lt;Email&gt;', $.trim($("#txtFeedbackEmail").val()));
                    }
                    else {
                        document.getElementById('8_lblFeedbackMsg').innerHTML = document.getElementById('8_hdnMsgExists').value.replace('&lt;Email&gt;', $.trim($("#txtFeedbackEmail").val()));
                    }
                    document.getElementById("Message").style.display = "block";
                    document.getElementById("PopUpForFeedBack").style.display = "none";
                    $("#txtfeedback").val('');
                    refreshLoginPanel();
                    return;
                }

                var strValMsg = "Please provide or correct the following information \n";
                $.each(dataFeedback.Errors, function (key, value) {
                    if (value != null) {
                        strValMsg = strValMsg + "- " + value[value.length - 1].ErrorMessage + "\n";
                    }
                });
                // Don't comment this alert message - this is needed for validation message
                alert(strValMsg);
            },
            error: function (xhr) {
                document.getElementById("divLoading").style.display = 'none';
                alert("Oops. We're sorry. An error occurred and we're on the case.");
            }
        });
    }
