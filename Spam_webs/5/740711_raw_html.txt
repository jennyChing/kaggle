/*Check touch screen and handle social icons*/
function is_touch_device() {
 return (('ontouchstart' in window)
      || (navigator.MaxTouchPoints &gt; 0)
      || (navigator.msMaxTouchPoints &gt; 0));
}

function check_touch_device() {
if (!is_touch_device()) {
$('.caption-sm').css('display','none');

    mouseenter: function () {
		$(this).find('.caption').slideDown(100);
    },
    mouseleave: function () {
        $(this).find('.caption').slideUp(100);
    }
}, ".thumbnail");

	$('.caption-sm').css('display','block');
	$('.wishlist-pc').css('display','none');
	$('.not-visible-on-touch-device').css('display','none');
