function updateLikeProducts(){
  for (var i = 1; i &lt;= 4; ++i){
	  var anchor=dojo.byId('likeProductAnchor' + i);
	  var img=dojo.byId('likeProductImg' + i);
	  var headline=dojo.byId('likeProductHeadline' + i);
	  var retailPrice=dojo.byId('likeProductRetailPrice' + i);
      var freeShip = dojo.byId('likeProductFreeShip' + i);
      var imghref = dojo.byId('likeProductImg' + i).src;
	  var likeSku = new LikeSku();
	  if ((i+likeSkusBegin)&lt;=likeSkusArray.length){
		  likeSku = likeSkusArray[i+likeSkusBegin-1];
          anchor.title=likeSku.tooltip;
          anchor.alt=likeSku.tooltip;
          anchor.href=likeSku.href;
          img.src=imghref;
          headline.innerHTML=likeSku.headline;
          retailPrice.innerHTML=likeSku.retailPrice;
          if (likeSku.freeShip) {
              freeShip.style.display = '';
          } else {
              freeShip.style.display = 'none';
          }
          img.style.visibility = '';
      }
      else {
          img.style.visibility = 'hidden';
          anchor.title='';
          anchor.alt='';
          anchor.href='javascript:void(0)';
          img.src='';
          headline.innerHTML='';
          retailPrice.innerHTML='';
          freeShip.style.display = 'none';
      }
	  console.debug('updateLikeProducts');
  }
}
