       <article class="article">
        Turmeric is a potent anti-infammatory spice which can help to decrease muscle soreness after an intense workout. It also neutralizes free radicals which can lead to wrinkles, cellulite, and fatigue. In addition, turmeric improves liver detoxification, so that pesticides and other environmental chemicals are safely removed from the body. It also turns on a gene that enhances the body's own production of antioxidants thereby acting as a powerful anti-aging nutraceutical. Its warm, peppery, flavor and golden color adds a richness and depth to smoothies, vegetables, legumes, nut-pates and eggs. Here are 10 easy ways to add turmeric to your life.

$('.twitter_sharequote_button').click(function(event){
    event.preventDefault();
    var by = '';
    
        by = ' by @foodcoachnyc';
        
    
    var maxTweetLen = 97 - by.length; // 140(max tweet length) - 42 - by.length - 1 (a space for an ending ellipsis).
    var link = 'https://twitter.com/intent/tweet?url=' + encodeURIComponent(MBGArticle.shortUrl);
    var text = $(this).parent().attr("data-text");
    var via = 'mindbodygreen';
    
    if(text.length &gt; maxTweetLen){
    	try{
    		var trimmedText = '';
    		var words = text.split(' ');
    		var counter = 0;
    		for(var i in words){
    			var w = words[i];
    			if(counter + w.length &gt;= maxTweetLen){
    				if(trimmedText.substr(trimmedText.length-1) === ' '){
    					trimmedText = trimmedText.substr(0, trimmedText.length-1);
    				}
    				trimmedText += ''; // finish the trimmed quote with an ellipsis.
    				break;
    			}else{
    				trimmedText += (w + ' ');
    				counter += (w.length + 1);
    			}
    		}
    		text = trimmedText;
    	}catch(err){
    		console.log(err);
    	}
    }
    
    link += '&amp;via=' + via + '&amp;text=' + encodeURIComponent(text) + by;
    window.open(link);
    try{
        ga('send', 'event', 'Article', 'Twitter Share Clicked', '9206');
    }catch(err){}
});

// article saving
$('.article_save_button').click(function(event){
    event.preventDefault();
    
    var getChannelsShowModal = function(response,error){
        if(error!==null){
            console.log(error);
            return;
        }
        if(response!==null){
            if(response===false){
                console.log('Login returned false');
                return;
            }else if(response===true){
                MBGUI.toggleUtilityMenuToMyMBG();                   
            }
        }
        $.ajax({
            url: MBGGlobals.getBaseUrl() + 'getchannels.action',
            traditional : true,
            cache : false
        }).done(function(data){
            if(data.success===true){
                if(data.loggedIn === true){
                    var html = "";
                    $(data.channels).each(function(index,item){
                        html += "&lt;input type='checkbox' name='channels' value='" + item.id + "' label='" + item.name + "'/&gt;&nbsp;&nbsp;" + item.name + "&lt;br&gt;";
                    });
                    if(html != ""){
                        $('#article_save_modal_channel_target').html(html);
                    } else {
                        $('#article_save_modal_channel_target').html('You have no saved channels');
                    }
                    $('#article_save_modal').foundation('reveal','open');
                }else{
                    // server thinks the user is not logged in but client does, so show log in screenset
                    MBGSocial.displayLoginModal(getChannelsShowModal);
                }
            }else{
                console.log('An error occcured getting channels: ' + data.message);
            }           
        }).fail(function(xhr,status,error){
            console.log('An error occurred getting channels ' + status + ', ' + error);
        });
                
    };
    
    if(MBGSocial.isUserLoggedIn()){
        try{
            ga('send', 'event', 'Article', 'Logged In Save Button', '9206');
        }catch(err){}
        getChannelsShowModal(null,null);
    }else{
        try{
            ga('send', 'event', 'Article', 'Logged Out Save Button', '9206');
        }catch(err){}
        // use register screenset
        MBGSocial.displayLoginModal(getChannelsShowModal,true);
    }
    $(this).children("img").attr("src", "//res.mindbodygreen.com/img/msc/article-save.svg");
});

$('#article_save_modal_save_button').click(function(event){
    event.preventDefault();
    var sub = {};
        sub.id = '9206';
        sub.objectType = 'post';
        var newChannel = $('#article_save_modal_new_channel').val().trim();
        var idCount = $('#article_save_modal input[name=channels]:checked').length;
        if (newChannel === '' &amp;&amp; idCount == 0) {
            $('#article_save_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! You need to select a channel or create a new one to save this article!');
            return false;
        }
        if (newChannel != '') {
            sub.newChannel = newChannel;
        }
        if (idCount &gt; 0) {
            var channels = [];
            $('#article_save_modal input:checked').each(function() {
                channels.push($(this).val());
            });
            sub.channelIds = channels;
        }
        
        var inputString = JSON.stringify(sub);
        $.ajax({
            type : 'POST',
            cache : false,
            url : MBGGlobals.getBaseUrl() + 'savechannelsubscription.action',
            data: {
                subscription : inputString
            }
        }).done(function(obj) {
            if(obj.success === true) {
                var successMessage = 'Your article has been saved!';
                if (newChannel !== '') successMessage = 'Your article and new channel have been saved!';
                $('#article_save_modal_response_target').removeClass('hide').removeClass('error').addClass('success').html(successMessage);
                setTimeout(function() {
                    console.log('closing modal');
                    $('#article_save_modal').foundation('reveal','close');
                    $('#article_save_modal_response_target').addClass('hide').html('');
                }, 3500);
                try{
                    ga('send', 'event', 'Article', 'Article Saved', '9206');
                }catch(err){}
            } else {
                $('#article_save_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! Something went wrong and we could not save your article. If this error persists, please write us at support@mindbodygreen.com.');
                console.log(obj.message);
            }
        }).fail(function(jqXHR, textStatus) {
            $('#article_save_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! Something went wrong and we could not save your article. If this error persists, please write us at support@mindbodygreen.com.');
            console.log(textStatus);
        });
        
        
});
