jQuery(document).ready(function() {
	var fixPinterestQueryString = function () {
		var imageUrl = jQuery('meta[property="og:image"]').attr('content');
		var description = jQuery('meta[property="og:title"]').attr('content');
		var originalHref = jQuery('.gig-button-container-pinterest-pinit a').attr('href');
		if (!originalHref) return;
		var correctedHref = originalHref.replace(/%25/g, '%');
		correctedHref = correctedHref.replace(/&amp;media=[^&amp;#$]*/, '&amp;media=' + imageUrl);
		correctedHref = correctedHref.replace(/&amp;description=[^&amp;#$]*/, '&amp;description=' + description);
		jQuery('.gig-button-container-pinterest-pinit a').attr('href', correctedHref);
	};

	// Wait for Gigya to load
	setTimeout(fixPinterestQueryString, 100);
	setTimeout(fixPinterestQueryString, 200);
	setTimeout(fixPinterestQueryString, 300);
	setTimeout(fixPinterestQueryString, 500);
	setTimeout(fixPinterestQueryString, 1000);
	setTimeout(fixPinterestQueryString, 2000);

});
