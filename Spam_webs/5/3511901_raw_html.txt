              var BDC = {
  init: function(){
    BDC.initFilter();
  },
  initFilter: function(){
    if(!$('#postFilter').length) return;
    BDC.initPagination();
    $('#postFilter a').click(function(){
      var target = $(this).attr('href');
      BDC.clearFilters();
      $(target).show();
      $(this).addClass('active');
      $($('.pagination .pageButton', target)[0]).click();
      return false;
    });
    $($('#postFilter a')[0]).click();
  },
  clearFilters: function(){
    $('.postsContainer').hide();
    $('#postFilter a').removeClass('active');
  },
  initPagination: function(){
    var postPerPage = 2;
    if(!$('.postsContainer').length) return;
    $('.postsContainer').each(function(){
      var scope = this;
      var postAmount = $('.post', this).length;
      var pages = Math.ceil(postAmount/postPerPage);
			
      if(pages &gt; 1){
        $(this).attr('data-currentpage', 1);
        var pagination = $('&lt;div class="pagination"&gt;&lt;/div&gt;');
        $(this).append(pagination);
        for(var a=0;a&lt;pages;a++){
          var pageButton = $('&lt;a href="#'+$(this).attr('id')+'" originalAttribute="href" originalPath="#'+$(this).attr('id')+'" originalAttribute="href" originalPath="#'+$(this).attr('id')+'" class="pageButton"&gt;'+(a+1)+'&lt;/a&gt;').click(function(){
            $('.post', scope).hide();
            $('.pageButton').removeClass('current');
            $(this).addClass('current');
            var page = parseInt($(this).html());
            var index = postPerPage*page;
            for(var b=(index-postPerPage);b&lt;index;b++){
              $($('.post', scope)[b]).show();
            }
            $(scope).attr('data-currentpage', page);
            if(page == 1){
              $('.prevButton', pagination).hide();
            } else {
              $('.prevButton', pagination).show();
            }
            if(page == pages){
              $('.nextButton', pagination).hide();
            } else {
              $('.nextButton', pagination).show();
            }
            return false;
          });
          pagination.append(pageButton);
        }
        pagination.prepend($('&lt;a class="prevButton"&gt;&lt;span&gt;Prev&lt;/span&gt;&lt;/a&gt;').click(function(){
          var target = parseInt($(scope).attr('data-currentpage'))-1;
          $($('.pageButton', pagination)[target-1]).click();
          return false;
        }));
        pagination.append($('&lt;a class="nextButton"&gt;&lt;span&gt;Next&lt;/span&gt;&lt;/a&gt;').click(function(){
          var target = parseInt($(scope).attr('data-currentpage'))+1;
          $($('.pageButton', pagination)[target-1]).click();
          return false;
        }));
        $($('.pageButton', pagination)[0]).click();
      }
    });
  }

             <p>
              Most entrepreneurs struggle to delegate responsibility. After years of building up their company and controlling all aspect of business, they find it difficult to let go. But trying to do too much can prove detrimental to business growth, employee engagement and your own stress levels.
             </p>
             <p>
              In this infographic, you'll discover 9key strategies to help you strike the right balance between maintaining control and delegating efficiently. From identifying the best managers to sharing information and developing repeatable processes, learn about key actions to make delegation work for your company.
             </p>
             <p>
              By applying these essential strategies, you'll help boost employee motivation, make your own life easier and create a more robust and healthier business to promote future growth.
