			ulLat:           $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=ulLat]'),
			ulLng:           $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=ulLng]'),
			zoom:            $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=zoom]'),
			mapViewType:     $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=mapViewType]'),
			mapDragType:     $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=mapDragType]'),
			allowMouseWheel: $('[data-wnt-map-name=' + $map.mapTracks("getMapName") + '][data-wnt-map-bind=allowMouseWheel]')
		};

		var setMapBindFields = function () {
			if ($map.mapTracks("getProvider") == "mapquest") {
				$bindingFields.centerLat.val($map.mapTracks("getCenter").lat);
				$bindingFields.centerLng.val($map.mapTracks("getCenter").lng);
				$bindingFields.mapType.val($map.mapTracks("getCurrentView"));
				$bindingFields.lrLat.val($map.mapTracks("getBounds").lr.lat);
				$bindingFields.lrLng.val($map.mapTracks("getBounds").lr.lng);
				$bindingFields.ulLat.val($map.mapTracks("getBounds").ul.lat);
				$bindingFields.ulLng.val($map.mapTracks("getBounds").ul.lng);
				$bindingFields.zoom.val($map.mapTracks("getZoomLevel"));
			}
			$bindingFields.centerLat.change();
			$bindingFields.centerLng.change();
			$bindingFields.mapType.change();
			$bindingFields.lrLat.change();
			$bindingFields.lrLng.change();
			$bindingFields.ulLat.change();
			$bindingFields.ulLng.change();
			$bindingFields.zoom.change();
		}

		setMapBindFields();


		// Track map usage
		var mapTrack = function () {
			var mapTrigger   = createGuid();
			var myMapTrigger = mapTrigger;
			
				var page_type = "search";
				window.setTimeout(function () {
					if (myMapTrigger == mapTrigger) {
						$.post("gateway.cfm?action=dbtrack", {
							page_type:  page_type,
							map_type:   $map.mapTracks("getCurrentView"),
							mt_version: 3
						});
					}
				}, 50);
			
		};

		mapTrack();


		// Toggle POI map status
		var togglePOIMapStatus = function (e, poiStatus) {
			$.ajax({
				type: "GET",
				url:  "gateway.cfm?action=togglePOIMapStatus&amp;poiStatus=" + poiStatus
			});
		}


		// Save the map control
		var persistMapControl = function () {

			var $map = $(this);

			if ($map.mapTracks("getMapName") != "poiMiniMap") {

				var controlData = {
					mapViewType:     $map.mapTracks("getCurrentView"),
					mapDragType:     $map.mapTracks("getDragType"),
					allowMouseWheel: $map.mapTracks("getAllowMouseWheel")
				}

				$bindingFields.mapViewType.val(controlData.mapViewType);
				$bindingFields.mapDragType.val(controlData.mapDragType);
				$bindingFields.allowMouseWheel.val(controlData.allowMouseWheel);

				persistRequest = $.ajax({
					type: "POST",
					url:  "gateway.cfm?action=persistMapControl",
					data: controlData
				});

			}

		}



		// Auto-resizing the map
		var autoResizeMap = function () {
			var newWidth = $map.parent().width()-2;
			if ($map.mapTracks('getSize').width != newWidth) {
				$map.mapTracks('setSize', newWidth, $map.mapTracks("getSize").height);
			}
		};

		autoResizeMap();


		// Bind the map events

		$map
			.on("zoomEnd moveEnd dragEnd", setMapBindFields)
			.on("moveEnd zoomEnd, currentView mapControlChanged", mapTrack)
			.on("poiStatusUpdated", togglePOIMapStatus)
			.on("currentView mapControlChanged", persistMapControl)
			.on('mapOpen', autoResizeMap);
		$(window).resize(autoResizeMap);

		$("[data-wnt-resetMap=]").click(function () {
			$map.mapTracks("resetCenter");
		});

	});
    </script>
