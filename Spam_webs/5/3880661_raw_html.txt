        function cloneImageAttributes(orig, newItem){
            newItem = jQuery(newItem);
            orig = jQuery(orig);

        function setStoryImage(url){
            jQuery("#story-image").addClass("offscreen");
            
            //jQuery("#story-image").attr("src","");
            var newImg = jQuery("&lt;img&gt;");
            
            var currImg = jQuery("#story-image");
            newImg = cloneImageAttributes(currImg, newImg);
            var parentContainer = currImg.parent();
            
            
            if(addLoad) {
                newImg.load(function() {
                    var aspect = (this.width/this.height);   
                    if(aspect&gt;1.77) {
                        var newWidth = 896;
                        var hewHeight = 896/aspect;
                        jQuery(this).attr('style', "width:" + newWidth + "px !important; height:" + hewHeight + 'px !important;');
                        jQuery(this).css('margin-top', (504/2 - hewHeight/2) + 'px');
                     }
                     else {
                        jQuery(this).attr('style', '');
                        var hScale = 504/this.height;
                        if(hScale!=1) {
                            this.width = this.width*hScale;
                            this.height = this.height*hScale;
                        }
                        jQuery(this).css('margin-left', (896/2 - this.width/2) + 'px');
                     
                     }
                    jQuery(this).removeClass("offscreen");
                    if(this.width&lt;896 || this.height&lt;504) {
                        addBackgroundImage(this.src.replace(currentImageExt,"_896_ghst") );

                    } else { 
                       hideItem('background-image');

                        if(currImg)currImg.remove();
                        parentContainer.append(newImg);
                        newImg.animate({opacity:1},250);

                newImg.error(function() { 
                    if(this.src.indexOf("_lg")&lt;0) {

            newImg.attr("src", url);
            jQuery("#story-image").attr("src",url);            
        
        }

           <span>
            The video below explains what it is difficult to see in many of the hunting pictures posted by trophy hunters, by giving a scenario of what really goes on in a trophy hunt. It is about power over an animal. It is important to consider the personality of those who are addicted to this. Less than 10% of hunters in our modern times really hunt to feed their families. The money to be made in the recreational hunting industry makes it very difficult to pass laws against cruelty to wildlife. As with the most diabolical side of human nature in any part of the world, this complete lack of protections makes easy victim of our wildlife, and there is a disturbing element that has an outlet for their cruelty.
