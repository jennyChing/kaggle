	if (qp.utm_source !== undefined &amp;&amp; qp.utm_campaign !== undefined) ubparam = "&amp;utm_source="+qp.utm_source+"&amp;utm_campaign="+qp.utm_campaign;

	if (ubparam !== undefined) ubparam = "refpage=" + "%2F" + ubparam;
	else ubparam = "refpage=" + "%2F";
	ubparam += "&amp;refer=" + escape(document.referrer);

	_spaq.push({
		'ref': 'ub',
		'url': '/auth/user_box.php',
		'params': ubparam,
		'callback': function (data) {
			if (data &amp;&amp; data.result) {
				initializeUserBox(data.result);
			}
		}
	});
	_spaq.push ({
		'ref': 'fb_sdk',
		'url': '/facebook/fbincludes/js_sdk_include.php',
		'params': ubparam,
		'callback': function (data) {
			$('fb-script').update(data['result']);
		}
	});
	_spaq.push ({
			'ref': 'google_sdk',
			'url': '/google/google_api_include.php',
			'callback': function (data) {
				if (data.result) {
					var googleScript = $j(data.result);
					$j("#google-script").append(googleScript);
				}
			}
		});
	_spaq.push({
		'ref': 'pc',
		'url': '/games/ajax/playcount.php',
		'callback': function (data) {
			if (data.result) {
				var result = JSON.parse(data.result);
				if (result.success) {
					$j('#footerCountValue').html(result.play_count);
					$j('#footerCount').show();
				}
			}
		}
	});
	
	var _spAjaxCalls = _spAjaxCalls || [];
	if (_spAjaxCalls.length) {
		_spaq.push(_spAjaxCalls);
	}


	document.observe('dom:loaded', function(){
		_spaq.send();
	});
