 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')

$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event

    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';

                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());

            }

            $(this).attr('ga-event-added', 'true');
        });

            http://www.quadibloc.com/maps/mpol05.htm has a nice picture of a polyconic projection.
            <br/>
            You choose a rectangle that is twice as wide as it is tall.
            <br/>
            Divide the wide side into six sections.
            <br/>
            Draw six arc sections that just touch in the middle.
            <br/>
            Mark an extra edge on one side of the arcs so there will be some overlap to glue.
            <br/>
            Cut out, remembering to leave the arcs connected as well as the tabs.
            <br/>
            Even if you don't get the arcs perfect, you will end up with a quite nice ball!
            <br/>

           How do you make paper?
          </a>
         </h1>
         <div class="content_text" ga-event="Feed_Click,index/question_answered/15700,f_q_1908084">
          To make paper, mix up vegetable fibers, and cook them in hot water until the fibers are soft but not dissolved. Add a base chemical such as lye to the hot water, as this softe
          <span class="hellip">
           
          </span>
          <span class="hidden">
           ns the fibers as they are cooking. Then, pass a screen-like material through the mixture, let the water drip off and/or evaporate, and then squeeze or blot out additional water. A layer of paper is left behind. Essential to the process are the fibers, which are never totally destroyed, and, when mixed and softened, form an interlaced pattern within the paper itself.    If you were to make traditional Asian paper back then, then follow the rules below:   1.Combine water, felt, and old clothes together (two gallons of water).   2. Mix thoroughly.   3. (optional) Put bleach in.   4. Let stand for a while ( the longer the better; usually one day).   5. Take some of the mixture, and put it in a blender.   6. Blend for about 5- 10 min.   7. Lay flat down.   8. Let it dry and you're done.
