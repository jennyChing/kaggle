           (function () {
        String.prototype.slugify = function () {
            return this.toString().toLowerCase()
                    .replace(/[\s']+/g, '-')           // Replace spaces with -
                    .replace(/[^\w\-]+/g, '')       // Remove all non-word chars
                    .replace(/\-\-+/g, '-')         // Replace multiple - with single -
                    .replace(/^-+/, '')             // Trim - from start of text
                    .replace(/-+$/, '');            // Trim - from end of text
        };

        var timer = null;
        var ajaxRequest = null;
        var baseSubmitUrl = "/actualites-regions-departements-villes/recherche";

        $('.rechercheVilleForm').each(function () {
            var $me = $(this);
            var $resultContainer = $me.children('.rechercheVilleResultContainer');
            var $input = $me.children('.rechercheVille');

            $(this).keyup(function (e) {

                if (timer !== null) {
                    clearTimeout(timer);
                }
                timer = setTimeout(function () {
                    if (ajaxRequest !== null) {
                        ajaxRequest.abort();
                    }
                    ajaxRequest = $.ajax({
                        type: 'GET',
                        cache: true,
                        url: '/search/commune' + '/' + $input.val(),
                        success: function (data) {
                            if (!data) return;

                            var slugDep, slugNom;
                            var html = document.createElement("ul");

                            for (var i = 0; i &lt; data.total; i++) {
                                if (data[i] !== undefined) {
                                    slugDep = data[i].nomDepartement.slugify() + '-' + data[i].codeDepartement;
                                    slugNom = data[i].slug;

                                    var para = document.createElement("li");

                                    var link = document.createElement("a");
                                    link.href = 'http://www.leparisien.fr/' + slugDep + '/' + slugNom + '?store=1';

                                    link.innerHTML = data[i].nom;

                                    var span = document.createElement("span");
                                    span.className = "droite";
                                    span.innerHTML = '(' + data[i].codePostal + ')';

                                    link.appendChild(span);
                                    para.appendChild(link);
                                    html.appendChild(para);

                                }
                            }


                            $resultContainer.html(html);

                            if (data.total &gt; 0) {
                                // Le premier lment renvoy correspond au nom exact de la ville
                                // On peut donc prendre le premier lment
                                slugDep = data[0].nomDepartement.slugify() + '-' + data[0].codeDepartement;
                                slugNom = data[0].slug;
                            }
                        },
                        error: function () {
                            $resultContainer.html('');
                        }
                    });
                }, 300);
            });
        });
    })();
