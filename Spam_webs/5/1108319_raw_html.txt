		}
		else {
			// Show grid -- it's hidden until this decision is made to avoid a flash of content
			$('.imageGrid').css('visibility', 'visible');
		}

		// Set up image hover
		$('.imageGrid .item .image').hover(
			function () {
				$(this).find('.hover').fadeIn('fast');
			},
			function () {
				$(this).find('.hover').fadeOut('fast');
			}
		);

		$well = $('.well');

		// Set up filter clicks
		$('.filterSort .main .btn').click(function () {
			switchWell('main', $(this));
		});
		$('.filterSort .types .btn').click(function () {
			switchWell('types', $(this));
		});

		// Set up callout click
		$('.well .callout').click(function() {
			$('.filterSort .main .btn').click();
		});

		// Show well?
		if (filterActive) {
			$well.find('.summary').show().addClass('s');
			$('.filterSort .controller .main .btn').addClass('s');
			showWellArrow($('.filterSort .controller .main .btn'));
			$well.show();
		}

		// QuickView
		$('a[rel~="quickview"]').colorbox({width:"1000px", height:"570px", scrolling: false, iframe:true});
	});
	function showWellArrow($alignWith) {
		var $arrow = $('.filterSort .arrow');
		$arrow.css('left', $alignWith.position().left + ($alignWith.outerWidth() / 2) - 10);
		$arrow.fadeIn(100);
	}
	function hideWellArrow() {
		$('.filterSort .arrow').fadeOut(100);
	}
	function hideWell() {
		$('.well .callout').show();
		$well.children('.wellPanel').removeClass('s').hide();
		$('.filterSort .controller .btn').removeClass('s');
		if (filterActive)
		{
			$well.find('.summary').addClass('s').show();
			$('.filterSort .controller .main .btn').addClass('s');
			showWellArrow($('.filterSort .controller .main .btn'));
			$('.well .callout').show();
		}
		else
		{
			hideWellArrow();
		}
		animateWell();	
	}
	function switchWell(activePanelClass, $clickedButton) {
		var $target = $well.children('.' + activePanelClass);
		if ($target.is(':visible'))
		{
			hideWell();
		}
		else
		{
			// Show panel
			$('.well .callout').hide();
			$well.children('.wellPanel').removeClass('s').hide();
			$target.show().addClass('s');
			if (typeof($clickedButton) != 'undefined') 
			{
				$('.filterSort .controller .btn').removeClass('s');
				$clickedButton.addClass('s');
				showWellArrow($clickedButton);
			}
			animateWell();
		}
	}
	function animateWell() {
		var $activeChild = $well.children('.s');
		if ($activeChild.length == 0)
		{
			$well.slideUp();
			return;
		}
		if (!$well.is(':visible'))
		{
			// Well was not shown at all
			$well.css('height', 1);
			$well.show();
		}
		var initialHeight = $well.height();
		if (initialHeight == 0) initialHeight = 1;
		$well.animate( { height: initialHeight }, 1, function() {
			var newHeight = parseInt($activeChild.outerHeight(true));
			if (newHeight == 0) newHeight = wellHeightClosed;
			$well.animate( { height: newHeight }, 'fast', 'easeOutQuad');
		});
	}
