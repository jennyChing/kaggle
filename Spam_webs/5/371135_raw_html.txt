function detectDipWidth()
{
    var ua = navigator.userAgent
    if( ua.match(/(iPad|iPhone|iPod)/g) )
    {
        // simply return screen.width because it's not subject to rotation on safari
        return screen.width;
    }
    else if( ua.match(/Android/g) )
    {
        // detect orientation
        // This represent the width in PORTRAIT orientation
        // On some device like Nexus, the device.height in landscape will not be equal to device.width in portrait
        // because of the Android software touch button.
        var deviceWidth = window.screen.width;
        if( window.orientation == 90 || window.orientation == -90 )
        {
            deviceWidth = window.screen.height;
        }

        if( ua.match(/Chrome/g) )
        {
            return deviceWidth;
        }

        return Math.floor(deviceWidth / window.devicePixelRatio);
    }

    return window.screen.width;
}
