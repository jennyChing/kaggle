siteSearch(['jquery','jhmContext','resjs'], function ( $, c, res) {
	if(typeof c.gsa !== 'undefined') {
		var searchOptDiv = document.createElement('div'),
			jhmInpOpt = document.createElement('input'),
			subSiteInpOpt = document.createElement('input'),
			sitesearchInpOpt = document.createElement('input');
		
		$(jhmInpOpt).attr({
			id: 'subSiteSearchOptMain',
			value: 'site',
			name: 'subSiteSearchFlag',
			type: 'radio'
		});
		
		$(subSiteInpOpt).attr({
			id: 'subSiteSearchOptSub',
			value: 'sitesearch',
			name: 'subSiteSearchFlag',
			type: 'radio'
		});
		
		if (typeof c.gsa.subsiteurl !== 'undefined') {
			$(sitesearchInpOpt).attr({
				value: c.gsa.subsiteurl,
				name: 'sitesearch',
				type: 'hidden'
			});
		}
		
		if(c.gsa.defaultToSubSite) {
			$(subSiteInpOpt).attr('checked',true);
		} else {
			$(jhmInpOpt).attr('checked',true);
		}
		
		if(typeof c.gsa.external !== 'undefined') {
			$('#jhmSubSiteSearch').attr('action',c.gsa.external);
		}
		
		if(typeof c.gsa.inputName !== 'undefined') {
			$('#subsite-search-field').attr('name',c.gsa.inputName);
		}
		
		if(typeof c.gsa.client !== 'undefined') {
			$('#jhmSubSiteSearch input[name="client"]').val(c.gsa.client);
		}
		
		if(typeof c.gsa.proxystyle !== 'undefined') {
			$('#jhmSubSiteSearch input[name="proxystylesheet"]').val(c.gsa.proxystyle);
		}
		
		if(typeof c.gsa.collection !== 'undefined') {
			$('#jhmSubSiteSearch input[name="site"]').val(c.gsa.collection);
		}
		
		$(searchOptDiv).addClass('search-options')
			.appendTo($('.search-field-wrapper'))
			.append(jhmInpOpt)
			.append('&lt;label&gt;Johns Hopkins Medicine&lt;/label&gt;')
			.append(subSiteInpOpt)
			.append('&lt;label&gt;' + c.gsa.display + '&lt;/label&gt;');
			
		$('#jhmSubSiteSearch').append(sitesearchInpOpt);
	}
	
	res.setupSearchFunctions();
});

          <h1 class="mainBodyContentTitle">
           <span class="">
            Primary (Non-Autistic) Motor Stereotypies
           </span>
          </h1>
          <div class="mainBodyContentData">
           <div>
            <a name="Skip">
            </a>
           </div>
           <p>
            Primary motor stereotypies (also called stereotypic movement disorder), are rhythmic, repetitive, fixed, predictable, purposeful, but purposeless movements that occur in children who are otherwise developing normally. Examples of primary motor stereotypies are flapping and waving of the arms, hand flapping, head nodding and rocking back and forth.
           </p>
           <div>
            <div class="flex-33-right">
             <img alt="motor stereotypies patient with Dr Singer" border="0" src="/sebin/x/a/motor-stereotypies.jpg"/>
            </div>
           </div>
           <p>
            These movements usually appear in the first three years of a childs life and often continue. Though primary motor stereotypies are fairly common, the cause is unknown.
           </p>
           <p>
            Motor stereotypies can also occur in children who have developmental problems, for example a child with
            <a href="http://www.kennedykrieger.org/patient-care/diagnoses-disorders/autism-spectrum-disorders-asd-and-pervasive-developmental-disorders-pdd" target="_blank" title="more information on autism">
             autism
            </a>
            , mental retardation, or vision or hearing impairment. In such cases where a behavioral or neurological disorder has been diagnosed, the movements are called secondary motor stereotypies.
           </p>
           <p>
            Thanks to generous support from the
            <strong>
             Nesbitt-McMaster Foundation
            </strong>
            , we offer this online resource for parents like you who are looking for answers.
           </p>
