						function serverTime() { 
						    var time = null; 
						    $.ajax({url: 'http://www.c4dlive.com/serverTime', 
						        async: false, dataType: 'text', 
						        success: function(text) { 
						            time = new Date(text); 
						        }, error: function(http, message, exc) { 
						            time = new Date(); 
						    }}); 
						    return time;
