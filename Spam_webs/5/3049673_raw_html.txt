           <p>
            Despite a decent range of handsets such as the Lumia 1520,
            <a href="http://www.thenational.ae/business/nokia">
             Nokia
            </a>
            has yet to break the stranglehold of
            <a href="http://www.thenational.ae/business/samsung-electronics">
             Samsung
            </a>
            and
            <a href="http://www.thenational.ae/business/apple">
             Apple
            </a>
            in the high end smartphone market.
           </p>
           <p>
            Now, the manufacturer is doubling down on the fast growing budget smartphone segment, with the launch of its new XL handset, following the launch of the X handset in the UAE in March.
           </p>
           <p>
            Perhaps unsurprisingly, the word that comes to mind when reviewing the XL is cheap, for better and for worse.
           </p>

    function UserDataUpdateMessage(AItemType)
  {
    switch(AItemType) 
    {
      case 1:
        return '&lt;p class="savecontent"&gt;topic is saved&lt;/p&gt;';
      break;
      case 2:
        return '&lt;p class="savecontent"&gt;event is saved&lt;/p&gt;';
      break;
      case 3:
        return '&lt;p class="savecontent"&gt;article is saved&lt;/p&gt;';
      break;
      case 4:
        return '&lt;p class="savecontent"&gt;invalid data&lt;/p&gt;';
      break;
      case 5:
        return '&lt;p class="savecontent"&gt;please, sign in&lt;/p&gt;';
      break;
    }
  }
  function UserDataUpdate(AItemAction,AItemType,AItem,AResult)
  {
    $.ajax(
    {
    type: 'POST',
    data: { 
      itemaction: AItemAction,
      itemtype: AItemType, 
      item: AItem  
    },
    url: '/misc/misc/MyNational.pbs' ,
    success: function(data){
      var LData = data || "no response text";
      if(AItemAction == 0)
      {
        console.log(LData);
        $('#'+AResult).hide();
        if(AItemType == 1)
          $('#currenttopic').hide();
        LTemp = $('#MyNationalCount_'+AItemType).html();
        LTemp = (LTemp.replace("(","")).replace(")","");
        LTemp = parseInt(LTemp)-1;
        $('#MyNationalCount_'+AItemType).html('('+ LTemp + ')');
      }
      else 
      {
        if(LData.search("saved") != -1)
          $('#'+AResult).html(UserDataUpdateMessage(AItemType));
        else if(LData.search("invalid") != -1)
          $('#'+AResult).html(UserDataUpdateMessage(4));
        else
          $('#'+AResult).html(UserDataUpdateMessage(5));
      }
    },
      error: function(){ console.log('Error, something went wrong...') }
    });
  }
