                            <div class="ERSName" itemprop="name">
                             The Perfect Soft Pretzels

                             <div class="ERSAuthor">
                              Author:
                              <span itemprop="author">
                               <a href="http://ohsweetbasil.com">
                                Oh, Sweet Basil
                               </a>
                              </span>

                             <div class="ERSHead">
                              Recipe type:
                              <span itemprop="recipeCategory">
                               appetizer, bread, snack
                              </span>
                             </div>
                             <div class="ERSClear">
                             </div>
                             <div class="ERSTimes">
                              <div class="ERSHead">
                               Prep time:
                               <time datetime="PT1H10M" itemprop="prepTime">
                                1 hour 10 mins
                               </time>
                              </div>
                              <div class="ERSHead">
                               Cook time:
                               <time datetime="PT12M" itemprop="cookTime">
                                12 mins
                               </time>
                              </div>
                              <div class="ERSHead">
                               Total time:
                               <time datetime="PT1H22M" itemprop="totalTime">
                                1 hour 22 mins
                               </time>
                              </div>
                             </div>
                             <div class="ERSClear">
                             </div>
                             <div class="ERSHead">
                              Serves:
                              <span itemprop="recipeYield">
                               8 servings
                              </span>
                             </div>
                             <div class="ERSClear">
                             </div>
                            </div>
                            <div class="ERSSummary" itemprop="description">
                             Hot, soft, chewy pretzels
                            </div>
                            <div class="ERSIngredients">
                             <div class="ERSIngredientsHeader ERSHeading">
                              Ingredients
                             </div>
                             <ul>
                              <li class="ingredient" itemprop="ingredients">
                               1 cups warm water
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               1 tablespoon sugar
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               2 teaspoons kosher salt
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               2 teaspoons active dry yeast
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               3 tablespoons unsalted butter, melted
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               4 cups all-purpose flour
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               6 Cups Water
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                                Cup Baking Soda
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               1 Egg yolk + 1 tablespoon water
                              </li>
                              <li class="ingredient" itemprop="ingredients">
                               Sea Salt For Topping
                              </li>
                             </ul>
                             <div class="ERSClear">
                             </div>
                            </div>
                            <div class="ERSInstructions">
                             <div class="ERSInstructionsHeader ERSHeading">
                              Instructions
                             </div>
                             <ol>
                              <li class="instruction" itemprop="recipeInstructions">
                               Combine the water and sugar in a standing mixing bowl and sprinkle the yeast on top. allow to sit for 10 minutes, until the mixture begins to foam and grow. add the butter and stir to combine.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               add the flour and salt together and whisk to combine. pour in the liquid and use the dough hook attachment to mix on low speed until all of the ingredients are combined. increase the mixer speed to medium and knead until the dough is smooth and pulls away from the side of the bowl, about 4 to 5 minutes.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               remove the dough from the mixer and place it in a clean, oiled bowl. cover with a kitchen towel and place in a warm spot until the dough has doubled in size, about an hour.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               preheat the oven to 450 degrees. line 2 baking sheets with parchment paper and drizzle with a little oil.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               fill a large pot with water and the baking soda. bring the mixture to a rolling boil.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               meanwhile, place the dough on a slightly oiled counter and divide into 8 equal pieces. roll out each piece of dough into a 24-inch rope, lay it on the counter, then holding each end in a hand make a u-shape with the rope, cross them over each other, bring it down, twist, and press onto the bottom of the u in order to form a pretzel. place onto the parchment-lined baking sheet.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               drop the pretzels into the boiling water, 1 or 2 at a time for about 30 seconds, constantly spooning the hot water over the dough. remove them from the water using a large, flat spatula and place right on the oiled baking sheet. brush the top of each pretzel with egg wash and sprinkle with coarse salt.
                              </li>
                              <li class="instruction" itemprop="recipeInstructions">
                               bake until dark golden brown in color, about 12 minutes. transfer to a cooling rack for at least 5 minutes before serving.
                              </li>
                             </ol>
