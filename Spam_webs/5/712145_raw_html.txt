   <script type="text/javascript">
    var current_panel = 1;
    var last = null;
    var carousel_panels_count = 0;
    var block_scroll = false;
    
    $(function(){
        
        last = $('.carousel-inside').children('.carousel-inside-panel:last-child')[0];
        carousel_panels_count = $('.carousel-inside-panel').length;
        
        $('a.carousel-left').click(function(){
            if(!block_scroll) {
                scrollCarouselRight();
            }
        });
        
        $('a.carousel-right').click(function(){
            if(!block_scroll) {
                scrollCarouselLeft();
            }
        });
        
    });
    
    function scrollCarouselLeft() {
        current_panel++;
        block_scroll = true;
        $('.carousel-inside &gt; .carousel-inside-panel').each(function(){
            $(this).animate(
                {
                    left: $(this).position().left - 890
                }, 
                250, 
                function(){
                    if($(this)[0] === last) {
                        block_scroll = false;
                    }
                }
            );
        });
        
        if(current_panel &gt;= carousel_panels_count) {
            $('a.carousel-right').css('visibility', 'hidden');
        } else {
            $('a.carousel-right').css('visibility', 'visible');
            $('a.carousel-left').css('visibility', 'visible');
        }
    }
    
    function scrollCarouselRight() {
        current_panel--;
        block_scroll = true;
        $('.carousel-inside &gt; .carousel-inside-panel').each(function(){
            $(this).animate(
                {
                    left: $(this).position().left + 890
                }, 
                250, 
                function(){
                    if($(this)[0] === last) {
                        block_scroll = false;
                    }
                }
            );
        });
        
        if(current_panel &lt;= 1) {
            $('a.carousel-left').css('visibility', 'hidden');
        } else {
            $('a.carousel-left').css('visibility', 'visible');
            $('a.carousel-right').css('visibility', 'visible');
        }
    }
   </script>
