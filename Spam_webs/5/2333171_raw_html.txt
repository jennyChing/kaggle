       <h1 class="entry-title">
        Chicken Tetrazzini
       </h1>
       <div class="post-info">
        <span class="date published time" title="2010-01-01T13:40:25+00:00">
         January 1, 2010
        </span>
        by
        <span class="author vcard">
         <span class="fn">
          <a class="fn n" href="http://thehappyhousewife.com/cooking/author/TheHappyHousewife/" rel="author" title="Toni Anderson">
           Toni Anderson
          </a>
         </span>
        </span>
       </div>
       <div class="entry-content">
        <p>
         <img alt="Chicken Tetrazzini at The Happy Housewife" class="alignnone wp-image-20609 size-full" height="375" src="http://thehappyhousewife.com/cooking/files/2010/01/chicken-tetrazzini.jpg" title="Chicken Tetrazzini at The Happy Housewife" width="500"/>
        </p>
        <div class="easyrecipe" itemscope="" itemtype="http://schema.org/Recipe">
         <link href="http://thehappyhousewife.com/cooking/files/2010/01/chicken-tetrazzini-300x225.jpg" itemprop="image"/>
         <div class="ERSSavePrint">
          <span class="ERSPrintBtnSpan">
           <a class="ERSPrintBtn" href="http://thehappyhousewife.com/cooking/easyrecipe-print/9423-0/" rel="nofollow" target="_blank">
            Print
           </a>
          </span>
         </div>
         <div class="ERSName" itemprop="name">
          Chicken Tetrazzini
         </div>
         <div class="ERSClear">
         </div>
         <div class="ERSDetails">
          <div class="ERSHead">
           Recipe type:
           <span itemprop="recipeCategory">
            Main Dish
           </span>
          </div>
          <div class="ERSClear">
          </div>
          <div class="ERSTimes">
          </div>
          <div class="ERSClear">
          </div>
          <div class="ERSClear">
          </div>
         </div>
         <div class="ERSIngredients">
          <div class="ERSIngredientsHeader ERSHeading">
           Ingredients
          </div>
          <ul>
           <li class="ingredient" itemprop="ingredients">
            1 box noodles
           </li>
           <li class="ingredient" itemprop="ingredients">
            2 cooked and chopped chicken breasts
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup shredded zucchini
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup shredded carrots
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup chopped onion
           </li>
           <li class="ingredient" itemprop="ingredients">
            3 tablespoons butter
           </li>
           <li class="ingredient" itemprop="ingredients">
            1 tablespoon of minced garlic
           </li>
           <li class="ingredient" itemprop="ingredients">
            1 can cream of chicken soup
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup sour cream
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup milk or heavy cream
           </li>
           <li class="ingredient" itemprop="ingredients">
             cup parm cheese
           </li>
           <li class="ingredient" itemprop="ingredients">
            salt and pepper to taste
           </li>
          </ul>
          <div class="ERSClear">
          </div>
         </div>
         <div class="ERSInstructions">
          <div class="ERSInstructionsHeader ERSHeading">
           Instructions
          </div>
          <ol>
           <li class="instruction" itemprop="recipeInstructions">
            Cook the noodles.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            While the noodles are boiling, melt butter on low heat in a frying pan. Add garlic, onion, zucchini and carrots. Saute for about 3 minutes.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Add soup, sour cream, milk, cheese and chicken. Stir over low heat, until noodles are done boiling.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Drain noodles and add to the mixture. Toss well and serve immediately.
           </li>
          </ol>
          <div class="ERSClear">
