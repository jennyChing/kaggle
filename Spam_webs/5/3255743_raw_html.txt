    var slider_item_count = 2;
	var cur_item = 0;
	var first = true;
	var to;
	var slide_completed = true;
		function slider_out(item_id) {
		if (slide_completed) {
			slide_completed = false;
			if (item_id == '') {
				item_id = cur_item;
				clearTimeout(to);
			}
			$("#slider-price-"+item_id).fadeOut(300);
			if ($("#offer-of-month-"+item_id).length &gt; 0)
				$("#offer-of-month-"+item_id).fadeOut(300);
			$("#slider-price-"+item_id).fadeOut(300);
			$("#slider-button-"+item_id).fadeOut(300, function() {
				if ($("#offer-of-month-"+item_id).length &gt; 0) {
					$("#offer-of-month-"+item_id).css('width',0);
					$("#offer-of-month-"+item_id).css('height',0);
					$("#offer-of-month-"+item_id).show();
				}
				$("#slider-description-"+item_id).animate({
					left:1000
					},200,function() {
						$("#slider-img-small-"+item_id+"-1").animate({
							top:95
							},200,function() {
								$("#slider-img-small-"+item_id+"-2").animate({
									top:95
									},200,function() {
										$("#slider-img-small-"+item_id+"-1").css('top',-95);
										$("#slider-img-small-"+item_id+"-2").css('top',-95);
										$("#slider-img-big-"+item_id).fadeOut(300, function(){
											slider('next');
										});
									});
							});
					});
			});
		}
	}
		function slider(direction, item_id) {
		slide_completed = false;
		if (direction == 'next') {
			if (cur_item == slider_item_count - 1)
				item_id = 0;
			else
				item_id = cur_item + 1;
		} else if (direction == 'prev') {
			if (cur_item == 0)
				item_id = slider_item_count - 1;
			else
				item_id = cur_item - 1;
		}
		$("#slider-img-big-"+item_id).fadeIn(300, function(){
			$("#slider-img-small-"+item_id+"-1").animate({
				top:0
				},200,function() {
					$("#slider-img-small-"+item_id+"-2").animate({
						top:0
						},200,function() {
							$("#slider-description-"+item_id).animate({
								left:731
								},200,function() {
									$("#slider-price-"+item_id).fadeIn(300);
									$("#slider-button-"+item_id).fadeIn(300, function(){
										if ($("#offer-of-month-"+item_id).length &gt; 0) {
											$("#offer-of-month-"+item_id).animate({
													height:87,
													width:87
												}, 200, function() {
													slide_completed = true;
																										to = setTimeout(function(){
														slider_out(item_id);
													},6000);
																								});
										} else {
											slide_completed = true;
																						to = setTimeout(function(){
												slider_out(item_id);
											},6000);
																					}
									});
							});
					});
			});
		});
		cur_item = item_id;
	}
	$(window).load(function() {
		setTimeout(function(){

			},100);

	function show_slider_arrows(side) {
		if (side == 'right') {
			$("#slider-nav-right").addClass('active');
			$("#slider-nav-left").addClass('inactive');
		} else if (side == 'left') {
			$("#slider-nav-left").addClass('active');
			$("#slider-nav-right").addClass('inactive');
		}
	}
	
	function hide_slider_arrows() {
		$("#slider-nav-right").removeClass('active');
		$("#slider-nav-left").removeClass('active');

		$("#slider-nav-left").removeClass('inactive');
	}
