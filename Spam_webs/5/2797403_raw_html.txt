        <p>
         Are you looking for something easy, delicious and a little bit different for tonights supper? This recipe for Corn Chowder will definitely fit the bill.
        </p>
        <p>
         You probably have all of the necessary ingredients for this chowder in your pantry and fridge right now. This recipe takes under half an hour from start to finish and it is a warm comforting meal to serve on a cold night!
        </p>
        <p style="text-align: center;">
         <img alt="Corn Chowder" class="aligncenter wp-image-19041" height="527" src="http://oldermommystillyummy.com/wp-content/uploads/2014/02/036-edited1.jpg" width="608"/>
        </p>
        <p style="text-align: center;">
         Fry bacon until well cooked, remove to paper towel to drain. Discard all but two tablespoons of the bacon fat.
        </p>
        <p style="text-align: center;">
         <img alt="Corn Chowder" class="aligncenter size-full wp-image-19039" height="458" src="http://oldermommystillyummy.com/wp-content/uploads/2014/02/041-edited.jpg" width="608"/>
        </p>
        <p style="text-align: center;">
         After sauteing onions until translucent, add cooked potatoes and both types of corn to the pot.
        </p>
        <p style="text-align: center;">
         <img alt="Corn Chowder" class="aligncenter size-full wp-image-19042" height="496" src="http://oldermommystillyummy.com/wp-content/uploads/2014/02/047-edited.jpg" width="606"/>
        </p>
        <p style="text-align: center;">
         Barely cover vegetables with milk or cream.
        </p>
        <p style="text-align: center;">
         <img alt="Corn Chowder" class="aligncenter size-full wp-image-19040" height="538" src="http://oldermommystillyummy.com/wp-content/uploads/2014/02/082-pinterst-wshadow.jpg" width="498"/>
        </p>
        <p style="text-align: center;">
         A delicious and satisfying bowl of corn chowder is perfect for supper on a cold evening.
        </p>
        <p style="text-align: left;">
         NOTE  When I make this chowder I always make biscuits to go with it. I have a really great recipe for the
         <strong>
          <a href="http://oldermommystillyummy.com/2013/04/flaky-homemade-biscuits.html" target="_blank">
           best flaky biscuits
          </a>
         </strong>
         and they are very easy to make.
        </p>
        <p>
        </p>

         <div class="divERSHeadItems">
          <div class="ERSAuthor">
           Author:
           <span itemprop="author">
            Monica
           </span>
          </div>
          <div class="ERSCategory">
           Recipe type:
           <span itemprop="recipeCategory">
            Soups/Chowders
           </span>
          </div>
          <div class="ERSServes">
           Serves:
           <span itemprop="recipeYield">
            6 - 8
           </span>
          </div>
         </div>
         <div class="ERSIngredients">
          <div class="ERSIngredientsHeader ERSHeading">
           Ingredients
          </div>
          <ul>
           <li class="ingredient" itemprop="ingredients">
            5 Strips of Bacon, diced
           </li>
           <li class="ingredient" itemprop="ingredients">
            2 Medium sized onions, finely diced
           </li>
           <li class="ingredient" itemprop="ingredients">
            5 Potatoes, cut in  -  inch dice {I prefer a yellow fleshed potato}
           </li>
           <li class="ingredient" itemprop="ingredients">
            1 Can, 341 ml (12oz) corn niblets (Do not drain)
           </li>
           <li class="ingredient" itemprop="ingredients">
            2 Cans, 398 ml each (14 oz) creamed corn
           </li>
           <li class="ingredient" itemprop="ingredients">
            Milk (2 to 3 cups)
           </li>
           <li class="ingredient" itemprop="ingredients">
            Salt &amp; Pepper
           </li>
           <li class="ingredient" itemprop="ingredients">
            Green onion to garnish, finely chopped (optional)
           </li>
          </ul>
          <div class="ERSClear">
          </div>
         </div>
         <div class="ERSInstructions">
          <div class="ERSInstructionsHeader ERSHeading">
           Instructions
          </div>
          <ol>
           <li class="instruction" itemprop="recipeInstructions">
            In a dutch oven or large pot, fry bacon until crispy.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Remove bacon to paper towel to drain, Set aside.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Discard all but 2 Tbsp. bacon fat.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Boil cubed potatoes until tender. Drain and keep warm.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Heat bacon fat remaining in pot, add diced onions and cook until onions are translucent.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Add corn niblets with juice, creamed corn and cooked potatoes.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Barely cover contents of pot with milk.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Add bacon to chowder.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Season to taste with salt and pepper.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Stir and heat contents until hot, but DO NOT BOIL.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            For a richer chowder you may substitute 18% cream for all or part of the milk.
           </li>
           <li class="instruction" itemprop="recipeInstructions">
            Serve in warmed bowls, garnish with finely chopped green onions.
           </li>
          </ol>

          (function() {
  var form = document.getElementById("mad_mimi_signup_form"),
      submit = document.getElementById("webform_submit_button"),
      email = /.+@.+\..+/;
      
  form.onsubmit = function(event) {
    var isValid = validate();
    if(!isValid) {
      for(var i = 0; i &lt; form.elements.length; ++i) {
        var input = form.elements[i];
        if(input.className.indexOf("required") &gt;= 0) {
          input.onchange = validate;
        }
      }
      return false;
    }
    if(form.getAttribute("target") != "_blank") {
      form.className = "mimi_submitting";
      submit.value = submit.getAttribute("data-submitting-text");    
  		submit.disabled = true;
  		submit.className = "disabled";
    }

    setTimeout(function() {
      for(var i = 0; i &lt; form.getElementsByTagName("input").length; ++i) {
        var input = form.getElementsByTagName("input")[i];
        if(input.getAttribute("type") == "text") {
          input.value = "";
        }
        if(input.id == "signup_email") {
          input.placeholder = "you@example.com";
        } else {
          input.placeholder = "";
        }
      }
    }, 3000);
  };
  
  function validate() {
    var isValid = true;
    
    for(var i = 0; i &lt; form.elements.length; ++i) {
      var input = form.elements[i],
          allDivs = input.parentNode.getElementsByTagName("div");

      if(input.className.indexOf("required") &gt;= 0) {
        if(input.id == "signup_email") {
          if(!email.test(input.value)) {
            emailErrorMessage(input, allDivs);
            isValid = false;
          } else {
            removeErrorMessage(input, allDivs);
          }
        } else {
          if((input.type == "checkbox" &amp;&amp; !input.checked) || input.value == "" || input.value == "-1") {
            fieldErrorMessage(input, allDivs);
            isValid = false;
          } else {
            removeErrorMessage(input, allDivs);
          }
        }
      }
    }
    
    form.className = isValid ? "" : "mimi_invalid";
    submit.value = isValid ? submit.getAttribute("data-default-text") : submit.getAttribute("data-invalid-text");
		submit.disabled = !isValid;
		submit.className = isValid ? "submit" : "disabled";
    
    return isValid;
  }

  function emailErrorMessage(input, allDivs) {
    if(input.getAttribute("data-webform-type") == "iframe") {
      input.className = "required invalid";
      input.placeholder = input.getAttribute("data-required-message") || "This field is required";
    } else {
      allDivs[0].innerHTML = input.getAttribute("data-invalid-message") || "This field is invalid";
    }
  }

  function fieldErrorMessage(input, allDivs) {
    if(input.getAttribute("data-webform-type") == "iframe") {
      input.className = "required invalid";
      input.placeholder = input.getAttribute("data-required-message") || "This field is required";
    } else {
      for(var i = 0; i &lt; allDivs.length; ++i) {
        var element = allDivs[i];
        if(element.className.indexOf("mimi_field_feedback") &gt;= 0) {
          return element.innerHTML = input.getAttribute("data-required-message") || "This field is required";
        }
      }
    }
  }

  function removeErrorMessage(input, allDivs) {
    if(input.getAttribute("data-webform-type") == "iframe") {
      input.className = "required";
    }

    for(var i = 0; i &lt; allDivs.length; ++i) {
      var element = allDivs[i];
      if(element.className.indexOf("mimi_field_feedback") &gt;= 0) {
        return element.innerHTML = "";
      }
    }
  }
})();
