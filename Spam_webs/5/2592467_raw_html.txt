    function validateEmail(email){
      var re = /^(([^&lt;&gt;()[\]\\.,;:\s@\"]+(\.[^&lt;&gt;()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
      if(re.test(email)){
        return true;
      }else{
        return false;
      }
    };

    function validatePassword(password){
      var re = /^(?=.*[A-Z])[A-Za-z\d$@$!%*#?&amp;^()-_+=`~|]{8,}$/;
      if(re.test(password)){
        return true;
      }else{
        return false;
      }
    };

  var parseErrorResponse = function(responseText) {
    try { return jQuery.parseJSON(responseText); } catch(err) { return {errors: {}}; }
  };

  var formatErrorMessage = function(key, error) {
    var fragment = key + ' ' + error;
    return fragment.charAt(0).toUpperCase() + fragment.slice(1);
  };

  var handleErrors = function(form, jqxhr) {
    var errorMessages = [];
    var response = parseErrorResponse(jqxhr.responseText);
    if (response.errors) {
      _.each(response.errors, function(errors, key) {
        if (key == 'base') {
          errorMessages = errorMessages.concat(errors);
        }
        else {
          errors.forEach(function(error){ errorMessages.push(formatErrorMessage(key, error));});
          var inputSelector = 'input[name*="[' + key + ']"]';
          form.find(inputSelector).parents('.control-group').addClass('error');
          $('#registration_email_error').hide();
          $('#registration_password_error').hide();
        }
      });
    }
    if (errorMessages.length == 0) {

      $('#registration_email_success').hide();
      $('#registration_password_error').hide();
      $('#registration_password_success').hide();
    }
    form.find('.generic-error').html(errorMessages.join("&lt;br /&gt;")).slideDown(250);
  };

    form.find('.control-group').removeClass('error');
