                this.pingTimerId = window.setTimeout(function () {
                    self.ping()
                }, this.pingTimeInterval);
            };

            this.ping = function () {
                this.pingServer();
                if (this.timeLimitReached()) {
                    this.stop();
                } else {
                    this.incrementPingTimeInterval();
                    this.runPingLoop();
                }
            };

            this.timeLimitReached = function () {
                return this.pingTimeInterval &gt; MAX_TRACKING_TIME_LIMIT;
            };

            /**
             * 2X time : from 5 sec to 10 min (5, 10, 20, 40, 80, 160, ...)
             * +10 min : from 10 min to 1 hour
             * +15 min : from 1 hour to MAX_TRACKING_TIME_LIMIT

            this.incrementPingTimeInterval = function () {
                if (this.pingTimeInterval &gt;= 10 * MINUTE) {
                    this.pingTimeInterval += 10 * MINUTE;
                } else if (this.pingTimeInterval &gt;= HOUR) {
                    this.pingTimeInterval += 15 * MINUTE;
                } else {
                    this.pingTimeInterval *= 2;
                }
            };

            this.pingServer = function () {
                var beaconPixel = new Image();
                // We use a timestamp to invalidate image cache
                var timestamp = new Date().getTime();
                // Setting the Image.src hits the server
                beaconPixel.src = PING_URL + '?timestamp=' + timestamp;
            };
        }

            window.__smoreTimeTracker.start();
