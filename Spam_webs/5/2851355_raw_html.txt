      function loadAfterAd () {
        var $after = $('.ad-after');

        if ( $after.length &gt; 0 ) {

                bWidth = $banner.find('iframe').width();

            if ( bWidth !== null &amp;&amp; bWidth &lt;= 728 ) {
                $after.empty();
                $after.dfp();
            }
        }

                $element.addClass('visible');
            }
        });

    /**

    function interstitialAd(adunit, adsize, divId) {

                'enableServices();googletag.display("'+divId+'");});';
