/**
* Initialize prettyPhoto.
*/
$.prettyPhoto.initialize = function() {
settings = video_settings;
if($.browser.msie &amp;&amp; parseInt($.browser.version) == 6) settings.theme = "light_square"; // Fallback to a supported theme for IE6
_buildOverlay(this); // Build the overlay {this} being the caller
setItem = (_getFileType($(this).attr('href')) != 'youtube') ? $(this).attr('href').toLowerCase() : $(this).attr('href');
setPosition = jQuery.inArray(setItem, video_images); // Define where in the array the clicked item is positionned
$.prettyPhoto.open();
return false;
}
/**
* Opens the prettyPhoto modal box.
* @param image {String,Array} Full path to the image to be open, can also be an array containing full images paths.
* @param title {String,Array} The title to be displayed with the picture, can also be an array containing all the titles.
* @param description {String,Array} The description to be displayed with the picture, can also be an array containing all the descriptions.
*/
$.prettyPhoto.open = function() {
if(typeof settings == "undefined"){ // Means it's an API call, need to manually get the settings and set the variables
settings = video_settings;
_buildOverlay(this); // Build the overlay {this} being the caller
video_images = $.makeArray(arguments[0]);
video_titles = (arguments[1]) ? $.makeArray(arguments[1]) : $.makeArray("");
video_descriptions = (arguments[2]) ? $.makeArray(arguments[2]) : $.makeArray("");
isSet = (video_images.length &gt; 1) ? true : false;
setPosition = 0;
}
if($.browser.msie &amp;&amp; $.browser.version == 6) $('select').css('visibility','hidden'); // To fix the bug with IE select boxes
if(settings.hideflash) $('object,embed').css('visibility','hidden'); // Hide the flash
_checkPosition($(video_images).size()); // Hide the next/previous links if on first or last images.
$('.video_loaderIcon').show();
// Fade the content in
if($ppt.is(':hidden')) $ppt.css('opacity',0).show();
$video_overlay.show().fadeTo(settings.animationSpeed,settings.opacity);
// Display the current position
$video_pic_holder.find('.currentTextHolder').text((setPosition+1) + settings.counter_separator_label + $(video_images).size());
// Set the description
$video_pic_holder.find('.video_description').show().html(unescape(video_descriptions[setPosition]));
// Set the title
(settings.showTitle &amp;&amp; video_titles[setPosition] != "") ? $ppt.html(unescape(video_titles[setPosition])) : $ppt.html('&nbsp;');
// Get the dimensions
movie_width = ( parseFloat(grab_param('width',video_images[setPosition])) ) ? grab_param('width',video_images[setPosition]) : settings.default_width.toString();
movie_height = ( parseFloat(grab_param('height',video_images[setPosition])) ) ? grab_param('height',video_images[setPosition]) : settings.default_height.toString();
// If the size is % based, calculate according to window dimensions
if(movie_width.indexOf('%') != -1 || movie_height.indexOf('%') != -1){
movie_height = parseFloat(($(window).height() * parseFloat(movie_height) / 100) - 100);
movie_width = parseFloat(($(window).width() * parseFloat(movie_width) / 100) - 100);
percentBased = true;
};
// Fade the holder
$video_pic_holder.fadeIn(function(){
imgPreloader = "";
// Inject the proper content
switch(_getFileType(video_images[setPosition])){
case 'image':
imgPreloader = new Image();
// Preload the neighbour images
nextImage = new Image();
if(isSet &amp;&amp; setPosition &gt; $(video_images).size()) nextImage.src = video_images[setPosition + 1];
prevImage = new Image();
if(isSet &amp;&amp; video_images[setPosition - 1]) prevImage.src = video_images[setPosition - 1];
$video_pic_holder.find('#video_full_res')[0].innerHTML = settings.image_markup;
$video_pic_holder.find('#fullResImage').attr('src',video_images[setPosition]);

};

};

break;

/**
* Change page in the prettyPhoto modal box
* @param direction {String} Direction of the paging, previous or next.
*/
$.prettyPhoto.changePage = function(direction){
currentGalleryPage = 0;
if(direction == 'previous') {
setPosition--;
if (setPosition &lt; 0){
setPosition = 0;
return;
};
}else if(direction == 'next'){
if(video_slideshow &amp;&amp; $('.video_arrow_next').is('.disabled')) {
setPosition = -1;
}else if($('.video_arrow_next').is('.disabled')){
return;
}
setPosition++;
}else{
setPosition=direction;
};
if(!doresize) doresize = true; // Allow the resizing of the images
_hideContent(function(){ $.prettyPhoto.open(); });
};
/**
* Change gallery page in the prettyPhoto modal box
* @param direction {String} Direction of the paging, previous or next.
*/
$.prettyPhoto.changeGalleryPage = function(direction){
if(direction=='next'){
currentGalleryPage ++;
if(currentGalleryPage &gt; totalPage){
currentGalleryPage = 0;
};
}else if(direction=='previous'){
currentGalleryPage --;
if(currentGalleryPage &lt; 0){
currentGalleryPage = totalPage;
};
};
// Slide the pages, logic ++
$video_pic_holder.find('.video_gallery li').each(function(i){
$(this).animate({
'left': (i * itemWidth) - ((itemsPerPage * itemWidth) * currentGalleryPage)
});
});
};
/**
* Start the slideshow...
*/
$.prettyPhoto.startSlideshow = function(){
if(typeof video_slideshow == 'undefined'){			$video_pic_holder.find('.video_play').unbind('click').removeClass('video_play').addClass('video_pause').click(function(){
$.prettyPhoto.stopSlideshow();
return false;
});
}else{
$.prettyPhoto.changePage('next');
};
video_slideshow = setTimeout($.prettyPhoto.startSlideshow,settings.slideshow);
}
/**
* Stop the slideshow...
*/
$.prettyPhoto.stopSlideshow = function(){		$video_pic_holder.find('.video_pause').unbind('click').removeClass('video_pause').addClass('video_play').click(function(){
$.prettyPhoto.startSlideshow();
return false;
});
clearTimeout(video_slideshow);
}
/**
* Closes the prettyPhoto modal box.
*/
$.prettyPhoto.close = function(){
clearTimeout(video_slideshow);	
$video_pic_holder.find('object,embed').css('visibility','hidden');
$('div.video_pic_holder,div.ppt,.video_fade').fadeOut(settings.animationSpeed,function(){ $(this).remove(); });
$video_overlay.fadeOut(settings.animationSpeed, function(){
if($.browser.msie &amp;&amp; $.browser.version == 6) $('select').css('visibility','visible'); // To fix the bug with IE select boxes
if(settings.hideflash) $('object,embed').css('visibility','visible'); // Show the flash
$(this).remove(); // No more need for the prettyPhoto markup
settings.callback();
doresize = true;
delete settings;
});
};
/**
* Set the proper sizes on the containers and animate the content in.
*/
_showContent = function(){
$('.video_loaderIcon').hide();
$ppt.fadeTo(settings.animationSpeed,1);
// Calculate the opened top position of the pic holder
projectedTop = scrollPos['scrollTop'] + ((windowHeight/2) - (correctSizes['containerHeight']/2));
if(projectedTop &lt; 0) projectedTop = 0;
// Resize the content holder
			$video_pic_holder.find('.video_area').animate({'height':correctSizes['contentHeight']},settings.animationSpeed);
// Resize picture the holder
$video_pic_holder.animate({
'top': projectedTop,
'left': (windowWidth/2) - (correctSizes['containerWidth']/2),
'width': correctSizes['containerWidth']
},settings.animationSpeed,function(){			$video_pic_holder.find('.video_hoverContainer,#fullResImage').height(correctSizes['height']).width(correctSizes['width']);
$video_pic_holder.find('.video_fade').fadeIn(settings.animationSpeed); // Fade the new content
// Show the nav
if(isSet &amp;&amp; _getFileType(video_images[setPosition])=="image") { $video_pic_holder.find('.video_hoverContainer').show(); }else{ $video_pic_holder.find('.video_hoverContainer').hide(); }
if(correctSizes['resized']) $('a.video_box_expand,a.video_content').fadeIn(settings.animationSpeed); // Fade the resizing link if the image is resized
settings.changepicturecallback(); // Callback!
});
if(isSet &amp;&amp; settings.overlay_gallery &amp;&amp; _getFileType(video_images[setPosition])=="image") {
itemWidth = 52+5; // 52 beign the thumb width, 5 being the right margin.
navWidth = (settings.theme == "facebook") ? 58 : 38; // Define the arrow width depending on the theme
itemsPerPage = Math.floor((correctSizes['containerWidth'] - 100 - navWidth) / itemWidth);
itemsPerPage = (itemsPerPage &lt; video_images.length) ? itemsPerPage : video_images.length;
totalPage = Math.ceil(video_images.length / itemsPerPage) - 1;
// Hide the nav in the case there's no need for links
if(totalPage == 0){
navWidth = 0; // No nav means no width!
$video_pic_holder.find('.video_gallery .video_arrow_next,.video_gallery .video_arrow_previous').hide();
}else{
navWidth = (settings.theme == "facebook") ? 58 : 38; // Define the arrow width
$video_pic_holder.find('.video_gallery .video_arrow_next,.video_gallery .video_arrow_previous').show();
};
galleryWidth = itemsPerPage * itemWidth + navWidth;
// Set the proper width to the gallery items
$video_pic_holder.find('.video_gallery')
.width(galleryWidth)
.css('margin-left',-(galleryWidth/2));
$video_pic_holder.find('.video_gallery ul').width(itemsPerPage * itemWidth);
// Reposition the LIs
$video_pic_holder.find('.video_gallery ul li').each(function(i){
$(this).css({
'position':'absolute',
'left': i * itemWidth
});
});
}else{
$video_pic_holder.find('.video_area').unbind('mouseenter mouseleave');
$video_pic_holder.find('.video_gallery').hide();
};		
if(settings.autoplay_slideshow) {
video_slideshow = clearTimeout(video_slideshow);
$.prettyPhoto.startSlideshow();
settings.autoplay_slideshow = false; // Because we only want it to autostart on the first opening.
};
};	
/**
* Hide the content...DUH!
*/
function _hideContent(callback){
// Fade out the current picture
$video_pic_holder.find('#video_full_res object,#video_full_res embed').css('visibility','hidden');
$video_pic_holder.find('.video_fade').fadeOut(settings.animationSpeed,function(){
$('.video_loaderIcon').show();	
callback();
});
};
/**
* Check the item position in the gallery array, hide or show the navigation links
* @param setCount {integer} The total number of items in the set
*/
function _checkPosition(setCount){
// If at the end, hide the next link
if(setPosition == setCount-1) {
$video_pic_holder.find('a.video_button_next').css('visibility','hidden');
$video_pic_holder.find('.video_nav a.video_arrow_next').addClass('disabled').unbind('click');
}else{ 
$video_pic_holder.find('a.video_button_next').css('visibility','visible');
$video_pic_holder.find('.video_nav a.video_arrow_next.disabled').removeClass('disabled').bind('click',function(){
$.prettyPhoto.changePage('next');
return false;
});
};
// If at the beginning, hide the previous link
if(setPosition == 0) {
$video_pic_holder.find('a.video_button_previous').css('visibility','hidden');
$video_pic_holder.find('.video_nav a.video_arrow_previous').addClass('disabled').unbind('click');
}else{
$video_pic_holder.find('a.video_button_previous').css('visibility','visible');
$video_pic_holder.find('.video_nav a.video_arrow_previous.disabled').removeClass('disabled').bind('click',function(){
$.prettyPhoto.changePage('previous');
return false;
});
};
(setCount &gt; 1) ? $('.video_nav').show() : $('.video_nav').hide(); // Hide the bottom nav if it's not a set.
};
/**
* Resize the item dimensions if it's bigger than the viewport
* @aram width {integer} Width of the item to be opened
* @param height {integer} Height of the item to be opened
* @return An array containin the "fitted" dimensions
*/
function _fitToViewport(width,height){
resized = false;
_getDimensions(width,height);		
// Define them in case there's no resize needed
imageWidth = width, imageHeight = height;
if( ((video_containerWidth &gt; windowWidth) || (video_containerHeight &gt; windowHeight)) &amp;&amp; doresize &amp;&amp; settings.allowresize &amp;&amp; !percentBased) {
resized = true, fitting = false;
while (!fitting){
if((video_containerWidth &gt; windowWidth)){
imageWidth = (windowWidth - 200);
imageHeight = (height/width) * imageWidth;
}else if((video_containerHeight &gt; windowHeight)){
imageHeight = (windowHeight - 200);
imageWidth = (width/height) * imageHeight;
}else{
fitting = true;
};
video_containerHeight = imageHeight, video_containerWidth = imageWidth;
};
_getDimensions(imageWidth,imageHeight);
};
return {
width:Math.floor(imageWidth),
height:Math.floor(imageHeight),
containerHeight:Math.floor(video_containerHeight),
containerWidth:Math.floor(video_containerWidth) + 40, // 40 behind the side padding
contentHeight:Math.floor(video_areaHeight),
contentWidth:Math.floor(video_areaWidth),
resized:resized
};
};
		
/**
* Get the containers dimensions according to the item size
* @param width {integer} Width of the item to be opened
* @param height {integer} Height of the item to be opened
*/
function _getDimensions(width,height){
width = parseFloat(width);
height = parseFloat(height);
// Get the details height, to do so, I need to clone it since it's invisible
$video_details = $video_pic_holder.find('.video_details');
$video_details.width(width);
detailsHeight = parseFloat($video_details.css('marginTop')) + parseFloat($video_details.css('marginBottom'));
$video_details = $video_details.clone().appendTo($('body')).css({
'position':'absolute',
'top':-10000
});
detailsHeight += $video_details.height();
detailsHeight = (detailsHeight &lt;= 34) ? 36 : detailsHeight; // Min-height for the details
if($.browser.msie &amp;&amp; $.browser.version==7) detailsHeight+=8;
$video_details.remove();
// Get the container size, to resize the holder to the right dimensions
video_areaHeight = height + detailsHeight;
video_areaWidth = width;
video_containerHeight = video_areaHeight + $ppt.height() + $video_pic_holder.find('.video_top').height() + $video_pic_holder.find('.video_bottom').height();
video_containerWidth = width;
}
function _getFileType(itemSrc){
if (itemSrc.match(/youtube\.com\/watch/i)) {
return 'youtube';
}else if (itemSrc.match(/vimeo\.com/i)) {
return 'vimeo';
}else if(itemSrc.indexOf('.mov') != -1){ 
return 'quicktime';
}else if(itemSrc.indexOf('.swf') != -1){
return 'flash';
}else if(itemSrc.indexOf('iframe') != -1){
return 'iframe'
}else if(itemSrc.substr(0,1) == '#'){
return 'inline';
}else{
return 'image';
};
};
function _centerOverlay(){
if(doresize &amp;&amp; typeof $video_pic_holder != 'undefined') {
titleHeight = $ppt.height(), contentHeight = $video_pic_holder.height(), contentwidth = $video_pic_holder.width();
projectedTop = (windowHeight/2) + scrollPos['scrollTop'] - (contentHeight/2);
$video_pic_holder.css({
'top': projectedTop,
'left': (windowWidth/2) + scrollPos['scrollLeft'] - (contentwidth/2)
});
};
};
function _getScroll(){
if (self.pageYOffset) {
return {scrollTop:self.pageYOffset,scrollLeft:self.pageXOffset};
} else if (document.documentElement &amp;&amp; document.documentElement.scrollTop) { // Explorer 6 Strict
return {scrollTop:document.documentElement.scrollTop,scrollLeft:document.documentElement.scrollLeft};
} else if (document.body) {// all other Explorers
return {scrollTop:document.body.scrollTop,scrollLeft:document.body.scrollLeft};
};
};
function _resizeOverlay() {
windowHeight = $(window).height(), windowWidth = $(window).width();
if(typeof $video_overlay != "undefined") $video_overlay.height($(document).height());
};
function _buildOverlay(caller){
// Find out if the picture is part of a set
theRel = $(caller).attr('rel');
galleryRegExp = /\[(?:.*)\]/;
