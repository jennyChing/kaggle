					$('a[id=LocationId_Id_clearAll]').css('display','block');
				}
			});
			
		// -- Clear All --
		window.$('a[id=LocationId_Id_clearAll]').bind("click", function(){				
				//Clear locationIds
				OnClearLocations_LocationId();
				
				//Clear display
				window.$( "input[id=LocationId_Id_display]" ).val('');
				// Hide clear btn
				$('a[id=LocationId_Id_clearAll]').css('display','none');				
		});

		$('input[id=LocationId_Id_display]').blur(function() {
			if( !$(this).val()) {
				$('a[id=LocationId_Id_clearAll]').css('display','none');	
			}
		});
		
		// -- Auto-complete
		window.$('input[id=LocationId_Id_display]').autocomplete({
            source: '/location-lookup/'
        });    
		
		window.$( "input[id=LocationId_Id_display]" ).bind( "autocompleteselect", function(event, ui) {

			//Clear existing selection
			OnClearLocations_LocationId();
			
			//Add location hidden Id field			
			window.$("div[id=LocationId_Id_selected]").append(
				'&lt;input type="hidden" name="LocationId" value="' + ui.item.locationId + '" /&gt;');

			
				var getNearLocationsUrl = '/location-near-facets/?LocationId=' + ui.item.locationId;
				
				
				
				$('input[id=LocationId_Id_display]').addClass('ui-autocomplete-loading');

				//Add near location facets
				$.getJSON(
					getNearLocationsUrl,
					function(data){
						for (var i = 0; i &lt; data.results.length; i++) {			
							var item = data.results[i];
							var label = item.label;
							if(item.jobCount &gt; 0)
								label = label + ' (' + item.jobCount + ')';
							
							OnAddLocation_LocationId(item.locationId, label, item.isDefaultSelection);
						}

						$('input[id=LocationId_Id_display]').removeClass('ui-autocomplete-loading');
					}
				);				

			
						
		}).bind("autocompletefocus", function(event, ui){			
			event.preventDefault();			
		});
	
		function OnClearLocations_LocationId()
		{				
			window.$('div[id=LocationId_Id_selected]').each(function(){
					$(this).empty();
			});			
			idx = 1;
		}
	
			
			function OnAddLocation_LocationId(locationId, label, isChecked)
			{
				if(idx == 1){					
					window.$("div[id=LocationId_Id_selected]").append(
						'&lt;p&gt;&lt;strong&gt;Show me jobs nearby&lt;/strong&gt;&lt;/p&gt;' + 
						'&lt;input type="hidden" name="NearFacetsShown" value="True" /&gt;' );
				}
				
				var pId = "LocationId_Id_selected_" + idx;
				var inputId = "LocationId_Id_input_" + idx;
				
				var checkedAttr = ' ';
				if(isChecked)
					checkedAttr = ' checked="checked" ';
				
				window.$("div[id=LocationId_Id_selected]").append('&lt;p id="' + pId + '"&gt;' + 
					'&lt;input id="' + inputId + '" type="checkbox" name="LocationId" value="' + locationId + '" ' +
						checkedAttr + '/&gt;' +
					'&lt;label for="' + inputId + '"&gt;' + label + '&lt;/label&gt;&lt;/p&gt;');
				
				idx = idx + 1;			
			}		
		
		
	});
