                        stringPrice1 += data.data.pricetax;
                        stringDiscount = '';
                        if(data.data.discount){
                            stringDiscount = '&lt;br /&gt;&lt;div class="discount"&gt;';
                            $.each(data.data.discount, function(discountKey, discountAmt) { 
                                stringDiscount += discountAmt+'&lt;br /&gt;';
                            });
                            stringDiscount += '&lt;/div&gt;';
                        }

                            }else{
