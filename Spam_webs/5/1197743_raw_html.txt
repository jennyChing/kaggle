        function setUrl(key, value) {

    uri = window.location.search;

    var re = new RegExp("([?&amp;])" + key + "=.*?(&amp;|$)", "i");
    var separator = uri.indexOf('?') !== -1 ? "&amp;" : "?";
    var qs;

    if (uri.match(re)) {
        qs = uri.replace(re, '$1' + key + "=" + value + '$2');
    } else {
        qs = uri + separator + key + "=" + value;
    }

    window.location.replace(window.location.href.split('?')[0] + qs);
}
