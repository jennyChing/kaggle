  function resizeEnd(callback) {
    var rtime = new Date(1, 1, 2000, 12,00,00);
    var timeout = false;
    var delta = 200;
    $(window).resize(function() {
      rtime = new Date();
      if (timeout === false) {
        timeout = true;
        setTimeout(resizeend, delta);
      }
    });

    function resizeend() {
      if (new Date() - rtime &lt; delta) {
        setTimeout(resizeend, delta);
      } else {
        timeout = false;
        callback();

    function removeDeletedTrees(listIdentifier) {
      $(listIdentifier).children(".removed").each(function() {

    function ajaxSortCall(container, sortParam, callback_on_complete) {
      var url = '/comments';
      var talkboxid = '1001-CRNEWS_49839';
      var loading_message = '&lt;div class="loading-message cf"&gt;&lt;div class="left strong"&gt;Loading...&lt;/div&gt;';
      var replyCount = '5';
      var threadlimit = '2';
      var pagelimit = '20';
      $("#"+container).children().remove();
      $("#"+container).prepend(loading_message);
      $.get(url, {pg:0, talkboxid:talkboxid, sort:sortParam, replycount:replyCount, threadlimit:threadlimit-1, pagelimit:pagelimit}, function(data) {
	data = jQuery.parseJSON(data);
        if (data) {
          $("#singleComment").tmpl(data.slice(0,parseInt(pagelimit)), {thread_limit: 0}).appendTo("#"+container);
          removeDeletedTrees("#"+container);
        }
        $("#"+container).children(".loading-message").remove();
        $(".reply-form").each(function() {
          Guestbook.AddPlaceholder(this);
        });

        if (typeof callback_on_complete === "function") {
          callback_on_complete();
        }
      });
      $(".more_comments_box").children().remove();
      $("#moreCommentsButton").tmpl({sortParams:sortParam}).appendTo(".more_comments_box");
    }

    function getNextPage(container, sortParams) {
      var url = '/comments';
      var talkboxid = '1001-CRNEWS_49839';
      var loading_message = '&lt;div class="loading-message cf"&gt;&lt;div class="left strong"&gt;Loading...&lt;/div&gt;';
      var pg = parseInt($('.paging-bar').attr('data-page')) + 1;
      var threadlimit = '1';
      var pagelimit = '20';
      $(function() {
        if (pg &gt; $('.paging-bar').attr('data-page-limit')) {
          $('.next-button-med').attr('class', 'control-button next-button-light');
          return;
        }
        $('.next-button-med').addClass('control-loading');
        $.get(url, {pg:pg, talkboxid:talkboxid, sort:sortParams, replycount:0, threadlimit:threadlimit, pagelimit:pagelimit}, function(data) {
          data = jQuery.parseJSON(data);
          if (!data) {
            return false;
          }
          if (data.length &lt;= parseInt(pagelimit)) {
            $('.next-button-med').attr('class', 'control-button next-button-light');
            $('.paging-bar').attr('data-page-limit', pg);
          }
          if (pg &gt; 0) {
            $('.prev-button-light').attr('class', 'control-button prev-button-med');
          }
          $('.paging-bar').attr('data-page', pg);
          $('.page-display').text('Page ' + (parseInt($('.paging-bar').attr('data-page'))+1).toString());
          $("#"+container).children().remove();
          $("#singleComment").tmpl(data.slice(0,parseInt(pagelimit)), {thread_limit: 0}).appendTo("#"+container);
          $('.control-loading').removeClass('control-loading');
          removeDeletedTrees("#"+container);
          $(".reply-form").each(function() {
            Guestbook.AddPlaceholder(this);
          });
        });
      });
    }
