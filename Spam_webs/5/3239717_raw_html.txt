/**
 * Calls {@link window.htmlAdWH} method with arguments defined by the members of a ads hash object.
 * @alias window.AUTOBLOG.ads.adCall
 * @param {object} ads Ads hash object.
 * @param {string} layout Sets the ad type. `desktop|mobile|tablet`
 * @example
 * AUTOBLOG.ads.adCall(AUTOBLOG.ads.desktop, 'desktop');
 */
AUTOBLOG.ads.adCall = function(ads, layout){
	atwLoaded = 1;
	

	if(typeof ads == 'string'){
		ads = AUTOBLOG.ads[ads];
	}else if(ads == null){
		ads = AUTOBLOG.ads.desktop;
	}

	AUTOBLOG.ads.adCleanup();

	for (var key in ads) {
		var obj = ads['mn'] !== undefined ? ads : ads[key];

		if(obj.hasOwnProperty('mn')) {
			if(document.getElementById(key)){
				adVis(key, 1);

				htmlAdWH(obj['mn'],obj['width'],obj['height'],obj['adType'],key);

				adVis(key, 0);

				if(layout){
					document.getElementById(key).setAttribute('data-adtype', layout);
				}
			}
		}
	}

	function adVis(key, value) {
		var adSetWhitelist = ['ad-tile', 'ad-tile-refresh', 'ad-btf-native', 'ad-banner-btf'];

		if(value === 1){
			if(adSetWhitelist.indexOf(key) &gt; -1){
				if(key === 'ad-tile' || key === 'ad-tile-refresh'){
					if(layout === 'mobile'){
						adSetInView('1');
					}
				}else{
					adSetInView('1');
				}
			}
		}else if(value === 0){
			if(adSetWhitelist.indexOf(key) &gt; -1){
				if(key === 'ad-tile' || key === 'ad-tile-refresh'){
					if(layout === 'mobile'){
						adSetInView('0');
					}
				}else{
					adSetInView('0');
				}
			}
		}
	}
}
