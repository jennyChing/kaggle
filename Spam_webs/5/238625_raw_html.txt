$(window).resize(function(){
	resize_drowdowns();
	resize_dropdown_items();

})

function resize_drowdowns() {
	$(".dropdown_item").each(function(event) {
		var total_width = $(this).parent().parent().parent().parent().width();
		var position = ($(this).parent().parent().width() - $(this).parent().width());
		var new_width = total_width - position;

		$(this).width(new_width);
	});
}

function resize_dropdown_items() {
	var counter = 0 ;
	var largest_height = 0;

	$('.thing-item').css('height','auto');

	if($(window).width() &gt; 690) {
		$(".things-row").each(function(event) {
			var current_row = $(this);
			largest_height = 0;

			current_row.children('.col-sm-4').each(function(event) {
				if($(this).children('.row').height() &gt; largest_height){ largest_height = $(this).children('.row').height(); }
			});

			current_row.children('.col-sm-4').each(function(event) {
				$(this).children('.row').height(largest_height);
			});
		});
	}

	//alert(largest_height);
}

function resize_dropdown_overlay() {
	$(".dropdown_overlay").each(function(event) {
		//Set Width
		$(this).width($(this).parent().width());

		//Set Height
		$(this).height($(this).children('.dropdown_overlay_inner').height());
	});
}

	window.location = url;
