    var clientWindow = null;
var lightbox = null;
var lightboxOverlay = null;
var lightboxClose = null;
var lightboxContentArea = null;
var lightboxTemplate = null;

function realignLightBox() {
	if (lightbox.attr("lb-type") != "asset") {
		lightbox.center();
	}
}

function loadLightBoxAsset(assetId) {
	var contentURL = "/on/demandware.store/Sites-burtsbees-Site/default/Page-Include?cid=REPLACETHISCONTENTID";
	
	jQuery.ajax({
		dataType: "html",
		url: contentURL.replace("REPLACETHISCONTENTID", assetId)
	}).done(function(html){
		lightboxContentArea.empty();
		lightboxContentArea.append(lightboxTemplate.children(":first-child").clone());
		lightbox.removeClass("loading").removeClass("printing");
		lightboxContentArea.find(".WFBcontentwrapper").append(html);
		realignLightBox();
		runLightBoxEvent(lightboxTemplate.attr("lb-onshow"), null);
	});
}

function openLightBox(options) {
	lightboxTemplate = jQuery("#" + options.lbTemplateId);
	lightbox.attr("lb-type", options.lbType).addClass(options.lbType);

	var customClass = lightboxTemplate.attr("lb-class");
	lightbox.addClass(customClass).attr("lb-class", customClass);	
	
	if (options.lbType == "asset") {
		lightbox.addClass("loading");
	}
	
	lightbox.attr("lb-onclose", lightboxTemplate.attr("lb-onclose"));
	runLightBoxEvent(lightboxTemplate.attr("lb-oninit"), null);

	switch (options.lbType) {
		case "video" :
			// Video Type Lightbox
			lightboxContentArea.append(lightboxTemplate.children(":first-child").clone());
			lightboxContentArea.find("iframe").attr("src", options.lbRef);
			runLightBoxEvent(lightboxTemplate.attr("lb-onshow"), null); 
			break;
		case "asset" :
			// Content Asset Type Lightbox
			loadLightBoxAsset(options.lbRef.substr(1));		// Remove 1st character
			break;	
		case "custom" :
			// Custom Type Lightbox
			lightboxContentArea.append(lightboxTemplate.children(":first-child").clone());
			runLightBoxEvent(lightboxTemplate.attr("lb-onshow"), null); 
			break;	
		default:
			// Do nothing
			break;
	}

	lightboxOverlay.show();
	lightbox.show();
	if (options.lbType == "asset") {
		lightbox.center({ vertical : false });
		lightbox.css("top", clientWindow.scrollTop() + "px");
	}
	else {
		realignLightBox();
	}	
}

function closeLightBox() {
	lightboxOverlay.hide();
	lightbox.hide();
	switch (lightbox.attr("lb-type")) {
		case "video" :
			lightbox.removeClass("video");	
			break;
		case "asset" :
			lightbox.removeClass("asset").removeClass("loading").removeClass("printing");	
			break;
		case "custom" :
			lightbox.removeClass("custom");	
			break;
		default:
			// Do nothing
			break;
	}
	lightboxContentArea.empty();
	var closeEventName = lightbox.attr("lb-onclose");
	lightbox.removeAttr("lb-onclose").removeAttr("lb-type").removeClass(lightbox.attr("lb-class")).removeAttr("lb-class");
	//runLightBoxEvent(closeEventName, null);
}

function runLightBoxEvent(eventName, arguments) {
	if (eventName == null) { return }
	var event = window[eventName];
	if (typeof event !== 'function') { return }
	event.apply(window, arguments);
}

jQuery(document).ready(function() {
	clientWindow = jQuery(window);
	lightbox = jQuery("#colorbox");
	lightboxOverlay = jQuery("#cboxOverlay");
	lightboxClose = jQuery("#cboxClose");
	lightboxContentArea = jQuery("#cboxLoadedContent");
	
	jQuery("a.lightbox").click(function(e){
		e.preventDefault();
		e.stopPropagation();
		var lightboxType = "";
		var lightboxTemplateId = "";
		var lbLink = jQuery(this);
		
		if (lbLink.hasClass("wfbVideo")) {
			lightboxType = "video";
			lightboxTemplateId = "videoboxtemplate";
		}
		else {
			if (lbLink.hasClass("lbasset")) {
				lightboxType = "asset";
				lightboxTemplateId = lbLink.attr("lb-templateid")
			}
		}	
		
		openLightBox({ 
			lbType : lightboxType, 
			lbTemplateId : lightboxTemplateId, 
			lbRef : lbLink.attr("href")
		});		
		return false;
	});
	
	clientWindow.resize(realignLightBox);
	clientWindow.scroll(realignLightBox);
	lightboxClose.click(closeLightBox);	
	lightboxOverlay.click(closeLightBox);
});
