       <script type="text/javascript">
        $(document).ready(function() {
    $('#loginform').bind("submit", function(el) {
        var errormsg = '';

        var username = $("#login_username", el).val();
        if (username.length == 0) {
            errormsg = "Please specify your email address.";
            $(".emailFieldError").html(errormsg).show();
        } else {
            $(".emailFieldError").hide();
        }
        
        var password = $("#login_password", el).val();
        if (password.length == 0) {
            errormsg = "Please specify your password.";
            $(".passwordFieldError").html(errormsg).show();
        } else {
            $(".passwordFieldError").hide();
        }

        if(errormsg.length &gt; 0) return false;
        return true;
    });
    });
       </script>
