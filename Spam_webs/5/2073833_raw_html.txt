var trackConversion = function(url, scope) {
  var segmentRecorded = false
  var googleRecorded  = false

  var didReceiveRecord = function(){
    if(googleRecorded &amp;&amp; segmentRecorded) {
      document.location = url
      console.log('-- Conversions finished : redirect now')
    }
