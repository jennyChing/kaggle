	function publishComment(theBody) {
		// hack the FB object because it doesn't update the session
		FB.Connect._singleton._userInfo.shortStorySetting = FB.FeedStorySetting.autoaccept;
		FB.Connect.streamPublish(theBody, {'name': '- ' + document.title, 'href': window.location.href}, [{'href': window.location.href, 'text': 'Visit Site'}], null, null, function(post_id, ex) { jQuery("#comment_form").submit() }, true, null);
	}

	function onSubmit(commentId) {
		var theBody = (typeof punymce != "undefined") ? eval(commentId).getContent({save : true}) : document.getElementById(commentId).value;

		if (!validateComment(theBody))
			return false;

		if (jQuery("#facebookPublish").attr('checked')) {
			FB.Facebook.apiClient.users_hasAppPermission("publish_stream", function(p) {
				if (p == "1") {
					publishComment(theBody);
				} else {
					FB.Connect.showPermissionDialog("publish_stream", function(perms) {
						if (perms != "")
							publishComment(theBody);
						else
							jQuery("#comment_form").submit();

					});
				}
			});

			return false;
		}

		return true;
	}
