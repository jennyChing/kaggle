   <script>
    function loadResponsiveMenu() {
		jQuery("#mobileMenu ul li").attr("class", "firstLevel");
		jQuery("#mobileMenu ul li ul li").attr("class", "secondLevel");
		jQuery("#mobileMenu ul li ul li ul li").attr("class", "thirdLevel");
		jQuery("#mobileMenu ul li ul li ul li ul li").attr("class", "fourthLevel");
		
		jQuery("#mobileMenu ul li").has("ul").prepend('&lt;img src="' + contextPath + '/etc/designs/cnic_main_styles/mobilelibs/images/cnic_icon_arrow_right.png" class="icon" style="float:right"/&gt;');
		
		jQuery(".mobile_menu_icon").on("click", function(e) {
			jQuery("#mobileMenu &gt; ul").toggle();
			
			if( (jQuery(this).css("background-color") == "#2f659a") || (jQuery(this).css("background-color") == "rgb(47, 101, 154)") ) {
				jQuery(this).css({"background-color":""});
			}
			else {
				jQuery(this).css({"background-color":"#2f659a"});
			}
			
			e.stopPropagation();
		});
		
		jQuery("#mobileMenu").on("click", ".icon", function(e) {
			jQuery(this).parent().find("ul").first().toggle();
			
			var currentUrl = window.location.host;
			var imagePathRight = contextPath + "/etc/designs/cnic_main_styles/mobilelibs/images/cnic_icon_arrow_right.png";
			var imagePathDown = contextPath + "/etc/designs/cnic_main_styles/mobilelibs/images/cnic_icon_arrow_down.png";
			var iEFirstPath = document.location.protocol + "//" + currentUrl + imagePathRight;
			
			if( (jQuery(this).attr("src") == imagePathRight) ||  (jQuery(this).attr("src") == iEFirstPath) ) {
				jQuery(this).attr("src", imagePathDown);
			}
			else {
				jQuery(this).attr("src", imagePathRight);
			}
			
			e.stopPropagation();
		});

		//disable mobile_menu_icon click event
		jQuery("div.header.menuRight a.mobile_menu_icon").click(function(e) {
			e.preventDefault();
           });
		
	}	
	
	//set document.ready trigger to invoke mobile menu based on type
	if (mobileMenuType == "mmenu") {	
		jQuery(document).ready(function() {
			//invoke mMenu
			jQuery("#mobileMenu").mmenu();
		});
	} else {
		jQuery(document).ready(function() {
			//invoke responsiveNav
			loadResponsiveMenu();
    	});
	}
