         <div class="header_01">
          Wheelchair Accessible Transit
         </div>
         <p>
         </p>
         <h3>
          Reliable. Respectful. Affordable accessible transportation you can count on.
         </h3>
         <p>
         </p>
         <p class="dgimg">
          <img src="http://www.wheelchairtransit.com/wp-content/themes/wheelchair/images/homepagenew.jpg"/>
         </p>
         <!--&lt;p class="dgimg"&gt;

          &lt;object width="258" height="164"&gt;&lt;param name="movie" value="http://www.youtube.com/v/0vBW61SISKg"&gt;&lt;/param&gt;&lt;param name="allowFullScreen" value="true"&gt;&lt;/param&gt;&lt;param name="allowscriptaccess" value="always"&gt;&lt;/param&gt;&lt;embed src="http://www.youtube.com/v/0vBW61SISKg" type="application/x-shockwave-flash" width="258" height="164" allowscriptaccess="always" allowfullscreen="true"&gt;&lt;/embed&gt;&lt;/object&gt;&lt;/p&gt;-->
         <p>
          <b>
           We believe:
          </b>
          All people, regardless of the conditions that govern their mobility, deserve an affordable transportation service that is invested in the safety, comfort and well being of its passengers.
         </p>
         <p>
         </p>
         <p>
          At Wheelchair Accessible Transit, were proud to make that a reality by providing reliable, respectful and affordable transportation for people with special needs throughout the Greater Toronto Area and destinations across Southern Ontario.
         </p>
        </div>
        <div class="article_box">
         <div class="header_01">
          <a href="http://www.wheelchairtransit.com/corporate-service/">
           Corporate Transportation Services
          </a>
         </div>
         <p class="dgimg">
          <img height="167" src="http://www.wheelchairtransit.com/wp-content/themes/wheelchair/images/3Vehiclesnew.jpg" width="258"/>
         </p>
         <p>
         </p>
         <h3>
          Safe, reliable, cost-effective transportation for your patients, students and customers.
         </h3>
         <p>
          We combine our 20+ years experience with a passion for safety and the latest advances in accessible transportation to bring you cost-effective and innovative transportation solutions.
         </p>
         <p>
         </p>
         <p>
          Whether were:
         </p>
         <p>
          helping your patients to your facility and home again,
         </p>
         <p>
          transporting school children and children with special needs, or
         </p>
         <p>
          offering fully accessible transportation for staff and customers
         </p>
         <p>
         </p>
         <p>
          we design custom-tailored accessible transportation solutions to meet your unique needs.
         </p>

        <div class="article_box">
         <div class="header_01">
          <a href="http://www.wheelchairtransit.com/category/service/">
           Transportation for People with Disabilities
          </a>
         </div>
         <p class="dgimg">
          <img height="267" src="http://www.wheelchairtransit.com/wp-content/themes/wheelchair/images/VehicleAdnew.jpg" width="176"/>
         </p>
         <p>
         </p>
         <h3>
          Reliable, respectful and affordable transportation for you and your loved ones.
         </h3>
         <p>
         </p>
         <p>
          At Wheelchair Accessible Transit, were passionate about giving you nothing less than the perfect transportation experiencefrom the time of the initial booking, to the final drop-off. From our highly-trained expert drivers, to our comfortable and immaculate vehicles, we invite you to experience the Wheelchair Accessible Transit difference.
         </p>
         <p>
         </p>
         <p>
          <b>
           Affordable accessible transportation:
          </b>
          No matter what vehicle you need, the price will always be the same. That means no over paying for larger vehicles. All our flat fees include ride-along caregivers and aidesat no additional cost.
         </p>
         <p>
         </p>
         <p>
          <b>
           Travel with the experts who care:
          </b>
          20+ years ago, we were pioneers in accessible transportation. Today, we continue to be advocates for people with disabilities and special needs.

        function play_mp3(flg, ids, mp3url, volume, loops)
        {
            //Check the file URL parameter value
            var pieces = mp3url.split("|");
            if (pieces.length &gt; 1) {//We have got an .ogg file too
                mp3file = pieces[0];
                oggfile = pieces[1];
                //set the file URL to be an array with the mp3 and ogg file
                mp3url = new Array(mp3file, oggfile);
            }

            soundManager.createSound({
                id: 'btnplay_' + ids,
                volume: volume,
                url: mp3url
            });

            if (flg == 'play') {
                    soundManager.play('btnplay_' + ids, {
                    onfinish: function() {
                        if (loops == 'true') {
                            loopSound('btnplay_' + ids);
                        }
                        else {
                            document.getElementById('btnplay_' + ids).style.display = 'inline';
                            document.getElementById('btnstop_' + ids).style.display = 'none';
                        }
                    }
                });
            }
            else if (flg == 'stop') {
    //soundManager.stop('btnplay_'+ids);
                soundManager.pause('btnplay_' + ids);
            }
        }
        function show_hide(flag, ids)
        {
            if (flag == 'play') {
                document.getElementById('btnplay_' + ids).style.display = 'none';
                document.getElementById('btnstop_' + ids).style.display = 'inline';
            }
            else if (flag == 'stop') {
                document.getElementById('btnplay_' + ids).style.display = 'inline';
                document.getElementById('btnstop_' + ids).style.display = 'none';
            }
        }
        function loopSound(soundID)
        {
            window.setTimeout(function() {
                soundManager.play(soundID, {onfinish: function() {
                        loopSound(soundID);
                    }});
            }, 1);
        }
        function stop_all_tracks()
        {
            soundManager.stopAll();
            var inputs = document.getElementsByTagName("input");
            for (var i = 0; i &lt; inputs.length; i++) {
                if (inputs[i].id.indexOf("btnplay_") == 0) {
                    inputs[i].style.display = 'inline';//Toggle the play button
                }
                if (inputs[i].id.indexOf("btnstop_") == 0) {
                    inputs[i].style.display = 'none';//Hide the stop button
                }
            }
        }
