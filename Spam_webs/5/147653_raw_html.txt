			var options = {
				dataType: 'json',
				success: addEmailSuccess,
				error: errorMethod,
				beforeSubmit: function() {
					// email verification
					var emailVal = $("#emailSubscribeAddressModal").val();
					var emailReg = /^([\w-\.]+@([\w-]+\.)+[\w-]{2,4})?$/;
					if (emailVal == '') {
						$("#emailSubscribeAddressModal").val("Please Enter Your Email");
						return false;
					} else
						if (!emailReg.test(emailVal)) {
							$("#emailSubscribeAddressModal").add().css("color","#990000");
							$("#emailSubscribeAddressModal").val('Please Enter a Valid Email');
							return false;
					}
				}
			};
			$("#emailSignUpModal").ajaxSubmit(options);
		});
	});
	function addEmailSuccess(responseText, statusText, xhr, $form) {
		// A session timeout returns a single json object having status=session_expired
		var status = responseText.status,
			$html = $('html');
		if(status == 'session_expired') {
			window.location = responseText.redirect_url;
		} else {
			if ($html.hasClass('first-visit')) {
				if ($html.hasClass('small-screen')) {
					fireTealiumEvent({
						"e_headerpopup_email_p" : "t",
						"lead_type" : "email header popup phone signup"
					});
				}
				else {
					fireTealiumEvent({
						"e_headerpopup_email" : "t",
						"lead_type" : "email header popup signup"
					});
				}
				$html.removeClass('first-visit');
				$('#emailSubscribeAddressModal').val('');
			}
			else {
				if ($html.hasClass('small-screen')) {
					fireTealiumEvent({
						"e_header_email_p" : "t",
						"lead_type" : "email header phone signup"
					});
				}
				else {
					fireTealiumEvent({
						"e_header_email" : "t",
						"lead_type" : "email header signup"
					});
				}
			}
			// passing value of newEmailSignUp back via JSON obj
			var newSignUp = responseText.newEmailSignUp;
			if (newSignUp == true) {
				$('#showEmailSignUpModalNew').trigger('click');
			} else {
				$('#showEmailSignUpModalExisting').trigger('click');
			};
		}
	}
