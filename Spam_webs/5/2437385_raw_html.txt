			google_login_signinCallback = function(authResult){
				//console.log('google_login_signinCallback');
				//console.log(authResult);
				google_login_obj.authResult = authResult;
				
				
				if(authResult['access_token']){
					// Successfully authorized
					//console.log('login worked');
					google_login_obj.token = gapi.auth.getToken();

				}else if(authResult['error']){
					// There was an error.

					//   "access_denied" - User denied access to your app

			google_login_obj.getUserName = function() {
				gapi.client.load(
					'plus', 
					'v1', 
					function() {
						var request = gapi.client.plus.people.get(
							{
								'userId': 'me'
							}
						);
						request.execute(
							function(resp) {
								//console.log(resp);
								//console.log('f name: ' + resp.name.givenName);
								//console.log('l name: ' + resp.name.familyName);
								google_login_obj.user.first_name = resp.name.givenName;
								google_login_obj.user.last_name = resp.name.familyName;
							}
						);
					}
				);
			}
	      
			// get the user's email, with an optional param to call the prime login when retrieved
			google_login_obj.getUserEmail = function(call_login) {
				gapi.client.load(
					'oauth2', 
					'v2', 
					function() {
						gapi.client.oauth2.userinfo.get().execute(
							function(resp) {
								// Shows user email
								//console.log(resp.email);
								google_login_obj.user.email = resp.email;
								if(call_login){
									google_login_obj.loginToPrime();
								}
							}
						)
					}
				);
				
			}

        function increaseFontSize() {   
            var p = document.getElementById('col2').querySelectorAll('p,ul,ol'); 
            for(i=0;i&lt;p.length;i++) {   
                if(p[i].parentNode.className.indexOf("actionLinks") &gt; -1 ) continue;

                if(p[i].style.fontSize) { 
                    var s = parseFloat(p[i].style.fontSize.replace("em","")); 
                } else {   
                    var s = 0.9; 
                } 
                
                if(s &lt; max ) {   
                    s += .1; 
                } 
                p[i].style.fontSize = s+"em"   
            }
        } 
        
        function decreaseFontSize() { 
            var p = document.getElementById('col2').querySelectorAll('p,ul,ol'); 
            for(i=0;i&lt;p.length;i++) {   
                if(p[i].parentNode.className.indexOf("actionLinks") &gt; -1 ) continue;

                if(p[i].style.fontSize) { 
                    var s = parseFloat(p[i].style.fontSize.replace("em","")); 
                } else {   
                    var s = 0.9; 
                } 
                
                if(s &gt; min) {   
                    s -= .1; 
                }

                p[i].style.fontSize = s+"em"
            } 
        }

function signupRedirect() {
	
	/*if (is_mustang == 1) {
		alert('signupRedirect signup_redirect_url = ' + signup_redirect_url);
	}*/
	
	
	
	// do we have a redirect url
	if (signup_redirect_url != '') {
		
		
		// set some vars that are strings we will search for
		var replace_site_url = '[site_url]';
		var replace_email = '[email]';
		
		// go through each string, if it is not empty
		if (replace_site_url != '') {
			// add the home page url for this site, urlencoded
			signup_redirect_url = signup_redirect_url.replace(replace_site_url, escape('http://www.mrfood.com'));
		}
		
		if (replace_email != '') {
			// add the email address the user signed up with
			signup_redirect_url = signup_redirect_url.replace(replace_email, signup_redirect_qry_params['email']);
		}
		
		
				if(post_newsletter_signup.is_new_signup == true){
			var addl_qry_str = '?';
			if(signup_redirect_url.indexOf("?") &gt; 0){
				// the url already has a query string, so just tack on to it
				addl_qry_str = '&amp;';
			}
			
			signup_redirect_url += addl_qry_str + 'pp=1';
		}
		
		
				
		if (is_mustang == 1) {
			var confirm_str = 'signupRedirect' + "\n\n";
			confirm_str += 'replace_site_url = ' + replace_site_url + "\n\n";
			confirm_str += 'replace_email = ' + replace_email + "\n\n";
			confirm_str += 'post_newsletter_signup.is_new_signup = ' + post_newsletter_signup.is_new_signup + "\n\n";
			
			confirm_str += 'signup_redirect_qry_params[email] = ' + signup_redirect_qry_params['email'] + "\n\n\n\n";
			
			confirm_str += 'signup_redirect_url = ' + signup_redirect_url + "\n\n";
			confirm_str += 'http_host = http://www.mrfood.com' + "\n\n";
			
			confirm_str += 'signup_redirect_id = ' + signup_redirect_id + "\n\n";
			
			confirm_str += 'Click OK to stay on this page.' + "\n";
			confirm_str += 'Click Cancel to go to Flow URL.' + "\n";
			
			if (confirm(confirm_str)) {
				return;
			}
		}
		
		// redirect the user
		window.location = signup_redirect_url;
		
		
		
	}
	
	
}
