 * Allow GA events via `ga-event` attributes on elements, both those
 * static on the page and those added in dynamically.
 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')
 *
 * Elements that have been processed on the page already will have a 'ga-event-added' attribute
 *
 */
$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event
    //  non-interactive (optional) - true for non-interactive events (ie. it shouldn't be used in bounce-rate calcs)
    //
    // var ga_event_handler = function(ev) {
    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';
                
            // Only add a delegated handler once for a given even `type'
            if (!_handlers[type]) {
                // Delegate to handle dynamic elements, track element ref via closure 
                $(document).on(type, selector, (function() {
                    return function(ev) {
                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());
                _handlers[type] = true;
            }

            // mark element as processed
            $(this).attr('ga-event-added', 'true');
        });

        // wait for 100ms and poll again (catch dynmically added elements)
        window.setTimeout(poll_ga_attr_events, 150);
    }


    // let's go, start processing ...
    poll_ga_attr_events();

});

function listen(evnt, elem, func) {
    if (elem.addEventListener) { // W3C DOM
        elem.addEventListener(evnt,func,false);
    }
    else if (elem.attachEvent) { // IE DOM
        var r = elem.attachEvent("on"+evnt, func);
        return r;
    }

        <img alt="Mc10:35" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/1/1/3/7/0/1137099222/b6f94d99caebe0f8f73b49ed246901bb599271c7.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Mc10:35
        </span>
       </h2>
       <div class="section_body">
        <p>
         The Mc10:35 is a traditional egg McMuffin with a burger patty stuffed inside. Although sausage may be better, the burger goes well with the ham, cheese, and egg. The interesting thing about this breakfast sandwich is that its availability falls only between 10:30 and 11:00 in the morning.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="7">
        <img alt="McCrepe" class="image" src="//file.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/6/9/5/2/6/695267910/6cd5ff0f381aab11bdfb6178fd084f50cba0b26a.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         McCrepe
        </span>
       </h2>
       <div class="section_body">
        <p>
         The McCrepe exists, and many individuals who have a sweet tooth in the morning order one. The McCrepe is a combination of pancake filled with a yogurt parfait. You roll up the pancake and top it with syrup and granola for a sweet crunch. This secret menu item is a do-it-yourself one, and consumers must order both the parfait and hot cakes to create it.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="8">
        <img alt="Root Beer Float" class="image" src="//file.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/1/1/0/2/6/110264357/0ac54754e83e896cdbed2bed12336d89ac93c0eb.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Root Beer Float
        </span>
       </h2>
       <div class="section_body">
        <p>
         Many McDonald's facilities no longer carry root beer as a drink option, so finding this secret menu item proves to be difficult for some. The root beer float is exactly how it sounds, a serving of fizzy root beer with McDonald's vanilla ice cream. If your McDonald's does not have root beer, ask for a coke float; it tastes just as good.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="9">
        <img alt="Monster Mac" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/1/0/0/3/2/1003252223/ee1d3ec2dd2cbbd6699f00149a089f909080520a.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Monster Mac
        </span>
       </h2>
       <div class="section_body">
        <p>
         If a Big Mac satisfies and tempts your taste buds, you need to try a Monster Mac. The name fits the sandwich well. The Monster Mac contains eight burger patties. Instead of the traditional two-patty Big Mac, the Monster Mac adds three more patties to each layer. If you really want to shake things up, quadruple up on the cheese and add more Big Mac sauce.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="10">
        <img alt="Steamed Bun" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1427116150/qdf8pfjlzfdvkndicf9s.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Steamed Bun
        </span>
       </h2>
       <div class="section_body">
        <p>
         Unlike every other bun on the menu, the Filet O Fish has a steamed bun. It's fresh, fluffy and sans grill marks. Just ask for your sandwich to come with the steamed bun option.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="11">
        <img alt="French Fries with Big Mac Sauce" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416504667/rw9gs1mi5ajiavsmizty.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         French Fries with Big Mac Sauce
        </span>
       </h2>
       <div class="section_body">
        <p>
         So, everyone knows McDonald's has the best fries around. If you don't want to disgrace them by adding mere ketchup, go the extra mile and order them with Big Mac sauce. Some locations may charge you extra for it, but you may get lucky at others and they'll slip in the sauce for free!
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="12">
        <img alt="McKinley Mac" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416504858/qvjuvrrt3fgpon5yx2l3.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         McKinley Mac
        </span>
       </h2>
       <div class="section_body">
        <p>
         If you want some extra protein included on your Mac, go for the McKinley. This one is your standard Big Mac, but instead of using regular meat patties, they upgrade you to the quarter pounder patties. That's right, a whole half pound of burger on your Mac!
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="13">
        <img alt="Biscuits and Gravy" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416505253/eeavombnjmadwpy59paq.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Biscuits and Gravy
        </span>
       </h2>
       <div class="section_body">
        <p>
         Sure, nothing is more American than McDonald's, but the southern locations lay claim to its Biscuits and Gravy option. It is, however, available at many more locations on the "secret menu."
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="14">
        <img alt="Grilled Cheese" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416505402/emhcm87lhycex0fjxma0.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Grilled Cheese
        </span>
       </h2>
       <div class="section_body">
        <p>
         Meat not your style? You can also opt for a grilled cheese, if you're in the mood. It's not just a cheeseburger minus the burger. If you ask, they will also toast the bun for you to give you that authentic grilled cheese experience.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="15">
        <img alt="Dipping Sauces" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416512021/ouo3jlmyzrucyuxw2dhf.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Dipping Sauces
        </span>
       </h2>
       <div class="section_body">
        <p>
         There are actually more dipping sauces at McDonald's than you realize. Sure, we all know about the standard Ranch, Sweet &amp; Sour, and BBQ. But did you know they also have Spicy Buffalo, Sweet Chili, Chipotle, and Hot Mustard? Oh yeah.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="16">
        <img alt="Poor Man's Big Mac" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1416512305/djizg5d8m1zzlt6e9vau.jpg"/>
       </span>
       <h2 class="section_title">
        <span>
         Poor Man's Big Mac
        </span>
       </h2>
       <div class="section_body">
        <p>
         So, this is an alternative version of the Big Mac with fewer calories and for half the price. The trick is, it's really a McDouble with extra lettuce and the Mac sauce instead of ketchup and mustard.
        </p>
       </div>
      </div>

             What is the secret about McDonald's?
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_q_130366262">
            that it's made entirely out of crap, yet people still eat it!  There was a study where the left out McDonalds french fries and hamburger for 100+ days compared to regular(home
            <span class="hellip">
             
            </span>
            <span class="hidden">
             cooked) french fries and hamburger. The Mcdonalds meal looked very similar to how it was when it was first bought while the regular meal got moldy. This shows that McDonalds probably has certain chemicals in its food.   McDonalds and other huge chains are also known for using every part of the cow in their meat. They use the regular parts of the meat(fat and muscle) but in addition use the hooves and other parts. They put those extra parts through a process of grinding it, mixing it with meat filler, and cleaning it with ammonia.   The movie Super Size Me shows how if eaten excessively, McDonalds could damage your liver and do harm to your body. But take note: Anything taken excessively is unhealthy.

           <h1 class="title " ga-event="Feed_Click,index/article/15514,f_g_1190537">
            <a class="title_link black" href="http://www.answers.com/article/1190537/10-secret-menu-items-you-oughta-try-from-jamba-juice">
             10 Secret Menu Items You Oughta Try from Jamba Juice
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1190537">
            Fans of strawberry-flavored pink Starburst candies can drink it up with the Pink Star Jamba Juice smoothie. For an original, or 24-oz., smoothie, combine 6 oz. soymilk, 6 oz.
            <span class="hellip">
             
            </span>
            <span class="hidden">
             lemonade, and ice in a blender. Add a 2-oz. scoop each of raspberry sherbet, frozen yogurt, and sorbet, then two 2-oz. scoops of strawberries before blending. Most Jamba Juice stores can make this secret menu smoothie; if not, they may be able to create it from this recipe.Formerly a regular menu item, Pina Colada Jamba Juice is a refreshing tropical treat. For an original size smoothie, combine and blend 12 oz. pineapple juice, a pinch of coconut, and two 2-oz. scoops each of pineapple sherbet and frozen yogurt. Add one 2-oz. scoop of banana and ice and blend. Stores that don't carry coconut can substitute coconut water for some of the pineapple juice. Each size smoothie uses a different color and size scoop, but the ratios are the same.This summery secret Jamba Juice combines the fresh lushness of peaches and creamy frozen yogurt. For an original size, combine 6 oz. each of peach juice and soymilk, add one 2-oz. scoop each of frozen yogurt and peaches, and top it off with one scoop of ice - refreshing and healthy.The original Andre is unknown, but his creation should have earned him Employee of the Year. A refreshing 24-oz. Andre's Surprise combines 9 oz. orange juice and 1 oz. raspberry juice. Add two scoops of orange sherbet to one scoop each of pineapple sherbet, mangoes, peaches, and ice. To enhance a smoothie's nutrition, add a soy protein, antioxidant, or vitamin boost.This fruit-filled treat must get its orchard name from the peaches, as the rest of the ingredients grow in other environments. You can ponder this mystery while sipping this cool blend. For a 24-oz. Orchard Oasis smoothie, combine 12 oz. peach juice; one 2-oz. scoop each of pineapple sherbet, blueberries, and strawberries; and ice.Strawberry Cheesecake is one of Jamba Juice's most-requested secret smoothies. For an original size sippable cheesecake treat, blend 12 oz. white grape juice with two 2-oz. scoops of frozen yogurt, one 2-oz. scoop of strawberries, and a scoop of ice. To complete the cheesecake illusion, pack in a few graham crackers to simulate the crust.The White Gummi smoothie is sweeter than much of the Jamba Juice secret menu, making it a good candidate to split up among the kids. An original size smoothie contains 6 oz. peach juice and 4 oz. soymilk. Add one 2-oz. scoop each of lime sherbet, raspberry sherbet, pineapple sherbet, orange sherbet, strawberries, mangoes, and ice.Although pies are not the ideal health food, their filling flavors are a great inspiration for yogurt and smoothies. The original size Banana Cream Pie entry on Jamba Juice's secret menu starts with 10 oz. blended soymilk and chocolate base. Add three 2-oz. scoops of frozen yogurt, two 2-oz. scoops of bananas, and one 2-oz. scoop of ice. You can add a scoop of strawberries or change the soymilk/chocolate base ratio to alter the smoothie's sweetness.One of several variations of the classic Orange Dream Machine, a 24-oz. Raspberry Dream Machine starts with 6 oz. raspberry juice and 6 oz. soymilk. Add a 2-oz. scoop each of raspberry sherbet, frozen yogurt, and ice for a sweet and tart refresher.This secret menu item is an easy hack at the Jamba Juice counter. Simply order the Chocolate Moo'd smoothie with a scoop of strawberries and a scoop of frozen yogurt. This secret smoothie is higher in calories than most items on the menu, making it perfect to split up among the kids. Active kids who need to replace calories expended during strenuous exercise and play will appreciate this treat.

           <h1 class="title " ga-event="Feed_Click,index/article/15514,f_g_1177088">
            <a class="title_link black" href="http://www.answers.com/article/1177088/10-items-you-need-to-try-from-the-taco-bell-secret-menu">
             10 Items You Need To Try From The Taco Bell Secret Menu
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1177088">
            Taco Bell is the fast food destination for many high schoolers, college students and adults who crave a quick meal. What a lot of us don't realize is that there are lots of th
            <span class="hellip">
             
            </span>
            <span class="hidden">
             ings that you can order that aren't on the "normal" menu. The Cheesy Gordita Crunch is an item that once appeared as a regular option on Taco Bell's menus across the nation. However, few found it appealing, and it has been confined these days to the secret menu. This taco has all the works, including beef, beans, cheese, and a layer of sour cream, guacamole, and scallions.This menu item drew inspiration from the movie "The Hulk." Unlike its media counterpart, however, the food item did not enjoy as much success. Taco Bell's Hulk comes in two types. The first is a bean and cheese burrito smothered with guacamole, and the other is a five-layer burrito with guacamole instead of cheese.The Superman created by Taco Bell is less impressive than the action character of Superman. The food version is essentially a burrito with potatoes, sour cream, crunchy tortilla chips, and a smattering of guacamole, which comes wrapped in a warm, soft flour tortilla.Some like it hot, and if you are one who can't get enough spice in your meals, the Chili Cheese Burrito is just for you. This burrito is more or less a spicier version of the standard bean and cheese burrito that Taco Bell offers. However, it is not for the faint-hearted, and comes topped with smoldering hot sauce.If your doctor has told you that you need to eat more calcium, which you can get through cheese, then this off-menu item will probably satisfy your nutritional needs for weeks. This enchilada comes stuffed with three types of cheese, beef, and beans. It comes covered with red sauce, chopped onions, and yes, more cheese.In addition to having short-lived meal-sized menu items taken off its menu, Taco Bell removed less popular condiments too. Green sauce, which looks as mysterious as its name suggests, is one that did not find a cult following. This sauce is generic green chili sauce that you can find in some locations.This secret menu, as the name implies, is just what it sounds like. These quesadillas do not appear on the standard Taco Bell menu, perhaps because they take too long to prepare, or because after weathering the grill twice, they come out looking a little under the weather.Cheesaritos once appeared as a regular item on Taco Bell's menu. Due to waning popularity, however, Taco Bell removed Cheesaritos from its standard menu. These ooey, gooey delights are available upon request, and come stuffed with cheese, scallions, and a healthy dose of taco sauce wrapped in a flour tortilla.Despite the addition of a few more items, and hundreds more calories, than its counterpart, the Incredible Hulk never made its way into the hearts of consumers. The Incredible Hulk is a still a five-layer burrito, complete with rice, beans, lettuce, and tomatoes, but contains guacamole rather than nacho cheese.This term refers to degree of spiciness. If you order something lava-style from Taco Bell, be prepared for a heat wave. The restaurant reserves its hottest sauce, the lava sauce, for the bravest souls.            <h1 class="title " ga-event="Feed_Click,index/article/15514,f_g_1185062">
            <a class="title_link black" href="http://www.answers.com/article/1185062/6-items-you-need-to-order-from-the-panera-secret-menu">
             6 Items You Need To Order From The Panera Secret Menu
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1185062">
            According to Brand Eating, Panera Bread began offering six new menu items in early 2013. Packed with more protein and fewer carbohydrates than their previous offerings, the ne
            <span class="hellip">
             
            </span>
            <span class="hidden">
             w items are grouped under the title "Power," and are marketed with an emphasis on lower calories and healthier food choices. The meat is all-natural and the lemon or lime juice is freshly squeezed. Baby spinach, loaded with essential minerals and vitamins, is used in four of the six Secret Menu items.Interested in enjoying eggs with your meal? Try the Power Breakfast Egg White Bowl with Roasted Turkey, or the Power Breakfast Egg Bowl with Steak. The egg white bowl is only 170 calories and filled with all-natural roasted turkey, egg whites, roasted peppers, baby spinach, and basil pesto, and topped off with a little salt and pepper. Best of all, the turkey is raised free of antibiotics, for those trying to be more conscious about their meat. The second egg bowl is prepared with two eggs, sirloin steak tip slices, avocados, and tomatoes, and contains a very reasonable 270 calories.Not interested in eggs with your sirloin steak? Perhaps a Power Sirloin Steak Lettuce Wrap with cucumbers, diced tomatoes, red onions, basil pesto and fresh lime juice on a leafy green lettuce base will appeal more to your palate. This meal clocks in at only 210 calories.Are your taste buds craving some olives? The Power Mediterranean Turkey salad is chock-full of roasted turkey, tomatoes, red onions, and kalamata olives. A bed of baby spinach and romaine lettuce with extra virgin olive oil and freshly squeezed lemon juice rounds out the ingredients of this appetizing salad. The Roasted Turkey Salad is loaded with goodness, yet only has 290 calories.Coming in at 300 calories is the Power Chicken Hummus Bowl, which features baby spinach topped with all-natural chicken, cucumbers, diced tomatoes, red onions, a jalapeno citrus hummus, and fresh lemon juice. This delicious menu item is a protein-rich hunger killer.The Power Mediterranean Chicken Salad is made with two greens, just like the turkey salad. Extra virgin olive oil and fresh squeezed lemon juice sit atop the salad, while baby spinach and romaine lettuce accompany tomatoes, diced eggs, chicken, and delicious applewood smoked bacon. Yes, bacon is included, yet this dish contains a mere 360 calories.
