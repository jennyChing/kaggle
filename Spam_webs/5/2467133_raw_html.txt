function update_data()
{
	 var reg = /^[_a-z0-9-]+(\.[_a-z0-9-]+)*@[a-z0-9-]+(\.[a-z0-9-]+)*(\.[a-z]{2,4})$/;
	var address = jQuery('#submit_email').val(); 
	
	
	  if (!reg.test(address)) {
	  
		jQuery('#err_email').css('display','block');
  		return false;
	}
	else
	{
		 jQuery('#err_email').css('display','none');
		 $.ajax({
				url: 'index.php?option=com_pcgajax',
				data: 'stremail='+address+'&amp;no_html=1&amp;cmd=SUBMITYOUREMAIL',
				type: 'POST',
				dataType: 'json',
				success: function(data)
				{ 
				//alert(data);
					if(data=='Thank You')
					{
						$.fancybox.close();
						return false;	
					}
					else
					{
						alert('Please Retry');
						return false;	
					}
				}
		});	
	}
}

         <span>
          <p>
           The broccoli salad with combination of colourful ingredients is versatile as ingredients can be upped or diminished as per your liking. Prepare easy summer broccoli salad with recipe given as follows.

        <h3>
         Ingredients
        </h3>
        <div class="product-detail">
         <ul>
          <p>
           <strong>
            For dressing
           </strong>
          </p>
          <ul style="list-style-type: disc;">
           <li>
            1/2 cup Buttermilk
           </li>
           <li>
            1/2 cup Mayonnaise
           </li>
           <li>
            1/2 cup sour Cream
           </li>
           <li>
            1 tsp minced Garlic
           </li>
           <li>
            2 tbsp chopped Parsley
           </li>
           <li>
            2 tbsp chopped Dill
           </li>
           <li>
            2 tbsp chopped Chives
           </li>
           <li>
            1/4 tsp Onion powder
           </li>
           <li>
            1/4 tsp kosher Salt
           </li>
           <li>
            1/8 tsp ground black Pepper
           </li>
           <li>
            1-1/2 tbsp apple cider Vinegar
           </li>
           <li>
            1 tsp Sugar
           </li>
          </ul>
          <p>
           <strong>
            For salad
           </strong>
          </p>
          <ul style="list-style-type: disc;">
           <li>
            3 cut into florets Broccoli heads
           </li>
           <li>
            1 cup grated Carrots
           </li>
           <li>
            1/2 cup chopped red Onion
           </li>
           <li>
            1/4 cup dried Cranberries
           </li>
           <li>
            1/4 cup toasted pine Nuts
           </li>
           <li>
            1/4 cup dry roasted Sunflower seeds
           </li>
           <li>
            3/4 cup grated Cheddar Cheese
           </li>
           <li>
            4 thick-cut slices chopped cooked Bacon
           </li>
          </ul>
         </ul>
        </div>
        <h3>
         How to make Summer Broccoli Salad
        </h3>
        <div class="description">
         <p>
          <strong>
           Step 1:
          </strong>
          For dressing: In a bowl, combine all the dressing ingredients together and mix well.
         </p>
         <p>
          <strong>
           Step 2:
          </strong>
          Add broccoli florets and all the salad ingredients in the bowl and toss until completely coated.
         </p>
         <p>
          <strong>
           Step 3:
          </strong>
          Keep the prepared salad in an airtight container and refrigerate to marinate for 24 hours.
         </p>
         <p>
          <strong>
           Step 4:
          </strong>
          Also, toss once or twice during marinating time, then serve chilled.
         </p>
