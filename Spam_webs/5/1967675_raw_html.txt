(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

                    <h3 class="post-title entry-title" itemprop="name">
                     Seafood Pizza
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-4162029618341184826" itemprop="description articleBody">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="http://2.bp.blogspot.com/-ABC9rxPrc3g/U2rwDn11XtI/AAAAAAAAATQ/hSmjzS5cJwo/s1600/Seafood+Pizza.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" height="201" src="http://2.bp.blogspot.com/-ABC9rxPrc3g/U2rwDn11XtI/AAAAAAAAATQ/hSmjzS5cJwo/s1600/Seafood+Pizza.jpg" width="400"/>
                      </a>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: center; vertical-align: baseline;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       For hearty slices, take your favorite seafood and place them right smack on top of the dough.
                       <br style="float: none;"/>
                       <em style="background-color: transparent; border: 0px; float: none; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                       </em>
                      </span>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <br/>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <br/>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <b>
                        Ingredients:
                       </b>
                      </span>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; text-align: justify; vertical-align: baseline;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <div style="background-color: white; border: 0px; float: none; line-height: 20px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       100 grams clams,
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       in their shells
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       100 gramsmussels,
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       in their shells
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       50 gramssquid,
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       cut into thin strips
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       1 1/4 cups water, for cooking
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       1pizza dough recipe
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       3/4 cup whole canned tomatoes,
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       crushed by hand and
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       drained thoroughly in a colander
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       1/2 teaspoon ground chili pepper
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       1 clove garlic, finely chopped
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       a handful of choppedfresh flat parsley
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       50 gramspeeled and cooked prawns,
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       preferably with the shell left on the tails
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       if you wish to eat with your hands
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       6 to 7 pieces unpeeled cooked prawns
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       salt and freshly ground pepper
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       1 tablespoon extra virgin olive oil
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <b>
                        Instructions:
                       </b>
                      </span>
                     </div>
                     <br/>
                     <div style="background-color: white; border: 0px; float: none; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </div>
                     <div style="line-height: 20px; text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <strong style="background-color: transparent; border: 0px; float: none; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                        1
                       </strong>
                       Rinse the clams and mussels carefully under running water to remove any sand. Remove any clams or mussels with broken shells and check that they are all closed. Tap gently on the sink any clams or mussels that are not closedif they close up, they can be used. Scrape the shells of the mussels clean with a small knife and remove any beards.
                      </span>
                     </div>
                     <div style="text-align: justify;">
                      <span style="line-height: 20px;">
                       <span style="font-family: Georgia, Times New Roman, serif;">
                        <br/>
                       </span>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <br/>
                     <div style="text-align: justify;">
                      <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                       <span style="font-family: Georgia, Times New Roman, serif;">
                        <strong style="background-color: transparent; border: 0px; float: none; font-weight: bold; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                         2
                        </strong>
                        <b>
                        </b>
                        Quickly boil the clams, mussels, and squid in water, with a pinch of salt, about 1 to 2 minutes. Throw away any clams, mussels, and squid that do not open. Drain.
                       </span>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <br/>
                     <div style="text-align: justify;">
                      <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                       <span style="font-family: Georgia, Times New Roman, serif;">
                        <strong style="background-color: transparent; border: 0px; float: none; font-weight: bold; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                         3
                        </strong>
                        <b>
                        </b>
                        Roll out the pizza dough into a large rectangle or any desired shape. Spread the tomatoes evenly over the dough, leaving the edge (about 3/4-inch) free of toppings. Sprinkle the chili pepper, chopped garlic, and half the parsley over the tomatoes. Then spread the clams, mussels, squid, and both the peeled and unpeeled prawns. Sprinkle over the rest of the parsley. Season with salt and pepper. Drizzle with olive oil.
                       </span>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <div style="text-align: justify;">
                      <span style="font-family: Georgia, Times New Roman, serif;">
                       <br/>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <br/>
                     <div style="text-align: justify;">
                      <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                       <span style="font-family: Georgia, Times New Roman, serif;">
                        <strong style="background-color: transparent; border: 0px; float: none; font-weight: bold; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                         4
                        </strong>
                        <b>
                        </b>
                        Bake in the center of the oven (preheated at 5ooF for 30 minutes before baking) for 10 to 15 minutes, or just until crispy.
                       </span>
                      </span>
                     </div>
                     <span style="background-color: transparent; border: 0px; float: none; line-height: 20px; margin: 0px; outline: 0px; padding: 0px; vertical-align: baseline;">
                     </span>
                     <div style="clear: both;">
                     </div>
