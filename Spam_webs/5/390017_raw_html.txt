ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.bcpUpdateLink = function(videoId, playlistId) {
  var playlistKey = "bclid";
  var videoKey = "bctid";
  var modSocial = ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.modSocial;
  var modVP  = ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.modVP;
  var tabBar = ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.tabBar;
  // If we are on the /multimedia page, then we use the playlist id, etc. else
  // we just shorten the url that we are on.
  var info = shortUrlAPI = null;
  var newLink = document.URL;

  if (!window.location.origin) {
    window.location.origin = window.location.protocol+"//"+window.location.host;
  }
  var baseURL = window.location.origin + "/multimedia";
  newLink = baseURL + "?" + playlistKey + "=" + playlistId + "&amp;" + videoKey + "=" + videoId;

  // Shorten the URL if we can.
  shortUrlAPI = Drupal.settings.basePath + 'brightcove-player/short-url?url=' + encodeURIComponent(newLink);
  // Fetch the short-url in synchronous manner.
  $.ajax({
    url: shortUrlAPI,
    async: false,
    dataType: "json",
    success: function(result) {
      info = result;
    }
  });
  if (info != null) {
    modSocial.setLink(info.short_url);
  }
  else {
    modSocial.setLink(newLink);
  }
}

    var tabBar = modExp.getElementByID("playlistTabs");
    ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.tabBar = tabBar;
    // If we have a channel passed in, then set focus to that tab.
    if (typeof multimedia_channel !== 'undefined') {
      var tabs = tabBar.getData();
      for (var i=0, len=tabs.length; i&lt;len; i++) {
        if (multimedia_channel == tabs[i].displayName.toLowerCase().replace(/[^a-z0-9]/g, '')) {

ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.onPlay = function(evt) {
  var mediaLength = evt.duration; //Required video duration
  ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.mediaLength = mediaLength; // Needed in another function.
  var mediaOffset = Math.floor(evt.position); //Required video position
  var mediaID = (evt.media.id).toString(); //Required video id
  var mediaFriendly = evt.media.displayName; //Required video title
  var mediaName = mediaID + ":" + mediaFriendly; //Required Format video name
  var mediaRefID = evt.media.referenceId; //Optional reference id
  var mediaPlayerType = ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.player.type; //Optional player type
  var mediaTagsArray = evt.media.tags; //Optional tags
  var mediaPlaylist = ""; //Optional Playlist
  var mediaTagsArray2 = new Array();
  var mediaPlayerName = ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.mediaPlayerName;
  // HTML5 Players have their tags flattend to a 1-dimensional array already.
  if (mediaPlayerType == 'html') {
    mediaTagsArray2 = mediaTagsArray;
  }
  else {
    for (var i = 0; i &lt; mediaTagsArray.length; i++) {
      mediaTagsArray2.push(mediaTagsArray[i]['name']);
    }
  }
  if (mediaPlayerType == "flash") { //Optional playlist id
    if (evt.media.lineupId != null) {
      mediaPlaylist = (evt.media.lineupId).toString();
    }

  } else {
    if (evt.media.playlistID !== null) {
      mediaPlaylist = (evt.media.playlistID).toString();
    }
  }
  /* Check for start of video */
  if (mediaOffset == 0) {
    /* These data points are optional. If using SC14, change context data variables to hard
    coded variable names and change trackVars above. */
    s.contextData['bc_tags'] = mediaTagsArray2.toString(); //Optional returns list of tags for current video. Flash only.
    s.contextData['bc_refid'] = mediaRefID; //Optional retursn reference id
    s.contextData['bc_player'] = mediaPlayerName; //Optional player name is currently hard coded. Will be dynamic in later releases.
    s.contextData['bc_playertype'] = mediaPlayerType; //Optional returns flash or html
    s.contextData['bc_playlist'] = mediaPlaylist; //Optional returns playlist number for current video.
    s.Media.open(mediaName, mediaLength, mediaPlayerName);
    s.Media.play(mediaName, mediaOffset);
  } else {
    s.Media.play(mediaName, mediaOffset);
  }
}

ECOmnitureBrightCoveHandlersc83f03aefa33b9895ade41559bcb07cf.onStop = function (evt) {

  }
}

      <h1 class="fly-title">
       Super-star programmers
      </h1>
      <h3 class="headline">
       Difference engine: Wired for speed
      </h3>
     </hgroup>
     <aside class="floatleft light-grey">
      <time class="date-created">
       Jun 15th 2012, 13:42
      </time>
      <span class="location">
       by N.V. | LOS ANGELES
      </span>
     </aside>
     <div class="block block-ec_components " id="block-ec_components-share_inline_header">

     <div class="main-content">
      <p>
      </p>
      <div class="content-image-float">
       <img alt="" src="http://cdn.static-economist.com/sites/default/files/imagecache/original-size/images/2012/06/blogs/babbage/20120616_stp511.jpg" title=""/>
      </div>
      <p>
       ARE geniuses just born with their brains wired differently? Or do their early experiences fashion a richer set of neuronal interconnections that let them view the world through a sharper lens? The literature is replete with accounts of people who went on to accomplish great thingsin the arts, sciences, philosophy or even politicsafter exhibiting little promise in their youth. It would be encouraging to think that, if nurturing does indeed play a crucial part, there could yet be hope for the rest of us.
       <br/>
       <br/>
       An outfit in San Francisco called tenXer has begun testing a service that aims to help people boost their mental accomplishments by up to tenfoldhence its name. That has made your correspondent wonder what distinguishes the truly talented from the journeymen of any trade. And what, if anything, the rest can do to improve their more menial lot.
       <br/>
       <br/>
       Several years ago, your correspondent wrote a column about innovative operating systems (see 
       <a href="http://www.economist.com/blogs/babbage/2010/06/techview_future_operating_systems">
        Heading for the clouds
       </a>
       , June 17th 2010). The software he admired the most was PC/GEOS, from a tiny firm in Berkeley, California, which crammed a full multitasking operating system and a whole suite of applications with scalable fonts and a stunning graphical interface into a couple of megabytes. The Microsoft equivalent of the day needed nearly ten times more memory and came with half the tools and none of the applications. In the early 1990s, GeoWorks Ensemble (as the program later became known) was the hare to the Mac and Windows tortoises.
       <br/>
       <br/>
       So how come PC/GEOS failed to take the personal-computer world by storm? As the company founder subsequently
       <a href="https://www.economist.com/user/3755826/comments">
        explained
       </a>
       , Microsoft got wind of what the upstart operating system could do and hastily tied up all the computer-makers who had licences to pre-install MS-DOS with exclusive two-year contracts to use only Windows. They simply froze us out of the market.
       <br/>
       <br/>
       The point of the anecdote is not simply to show that big generally trumps best, but that a small team of talented folk will invariably outperform an army of sloggers. The whole PC/GEOS suite was written by a handful of programmersa feat that took Microsoft hundreds of man-years to match. The former were all ten-xers, while the latter comprised hordes of one- and two-x programmers.
       <br/>
       <br/>
       In his delightful guide to recruiting top talent (Smart and Gets Things Done), Joel Spolsky, a company founder and ten-x programmer himself, as well as a former paratrooper in the Israeli army, wrote in 2007 that the trouble with using a lot of mediocre programmers instead of a couple of good ones is that no matter how long they strive, they will still produce something mediocre. Five Antonio Salieris won't produce Mozart's Requiem. Ever. Not if they work for 100 years.
       <br/>
       <br/>
       The idea that top-tier programmers are typically an order of magnitude more effective at writing computer code than their run-of-the-mill cohorts is not exactly news. In study after study since the 1960s, researchers have noticed that the productivity of individual programmers with similar levels of experience varies by a factor of at least ten-to-one.
       <br/>
       <br/>
       More interesting still, in Peopleware, published in 1987, Tom DeMarco and Timothy Lister found that not only did the best programmers generally outperform the worst by a factor of ten, but that there was at least a tenfold difference in productivity among software organisations. Within individual firms, the difference in performance was only 20% or so. Clearly, the brightest programmers tended to congregate in places that had a reputation for attracting talented people; where the challenges were enticing, and the conditions conducive to good work. In many cases, that meant leaving large software companies to join smaller ones or to start their own.
       <br/>
       <br/>
       But what exactly do ten-xers do to be so productive? It is not as though they write more lines of software per day. What little evidence there is suggests that, over any given period, all experienced programmersgood, bad or indifferenttend to produce much the same amount of code.
       <br/>
       <br/>
       The big difference is that the best coders keep more of what they have produced, while the worst constantly have to rework whole sections. The high-achievers also make each line of code achieve more. And they know how to avoid writing unnecessary codeby editing routines they have written in the past to accomplish similar things.
       <br/>
       <br/>
       Above all, elite programmers understand how to manage complexity. The complexity of software rises exponentially with size. A program comprising 100,000 lines is thousands of times more complicated than one with 10,000 lines, which, in turn, is hundreds of times more complicated than one with a mere 1,000 lines. To give some idea of the problem, a typical operating system today contains upwards of 50m lines of code. Programmers who can hold in their heads all the essential elements of such a piece of software are truly in a league of their own.
       <br/>
       <br/>
       Can talent on such a scale be fostered? Probably not. Apart from needing to be endowed with more than adequate mental horsepower, such people tend to be autotelic by natureie, internally driven, like missionaries. As such, they are not attracted by the material rewards that motivate ordinary folk. Their work alone provides them with all the satisfactions they need. But there is much that can be done to nurture lesser minds. And that is true not just for computer programmers.
       <br/>
       <br/>
       An
       <a href="http://www.tempobook.com/2011/10/25/thrust-drag-and-the-10x-effect">
        essay
       </a>
       written by Venkatesh Rao, author, IT consultant and programmer, caused a stir in the blogosphere last year by suggesting that as many as one in four people could potentially transform themselves into ten-xers. Mr Rao divided daily tasks into short, low-value chores (answering e-mail, paying bills, going to the post office, etc) and lengthier, high-value activities (eg, coding software, composing music, writing books) that required a person's full creative juices to flow. The former he likened to aerodynamic drag; the latter to engine thrust. According to Mr Rao, boosting performance was simply a matter of increasing thrust and reducing drag. The greater the difference, the faster the mental processes would fly.
       <br/>
       <br/>
       All easier said than done. While managing the dreary drag work is something that can be mastered (or farmed out) without difficulty, thrust work is much harder to speed up. Creative tasks simply take longer to get started. But once underway, their momentum builds exponentially, until tiredness finally sets in and productivity falls. The creative momentum a person builds during a four-hour session is at least four times the peak momentum achieved in a two-hour session. Best, then, to schedule the longest blocks of uninterrupted time possible strictly for thrust work.
       <br/>
       <br/>
       Where Mr Rao's aeronautical metaphor stalls is when it comes to defining the engine for generating thrust. His tentative conclusion is that it has two components. One is a person's natural talent (for logic arguments, perfect pitch, or juxtaposition of words)which may well be acquired in the womb. Lucky the few who are born with it.
       <br/>
       <br/>
       The other component is a person's capacity for metacognition. This psychological term describes the tricks people develop to control their own thinking processes. Figuring out how to do a particular task, and then making sure it is done correctly, involves cognitive processes of planning, evaluating and monitoring various activitiesall of which people seem to get better at with practice.
       <br/>
       <br/>
       So, for people with the mental tools, tuning their engines and reducing their drag becomes simply a matter of clearing their calendars of chores and getting their heads down for some lengthy, uninterrupted sessions of creative thinking. Mr Rao admits it is not easy getting from low-x to ten-x performance. You have to get through an initial starter-motor phase, hit an ignition point, and then keep the engine running until it wears out.
       <br/>
       <br/>
       For those wishing to have a go,
       <a href="http://www.tenxer.com/">
        tenXer
       </a>
       in San Francisco is currently offering a beta version of its tracking software to help people analyse their own performance, set goals for themselves and improve their lot. At present, the tracking software works only for programmers. But down the road, the same techniques could just as readily be applied to other occupations. Your dilatory correspondent would love to be able to cram a day's worth of thrust work into a couple of hours. He would then have more time to lounge in the garden, tinker in the garage, watch the playoffs, and do all the other enjoyable things some call a drag.
      </p>

        <div class="comment-body">
         <p>
          The greatest innovation center in the past generation was Steven Jobs sitting on the floor meditating yoga-style in his sparsely furnished home.
         </p>
         <p>
          Einstein worked at a creaky wooden Government patent desk, but outclassed all of Europe's research labs.  His experiments were done in his head  daydreaming.
         </p>
         <p>
          After the the Bubonic Plague closed Cambridge University, Issac Newton moved back into his father's small farm. There he invented Physics and Calculus.  He was a lousy farmer.
         </p>
         <p>
          Thomas Edison was the single greatest mind of his generation. His home and barnyard became his work bench. And that became his lab and it resulted in our electrical world and power grid.
         </p>
         <p>
          There are maybe a handful of major innovative thinkers in the world. And they are international distributed thorough out the world populations. The next Steve Jobs may be Chinese.
         </p>
         <p>
          I think a program like the MacArthur Genius Grants are a better source of identifying the talent. Or Nobel Laureates. Or Westinghouse/ Siemens Science Competitions. Or the Million Dollar winners from Jeopardy.
         </p>
         <p>
          Alternatively, identify the top 1% most ambitious and innovative entrepreneurs. Sponsor them to live and free their lives from petty concerns.
         </p>
         <p>
          An alternative stimulus program is pay these geniuses to move to a campus, give them free rent, special free parking license plates, and set up free lab and library space.
         </p>
         <p>
          You need to become a brain magnet for the Elite 1%...the trick is identifying this tiny minority.
         </p>
         <p>
          Silicon and Technology as a category are too limited. We need more cross disciplinary pollination spanning EVERYTHING.
         </p>
         <p>
          There is genius in every endeavor. And a gold mine in every mind.
         </p>
         <p>
          Instead of bringing 50,000 semiconductor microchip researchers to Silicon Valley, Alley or Village, find the nEXT Steve Jobs.
         </p>
         <p>
          It is more a man hunt than attempting to found a cutting edge university research team or research institute.
         </p>
         <p>
          The Solution is Simple, but is as easy as winning the lottery. Or meeting the next Einstein Jr on a playground.
         </p>
         <p>
          These minds occur randomly in the gene pool. Our education system sometimes labels them as abnormal, retarded, autistic, or mentally ill like they did Einstein. Or misanthropes like the Wright Brothers or Thomas Edison.  Or distracted bug collectors like Charles Darwin.
         </p>
         <p>
          It is a manhunt. Find and identify them. It is like the Tibetan Buddhist and their quest to find the next reincarnation of the Dalai Lama.  The other 99.999% are worker drones.
         </p>
         <p>
          Find the individual and nurture and reward them.  The nEXT Steve Jobs is out there.
         </p>

        <div class="comment-body">
         <p>
          Although there is certainly merit to the argument that at least some cognitive ability is genetic, there are also powerful arguments that much of adult ability stems from childhood nurturing.  In my experience, I've noticed that, even with children going to the same mediocre school, one child whose parents spend some extra time to assist him in reviewing and understanding concepts learned in class will excel academically while another child, in the same class, will fail when there is little to no parental involvement.
         </p>
         <p>
          Using these methods, my son, who at first struggled heavily with reading, now reads at 2-3 levels above his current grade and continuously wants to learn more (he still hates homework, of course!).  The same goes with math and computing skills, too.
         </p>
         <p>
          The influence of an involved parent cannot be overstated.  Whether it appears so or not, the fact is that kids will retain a much higher percentage of what they learn from their parents than from any other adult.
         </p>
         <p>
          In adulthood, I personally find that working with others who share my interests and have similar capabilities forces me to work harder to excel at my job -- increasing productivity significantly.  On the contrary, when I worked in an environment where I was the sole tech expert and most of my coworkers were of lower... capacity, we shall say, I would naturally reach a state of contentness, and stop pushing myself.
         </p>
         <p>
          It's not that smart or focused people are mystically attracted to work together so much as, once they do find each other, they are mutually nurturing and challenging, which itself causes the exponential rise in productivity and quality.  Simply being around someone who cares about the same project you care about makes a big difference.
         </p>

          If by create or transform, one means unleash what is already there, then fine. It takes training to make an artist, but the potential for the artist is built in.
         </p>
         <p>
          If you spend any time with people who calculate with extraordinary ease, you know it comes from a place that others can't access. Music is the same.
         </p>
         <p>
          I love the Mozart comparisons but, remember, he was the exception among the giants. Beethoven scratched out much of what he wrote, moved parts around, and rewrote all the time. Genius takes different forms. If we could identify one common thing, it is an ability to find the core. In programming - like in most things - that means identifying the roots of dependencies so the trees of possibility make sense and can be rendered. Normal people can't do that well. It can take time to find the keys and to unravel the dependencies sufficiently to make sure you're right, but the core insight process is the same in all "genius".
         </p>
         <p>
          An example is Newton. Follow his papers. He describes in each one a process, almost always undertaken over a period of years. He cuts away all the wrong turns and mistakes to lay out what he found, even when and especially when that means identifying what he did not find. He carefully pieces together a picture of a core insight that he fleshed out over time. You can see sometimes where the insight occurs: a guess driven by an observation, a sense that maybe there is something in this which can be unraveled. He never "discovers" but instead "uncovers"; his process illuminates for himself a problem, sometimes what he thinks is the answer, until he's sure he understands to the limit of his capacity. His great brilliance shows in the way he recognizes  the limits of what he can do: he polishes glass and that merely scratches it further and that becomes a conflict between wave and particle he can never resolve.
         </p>

        <div class="comment-body">
         <p>
          Very interesting article. And thank you for covering an issue that is rarely, if ever, talked about in the business press. Well done.
         </p>
         <p>
          Accolades aside, I must take issue with some of the points you made. Firstly, if I may use a cheesy metaphor from the film Moneyball: I think that hiring a star programmer is bad baseball. That top 20 percent programmer, while he might be able to hold 50 million lines of code in his head, will actually be dragging down the rest of a development team, because the organization will have to hire more mediocre programmers to make the budget support his superstar contract.
         </p>
         <p>
          There is no question that, in any field, theres a finite number of geniuses. You can teach anyone to play piano, but Mozarts are born, not made. And you cant run a company with a room full of Mozarts. Number one, theyre way too expensive; and number two, geniuses dont like to follow orders.
         </p>
         <p>
          In application development, there is a concept known as the coding cowboy. These are the Mark Zuckerbergs and Bill Gates of the world, who can write code like Mozart could write music. In a company thats building the next great consumer app, its entirely conceivable that a single coding cowboy could produce an app end-to-end and hand it over to a team to take to the next level. Thats the case at Facebook.
         </p>
         <p>
          However, when youre talking about mission-critical applications that run our world  our banking, hospital, government, military, and factory computing systems, and so on  what you really need is a superstar leader, not a lone-wolf superstar coder. You need a disciplined engineering team that includes many of the same roles youd see in a team that is engineering a bridge: architects, designers, construction men and women (developers), testers, and even regulators that enforce discipline and standards to ensure the resulting product is structurally sound and capable of supporting the role it was designed for.
         </p>
         <p>
          The real geniuses in software development today are the people who can build disciplined teams of software engineers who measurably produce software quickly and efficiently, and provide easy to understand KPIs that are useful to the CIOs, CFOs, and CEOs of their organizations.
         </p>
