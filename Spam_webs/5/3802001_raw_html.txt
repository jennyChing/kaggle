     $('#language-selector').on('change', function(){
		var localeHostname, val = $(this).val()
		var localeMatcher = new RegExp('^en\.');

		if (val === 'en') val = 'www';

		if (localeMatcher.test(window.location.hostname)) {
			localeHostname = window.location.hostname.replace(localeMatcher, val + '.');
		} else if (/^www/.test(window.location.hostname)){
			if (val === 'www') return;
			localeHostname = window.location.hostname.replace(/^www\./, val + '.');
		} else {
			localeHostname = val + '.' + window.location.hostname;
		}
		window.location.hostname = localeHostname;
	})
