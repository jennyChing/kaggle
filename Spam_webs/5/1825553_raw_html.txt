	function addOrReplaceQueryStringParam(url, param, value) {
	    // Using a positive lookahead (?=\=) to find the
	    // given parameter, preceded by a ? or &amp;, and followed
	    // by a = with a value after than (using a non-greedy selector)
	    // and then followed by a &amp; or the end of the string
	    var re = new RegExp('(\\?|\\&amp;)' + param + '=.*?(?=(&amp;|$))', "gi");
	    var qstring = /\?.+$/;
	    
	    // Check if the parameter exists
	    if (re.test(url)) {
	        // if it does, replace it, using the captured group
	        // to determine &amp; or ? at the beginning
	        url = url.replace(re, '$1' + param + '=' + value);
	    } else if (qstring.test(url)) {
	        // otherwise, if there is a query string at all
	        // add the param to the end of it
	        url = url + '&amp;' + param + '=' + value;
	    } else {
	        // if there's no query string, add one
	        url = url + '?' + param + '=' + value;
	    }
	    return url;
	}

	function removeQueryStringParam(url, param) {
	    var re = new RegExp('(\\?|\\&amp;)' + param + '=.*?(?=(&amp;|$))', "gi");
	    if (re.test(url)) {
	    	url = url.replace(re,'');
	    }
	    return url;
	}

           <a href="/search-browse-results?searchText=Microplane+Adjustable+Coarse+Grater&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Microplane Adjustable Coarse Grater
           </a>
          </li>
          <li>
           Using
           <a href="/search-browse-results?searchText=Flour+Sugar+Shaker&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Flour/Sugar Shaker
           </a>
           , generously sprinkle cornmeal over clean
           <a href="/search-browse-results?searchText=Large+Grooved+Cutting+Board&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Large Grooved Cutting Board
           </a>
           . Unroll pizza crust over cornmeal. Roll crust into 16x8-in. (40.5x20-cm) rectangle with
           <a href="/search-browse-results?searchText=Baker+s+Roller&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Bakers Roller
           </a>
           . Lightly brush top of crust with remaining oil using
           <a href="/search-browse-results?searchText=Chef+s+Silicone+Basting+Brush&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Chefs Silicone Basting Brush
           </a>
           . Sprinkle with additional cornmeal.
          </li>
          <li>
           While leaving crust on cutting board, fold crust in half from short end. Gently lift and place in one half of pan; carefully unfold and stretch to fit bottom of pan. Cook 3-5 minutes or until dark grill marks appear (see Cooks Tips). Using
           <a href="/search-browse-results?searchText=Chef+s+Tongs&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Chefs Tongs
           </a>
           , turn crust over; sprinkle with cheese. Cook an additional 3-5 minutes.
          </li>
          <li>
           Using Chefs Tongs and
           <a href="/search-browse-results?searchText=Nylon+Turner&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Nylon Turner
           </a>
           , slide crust out and transfer to cutting board. Mix sausage and tomato mixture; spread mixture over crust. Using
           <a href="/search-browse-results?searchText=Pizza+Cutter&amp;GlobalHeaderSearch=GlobalHeaderSearch&amp;itemType=TPCProduct&amp;categoryType=Product&amp;secondaryCategoryType=Recipe&amp;userSelection=false" style="-moz-user-select: none;">
            Pizza Cutter
           </a>
           , cut into 8 rectangle pieces. Cut each piece into a triangle.
          </li>
         </ol>
         <p>
          <strong>
           Yield:
          </strong>
          8servings of2 pieces
         </p>
