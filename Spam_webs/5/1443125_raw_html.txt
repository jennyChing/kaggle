function voteVid(obj, vote) {
    holder     = jQuery('.vote-update');
    count      = holder.text();
    vid_id     = jQuery(obj).attr('data-vid-id');
    obj_id     = jQuery(obj).attr('id');

    if ( jQuery(obj).hasClass('active') ) {
    } else {
        jQuery('.vote-toggle').removeClass('active');
        jQuery(obj).addClass('active');
        
        jQuery.ajax({
            url: "https://www.pokertube.com/ajax/voteVideo"+"/id/"+vid_id,
            type:"post",
            data: {'vote': obj_id, 'ajax': 1},
            dataType: 'json',
            success: function(data, textStatus) {
                if(data.update) {
                    var likes = (data.likes);
                    var dislikes = (data.dislikes);
                    
                    var total = likes + dislikes;

                    if (likes == 0)
                        var fraction = 0;    
                    else
                        var fraction = likes / total;

                    var percent = Math.round(fraction * 100);
                    
                    holder.text( percent + '% Like this');
                    jQuery.growl.notice({ title: "Update:", message: data.message });
                    
                    jQuery('.star-box span').removeClass('active');
                    jQuery(obj).addClass('active');
                    
                } else {
                    if(data.login) {
                        jQuery('#trigger-forced-signup').reveal();
                    }
                    if(data.message) {
                        jQuery.growl.error({ title: "Update:", message: data.message });
                    }
                    jQuery('.like-box a').removeClass('active');
                }
            },
            error: function(xhr, textStatus, errorThrown){
            }
        });
    }
    return false;   
}

function voteComment(target,likes,type) {

        jQuery('#comment-div-'+target).find('.vote-toggle.up').addClass('active');
    }else{
        jQuery('#comment-div-'+target).find('.vote-toggle.down').addClass('active');
    }

}
