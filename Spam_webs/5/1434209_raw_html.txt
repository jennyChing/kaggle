function qsearch_onSubmit()
{
	qs_enginename = document.qsearch_form.qsearch_select.value;
	qs_keywords = document.qsearch_form.keywords.value;
	switch( qs_enginename )
	{
	case 'site':
		break;
	case 'author':
		window.open('search.php?author=' + qs_keywords, '_self', '');
		return false;
	case 'google':
		window.open('http://www.google.com.tr/search?q=' + qs_keywords, '_google', '');

		return false;
	case 'altavista':
		window.open('http://www.altavista.com/web/results?itag=ody&amp;q=' + qs_keywords + '&amp;kgs=0&amp;kls=0', '_altavista', '');
		return false;

		if( (i = qsearch_findEngine(qs_enginename)) &gt;= 0 )
		{
			window.open(qsearch_engines[i].url + qs_keywords, '_blank', '');
			return false;
		}
		break;
	}
	return true;
