        /**
        * Menu animation function
        **/
        function nice_menu(element, link) {
            var $theDiv = $(element);
            var $visibleSiblings = $theDiv.siblings(':visible');
            if ($visibleSiblings.length) {
                // underline the clicked link
                $('#primary a').removeClass("active");
                $(link).addClass("active");
                // hide element
                // hide the siblings and wait till ready
                $visibleSiblings.animate({
                }, 200, 'swing', function () {
                    if ($visibleSiblings.height() == $theDiv.height()) {
                        // if no height change needed quickly change the height to avoid line jitter
                        // take siblings out
                        $visibleSiblings.animate({
                            "height": "hide"
                        }, 1, 'swing', function () {
                        });
                        // show element
                        $theDiv.animate({
                            "height": "show"
                        }, 1, 'swing', function () {
                        });
                    }
                    else {
                        // if height change IS needed transition the height
                        // take siblings out
                        $visibleSiblings.animate({
                            "height": "hide"
                        }, 400, 'swing', function () {
                        });
                        // show element
                        $theDiv.animate({
                            "height": "show"
                        }, 400, 'swing', function () {
                        });
                    }
                });
            }
            else {
                // toggle the element
                $theDiv.animate({
                    "height": "toggle"
                }, 400);
                // underline handling
                $(link).toggleClass("active");
            }
        }
    </script>
