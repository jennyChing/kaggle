   (function(document, window, $, undefined) {

      var Data = {
        dummy_image_url_400 : '/i/no_image_400x400.png',
        product_id : '164385',
        promotion_id : '',
        countdown_timer_end : 0,
        related_product_ids : []
      };
      var Routes = {
        store_cart : 'https://www.crunchyroll.com/store/cart',
        ajax : '/ajax/',
        wishlist : '/',
        subtype_product_ids_to_routes : {}
      };
      var State = {
        add_button_frozen : false
      };
      var Messages = {
        something_went_wrong : 'Something went wrong.'
      };
      var Functions = {
        startCountdownTimer : function() {
          if ($('#deals_countdown').length) {
            var config = {
              'type': 'letters',
              'renderTo': 'deals_countdown',
              'targetDate': Data.countdown_timer_end,
              'allowSingleDigits': false
            };
            (new CountdownTimer(config)).start();
          } else {
            ; // No-op; no countdown span is set.
          }
        }
      };
      var Actions = {
        /**
         * Listens for a click on an image thumb; sets the main (large) image
         * to the image thumb that is clicked.
         */
        setMainImage : function(event) {
          var element = this;
          var image_large_url = $(element).data('large') || Data['dummy_image_url_400'];
          $('#main_image').attr('src', image_large_url);
        },

        /**
         * Event fired upon adding the current item to the cart
         */
        addThisProductToCart : function(event) {
          if (State.add_button_frozen) {
            return;
          }
          State.add_button_frozen = true;
          var element = this;

          var quantity = $('#store_product_contents').find('select[name=quantity]').val();
          if (!quantity || quantity &lt; 1) {
            quantity = 1;
          }
          var is_gift = $('#gift_checkbox').is(':checked') ? 1 : 0;
          var params = {
            'req' : 'RpcApiEcommerce_AddToCart',
            'quantity' : quantity,
            'is_gift' : is_gift,
            'currency' : 'USD',
            'product_id' : Data.product_id,
            'promotion_id' : Data.promotion_id
          };
          $.post(
            Routes.ajax,
            params,
            function(resp) {
              var resp = Rpc.parseJSON(resp);
              if (resp.result_code == Rpc.RESULT_SUCCESS) {
                window.location = Routes.store_cart;
              } else {
                Rpc.printMessage(resp);
              }
              State.add_button_frozen = false;
            }
          );
        },

        /**
         *
         */
        addThisProductToWishList : function(event) {
          if (State.add_button_frozen) {
            return;
          }
          State.add_button_frozen = true;

          var params = {
            'req' : 'RpcApiEcommerce_AddToCart',
            'quantity' : 1,
            'is_gift' : 0,
            'currency' : 'USD', //Doesn't really matter.
            'cart_type' : 'wishlist',
            'product_id' : Data.product_id,
            'promotion_id' : Data.promotion_id
          };

          $.post(Routes.ajax,
            params,
            function (resp) {
              var resp = Rpc.parseJSON(resp);
              if (resp.result_code == Rpc.RESULT_SUCCESS) {
                window.location = Routes.wishlist;
              } else {
                Rpc.printMessage(resp);
              }
              State.add_button_frozen = false;
            }
          );
        },

        /**
         *
         */
        changeProductSubtype : function(event) {
          var $subtype_element = $(this);

          var product_id = $subtype_element.data('subtype-product_id');
          if (product_id) {
            var route = Routes.subtype_product_ids_to_routes[product_id];
            if (route) {
              window.location = route;
            }
          }
        },

        /**
         *
         */
        addAllRelatedToCart : function(event) {
          if (!Data.related_product_ids) {
            return;
          }
          if (State.add_button_frozen) {
            return;
          }
          State.add_button_frozen = true;

          var temp = Data.related_product_ids;
          temp.push(Data.product_id);

          var params = {
            'req': 'RpcApiEcommerce_AddToCart',
            'currency': 'USD', //Doesnt matter
            'product_ids': temp
          };
          $.post(Routes.ajax,
            params,
            function (resp) {
              var resp = Rpc.parseJSON(resp);
              if (resp.result_code == Rpc.RESULT_SUCCESS) {
                window.location = Routes.store_cart;
              } else {
                Rpc.printMessage(resp);
              }
              State.add_button_frozen = false;
            }
          );
        }

      };

      $(document).ready(function() {
        // Bind event listeners
        $('.actioncontainer-setmainimage').on('click', '.main_image_thumb', Actions.setMainImage);
        $('.action-addtocart').on('click', Actions.addThisProductToCart);
        $('.action-addtowishlist').on('click', Actions.addThisProductToWishList);
        $('.actioncontainer-changesubtype').on('click', 'li', Actions.changeProductSubtype);
        $('.action-addalltocart').on('click', Actions.addAllRelatedToCart);

        // Initialize the first-shown main image
        $('.actioncontainer-setmainimage').find('.main_image_thumb').first().click();

        // Start the countdown timer
        Functions.startCountdownTimer();

        // Image thumbnail hover shadow effects!
        $('#store_product_images').find('.more-images').find('li.main-image-setter')
          .hover(function () {
            $(this).addClass('container-shadow-dark').removeClass('container-shadow');
          },
          function () {
            $(this).addClass('container-shadow').removeClass('container-shadow-dark');
          });

        // Put the play icon
        play_icon_overlay('.more-images li.media-type-video');
      });

    })(document, window, jQuery);

          function zoomInCurrentImage(x, y, scale, type) {
            var active_index = myGallerySwiper.activeIndex;
            var $container = $('.swiper-container-gallery').find('.more-images');

            var img = $container.find('.main_image_thumb:eq('+active_index+')');

            current_img_max_height = img.css('max-height');

            freezeSwiping();

            // improves scale and translate animations
            img.addClass('smooth-transform');

            current_slide.css({
              'width' : current_slide_dim.width*scale,
              'height' : current_slide_dim.height*scale,
              'transform-origin' : current_transform_origin.x + 'px ' + current_transform_origin.y + 'px',
              '-ms-transform-origin' : current_transform_origin.x + 'px ' + current_transform_origin.y + 'px',
              '-webkit-transform-origin' : current_transform_origin.x + 'px ' + current_transform_origin.y + 'px'

            if (type === 'tap') {
              scaleImage(scale, current_slide);
            }
            else if (type === 'pinch') {
              scaleImagePinch(scale, current_slide);
            }
            else {
              console.log('Zoom in: Not a valid scale type.');

            in_zoom = true;

          function scaleImage(scale, img) {
            img.css({
              'text-indent': current_scale

            img.css({

          function calculateTransformOrigin(x, y, scale, type, img) {
            var x_orig_offset = (img.outerWidth(true)-img.outerWidth())/2;
            var y_orig_offset = slide_margin_top;
            var x_offset, y_offset;

            if (type === 'tap') {
              x_offset = x/3;
              y_offset = y/3;
            }
            else if (type === 'pinch') {
              x_offset = 0;
              y_offset = 0;
            }

            var scaled_x = modifyScaledX((x-x_orig_offset-x_offset)*scale/current_scale, img);
            var scaled_y = modifyScaledY((y-y_orig_offset-y_offset)*scale/current_scale, img);
            return {ox: scaled_x, oy: scaled_y, x: scaled_x, y: scaled_y};
          }

          function modifyScaledX(x, img) {
            var min_x = img.outerWidth(true)-img.outerWidth();
            var max_x = img.innerWidth();

            if (max_x &lt; min_x) {
              x = $(window).width() / 2;
            }
            else if (x &gt; max_x) {
              x = max_x;
            }
            else if (x &lt; min_x) {
              x = min_x;
            }
            return x;
          }

          function modifyScaledY(y, img) {
            var min_y = 0;
            var max_y = img.innerHeight() - slide_margin_top * 2;

            if (max_y &lt; min_y) {
              y = $(window).height() / 2;
            }
            else if (y &gt; max_y) {
              y = max_y + slide_margin_top;
            }
            else if (y &lt; min_y) {
              y = min_y + slide_margin_top;
            }
            else {
              y += slide_margin_top;
            }
            return y;
          }

          function allowImagePan(img, current_slide, scale) {
            var img_offset = {left: img.offset().left, top: img.offset().top + parseInt(img.css('padding-top')) + parseInt(img.css('padding-bottom'))}
            var dragged = false;

            img.hammer().off();
            current_slide.hammer().off();

            img.hammer().on('touch', function(event_touchposition) {
              var scaled_img_width_bounds = img.outerWidth() * scale;
              var scaled_img_height_bounds = img.outerHeight() * scale;

              var last_touch_point_x = 0;
              var last_touch_point_y = 0;

              if (img.data('lastMove')) {
                last_touch_point_x = img.data('lastMove').x;
                last_touch_point_y = img.data('lastMove').y;
              }

              var mouse_pos_in_image_x = event_touchposition.gesture.center.pageX - last_touch_point_x;
              var mouse_pos_in_image_y = event_touchposition.gesture.center.pageY - last_touch_point_y;

              current_slide.hammer().on('drag', function(event_newtouchposition) {
                event_newtouchposition.gesture.preventDefault();

                img_offset = {left: img.offset().left, top: img.offset().top + parseInt(img.css('padding-top')) + parseInt(img.css('padding-bottom'))}

                var move_x = event_newtouchposition.gesture.center.pageX - mouse_pos_in_image_x;
                var move_y = event_newtouchposition.gesture.center.pageY - mouse_pos_in_image_y;

                // negate panning if drag exceeds image's bounds
                if (img.data('lastMove')) {
                  if ((img_offset.left &gt; 0 &amp;&amp; move_x &gt;= img.data('lastMove').x)
                    || (scaled_img_width_bounds &lt; $(window).innerWidth() - img_offset.left &amp;&amp; move_x &lt; img.data('lastMove').x)) {
                    move_x = img.data('lastMove').x;
                  }

                  if (img_offset.top &gt; 0 &amp;&amp; move_y &gt;= img.data('lastMove').y
                    || (scaled_img_height_bounds &lt; $(window).innerHeight() - img_offset.top &amp;&amp; move_y &lt; img.data('lastMove').y)) {
                    move_y = img.data('lastMove').y;
                  }
                }

                var translate_x = move_x/scale;
                var translate_y = move_y/scale;

                img.css({
                  'transform': 'translate3d(' + translate_x + 'px, ' + translate_y + 'px, ' + 0 + 'px)',
                  '-ms-transform': 'translate(' + translate_x + 'px, ' + translate_y + 'px)',
                  '-webkit-transform': 'translate3d(' + translate_x + 'px, ' + translate_y + 'px, ' + 0 + 'px)'
                });
                img.data('lastMove', {x: move_x, y: move_y});
                dragged = true;
              });

            });
            current_slide.hammer().on('release',function(event){
              $(this).hammer().off('drag');
              if (img.data('lastMove') &amp;&amp; dragged) {
                current_transform_origin.x = current_transform_origin.ox - img.data('lastMove').x / (scale/2);
                current_transform_origin.y = current_transform_origin.oy - img.data('lastMove').y / (scale/2);
                dragged = false;
              }
            });
          }

          function disableImagePan(img, current_slide) {
