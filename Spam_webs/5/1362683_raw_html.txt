         String.prototype.myLastIndexOf = function(startString, startpos) {
        var stringToWorkWith = this.toString();
        var exploded = stringToWorkWith.split(";");
        var qString;
        var isAtEnd = false;
        for (var i=0; i&lt;exploded.length; i++) {
            if (exploded[i].indexOf(startString) &gt; -1) {
                qString = exploded[i];
                if (i == exploded.length - 1) isAtEnd = true;
                break;
            } 
        }
        var lastIndexOf = -1;
        if (qString != null) {
            lastIndexOf = (isAtEnd) ? qString.length + 1 : qString.length - 3;
        }
        return lastIndexOf;
    }

    String.prototype.replaceAt=function(startIndex, endIndex, newString) {
        var stringStart = this.substr(0, startIndex);
        var stringEnd = this.substring(endIndex, this.length);
        return stringStart + newString + stringEnd;
    }

    var trimIAndOthers = function(urlQueryString){
        if(urlQueryString.indexOf('i=1') &gt; -1)
        {   
            var startd = urlQueryString.indexOf("i=1");
            var startd1 = urlQueryString.indexOf("i=1") + 3;            
            var endd = urlQueryString.myLastIndexOf("i=1", startd1) + startd1;
            urlQueryString = urlQueryString.replaceAt(startd, endd, decodeURIComponent(""));

            //remove x1 &amp; q1
            if(urlQueryString.indexOf('x1=') &gt; -1)
            {
                var starte = urlQueryString.indexOf("x1=");
                var starte1 = urlQueryString.indexOf("x1=") + 4;            
                var ende = urlQueryString.myLastIndexOf("x1=", starte1) + starte1;
                urlQueryString = urlQueryString.replaceAt(starte, ende, decodeURIComponent(""));

                if (urlQueryString.indexOf("q1") &gt; -1) {
                    var q1Index = urlQueryString.indexOf("q1");
                    var q1String = urlQueryString.substring(q1Index);
                    var q1SemicolonIndex = q1String.indexOf(";") + 1; //plus 1 for semicolon

                    urlQueryString = urlQueryString.replace(q1String.substring(0, q1SemicolonIndex), "");
                }
            }

            //remove x2 &amp; q2
            if(urlQueryString.indexOf('x2=') &gt; -1)
            {
                var starti = urlQueryString.indexOf("x2=");
                var starti1 = urlQueryString.indexOf("x2=") + 4;            
                var endi = urlQueryString.myLastIndexOf("x2=", starti1) + starti1;
                urlQueryString = urlQueryString.replaceAt(starti, endi, decodeURIComponent(""));

                if (urlQueryString.indexOf("q2") &gt; -1) {
                    var q2Index = urlQueryString.indexOf("q2");
                    var q2String = urlQueryString.substring(q2Index);
                    var q1SemicolonIndex = q2String.indexOf(";") + 1; //plus 1 for semicolon

                    urlQueryString = urlQueryString.replace(q2String.substring(0, q1SemicolonIndex), "");
                }
            }

            if(urlQueryString.indexOf('sort=') &gt; -1)
            {
                var startj = urlQueryString.indexOf("sort=");
                var startj1 = urlQueryString.indexOf("sort=") + 5;            
                var endj = urlQueryString.myLastIndexOf("sort=", startj1) + startj1;
                urlQueryString = urlQueryString.replaceAt(startj, endj, decodeURIComponent(""));
            }
        }
        return urlQueryString;
    }

    function fillVar(inputs1, inputValuesArray) {
        var n = 0;      

        inputs1.each(function(){
            var name = jQuery(this).attr('name');
            if(jQuery(this).is(":checked")){              
                if(name == "Authors"){
                    isAuthor = true;
                } else if (name == "Exclusive") {
                    isExclusive = true;
                } else {
                    if (cmoInsiderArray.indexOf(name) &lt; 0 &amp;&amp; name != "CMO Insider") {
                        //if the name already exists somewhere else in the array, get rid of it, because we're placing it in a specific position now
                        if (inputValuesArray.indexOf(name) &gt; -1) {
                            inputValuesArray.splice(inputValuesArray.indexOf(name), 1);
                        }
                        inputValuesArray[n] = name;
                        n++;
                    } else {
                        jQuery("#site-section input.hiddenInputs").each(function(i) {
                            inputValuesArray[n] = jQuery(this).attr('name');
                            n++;
                        });
                    }                   
                }                
            } else {
                //when clicked checkbox is not checked, empty the array element variables and/or the second input variable
                if(name == "Authors"){
                    isAuthor = false;
                } else if (name == "Exclusive") {
                    isExclusive = false;
                } else {
                    if(inputValuesArray[n] == name) {
                        for (var i=0;i&lt;inputValuesArray.length;i++) {
                            inputValuesArray.splice(n,1);
                            if (name == "CMO Insider") {
                                //remove cmo insider children
                                var startIndex = inputValuesArray.indexOf("Ask the Headhunter");
                                var endIndex = inputValuesArray.indexOf("Digital Disruption");
                                inputValuesArray.splice(startIndex, endIndex + 1);
                            }
                        }
                    }
                }
            }
        })
    }

    function addExclusive() {
        if (isExclusive) {
            if (queryString.indexOf("q2=") &lt; 0) {
                queryString += "q2=true;x2=exclusive;";
            }
        } else {
            if (queryString.indexOf("q2=true;x2=exclusive;") &gt; -1) {
                queryString = queryString.replace("q2=true;x2=exclusive;", "");
            }
        }
    }

    function addAuthor() {
        if (isAuthor)
        {    
            //get search query value from input field
            var q = "";
            if(jQuery('#q').val() != null){         
                q = jQuery('#q').val();            
            } 

            if(queryString.indexOf("article_author_name") &lt; 0) {
                if (queryString.indexOf("q=") &lt; 0) {
                    queryString = "q=article_author_name:" + q + ";";
                } else {
                    queryString = "q=article_author_name:" + queryString.substr((queryString.indexOf("q=")+2),queryString.length);
                }
            }
            //if no q1 and no article_author_name, add the article_author_name to q
    //        if(queryString.indexOf('q1=') &lt; 0 &amp;&amp; queryString.indexOf("article_author_name") &lt; 0) {     
    //            queryString = "q=article_author_name:" + queryString.substr((queryString.indexOf("q=")+2),queryString.length);    
    //        }          
        } else {
            if (queryString.indexOf("article_author_name:") &gt; -1) {
                queryString = queryString.replace("article_author_name:", "");
            }
        }
    }

    function addSections() {
        if (inputValuesArray.length &gt;= 1) {
            var sectionName = getSectionName();

            //In case i=1 is at the beginning of the query string.           
            queryString = trimIAndOthers(queryString); 

            if (queryString.indexOf("q1=") &lt; 0) {
                queryString += "q1=" + decodeURIComponent(sectionName) + ";x1=t1;view=xml;";
            } else {
                var start2b = queryString.indexOf("q1=");
                var start2aa = queryString.indexOf("view=xml;");
                var end2aa = queryString.myLastIndexOf("view=xml;", start2aa) + start2aa;
                queryString = queryString.replaceAt(start2b, end2aa, "");                       
                queryString +="q1=" + decodeURIComponent(sectionName) + ";x1=t1;view=xml;";
            }
        } else if (queryString.indexOf("q1=") &gt; -1) {
            var start2b = queryString.indexOf("q1=");
            var start2aa = queryString.indexOf("view=xml;");
            var end2aa = queryString.myLastIndexOf("view=xml;", start2aa) + start2aa;
            queryString = queryString.replaceAt(start2b, end2aa, "");
        }
        removeDuplicates();

    }

    function getSectionName() {
        //set sectionName to all the checked inputValues (except author &amp; exclusive)
        var sectionName = "";
        for (var i=0;i&lt;inputValuesArray.length;i++)
        {                                                                   
            sectionName += (i &gt; 0) ? "|" + inputValuesArray[i] : inputValuesArray[i];
        }
        if(sectionName.indexOf(" ") &gt; 0) sectionName = encodeURIComponent(sectionName.replace(/ /g,"+"));
        return sectionName;
    }

    function removeDuplicates() {
        if (queryString.indexOf(";;")) {
            queryString = queryString.replace(";;", ";");
        }

        if (queryString.indexOf("view=xml") != queryString.lastIndexOf("view=xml")) {
            var firstIndex = queryString.indexOf("view=xml");
            queryString = queryString.replaceAt(firstIndex, firstIndex + 9, "");
        }
    }

    queryString = decodeURI(window.location.search);

        } else  {

        }

         <p>
          Soccer fans are considered to be among the most passionate of sports fansas in
          <i>
           fanatics
          </i>
          . Analysis from Adobe Digital Index (ADI) confirms that notion: The 2014 FIFA World Cup, which begins June 12, has already accumulated more than 19 million social mentions since June 2013.
         </p>
         <p>
          Soccer fans around the world are voicing their excitement for the event, with the World Cup receiving more social mentions in the month leading up to it than either the Super Bowl or Sochi Olympics. In fact, based on this social buzz data its seeing, ADI predicts that, in terms of total mentions, the World Cup will be the most social sporting event ever.
         </p>
         <p>
          <img height="257" src="/content/dam/CMO_Other/ADI/WorldCup_2014/World_Cup_Social_Buzz.png" width="430"/>
         </p>
         <p>
          ADI captured data through the Adobe Social listening tool, collecting buzz from blogs, Facebook, Google+, Reddit, Twitter, Dailymotion, Flickr, Instagram, Tumblr, VK, Disqus, Foursquare, Metacafe, Wordpress, and YouTube. The results are based on more than 69 million social mentions, across 230 countries and territories, talking about World Cup. (Note: Adobe is CMO.coms parent company.)
         </p>
         <p>
          So far, ADI reports, 90% of the world has already contributed to the World Cup conversation, which is more than the Olympics (84%) and the Super Bowl (78%). The majority of World Cup buzz is coming from the APAC region, which has contributed 48% of related conversations on social media. EMEA came in second, with 32%, and the Americas came in third, with 20%.
         </p>
         <p>
          When looking at specific countries, Japan is talking most about the event via social channels, accounting for 37% of World Cup mentions. The U.K. is in second place, clocking in at 11%. Brazil (9%), Germany (8%), and the U.S. (8%) round out the top five.
         </p>
         <p>
         </p>
         <p>
          To date, the conversations around World Cup have been mostly positive, according to ADI. Fifty-nine percent of the social buzz for World Cup relates to admiration, joy, or anticipation for the eventeverything youd expect from loyal football fans.
         </p>
         <p>
          <img height="252" src="/content/dam/CMO_Other/ADI/WorldCup_2014/World_Cup_Top_10_Countries.jpg" width="430"/>
         </p>
         <p>
          On the other hand, Brazilians
          <a href="http://www.punchng.com/sports/brazil-2014-a-world-cup-tainted-by-protests-violence/" target="_blank">
           arent exactly thrilled
          </a>
          about their country spending $500 million to host the World Cup. ADI found that 42% of social buzz from Brazilians has expressed sadness, anger, or disgust.
         </p>
         <p>
          <img height="257" src="/content/dam/CMO_Other/ADI/WorldCup_2014/_Brazil_Sentiment_World_Cup.jpg" width="430"/>
         </p>
         <p>
          World Cup players are getting a lot of buzz, as well. Unsurprisingly, Cristiano Ronaldowho has 26.5 million followers on Twitteris the most-mentioned player on the World Cup roster. He had more than 1.5 million mentions in May 2014 alone.
         </p>
         <p>
          <img height="272" src="/content/dam/CMO_Other/ADI/WorldCup_2014/World_Cup_Player_Mentions.jpg" width="431"/>
         </p>
         <p>
          During the course of the World Cup, the ADI team will track broadcast viewing data of the event, sponsor/advertiser buzz lift, and, for the first time ever, in-stadium tweeting and Foursquare check-ins. Follow the ADI team on Twitter for updated World Cup stats and insights at
          <a href="https://twitter.com/adobeindex">
           @AdobeIndex
          </a>
          .
         </p>
         <p>
          The World Cup is a global event, as you can see in the data, said Joe Martin, an analyst at ADI. This analysis shows the types of insights
          <a href="http://www.cmo.com/articles/2014/6/9/world-cup-marketing--global-push-from-mega-brands.html" target="_blank">
           marketers
          </a>
          can glean from social media. Companies can see where in the world people are talking about them and what they are saying.
         </p>
