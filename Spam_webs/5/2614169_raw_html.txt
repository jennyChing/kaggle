  <script type="text/javascript">
   function socialiteOnScroll(){
    window.socialised = window.socialised || {};
    var articles = $('.post');
    var win = $(window), updateArticles, onUpdate, updateTimeout;

    updateArticlesList = function(){
      articles = $('.post');
    }

    updateArticles = function()
    {
      // viewport bounds
      var wT = win.scrollTop(),
        wL = win.scrollLeft(),
        wR = wL + win.width(),
        wB = wT + win.height();
      // check which articles are visible and socialise!
      for (var i = 0; i &lt; articles.length; i++) {
        if (socialised[i]) {
          continue;
        }
        // article bounds
        var art = $(articles[i]),
          aT = art.offset().top,
          aL = art.offset().left,
          aR = aL + art.width(),
          aB = aT + art.height();
        // vertial point inside viewport
        if ((aT &gt;= wT &amp;&amp; aT &lt;= wB) || (aB &gt;= wT &amp;&amp; aB &lt;= wB)) {
          // horizontal point inside viewport
          if ((aL &gt;= wL &amp;&amp; aL &lt;= wR) || (aR &gt;= wL &amp;&amp; aR &lt;= wR)) {
            socialised[i] = true;
            Socialite.load(articles[i]);
          }
        }
      }
    };

    onUpdate = function()
    {
      if (updateTimeout) {
        clearTimeout(updateTimeout);
      }
      updateTimeout = setTimeout(updateArticles, 100);
    };

    win.on('resize', onUpdate).on('scroll', onUpdate);

    setTimeout(updateArticles, 100);
  }
  </script>

       </a>
