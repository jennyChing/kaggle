       function UpdateSearchField(terms) {
    	    
            // Limit search to 50 characters here as well in case the user somehow got past the textbox control limiter
            if (terms.length &gt; 50)
    	    {
    	        terms = terms.substring(50);
    	    }   // if

    	    // Encode search terms
    		$('#ucNav_hfSearch').val(encodeURI(terms));

    		if ($('.search-input').val() == 'type your search here...' || $('.search-input').val() == '')
    		{
    			return false;
    		};

    	function autocompleteSearch() {
    	    var ajaxUrl = "/SearchResults.aspx/GetAutoSuggest";
            var val = $(".search-input").val().replace(/'/g, "\\'");

            $.ajax(
                {
                    type: "POST",
                    url: ajaxUrl,
                    async: true,
                    username: null,
                    password: null, 
                    data: "{'query':'" + val + "'}",
                    contentType: 'application/json; charset=utf-8',
                    dataType: 'json',
                    success: function (msg) {
                        var newList = (msg.d) ? msg.d.split("||") : [];
                        var ajaxSource = [];
                        for (var i = 0; i &lt; newList.length; i++) {
                            var itemstring = newList[i];
                            var itemobj = { label: itemstring, value: itemstring };
                            ajaxSource.push(itemobj);
                        }
                        $(".search-input").autocomplete("option", "source", ajaxSource);
                    }, // success
                    error: function (jqxhr, txtstatus, errthrown) {
                       // alert?
                    } // error
                }  // ajax
          )
        }
        $(document).ready(function () {
           
            $(".search-input").keyup(function () {
                 autocompleteSearch();

                response: function (event, ui) {
                    // override to match everything in source
                    ui.content.splice(0, ui.content.length); // ditch all, but keep object
                    var src = $(this).autocomplete("option", "source")
                    for (var i = 0; i &lt; src.length; i++) {
                        ui.content.push(src[i]);
                    }
                }
            });

           Review of Album By Unknown Author 2011:
           <br/>
           Agrupacin Mark 16:15 perform a brand of Christian pop that draws mainly from vintage guitar pop and Mariachi vocals, with a vaguely psychedelic aftertaste at times. A Google search yields nothing about them, or their label, but Ensame A Amar is a pleasant surprise of a listen- thoroughly quirky and really quite charming. The production style is rather bare-bones (though this is hard to judge accurately, as the right channel of the stereo mix is practically inaudible), and with no insert in the case and no date printed on the cassette label, theres little hope of dating it accurately. Id estimate it could have been recorded anywhere between 1981 and 1996, give or take a few years.
           <br/>
           <br/>
           Most of the songs are sung in Spanish, with a handful of English songs thrown in to round out Side A. The Spanish-language tracks are principally sung in a Mariachi-ish duet, and the English language tracks veer a little more towards a discount brand of soul/R&amp;B- inflected pop. Theres a great cheapo string synth sound throughout the record, and a few of the songs feature some surprisingly off-kilter chord structures and odd production flourishes (the out-of-time hammer sound on Escuche El Sonido- is that Christ being nailed to the cross?) but what will definitely catch your attention most is the crazy lead guitar work. Whoever this session player was, he pulled off some searing leads and runs all over the place- check out the absurd scale run that starts off Solo Jesucristo Puede"
           <br/>
           If I have to cherry-pick my favorites (which is kind of difficult, as Ive developed quite a soft spot for this tape), Id recommend Solo Jesucristo Puede, Look For The Lord, I Love You Jesus, Cristo Viene, and Cuan Grande Su Amor. But really, give the whole thing a listen- its pretty great from end to end.

		function HideNoPreviewPlayButton()
		{
			$("tr[itemprop*='track']").each(function() {
				if ($(this).find(".jplay-button").length === 0) {
					var playIcon = $(this).find(".play-track");
					playIcon.hide();
				}
			});
		}

        function ShowShareSongMenu()
        {	
        	// Load menu on click
        	$('.custom_button').click(function() {
        		$(this).parent().parent().find('.hover_menu').fadeIn('fast');
        		$(this).data('in', true);
        		$(this).parent().parent().find('.hover_menu').data('hidden', false); 
        		$(this).addClass('present');         		
			});

			// Fix cursor issue
        	$('.hover_menu_close').mouseover(function()	{document.body.style.cursor='pointer';})
			.mouseleave(function() 	{ document.body.style.cursor='default'; });

			// Hide menu on click
        	$('.hover_menu_close').click(function ()
        	{
        		$(this).parent().hide();					        		
        	});     	
        }	// ShowShareSongMenu()
			
        function LoadSingleSongFunctions()
        {
        	ChangeSingleSongURL();
        	ChangeSingleSongMetaTitle();
        	PlaySingleSong();
        }	// LoadSingleSongFunctions()

        function PlaySingleSong()
        {
        	var trackInURL = window.location.pathname.split('/'); 
        	var correcTrack = trackInURL[trackInURL.length-1];
			
        	$('.track-title-meta').each(function ()
        	{
        		var currentTracks = $(this).attr("content");

        		currentTracks = currentTracks.replace(/\s+/g, '').replace(/[^a-z0-9\s]/gi,'');
								
        		if(currentTracks == correcTrack)
        		{
        			$(this).parent().parent().find('.play').click();
        		}	// if
        	});
        }  // CheckAndPlaySingleSong()

			
        function ChangeSingleSongURL()
        {
        	$(".custom_button").each( function()
        	{        		
        		var trackName = $(this).parent().parent().parent().parent().find('.track-title-meta').attr("content");	
		
        		if(trackName != undefined)
        		{   	
        			// Trim down the track name
        			trackName = trackName.replace(/\s+/g, '').replace(/[^a-z0-9\s]/gi,'');

        			var tempUrl = document.URL.split('#'); 

        			// Make sure we only attach one song title
        			var url = tempUrl[0].split('/');
        			var newURL = url[0] + "/" + url[1] + "/" + url[2] + "/" + url[3] + "/" + url[4] + "/"+ trackName;
					        			
        			$(this).parent().parent().attr("addthis:url",newURL);
        		}	// if
        	});
			        	
        	$(".addthis_button_compact").each(function()
        	{
        		$(this).parent().attr("addthis:url",document.URL);
        	});

        }	// ChangeAddThisURL()
		
        function ChangeSingleSongMetaTitle()
        {
        	var orgTitle = $('meta[property="og:title"]').attr('content');
        	var titlePieces =  orgTitle.split('-'); 
        	var twitterTitle;
        	var updatedTemplate;
        	var twitterTemplate = addthis_share.templates.twitter.split(';');
			
        	// Share songs 
        	$(".custom_button").mouseover(function ()
        	{
        		var trackName = $(this).parent().parent().parent().parent().find('.track-title-meta').attr("content");	
				        		
        		var ogTitle = ' ' + trackName + ' - ' + titlePieces[1];
        					
        		if(trackName != 'undefined')
        		{
        			var ogTitle = ' ' + trackName + ' - ' + titlePieces[1];
        		}	// if
        		else
        		{
        			var ogTitle = orgTitle;
        		}	// else
				
        		// Change meta titles				
        		$('meta[property="og:title"]').attr('content', ogTitle);
        		
        		updatedTemplate = twitterTemplate[0] + ' ' + trackName + ' - ' + twitterTemplate[1];
        		addthis_share.templates.twitter = updatedTemplate; 
        	});

        	// Share album
        	$(".addthis_button_compact").mouseover(function ()
        	{
        		updatedTemplate = twitterTemplate[0] + ' ' + twitterTemplate[1];
        		addthis_share.templates.twitter = updatedTemplate; 
        		
        		// Put meta back to orginal 
        		$('meta[property="og:title"]').attr('content', orgTitle);
        	});
        }	// ChangeSingleSongMetaTitle()
