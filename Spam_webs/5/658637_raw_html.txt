jQuery('span[type=attr-value]').click(function(){
	var select = jQuery(this).attr('select');

	var listBox = jQuery(this).parents('div').eq(0);
	var selectedBox = listBox.find('*[type=attr-value][select=1]');
	if (selectedBox.length &gt; 0){
		DoUnSelect(selectedBox[0]);
	}

	if (select == 1){
		DoUnSelect(this);
	}else{
		DoSelect(this);
	}

	function DoSelect(obj){
		jQuery(obj).addClass('select');
		jQuery(obj).attr('select', 1);
		jQuery(obj).removeClass('mouseover');
		jQuery(obj).addClass('selected');
		
		jQuery('#colorName').val(jQuery(obj).attr('value_id'));
		jQuery('#colorName').attr('append', jQuery(obj).attr('append'));
		jQuery('#colorName').attr('attr_value_name', jQuery(obj).attr('value_name'));
		//jQuery('.pdp_productInfoContainer .attribute').html('Color:'+jQuery(obj).attr('value_name'));
	}

	function DoUnSelect(obj){
		jQuery(obj).removeClass('select');
		jQuery(obj).attr('select', 0);
		jQuery(obj).removeClass('selected');
		jQuery(obj).removeClass('mouseover');

		jQuery('#colorName').val('');
		jQuery('#colorName').attr('append', '');
		jQuery('#colorName').attr('attr_value_name', '');

		//jQuery('.pdp_productInfoContainer .attribute').empty();
	}
}).mouseover(function(){
	if (jQuery(this).attr('select')!=1){
		jQuery(this).addClass('mouseover');
		if (jQuery.browser.msie &amp;&amp; jQuery.browser.version!='8.0'){
			jQuery('.pdp_productInfoContainer').css('z-index', '80');
		}
	}
	jQuery(this).find('div').show();
	jQuery(this).find('div').mouseover(function(){
		var $this = jQuery(this);
		setTimeout(function(){
			$this.hide();
		}, 100);
	});
}).mouseout(function(){
	jQuery(this).removeClass('mouseover');
	jQuery(this).find('div').hide();
	if (jQuery.browser.msie &amp;&amp; jQuery.browser.version!='8.0'){
		jQuery('.pdp_productInfoContainer').css('z-index', '');
	}
});
