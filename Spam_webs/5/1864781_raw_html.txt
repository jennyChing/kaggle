function wrapAnchors(box) {
	var anchors = $$(box + ' ul li a');
	if (null != anchors) {
		anchors.each(function(alpha) {
			spanIn = new Element('span');
			spanIn.wraps(alpha);
			spanEx = new Element('span');
			spanEx.wraps(spanIn);
		});
	}
}

function wrapListItems(box) {
	var listItems = $$(box + ' ul li');
	if (null != listItems) {
		listItems.each(function(li) {
			spanIn = new Element('span');
			spanIn.set('html',li.get('html'));
			spanEx = new Element('span');
			spanEx.adopt(spanIn);
			li.set('html','');
			li.adopt(spanEx);
		});
	}
}

             <h2>
              <a href="http://seawayblog.blogspot.com/2009/03/marrying-frogs-to-please-rain-god.html">
               Marrying frogs to please the rain god
              </a>
             </h2>
             <div class="belowtitles">
              By
              <span class="fn">
               Guido Trombetta
              </span>
              on 7:11 AM
              <a href="http://seawayblog.blogspot.com/search/label/News" rel="tag">
               News
              </a>
              ,
              <a href="http://seawayblog.blogspot.com/search/label/Photos" rel="tag">
               Photos
              </a>
             </div>
            </div>
            <div class="entry">
             <img src="http://www.sea-way.org/blog/FrogMarriage.jpg"/>
             <br/>
             <br/>
             A frog marriage ceremony takes place as a ritual to the rain gods to bring rain to the area!
             <br/>
             Elaborate preparations are believed to be the only solution to an unprecedented
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Drought" rel="wikipedia" target="_blank" title="Drought">
              drought
             </a>
             being faced by
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Assam" rel="wikipedia" target="_blank" title="Assam">
              Assam
             </a>
             ,
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/India" rel="wikipedia" target="_blank" title="India">
              India
             </a>
             , complete with a priest, shamianas, big feast and drums.
             <br/>
             <br/>
             The only irony is that the bride and bridegroom are a pair of male and female frogs. This is a customary ritual to please the rain god with such marriages taking place in several villages in Assam in the past one week as the State and the entire Northeastern region is suffering from unprecedented drought. According to the
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/India_Meteorological_Department" rel="wikipedia" target="_blank" title="India Meteorological Department">
              Indian Meteorological Department
             </a>
             (IMD),
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Nagaland" rel="wikipedia" target="_blank" title="Nagaland">
              Nagaland
             </a>
             ,
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Mizoram" rel="wikipedia" target="_blank" title="Mizoram">
              Mizoram
             </a>
             ,
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Manipur" rel="wikipedia" target="_blank" title="Manipur">
              Manipur
             </a>
             and
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Tripura" rel="wikipedia" target="_blank" title="Tripura">
              Tripura
             </a>
             are facing 99 per cent deficiency while
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Assam" rel="wikipedia" target="_blank" title="Assam">
              Assam
             </a>
             and
             <a class="zem_slink" href="http://en.wikipedia.org/wiki/Meghalaya" rel="wikipedia" target="_blank" title="Meghalaya">
              Meghalaya
             </a>
             are experiencing 89 per cent deficiency in the average rainfall.
