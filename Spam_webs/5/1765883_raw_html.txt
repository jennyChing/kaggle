            <h1 itemprop="name">
             74 shootings took place in US schools since Newtown massacre
            </h1>
           </div>
           <div class="white-border-caption">
            <div class="caption-img">
             <img alt="74 shootings took place in US schools since Newtown massacre" itemprop="image" src="http://cdn.ruvr.ru/2014/06/11/1507460079/9o_715924.jpg" title="74 shootings took place in US schools since Newtown massacre"/>
             <div style="display: none; float:left">
             </div>
             <div class="img-copyright">
              <span>
               <p>
                 Photo:
                <noindex>
                 <a href="http://www.vesti.ru" rel="nofollow">

            <h2>
             <span class="text-caption">
              <p>
               The school shooting in Oregon on Tuesday became the 74th instance of firearms discharge on school grounds or in school buildings since the late-2012 elementary school shooting in Newtown, Conn., according to a list maintained by the group Everytown for Gun Safety, according to The Washington Post.
              </p>
             </span>
            </h2>
           </div>
           <div class="diff-text-block">
            <p id="p_10">
             Everytown identifies a school shooting as any instance in which a firearm was discharged within a school building or on school grounds, sourced to multiple news reports per incident. Therefore, the data includes not just mass shootings, but also assaults, homicides, suicides and even accidental shootings.
            </p>
            <p id="p_20">
             In 2014 there have been at least 37 shootings on school grounds in the US, and nearly one shooting per week in the year and a half since Newtown. Ten shootings were reported in Georgia, the fact that some attribute to an expansive pro-gun law that was passed in the state this year. Seven shootings took place in Florida, five in Tennessee and four in North Carolina and California each. Atlanta was the only city that had three such shootings. Six other cities had two shootings.
            </p>
            <p id="p_30">
             All told, 31 states are represented on the list of shootings in schools or on school campuses or grounds. Of the shootings, 35 took place at a college or university, while 39 took place in K-12 schools.
            </p>
            <p id="p_40">
             A February analysis by the group of a list of school shootings since Newtown (which was later expanded) found that nearly half resulted in at least one death. Three in four shooters obtained guns at home, at least in instances in which the firearms source could be determined.
            </p>

   function login_errors(error) {
    if(error == 'captcha') return $.labels('reg_err_captcha', 'd13');
    if(error == 'empty')   return $.labels('reg_err_empty', 'd13');
    if(error == 'account_dup')   return $.labels('reg_err_account_dup', 'd13');
    if(error == 'email_dup')     return $.labels('reg_err_email_dup', 'd13');
    if(error == 'no_registration') return $.labels('reg_err_no_registration', 'd13');
    if(error == 'invalid_email') return $.labels('reg_err_invalid_email', 'd13');
    if(error == 'badlogin') return $.labels('reg_err_badlogin', 'd13');
    if(error == 'badpassword') return $.labels('reg_err_badpassword', 'd13');
    return $.labels('msg_bad_login', 'd13');
}

		var error = [];

		lerror.hide();
		login.removeClass(cierror);
		password.removeClass(cierror);

		if (!login.val()) {
			login.addClass(cierror);
			error[0] = $.labels('msg_bad_login', 'd13');
		}

		if (!password.val()) {
			password.addClass(cierror);
			error[0] = $.labels('msg_bad_login', 'd13');
		}

		if (error.length) {
			$('div', lerror).html(error.join('&lt;br/&gt;'));
			lerror.show();
			return false;
