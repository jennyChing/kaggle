var max = 2000;

	if(this.value.length &gt;= max)  return false;

function processForm() {


	var content = jQuery("#write-us-content").val().trim();
	var user_email = jQuery("#useremail").val().trim();


	if(content == false || content == 'Write Us') {
		jQuery('#write-us-notice').html('&lt;br/&gt;&lt;br/&gt;Please Enter Your Suggestions!');
		return false;
	}

	jQuery.ajax({
		type: 'POST',
		url: 'http://www.petrolicious.com/index.php?option=com_writeus&amp;task=processForm',
		data: {
			content: content,
			email : user_email
		},
		success: function(data) {

				if(data) {
					jQuery('#write-us-notice').html(data);
					jQuery("#write-us-content").val('');

					setTimeout(ToggleBox, 2000);

				}
		}
	});

	return false;

}

function ToggleBox() {
  jQuery('#txt-box').slideToggle('', function() {
		jQuery('.title-expand').toggleClass('up', jQuery(this).is(':visible'));
 });
