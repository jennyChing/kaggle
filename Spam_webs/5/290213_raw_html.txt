    var map = new Datamap({
scope: 'world',
element: document.getElementById('map-container'),
projection: 'mercator',
fills: {
    HIT: 'black',
    defaultFill: '#444',
},
data: {
},
geographyConfig: {
    borderWidth: 1,
    highlightBorderWidth: 1,
    borderColor: 'black',
    highlightFillColor: '31c6ed',
    popupOnHover: false, //disable the popup while hovering
    popupTemplate: function(geo, data) {
        if (data) {
            place_name = geo.properties.name + ' (' + geo.id + ')'
            return ['&lt;div class="hoverinfo"&gt;&lt;strong&gt;',
            data.val,
            '&lt;br/&gt;exposure in ' + place_name +  '&lt;/strong&gt;&lt;/div&gt;'].join('');
        } else {
            return '';
        }
    }
},
bubbleConfig: {
    popupOnHover: false,
    borderColor: 'black',
    highlightBorderWidth: 1,
    borderWidth: 1
}
});

g = map.svg.select("g.datamaps-subunits").selectAll("path").on("click", clicked);
width= $('#map-container').width()
height = $('#map-container').height()
active = d3.select(null);
var path = d3.geo.path()
    .projection(map.projection);

C = 4

function clicked(d) {
  gbubs = map.svg.select("g.bubbles")
  if (active.node() === this) return reset();
  active.classed("active", false);
  active = d3.select(this).classed("active", true);
  var bounds = path.bounds(d),
      dx = bounds[1][0] - bounds[0][0],
      dy = bounds[1][1] - bounds[0][1],
      x = (bounds[0][0] + bounds[1][0]) / 2,
      y = (bounds[0][1] + bounds[1][1]) / 2,
      scale = .9 / Math.max(dx / width, dy / height),
      translate = [width / 2 - scale * x, height / 2 - scale * y];

  map.options.geographyConfig.highlightBorderWidth = C / scale + "px"
  map.options.geographyConfig.borderWidth = C / scale + "px"
  g.transition()
      .duration(750)
      .style("stroke-width", C / scale + "px")
      .attr("transform", "translate(" + translate + ")scale(" + scale + ")");

  gbubs.transition()
      .duration(750)
      .attr("transform", "translate(" + translate + ")scale(" + scale + ")")
      .selectAll("circle")
      .style("stroke-width", C / scale + "px")
      .attr("r", C / scale + "px");
}

function clickedElement(d, that) {
  gbubs = map.svg.select("g.bubbles")
  if (active.node() === that) return reset();
  active.classed("active", false);
  active = d3.select(that).classed("active", true);
  var bounds = path.bounds(d),
      dx = bounds[1][0] - bounds[0][0],
      dy = bounds[1][1] - bounds[0][1],
      x = (bounds[0][0] + bounds[1][0]) / 2,
      y = (bounds[0][1] + bounds[1][1]) / 2,
      scale = .9 / Math.max(dx / width, dy / height),
      translate = [width / 2 - scale * x, height / 2 - scale * y];

  map.options.geographyConfig.highlightBorderWidth = C / scale + "px"
  map.options.geographyConfig.borderWidth = C / scale + "px"
  g.transition()
      .duration(750)
      .style("stroke-width", C / scale + "px")
      .attr("transform", "translate(" + translate + ")scale(" + scale + ")");

  gbubs.transition()
      .duration(750)
      .attr("transform", "translate(" + translate + ")scale(" + scale + ")")
      .selectAll("circle")
      .style("stroke-width", C / scale + "px")
      .attr("r", C / scale + "px");
}

function reset() {
  gbubs = map.svg.select("g.bubbles")

  active.classed("active", false);
  active = d3.select(null);

  g.transition()
      .duration(750)
      .style("stroke-width", "1px")
      .attr("transform", "");
  gbubs.transition()
      .duration(750)
      .attr("transform", "")
      .selectAll("circle")
      .style("stroke-width", "2px")
      .attr("r", "3px");
}

$.getJSON("/monitor/mapdata.json?type=exposure", function(data) {
map.updateChoropleth(data);
});
$.getJSON("/monitor/mapdata.json?type=hits", function(data) {
map.bubbles(data, {fillOpacity: 0.95, popupOnHover: false, highlightFillColor: 'black', borderWidth: 1});
});
