            <p>
             Managing diabetes is a full time job, but with a little bit of humor (and plenty of supplies) we can take it all in stride. Here are 29 things only a person living with diabetes would understand.
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               1. Every paper cut is an opportunity to test your blood sugar.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/1_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               2. You have an entire drawer, dresser, or closet devoted to diabetes supplies.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/2_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               3. You have hundreds of lancets and only a few test strips. But on the plus side, your health insurance company is willing to pay for more lancets!
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/3_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               4. When its time to test, all you have to do is squeeze your finger.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/5_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               5. The phrase once in a blue moon is a reminder that its time to change your lancet.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/25_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               6. You hesitate to wear white in case you prick your finger and hit a gusher.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/6_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               7. Your fingers appear to spell something in braille.
              </h2>
              <p>
               <span class="imageAreaBordered ">
                <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/7_1.jpg" title="image" width="642"/>
               </span>
              </p>
             </div>
            </div>
            <div class="wrap">
             <div class="content-head">
              <h2>
               8. Being high means something completely different to you than it does to most people.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/8_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               9. You can calculate the carbohydrate total of every meal in your head without breaking a sweat.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/9_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               10. You should test your blood sugar six times a day, but insurance only approved you for one strip a week.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/4_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               11. You can put a mathematician to shame: insulin on board, carb factors, insulin to carb ratio, no problem!
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/10_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">

               12. Well-meaning friends have offered you every diabetes remedy under the sun, from cinnamon to birdseed milk.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/11_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               13. Youve heard, But you dont look like a diabetic!
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/12_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               14. Youre familiar with all the diabetes horror stories of the relatives of anyone youve ever met.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/13_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               15. Youve heard, You cant eat that! too many times.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/14_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               16. Everyone wants to know where you got your cool pager.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/16_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               17. You find used test strips in your refrigerator but dont know how they got there.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/15_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">
              <h2>
               18. You have a pile of diabetes cookbooks holding up your sofa.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/17_1.jpg" title="image" width="642"/>
             </span>
            </p>
            <div class="wrap">
             <div class="content-head">

               19. You own 15 glucose meters, but you only use one.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">

             </span>
            </p>
            <div class="wrap">
             <div class="content-head">

               20. CSI would have a very hard time investigating the scene at your house.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/19_1.jpg" title="image" width="642"/>

             </div>
            </div>

             <span class="imageAreaBordered ">

             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/21_1.jpg" title="image" width="642"/>

             <div class="content-head">

               23. The pharmacy is number one on your speed dial, and youre on a first name basis with the pharmacist.
              </h2>
             </div>
            </div>
            <p>
             <span class="imageAreaBordered ">
              <img alt="image" height="493" src="http://www.healthline.com/hlcmsresource/images/29_things_diabates/img/22_1.jpg" title="image" width="642"/>
             </span>

    var fireOnce = false;
  require(['base/async'], function( Async ) {

    var waitForSignal = false;

    function GPTRefreshAllAds() {
      // KIMPS
      for( var i = 0; i &lt; adsArrayForKIMP.length; i++) {
        if( window.HL_ga_sendTrackEvent &amp;&amp; adsArrayForKIMP[i] ) {
          try {
            HL_ga_sendTrackEvent({
              gaEventCategory : 'Any Page',
              gaEventAction : 'Key Impression',
              gaEventLabel: adsArrayForKIMP[i] + ": hn.us.hl.lc.x.x.x",
              gaEventNonInteraction: true
            });
          } catch (e) {
            // if an undefined value in array exists, do no log in GA
            // this does not affect ad refresh
          }
        }
      }
      // THIS IS THE ONE CALL FOR ENTIRE SITE
      googletag.cmd.push( function() {
        googletag.pubads().refresh( adsArrayToRefresh );
      });
    }

    Async( arrayPromiseFunctions, function( PromiseFunction, callback) {
      PromiseFunction().then( function( dataObj ){
        if( dataObj ){
          adsArrayToRefresh.push( dataObj.slotObj );
          adsArrayForKIMP.push( dataObj.slotName );
        } else {
          waitForSignal = true;
        }
        callback();
      }, function() {
        callback();
      });
    }, function(){
      if ( waitForSignal ) {
        $( document ).on( 'wslAdNotShow', function() {
          if( !fireOnce ) {
            GPTRefreshAllAds();
            fireOnce = true;
          }
        });
      } else {
        GPTRefreshAllAds();
      }
    });
  });
   </script>
