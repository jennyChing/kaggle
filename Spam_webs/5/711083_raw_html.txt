        function LoadArticles(target, year, month, category) {
            $.ajax({
                type: "POST",
                url: Vars.data.baseURL + "Blogs.aspx/LoadArticles",
                data: "{year:" + year + ",month:" + month + ",category:" + category + "}",
                contentType: "application/json; charset=utf-8",
                dataType: "json",
                error: function (xhr, ajaxOptions, thrownError) {
                    $.showXhrErrorDialog(xhr, ajaxOptions, thrownError);
                    return false;
                },
                success: function (response) {
                    response = $.cleanAjaxResponse(response);
                    articleData = jQuery.parseJSON(response.genericMessage);
                    ShowArticleHeaders(target, articleData);
                    return;
                }
            });
        }

        function ShowArticleHeaders(target, articleData) {

                lis = lis + "&lt;li&gt;&lt;a href=\"" + url + "\"&gt;" + urlName + "&lt;/a&gt;&lt;/li&gt;"

            //console.log(lis);
            target.append("&lt;ul class=\"links-plus-large second\"&gt;" + lis + "&lt;/ul&gt;")
