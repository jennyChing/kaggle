        function checkLength( o, n, min, max ) {
            if ( o.val().length &gt; max || o.val().length &lt; min ) {
                o.addClass( "ui-state-error" );
                updateTips( "Duina " + n + " mora biti izmeu " +
                    min + " i " + max + " slova " );
                return false;
            } else {
                return true;
            }
        }
 
        function checkRegexp( o, regexp, n ) {
            if ( !( regexp.test( o.val() ) ) ) {
                o.addClass( "ui-state-error" );
                updateTips( n );
                return false;
            } else {
                return true;
            }
        }

	function register(){
	
		var dataString = 'type=ur&amp;name=' + document.getElementById('name').value + '&amp;pass=' + document.getElementById('password').value + '&amp;email=' + document.getElementById('emailR').value + '&amp;username=' + document.getElementById('uname').value;
		
		$.ajax({
			type: 'POST',
			url: '/generalUse/userHandler.php',
			data: dataString,
			cache: false,
			success: function(html){
				if(html=='1'){
					pozivL(1);
					$("#dialog-form").dialog( "close" );
				}
				else if(html=='2')
					pozivL(2);
				else if(html=='3')
					pozivL(9);
				else pozivL(5);
			}
		});	
	}
	
	function userLogin(data){
		if(data=='')
			var dataString = 'type=ul&amp;email=' + document.getElementById('emailL').value + '&amp;pass=' + document.getElementById('passwordL').value;
		else var dataString = data;
		
		$.ajax({
			type: 'POST',
			url: '/generalUse/userHandler.php',
			data: dataString,
			cache: false,
			success: function(html){
				pozivL(html);
				if(html=='3'){
					$("#dialog-form").dialog( "close" );
					window.location.reload();
					}
			}
		});	
	}
