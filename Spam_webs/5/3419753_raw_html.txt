         <h1>
          The Recipe
         </h1>
         <meta content="Deseree Kazda" itemprop="author"/>
         <meta content="A chewy decadent chocolate chip skillet cookie with dark chocolate chips and vanilla pudding mixed in. " itemprop="description"/>
         <meta content="http://www.lifesambrosia.com/wp-content/uploads/Vanilla-Pudding-Chocolate-Chip-Skillet-Cookie-562x848.jpg" itemprop="image"/>
         <meta content="Desserts" itemprop="recipeCategory"/>
         <meta content="American" itemprop="recipeCuisine"/>
         <ul class="amount_time_display">
          <li class="value_container" itemprop="recipeYield">
           <i class="icon-food">
           </i>
           Serves 8 - 10
          </li>
          <li class="value_container">
           <i class="icon-clock">
           </i>
           Prep time:
           <time datetime="PT0H15M" itemprop="prepTime">
            15 min
           </time>
          </li>
          <li class="value_container">
           <i class="icon-clock">
           </i>
           Cook time:
           <time content="PT0H40M" itemprop="cookTime">
            40 min
           </time>
          </li>
         </ul>
        </div>
        <div class="ingredients">
         <div class="ingredient_group">
          <p class="recipeheadings">
           Ingredients:
          </p>
          <ul>
           <li itemprop="ingredients">
            1/2 cup (1 stick) butter, softened
           </li>
           <li itemprop="ingredients">
            3/4 cup dark brown sugar
           </li>
           <li itemprop="ingredients">
            3/4 cup granulated sugar
           </li>
           <li itemprop="ingredients">
            2 eggs
           </li>
           <li itemprop="ingredients">
            2 1/4 cups all purpose flour
           </li>
           <li itemprop="ingredients">
            1 teaspoon baking soda
           </li>
           <li itemprop="ingredients">
            1/4 teaspoon salt
           </li>
           <li itemprop="ingredients">
            1 (3.4 ounce) package instant vanilla pudding mix
           </li>
           <li itemprop="ingredients">
            1 (10ounce) bag dark chocolate chips
           </li>
           <li itemprop="ingredients">
            Vanilla ice cream
           </li>
          </ul>
         </div>
        </div>
        <div class="instructions">
         <p class="recipeheadings">
          Instructions:
         </p>
         <ol>
          <li itemprop="recipeInstructions">
           Preheat oven to 350 degrees.
           <br/>
          </li>
          <li itemprop="recipeInstructions">
           Beat together butter, brown sugar and granulated sugar until light and fluffy. Beat in eggs one at a time.
           <br/>
          </li>
          <li itemprop="recipeInstructions">
           Add in flour, baking soda, salt and instant vanilla pudding mix. Mix until well combined. Fold in chocolate chips.
           <br/>
          </li>
          <li itemprop="recipeInstructions">
           Press cookie dough evenly into a well seasoned 10-inch cast iron skillet. Bake in preheated oven 30 - 40 minutes or until golden brown. Allow to cool several minutes before slicing. Serve with vanilla ice cream
          </li>
         </ol>
         <p class="note">
          Enjoy!
         </p>
