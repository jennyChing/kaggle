function checkMouseLeave(element, evt) {
	if(element.contains &amp;&amp; evt.toElement) {
	  return !element.contains(evt.toElement);
	}
	else if(evt.relatedTarget) {
	  return !containsDOM(element, evt.relatedTarget);
	}
}


function containsDOM(container, containee) {
	var isParent = false;
	do {
	 if((isParent = container == containee))
	  break;
	  containee = containee.parentNode;
	}
	while(containee != null);
	return isParent;
}

         <h2>
          #ThrowbackThursday Restaurant Special at De Grendel
         </h2>
         <h4>
         </h4>
         <h3>
          <strong>
           Tuck into old-school classics with a new-school twist at this Durbanville wine estate
          </strong>
         </h3>
         <p>
          With roots that reach back to 18th century South Africa, De Grendel winery knows a thing or two about history, and with the recent launch of its restaurants #ThrowbackThursday lunch and dinner special, the Durbanville estate looks to share its fancy for nostalgia in a fun way with visitors.
         </p>
         <p>
          The plan of action is that every Thursday chef Ian Bergh will conjure up a classic dish of past and add an appropriately De Grendel-esque modern spin. So, for example, the inventive kitchen whizz may reincarnate the avocado ritz - that famed 1970s tower of shrimp, mayonnaise and avo - as...wait for it...ice cream!
         </p>
         <p>
          Other culinary timepieces up for possible reinvention include the likes of grandmas beef wellington or oupas traditional tripe. But as the #ThrowbackThursday dish of the day will change weekly in line with the eaterys commitment to sourcing fresh, seasonal local produce, its best to check the De Grendel Facebook page to find out whats on the menu.
         </p>
         <p>
          Whats more, apart from freshening up old-school favourites, the restaurant is giving ordinary glasses of wine a tinge of present-day attitude. That means that diners in the mood for something other than the farms award-winning vino can do a meal or sundowners with a wine cocktail in hand. De Grendel has a huge range, from a chilli-ginger Ros concoction to a Pinot Gris litchi-sage mix.
         </p>
         <p>
          And best of all, CapeTownMagazine readers who mention our publication when ordering the #ThrowbackThursday special get a free Sauvignon Blanc cocktail (a delicious elderflower, fresh mint and soda water infusion).
         </p>
         <p>
          Of course, it goes without saying that the Northern Suburbs dining hotspot will also serve its popular contemporary menu, an offering thats impressively modern but that doesnt aspire to fine dining pretention. Diners can look forward to inclusions like chicken ravioli with gorgonzola cream and pea pure as well as a squid and chorizo salad that masterfully balances lemon chilli with a touch of smoke.
         </p>
         <p>
          No matter whether youre a fan of days gone by or the here and now, one things for sure though: the view of Table Mountain from the De Grendel restaurant is unbelievable.
         </p>
         <p>
          <strong>
           Costs and Booking Info for #ThrowbackThursday Restaurant Special at De Grendel
          </strong>
         </p>
         <p>
          Costs vary for the #ThrowbackThursday restaurant special but are in line with the restaurants normal prices (starters cost between R50 and R70 and mains begin from R110). Dont forget to mention CapeTownMagazine when ordering, and you get a free Sauvignon Blanc cocktail.
         </p>
         <p>
          Booking is recommended as the eatery gets busy toward the end of the week. To do so, phone +27 (0) 21 558 7035. Do note, the eatery is open for lunch from 12pm to 2:30pm and for dinner from 7pm to 9:30pm.
         </p>
         <p>
          ---
         </p>
         <p>
          Interested in learning more about how this eatery straddles two worlds: the old and the new? Read our review of
          <a href="http://www.capetownmagazine.com/de-grendel-restaurant">
           De Grendel restaurant
          </a>
          .
         </p>
