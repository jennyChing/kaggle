	function isLeapYear(yrVal)
	{
		var leapYear=false;
		var year = parseInt(yrVal, 10);
	
		if (year%4 == 0)
			{
			leapYear=true;
			}
		return leapYear;
	}
	
	function getDaysInMonth(monthVal, YrVal)
	{
		var maxDays=31
		if (monthVal==1)
		{
			if (isLeapYear(YrVal))
				{
				maxDays=29;
				}
			else
				{
				maxDays=28;
				}
		}
		if (monthVal==3 || monthVal==5 || monthVal==8 || monthVal==10)
			{
			maxDays=30;
			}
		return maxDays;
	}
