            function btf_calculate_height(posts_html){
                var $posts_html = $(posts_html);
                $posts_html.each(function(index, element){

                    if($(element).hasClass('clear-bordered')){
                        if(single_top_posts_elements.length &gt; 0){

                            }

                        single_top_post_title_height = 0;
                        single_top_posts_elements = [];
                    }

                    if($(element).hasClass('post-small') &amp;&amp; typeof $(element).attr('style') === 'undefined'){

                        single_top_post_title_height = $(element).find('div.title').height() &gt; single_top_post_title_height ? $(element).find('div.title').height() : single_top_post_title_height;

            function btf_get_post_shares(post_html){
                var $post_html = $(post_html);
                $post_html.each(function(index, element){
                    var shares = 0;
                    if(typeof $(element).find('a.link').attr('href') !== 'undefined'){
                        $.ajax("http://graph.facebook.com/" + encodeURIComponent($(element).find('a.link').attr('href')),
                                {
                                    method : 'GET',
                                    dataType : 'json',
                                    async : false,
                                    success: function(data){
                                        if (typeof data.shares !== 'undefined' &amp;&amp; data.shares !== null &amp;&amp; data.shares &gt; 0) {
                                            shares = data.shares &gt; 20 ? btf_format_shares(parseInt(data.shares) + 1000) : btf_format_shares(data.shares);
                                        }
                                        if(parseInt(shares) &gt; 0){
                                            $(element).find('.link .post-meta .shares').text(shares);
                                        }else{
                                            $(element).find('.link .post-meta .shares').remove();
                                        }

                                    }
                                });
                    }
                });
                return $post_html;
            }

            function btf_format_shares(n) {
                var ranges = [
                    { divider: 1e9 , suffix: 'G' },
                    { divider: 1e6 , suffix: 'M' },
                    { divider: 1e3 , suffix: 'K' }
                ];

                for (var i = 0; i &lt; ranges.length; i++) {
                    if (n &gt;= ranges[i].divider) {

                        var num = n / ranges[i].divider;
                        // if(Math.round(num) !== num) {
                        num = num.toFixed(1);

                        // }
                        return num.toString() + ranges[i].suffix;
                    }
                }

                return n;
            }

            }
        });
