          <h3>
           Walt Disney Parks &amp; Resorts Careers
          </h3>
          <p>
           Every day at Walt Disney Parks and Resorts Segment, the job of the cast member is to help make guests dreams come true. More than 50 years ago, Walt Disney had a visionary new idea for family entertainment: a place where families can enjoy a fully-immersive atmosphere rich in detail. The secret to his success was placing the cast member at the center of everything. Theyre the ambassadors of the brand, they bring the magic to life, and they truly are Disney.
          </p>
          <p>
           Today, there are 11 Disney theme parks located in 5 of the worlds leading family vacation destinations. If youre interested in building a career with the Disney Company, our enterprise includes ABC Television Studios, ESPN, Disney Cruise Line, and a host of other amazing, successful organizations. Could it be your job to make dreams come true? Could it be your job to make the magic happen? If you choose to build a career with Disney, yes it could.
          </p>

    function Validatecaptcha() {

        if (document.getElementById("txtCaptcha") != null &amp;&amp; document.getElementById("imgCaptcha") != null &amp;&amp; document.getElementById("TB_ajaxContent") != null) {

            var captchaText = null;

            var searchEles = document.getElementById("TB_ajaxContent").children;

            if (searchEles.length &gt; 0) {

                for (var i = 0; i &lt; searchEles[0].childNodes.length; i++) {

                    if (searchEles[0].childNodes[i].tagName == "INPUT") {

                        if (searchEles[0].childNodes[i].id == "txtCaptcha") {

                            captchaText = searchEles[0].childNodes[i].value;
                            break;

                        }

                    }

                }

            }

            if (captchaText != null &amp;&amp; captchaText != "") {

                WebContent.SeoSiteWebContent.ValidateCapatcha(captchaText, onValidateCaptchaSuccess, onValidateCaptchaFailed);

            } else {

                alert("Please enter valid text");
                refreshCaptcha();
                return false;

            }

        }

    }

    function onValidateCaptchaSuccess(result) {

        if (result == true) {

            if (IsValidCaptcha == false) {

                if (document.getElementById("txtEmail") != null &amp;&amp; document.getElementById("txtEmail").value != "") {

                    Alert.SubscribeAlert.isuserexists(document.getElementById("txtEmail").value, onIsSEOSitesUserExistsSuccess, onIsSEOSitesUserExistsFailed);

                } else {

                    alert('Email address is required');

                }

            }

            else if (IsValidCaptcha == true) {

                tb_remove();
                document.forms["form1"].submit();

            } else {

                refreshCaptcha();
                document.getElementById("txtCaptcha").focus();
                return false;

            }

        } else {

            alert("Please enter valid text");

            // Reload captcha in case it fails                

            refreshCaptcha();
            return false;

        }

    }

    function onValidateCaptchaFailed(result) {

        alert('an error has occured');
        return false;

    }

    function refreshCaptcha() {

        // Get a reference to CAPTCHA image

        var img = document.getElementById("imgCaptcha");

        if (img != null &amp;&amp; img != '') {

            // Change the image

            var newCaptcha = $('#imgCaptcha').attr({ src: '/Handlers/CaptchaImageHandler.ashx' + '?' + (new Date()).getTime() }).load(function () {

            });

            if (document.getElementById("TB_ajaxContent") != null) {

                document.getElementById("TB_ajaxContent").innerHTML = document.getElementById("divCaptcha").innerHTML;

            }

            document.getElementById("txtCaptcha").value = '';

            

            $("#TB_window input:first").blur().focus().focus();

            

        }

    }
