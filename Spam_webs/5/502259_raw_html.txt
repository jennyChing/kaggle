Function GetCustomConnection(CustomConnection)
	On Error Resume Next
	Err.Clear
	Dim dbConnection
	Set dbConnection = Server.CreateObject("ADODB.Connection")
	dbConnection.ConnectionTimeout = 3
	dbConnection.ConnectionString = CustomConnection
	dbConnection.Open 
	if dbConnection.State  1 Or dbConnection.Errors.Count &gt; 0 Then
		Dim errorString
		errorString="Unable to create public connection"
		subject = "SQL Server Connection Error at " &amp; CStr(Now())
		SendEmail ERROR_EMAIL_RECIPIENT, subject , errorString
		AddToADOLog errorString &amp; vbcrlf &amp; DisplayConnectionInfo(dbConnection),"ERROR"
		'Response.Write "
  <pre>
   " &amp; connectionInfo &amp; "
  </pre>
  "
		Set GetPublicConnection = Nothing
	Else
		Set GetPublicConnection = dbConnection	
	End If
	On Error GoTo 0
End Function
Sub CloseConnection(objConn)
	On Error Resume Next
	Dim errorString
	errorString=DisplayConnectionInfo()
	If errorString  "" Then
		AddToADOLog errorString
	End If
	if objConn.State  0 Then	objConn.Close()
	On Error Goto 0
End Sub

Function DisplayConnectionInfo(objConn)
	On Error Resume Next
	Dim connectionInfo
	'connectionInfo = objConn.ConnectionString
	'connectionInfo = connectionInfo &amp; objConn.ConnectionString &amp; vbcrlf
	connectionInfo = connectionInfo &amp; DisplayConnectionErrorInfo(objConn) &amp; vbcrlf
	DisplayConnectionInfo = connectionInfo
	On Error Goto 0
End Function

Function DisplayConnectionErrorInfo(objConn)
	Dim errorString
	errorString=""
	for each objErr in objConn.Errors
	  errorString = errorString &amp; "Description: " &amp; objErr.Description &amp; vbcrlf
	  'errorString = errorString &amp; "Help context: " &amp; objErr.HelpContext &amp; vbcrlf
	  'errorString = errorString &amp; "Help file: " &amp; objErr.HelpFile &amp; vbcrlf
	  'errorString = errorString &amp; "Native error: " &amp; objErr.NativeError &amp; vbcrlf
	  errorString = errorString &amp; "Error number: " &amp; objErr.Number &amp; vbcrlf
	  errorString = errorString &amp; "Error source: " &amp; objErr.Source &amp; vbcrlf
	  errorString = errorString &amp; "SQL state: " &amp; objErr.SQLState &amp; vbcrlf
	next
	DisplayConnectionErrorInfo=errorString
End Function
