function changeRatingView(id) {
	for (i = 1; i &lt;= 10; i++) {
		img = document.getElementById('v'+i);
		if (i &lt;= id) {
			img.style.opacity = 1;
		} else {
			img.style.opacity = 0.2;
		}
	}
}
function resetRatingView() {
	for (i = 1; i &lt;= 10; i++) {
		img = document.getElementById('v'+i);
		if (i &lt;= avgvote) {
			img.style.opacity = 1;
		} else {
			img.style.opacity = 0.5;
		}
	}
}
function submitRating(v) {
	$.getJSON('/vote/'+gameid+'/'+v,
		function(data){
			avgvote = data.voteavg;
			document.getElementById('gameratingtxt').innerHTML = avgvote;
			resetRatingView();
			if (data.status == -1) {
				$("#gameratingresult").html('Your have already voted for this game!&lt;br/&gt;');
			} else {
				$("#gameratingresult").html('Your vote has been submitted!&lt;br/&gt;');
			}
			if (!me) {
				$("#gamepowerresult").html('Your voting power is 1.00 at the moment.&lt;br/&gt;Please register or log in to raise it!');

				$("#gamepowerresult").html('Thank you for voting. Your voting power is &lt;b&gt;' + data.power + '&lt;/b&gt; at the moment.&lt;br/&gt;Voting power lasts for &lt;b&gt;'+data.votesleft+'&lt;/b&gt; votes more.');
			} else if (data.moretovote &gt;= 0) {
				$("#gamepowerresult").html('Thank you for voting. Vote for &lt;b&gt;'+data.moretovote+'&lt;/b&gt; more games to raise your voting power.');
			} else if (data.votesleft &lt; 0) {
				$("#gamepowerresult").html('You have used all your voting power.&lt;br/&gt;Vote again tomorrow to raise your voting power');
			}
		}
	);
}
function submitFavorite(id, act) {
	$.getJSON('/favorites/'+act+'/'+id,
		function(data){
			if (data.isfavorite == true) {
				$("#addfav").css('display', 'none');
				$("#removefav").css('display', 'block');
			} else {
				$("#addfav").css('display', 'block');
				$("#removefav").css('display', 'none');
			}
		}
	);
}
