function getOffset(el) {
    if (el.getBoundingClientRect)
        return el.getBoundingClientRect();
    else {
        var x = 0, y = 0;
        do {
            x += el.offsetLeft - el.scrollLeft;
            y += el.offsetTop - el.scrollTop;
        } 
        while (el = el.offsetParent);

        return { "left": x, "top": y }
    }

function getCanvasWidthHeight(image){
	var windowWidth = 303;//parseInt($(window).width());
	var windowHeight = 190;//parseInt($(window).height());
	var height = windowHeight; //parseInt((windowHeight-60) * 0.8);
	var width = windowWidth; //parseInt(windowWidth * 0.92);
	var naturalWidth = image.naturalWidth? image.naturalWidth : image.width;
	var naturalHeight = image.naturalHeight? image.naturalHeight : image.height;
	var imageWidth = naturalWidth;
	var imageHeight = naturalHeight;

	// fit image into container
	if ((width/height) &gt; (imageWidth/imageHeight)) {
		animation = "down";
		imageWidth = width;
		imageHeight = parseInt(naturalHeight * (imageWidth/naturalWidth));
	} else {
		animation = "left";
		imageHeight = height;
		imageWidth = parseInt(naturalWidth * (imageHeight/naturalHeight));
	}
	
	return {
		"width":width,
		"height":height,
		"animation":animation,
		"imageWidth":imageWidth,
		"imageHeight":imageHeight
		};
}

function updateSong() {
	if (!parent.metaIsWorking &amp;&amp; !playerIsVisible()) {
		startUpdateSong();
		return;
	}
	parent.songIsChanged = false;
	if (playerType == 'program') showProgramInfo();
	else {
		try {
			listenLogData = {
				"cmd":"getLastPlayedSongs",
				"stationID": "30",
				"mode": "radio",
				"playerType": "website",
				"includeNowPlaying": 1,
				"lastPlayedLimit": 6,
				"cacheKill": Math.random()
			};
			
			if (typeof moreClicks !== 'undefined' &amp;&amp; typeof numSongsPerPage !== 'undefined') {
				listenLogData.lastPlayedLimit = (moreClicks*numSongsPerPage)+1;
			}
			
			if (parent.metaIsWorking) {
				listenLogData.artistName = meta_artist;
				listenLogData.songName = meta_song;
			}
					  
			$.ajax({
				type: 'GET',
				dataType: 'json',
				url: '/player/servicePlayer',
				data: listenLogData,
				timeout: LISTEN_LOG_TIMEOUT,
				success: function(data) {
					if (data.lastPlayedSongs !== undefined) {
						var lastPlayedSongs = data.lastPlayedSongs;
						var last_played = '&lt;div class="last-played-item first"&gt;&lt;h2&gt;Last Played&lt;/h2&gt;&lt;/div&gt;';
						if (lastPlayedSongs.lastPlayedTemplate !== undefined &amp;&amp; lastPlayedSongs.songs !== undefined &amp;&amp; lastPlayedSongs.songs.length &gt; 0) {
							for (var i=0; i&lt; lastPlayedSongs.songs.length; i++) {
								last_played += lastPlayedSongs.lastPlayedTemplate.replace(/{SONG_URL}/g,lastPlayedSongs.songs[i].song_url).replace(/{THUMB_IMG}/g,lastPlayedSongs.songs[i].thumb_img).replace(/{SHORT_ARTIST_NAME}/g,lastPlayedSongs.songs[i].short_artist_name).replace(/{SHORT_SONG_NAME}/g,lastPlayedSongs.songs[i].short_song_name).replace(/{LAST_TIME_WORDS}/g,lastPlayedSongs.songs[i].last_played_words);
							}
						} else {
							last_played += '&lt;div class="last-played-item"&gt;No songs found.&lt;/div&gt;';
						}
						
						$('#last-played').html(last_played);
						$('#last-played .last-played-item').last().addClass('last');
					}
					
					if (data.nowPlaying !== undefined) {
						radioSetNowPlaying(data.nowPlaying);
					}
				},
				complete: function(data) {
					startUpdateSong();
				}
			});
		} catch(e) {
			//if(window.console) console.log('UpdateSong Error: ',e);
		}
	}
}

function parseMeta(event) {
	if (playerType == 'program') return;
	if(window.top === window.self &amp;&amp; $("iframe#socast").length &gt; 0) {
		var iframe = document.getElementById("socast");
		iframe.contentWindow.parseMeta(event);
		return;
	}
	if (parent.windowIsBlurred) {
		parent.songIsChanged = true;
	}
	//if (window.console) console.log(event.metadata);
	if (event.metadata.title != undefined &amp;&amp; event.metadata.title != '') {
		parent.metaIsWorking = true;
		meta_title = event.metadata.title.split(' - ');
		if (meta_title.length == 2) {
			meta_artist = meta_title[1];
			meta_song = meta_title[0];
			//showSongInfo({});
		} else {
			meta_artist = '(CFPL AM) AM 980';
			meta_song = '';
		}
		//updateListenLog();
		if (!parent.windowIsBlurred) updateSong();
		setTimeout(clearBuffer,3000);
	} else {
		meta_artist = '(CFPL AM) AM 980';
		meta_song = '';
	}
	$('#radio-player-artist-name').html(meta_artist);
	$('#radio-player-song-name').html(meta_song);
}

            <h1 class="entry-title">
             Brown, Cheng, Swan Face Off In Heated Mayoral Debate
            </h1>
            <div class="news-meta">
             <div class="city-station">
              London, Ontario, Canada / (CFPL AM) AM 980
             </div>
             <div class="author">
              <a href="mailto:">
              </a>
             </div>
             <div class="post-date">
              October 15, 2014 09:55 am
             </div>
            </div>
            <img alt="Brown, Cheng, Swan Face Off In Heated Mayoral Debate" class="header-image full" src="http://d2x3wmakafwqf5.cloudfront.net/wordpress/wp-content/blogs.dir/104/files/2014/10/debate.png"/>

           <div class="entry-content">
            <p>
             The top three candidates vying to become Londons next Mayor squared off today on AM980.
            </p>
            <p>
             Matt Brown, Paul Cheng and Joe Swan took questions from AM980s Craig Needles as well as each other.
            </p>
            <p>
             The hour long debate had some heated moments with Brown, Cheng and Swan all trading barbs and trying to poke holes in each others platforms.
            </p>
            <p>
             Matt Brown took aim at the proposed over and underpasses Paul Cheng would like to construct at Londons rail crossings. Brown said there are 26 such crossings in London and he wanted to know how Cheng planned to fund the work when each project would cost an estimated $25 million.
            </p>
            <p>
             Cheng fired back that Brown should read his platform before commenting on it.
            </p>
            <p>
             I said we need to make a
             <em>
              plan,
             </em>
              Cheng said stressing that he wants to prepare for the crossings to happen, but not necessarily break ground on them.
            </p>
            <p>
             But in the last four years, why havent the studies been done for the Adelaide railroad crossing? What are you guys waiting for? How many more years do you want to wait? Cheng questioned.
            </p>
            <p>
             Joe Swan was quick to jump on Cheng for the perceived flip-flop.
            </p>
            <p>
             In one sentence you just contradicted yourself, he said. First of all you said youre against plans now you want to plan. Which one is it, Paul?
            </p>
            <p>
             I see things as what we need. What you folks see is, Its always been done that way, we need more studies so well do more studies. But I want action, Cheng asserted.
            </p>
            <p>
             Sorry, Paul, thats just a load of hogwash, Swan shot back.
            </p>
            <p>
             Swan also took a swipe at Chengs assertion that the skills he honed while managing overseas oil rigs will serve him well should he become mayor.
            </p>
            <p>
             Could one then presume if Im not successful (in the mayoral race) I could run an oil rig in Peru? Swan asked to which Cheng replied, no.
            </p>
            <p>
             Oh, so theyre not transferable skills then? Swan asked. No it doesnt go the other way, Cheng said as Swan guffawed.
            </p>
            <p>
             Management skill is transferable, but you dont have the technical know-how, Cheng went on to say.
            </p>
            <p>
             And you think that at City Hall you dont need technical knowledge of legislation, planning acts, human resources, finance, budgeting, strategic planning? Youre presuming then that you can just come off an oil rig and just take the mantle and lead? Swan said.
            </p>
            <p>
             I think you may need to re-think that, Paul.
            </p>
            <p>
             Other questions the candidates faced included their stance on taxes and for Brown and Cheng specifically, the cost of their campaign promises.
            </p>
            <p>
             Brown has been taken to task by his fellow candidates for failing to provide specific financial details on his plan for Bus Rapid Transit and a high-speed fibre network.
            </p>
            <p>
             In response to Wednesdays question, Brown said the BRT initiative would receive grants of $100 million from both the provincial and federal governments as well as $75 million in development charges. As for the cost of the fibre optic network, Brown said the answer is simple.
            </p>
            <p>
             Ive heard some outrageous figures. The fact of the matter is wed be looking at a $50,000 investment to do this, $50,000 to run a pilot in our downtown, Brown said.
            </p>
            <p>
             Were behind on this, theres no question. Stratford has a program up and running, Exeter has a program up and running. This is something we need to do to attract and keep those high-tech businesses here in London.
            </p>
            <p>
             If you werent able to catch Wednesdays debate live, you can listen to it in its entirety below:
