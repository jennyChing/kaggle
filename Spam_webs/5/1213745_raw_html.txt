function blenza_doSubmit( form, op, action )
{
  var xurl;
  if ( form.ml_xurl ) {
    xurl = form.ml_xurl.value + "/" + action + ".php";
  }
  else if ( form.xurl ) {
    xurl = form.xurl.value + "/" + action + ".php";
  }
  else {
    xurl = "http://{$blenza_host}{$blenza_root}/" + action + ".php";
  }
  var data = new Array();
  for ( i = 0; i &lt; form.elements.length; i++ ) {
    var name = form.elements[i].name;
    if ( name == "" ) {
      name = "xyzzy";
    }
    if ( name.substring(0,3) == "ml_" ) {
      name = name.substring(3);
    }
    var val;
    if ( form.elements[i].type == "checkbox" ) {
      if ( form.elements[i].checked ) {
        val = form.elements[i].value;
      }
      else {
        val = "";
      }
    }
    else {
      val = form.elements[i].value;
    }

    data.push( { "name": name, "value": val } );
  }
  blenza_submitdata( data, xurl );
}

function blenza_doEnter( form )

  if ( form.ml_name ) {
    name = form.ml_name.value;
  }
  else {
    name = form.name.value;
  }
  var url;
  if ( form.ml_url ) {
    url = form.ml_url.value;
  }

    return false;

  blenza_doSubmit( form, "Entry", "entry_do" );
  var pi;
  if ( form.ml_postid ) {
    pi = form.ml_postid.value;

  else {
    pi = form.postid.value;
  }
  blenza_setNextone( pi, name, url );
  blenza_switcheroo( "theform-" + pi, "theform2-" + pi );
  return false;
}

  if ( name.length == 0 ) {

function blenza_checkUrl( url )
{
  if ( url.length == 0 ) {
    alert( "Please enter your link URL." );
    return false;
    return true;
  }
  var test = "http://"

    alert( "Please enter your link URL." );
    return false;
  }
  return true;
}

  }
  var data = new Array();
  for ( i = 0; i &lt; form.elements.length; i++ ) {
    var name = form.elements[i].name;
    if ( name == "" ) {
      name = "xyzzy";
    }
    if ( name.substring(0,3) == "ml_" ) {
      name = name.substring(3);
    }
    var val;
    if ( form.elements[i].type == "checkbox" ) {
      if ( form.elements[i].checked ) {
        val = form.elements[i].value;
      }
      else {
        val = "";
      }
    }
    else {
      val = form.elements[i].value;
    }

    data.push( { "name": name, "value": val } );
  }
  blenza_submitdata( data, xurl );
}

function blenza_switcheroo( off, on )
{
  var ele1 = document.getElementById( off );
  ele1.style.display = "none";
  var ele2 = document.getElementById( on );
  ele2.style.display = "block";
}

function blenza_setNextone( pi, name, url )
{
  var nx = document.getElementById( "nextone-" + pi );
  if ( nx != null ) {
    nx.innerHTML = '&lt;a class="blenza-link" href="' + url + '"&gt;' + name + '&lt;/a&gt;';
  }
}

function blenza_doEnter( form )
{
  var name;
  if ( form.ml_name ) {
    name = form.ml_name.value;
  }
  else {
    name = form.name.value;
  }
  var url;
  if ( form.ml_url ) {
    url = form.ml_url.value;
  }
  else {
    url = form.url.value;
  }
  if ( ! blenza_checkName( name ) ) {
    return false;
  }
  if ( ! blenza_checkUrl( url ) ) {
    return false;
  }
  blenza_doSubmit( form, "Entry", "entry_do" );
  var pi;
  if ( form.ml_postid ) {
    pi = form.ml_postid.value;
  }
  else {
    pi = form.postid.value;
  }
  blenza_setNextone( pi, name, url );
  blenza_switcheroo( "theform-" + pi, "theform2-" + pi );
  return false;
}

function blenza_checkName( name )
{
  if ( name.length == 0 ) {
    alert( "Please enter your name." );
    return false;
  }

  return true;
}

function blenza_checkUrl( url )
{
  if ( url.length == 0 ) {
    alert( "Please enter your link URL." );
    return false;
    return true;
  }
  var test = "http://"
  if ( url.substring( 0, test.length ) != test ) {
    alert( "Link URL must begin with 'http://'" );
    return false;
  }
  if ( url == "http://www." ) {
    alert( "Please enter your link URL." );
    return false;
  }
  return true;
}
