    <div class="main-body">
     <strong>
      A YOUNG EMPEROR PENGUIN
     </strong>
     has taken a rare wrong turn and ended up stranded on a New Zealand beach.
     <br/>
     <br/>
     Emperor penguins typically spend their entire lives in Antarctica and almost never make landfall near humans. It has been 44 years since a wild Emperor penguin was last sighted in New Zealand.
     <br/>
     <br/>
     Conservation experts say the penguin is about 10 months old, and stands about 80cm high. It likely took to the water several months ago to search for squid and krill and lost its way.
     <br/>
     <br/>
     Experts say the bird is healthy but will need to find its way back south if it is to survive.
     <br/>
     <br/>
     A resident discovered the bird while walking her dog on Monday evening at Peka Peka Beach on the southwest coast of the North Island, 60km north of Wellington.

		window.addEventListener('message', function(e) {		
			var $iframe =  $('.popup-modal:visible');
			$iframe.each(function(index){        		
				iframeEvent($(this),e);        			
			});
		}, false);   
		
		function iframeEvent(iframe,e)
		{
			var eventName,height, array=[]; 
			if ($.isArray(e.data)){
				array = e.data;			 
			}
			else {
				array = e.data.split(',');
			}
			
			eventName = array[0];
			height = array[1];
			
			switch(eventName) {
				case 'setHeight':
					iframe.height(height);
					break; 
				case 'loading':        			
					var $parent = iframe.closest('.reveal-modal')
					$parent.append('&lt;div class="overlay" style=""&gt;&lt;/div&gt;')
					break;
				case 'scroll':					
					$('html, body').animate({
						scrollTop: (iframe.offset().top - 100)
					}, 1000);
					break;
			}
		}

		function closePupop() {
			var $close =$('#AccBlock').find('.close-pop');
			if ($close.length &gt; 0) {
				window.location.href = $close.attr('href');
			}
		}

		$(document).ready(function() {
			$('.reveal-modal-bg').on('click', function() {
				closePupop();
			});
		});
