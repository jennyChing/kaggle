      function form_check() {
  var result = true;
  var error = '1px solid #ff0000';

  var email_check = new RegExp('^[a-z0-9._-]+@[a-z0-9.-]+[.][a-z]{2,4}$','gi');
  var code_check = new RegExp('^[0-9]{4}$','gi');

  var recipient = document.getElementById('recipient');
  var sender = document.getElementById('sender');
  var name = document.getElementById('name');
  var subject = document.getElementById('subject');
  var message = document.getElementById('message');
  var code = document.getElementById('code');

  if(!recipient.value.match(email_check)) {
    recipient.style.border = error;
    result = false;
  } else {
    recipient.style.border = '';
  }

  if(!sender.value.match(email_check)) {
    sender.style.border = error;
    result = false;
  } else {
    sender.style.border = '';
  }

  if(name.value == '') {
    name.style.border = error;
    result = false;
  } else {
    name.style.border = '';
  }

  if(subject.value == '') {
    subject.style.border = error;
    result = false;
  } else if(subject.value.length &lt; 10) {
    subject.style.border = error;
    result = false;
    alert('Subject is too short! ;)');
  } else {
    subject.style.border = '';
  }

  if(message.value == '') {
    message.style.border = error;
    result = false;
  } else if(message.value.length &lt; 20) {
    message.style.border = error;
    result = false;
    alert('Your message is too short! (minimum 20 chars)');
  } else {
    message.style.border = '';
  }

  if(!code.value.match(code_check)) {
    code.style.border = error;
    result = false;
  } else {
    code.style.border = '';
  }

  return result;
}

function code_empty() {
  document.getElementById('code').value = '';
}
