            // now try to match the url by looping through it and each time removing the last '/urlpart/' untill there are no more url parts
            var indexOfLastSlash = url.lastIndexOf("/");

            //debugger;

            while ((indexOfLastSlash != -1) &amp;&amp; (matched == false)) {
                if (indexOfLastSlash != 0) {
                    url = url.substring(0, indexOfLastSlash);
                    matched = testUrlAgainstMenuJsons(url);
                    indexOfLastSlash = url.lastIndexOf("/");
                }
                else {
                    break;
                }
            }
        }
    }

    function testUrlAgainstMenuJsons(url) {
        if (url.substring(url.length - 1) == "/")
            url = url.substring(0, url.length - 2); // remove the last '/'

        // parse through the menu json array loaded into litMenuJsonArray
        for (i = 0; i &lt; menuJsonArray.length; i++) {
            var menuItem = menuJsonArray[i];
            var menuItemUrl = menuItem.Url;

            if (menuItemUrl.substring(menuItemUrl.length - 1) == "/")
                menuItemUrl = menuItemUrl.substring(0, menuItemUrl.length - 2); // remove the last '/'

            if (menuItemUrl.toUpperCase() == url.toUpperCase()) {
                defaultTabId = menuItem.TabLinkToActivate;  // set the defaultTabId to be this menu items tab to activate
                return true;
            }
        }
