	function  add2FavoritePost(id)
	{
			var cookieName = 'favorite';
			var cookieOptions = {expires: 7, path: '/'};
			
			//Load the items or a new array if null.
			var items = jQuery.cookie(cookieName) ? jQuery.cookie(cookieName).split(/,/) : new Array();
						
			indx = items.indexOf(id); 
			if(indx!=-1)
			{	items.splice(indx, 1); 
				jQuery("#item-favs span").removeClass("favored");					
				jQuery.cookie(cookieName, items.join(','), cookieOptions);
			}
			else // not exist
			{
				items.push(id);			
				jQuery("#item-favs span").addClass("favored");
				jQuery.cookie(cookieName, items.join(','), cookieOptions);
			}			
	}
