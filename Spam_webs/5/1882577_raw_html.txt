      function checkListAgainstHref(href, domainList)
{
	for(var i = 0; i &lt; domainList.length; i++)
	{
		//Use values from the array to match against the href value
		if(href.search(domainList[i]) &gt; -1)
		{
			//if the one found is the current domain, don't add
			var currentDomain = window.location.href;
			if(currentDomain.search(domainList[i]) &gt; -1)
			{
				return false;
			}
			else
			{
				return true;
			}
		}
	}
	return false;
};


function addGACodeToATags(domainList)
{
try{
	//Get all &lt;a&gt; tags on the page. If the destination domain is in the domain list, add the onClick event
	var aTags = document.getElementsByTagName('a');
	var thisHref = "";
	var currentOnClick = "";
	var aTagGACode = "newTracker._link(this.href); return false;";
	for(var i = 0; i &lt; aTags.length; i++)
	{
		thisHref = aTags[i].href;
		if(checkListAgainstHref(thisHref, domainList) == true)
		{
			//First, get the current onClick event
			currentOnClick = (aTags[i].getAttribute('onClick'));
			if(currentOnClick != null &amp;&amp; (currentOnClick.toLowerCase()).search("return") &gt; -1)
			{
				//currently existing return function, this is going to be weird. For now, just append to the end of the string
				currentOnClick += aTagGACode;
			}
			else
			{
				//no currently existing return
				currentOnClick = aTagGACode;
			}
			
			aTags[i].setAttribute('onClick', currentOnClick);
		}
	}
	}catch(ex)
	{
	//donothing
	};
};

	var formTagGACode = "javascript:newTracker._linkByPost(this);";
	for(var i = 0; i &lt; formTags.length; i++)

	}

function lt_getCurrentPage()
{
    var page = { type:'', subType:'', stage:''};
    var pageURL = window.location.href;

    ///////////////////////////////////////////
    //Item Category Page///////////////////////
    if($('#itemList').length &gt; 0 &amp;&amp; $('#multiAdd').length &gt; 0){
        page.type = lt_pageTypes.category;
        page.stage = lt_stages.other;
    }
    ///////////////////////////////////////////

    ///////////////////////////////////////////
    //Item Page (Item Drill Down)//////////////
    if(!page.type){
        // If this variable exists on the page, then we are on an item page. This variable is defined on the item drilldown template
        if(typeof lt_singleItemData !== 'undefined' &amp;&amp; lt_singleItemData){
            page.type = lt_pageTypes.product;
            page.stage = lt_stages.other;
        }
    }
    ///////////////////////////////////////////

    ///////////////////////////////////////////
    //Cart/////////////////////////////////////
    // If this attribute lives on the page with content in it
    if(!page.type &amp;&amp; $('#cart').length &gt; 0){
        page.type = lt_pageTypes.cart;
        page.stage = lt_stages.cart;
    }
    ///////////////////////////////////////////

    ///////////////////////////////////////////
    //Checkout Step////////////////////////////
    if(!page.type) {
        // If this variable exists on the page, then we are on a checkout page. This variable is defined on each individual checkout step
        if(typeof lt_checkoutstep !== 'undefined' &amp;&amp; lt_checkoutstep) {
            page.type = lt_pageTypes.checkout;
            if(lt_checkoutstep == lt_checkoutPageSteps.loginRegister){
                page.subType = lt_checkoutPageSteps.loginRegister;
                page.stage = lt_stages.loginRegister;
            }
            else if(lt_checkoutstep == lt_checkoutPageSteps.newCustomer){
                page.subType = lt_checkoutPageSteps.newCustomer;
                page.stage = lt_stages.loginRegister;
            }
            else if(lt_checkoutstep == lt_checkoutPageSteps.billing){
                page.subType = lt_checkoutPageSteps.billing;
                page.stage = lt_stages.billing;
            }
            else if(lt_checkoutstep == lt_checkoutPageSteps.shipping){
                page.subType = lt_checkoutPageSteps.shipping;
                page.stage = lt_stages.shipping;
            }
            else if(lt_checkoutstep == lt_checkoutPageSteps.payment){
                page.subType = lt_checkoutPageSteps.payment;
                page.stage = lt_stages.payment;
            }
            else if(lt_checkoutstep == lt_checkoutPageSteps.reviewSubmit){
                page.subType = lt_checkoutPageSteps.reviewSubmit;
                page.stage = lt_stages.reviewSubmit;
            }
        }
    }
    ///////////////////////////////////////////

    ///////////////////////////////////////////
    //Order Confirmation///////////////////////
    if(!page.type &amp;&amp; (pageURL.indexOf('thank-you') &gt; 0 || pageURL.indexOf('&amp;category=thanks') &gt; 0)) {
        page.type = lt_pageTypes.orderConfirmation;
        page.stage = lt_stages.orderConfirmation;
    }
    ///////////////////////////////////////////

    ///////////////////////////////////////////
    //Default Other////////////////////////////
    if(!page.type){
        page.type = lt_pageTypes.other;
        page.stage = lt_stages.other;
    }
    ///////////////////////////////////////////

    return page;
}

            item.itemRate = lt_getItemRate(page.type, itemForm.find('[ea-quantityprice*=]').attr('ea-quantityprice'), item.itemQty);
            item.itemImageURL = itemForm.find('.itemListImage a').attr('href');
            item.itemPage = lt_getPageURL();
            items.push(item);
        }
        //window.console &amp;&amp; console.log("Category Page Item Data: " + JSON.stringify(items));
    }
    else if(page.type == lt_pageTypes.cart){
        var tableRows = $('td [id*="carttablerow"]');
        for(var x = 0; x &lt; tableRows.length; x++)  {
            var rowObj = $(tableRows[x]);
            // Retrieve the row object ID attribute. Determine if this row contains a numeric value at the end of the
            // ID Value. If it does, this is an item row. If not, then it is another row, such as the subtotal or tax.
            // We only want to process rows that contain item data.
            var rowId = rowObj.attr('id');
            if(rowId &amp;&amp; !isNaN(rowId.replace('carttablerow', ''))){
                item = instItemObj();
                item.itemId = $(rowObj.children()[lt_cartTableIndex.itemId]).children()[0].text;
                item.itemName = $(rowObj.children()[lt_cartTableIndex.itemName]).html().split('&lt;')[0];
                item.itemQty = $($(rowObj.children()[lt_cartTableIndex.itemQty]).children()[0]).val();
                item.itemRate = $(rowObj.children()[lt_cartTableIndex.itemRate]).text().replace('$', '');
                item.itemImageURL = $($(rowObj.children()[lt_cartTableIndex.itemImageURL]).children()[0]).attr('src');
                item.itemPage = lt_getPageURL();
                items.push(item);
            }
        }
        //window.console &amp;&amp; console.log("Cart Page Item Data: " + JSON.stringify(items));
    }

    lt_req_addItemWithLinks(page.stage, items);
}

function lt_getItemRate(pageType, rateListStr, qty)
{
    if(!rateListStr &amp;&amp; !qty) {return 0;}

    var itemRate = 0.00;
    if(pageType == lt_pageTypes.category || pageType == lt_pageTypes.product)
    {
        if(pageType == lt_pageTypes.product){
            // Single Rate Price for Item
            // Some Product Items do not have a list rate, and only have a single item price.
            // If there is a dollar sign present on the value and no commas present, then identified as rate
            // already being provided. Do not need to parse data so return the value provided.
            if(rateListStr.indexOf('$') == 0 &amp;&amp; rateListStr.indexOf(',') &lt; 0){
                return rateListStr.replace('$', '');
            }

            // List Rate Price for Item
            var listRows = $(rateListStr).find('tr');
            var str = '';
            if(listRows){
                for(var t = 1; t &lt; listRows.length; t++){
                    if(t!=1){str+=';';}
                    var tempQty = parseInt($($(listRows[t]).children()[0]).text()); // First tr is qty
                    var tempPrice = $($(listRows[t]).children()[1]).text().replace('$',''); // Second tr is the price
                    str += tempQty +','+tempPrice;
                }
            }
            rateListStr = str;
        }
        //////////////////////////////////////////////////////////
        //STANDARDIZED LOGIC FOR PARSING LIST RATES STRING////////
        //EX:"0,12.95;5,11.25;10,10.95;"//////////////////////////
        var rates = [];
        var rateList = rateListStr.split(';');
        for(var i = 0; i &lt; rateList.length;i++) {
            var rateArr = rateList[i].split(',');
            rates.push( {qty: rateArr[0], rate:rateArr[1]} );
        }
        rates = rates.sort(function(a,b){return a.qty-b.qty}); //Sort by QTY
        for(var x = 0; x &lt; rates.length; x++) {
            if(typeof rates[x+1] !== 'undefined'){
                if(parseInt(qty) &gt;= parseInt(rates[x].qty) &amp;&amp; parseInt(qty) &lt; parseInt(rates[x+1].qty)) {
                    itemRate = rates[x].rate;
                    break;
                }
            }
            else{
                if(parseInt(qty) &gt;= parseInt(rates[x].qty)){
                    itemRate = rates[x].rate;
                    break;
                }
            }
        }
        //////////////////////////////////////////////////////////
    }
    return itemRate;
}

function lt_cleanItemStr(arr){
    var itemSplitIndex = '5';
    var returnArr = [];
    try {
        if(arr){
            var tempCount = 0;
            for(var o = 0; o &lt; arr.length; o++){
                if(tempCount != itemSplitIndex){
                    returnArr.push(arr[o]);
                    tempCount++;
                }
                else{
                    tempCount = 0;
                }
            }
        }
    }
    catch(e){}
    return returnArr;
}
function lt_getOrderConfirmItemData(itemDataString)
{
    // Example Item Str: id|name|options|rate|qty||
    // NetSuite provides this data as one string for all items. The identifier to distinguish
    // between items is the double pipeline without a space in-between
    var itemCollection = [];
    try
    {
        var itemScramble = itemDataString.split('|');
        var itemsArr = lt_cleanItemStr(itemScramble);
        var numOfIterations = parseInt(itemsArr.length) / 5;
        if(itemsArr){
            var countr = 0;
            for(var y = 0; y &lt; numOfIterations; y++){
                var tempObj = {};
                tempObj.id = itemsArr[countr];
                tempObj.name = itemsArr[countr+1];
                tempObj.options = itemsArr[countr+2];
                tempObj.rate = itemsArr[countr+3];
                tempObj.qty = parseInt(itemsArr[countr+4]);
                itemCollection.push(tempObj);
                countr+=5;
            }
        }
    }
    catch(e){}
    return itemCollection;
}
