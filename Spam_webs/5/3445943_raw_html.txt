    //convert lat long to zip
    function getZip(lat, long){
        $.ajax({
            url: 'http://maps.googleapis.com/maps/api/geocode/json?latlng=' + lat + ',' + long + '&amp;sensor=true',
            success: function(data) {
                //find the postal from the result
                for (var h=0; h&lt;data.results.length; h++){
                    for (var i=0; i&lt;data.results[h].address_components.length; i++){
                        for (var j=0; j&lt;data.results[h].address_components[i].types.length; j++){
                            if (data.results[h].address_components[i].types[j] == "postal_code"){
                                sho.zip = data.results[h].address_components[i].long_name;
                            }
                        }
                    }
                }
                
                $("#openLocalZipCode").html(sho.zip);
                $.cookie("customer_zip", sho.zip, { expires:  expireDate, path: '/', domain: 'searsoutlet.com' });
                if( storeSet !== true &amp;&amp; sho.myStore === undefined ){   //after zip is set but no store  
                    getClosestStore();
                } 
            },
            error: function() {
                console.log('Error Getting Zip: There was a problem detecting your location');
            }
        }).done(function (data) {});    
    }

    function rewriteUrl(pName, pValue, pathname, addToBrowserHistory) {
    
        pValue = pValue == null ? "" : pValue;
        var loc = window.location;
        var query = new QueryString();
        var newPathname = loc.pathname;
        if(pathname != null || pathname != 'null'){
            newPathname = pathname;
        }
       
        //&lt;protocol&gt;//&lt;host&gt;[:&lt;port&gt;]/&lt;pathname&gt;[&lt;hash&gt;][&lt;search&gt;]
        var newUrl =loc.protocol + "//" + loc.host + newPathname ;
        if(newUrl.indexOf("?") == -1) {
            newUrl =  newUrl + "?" ;
        }else{
            newUrl =  newUrl + "&amp;" ;
        }
        
        var params = query.getParameterNames();
        for (var i=0; i &lt; params.length; i++) {
            
            if (pName == "ps") {
                if (params[i] != pName &amp;&amp; params[i] != 'pn') {
                        var vals = query.get(params[i]);
                        for (var v=0; v &lt; vals.length; v++)
                            if(params[i]!="disp"){
                            newUrl += params[i] + "=" + vals[v] + "&amp;";
                            }
                }        
            } else {        
            if (params[i] != pName) {
                var vals = query.get(params[i]);
                for (var v=0; v &lt; vals.length; v++)
                    if(params[i]!="disp"){
                    newUrl += params[i] + "=" + vals[v] + "&amp;";
                    }
            }
        }
        }
        if (pValue != null &amp;&amp; pValue != "null") {
            newUrl += pName+"="+pValue;
        }
    
        if (addToBrowserHistory)
            window.location = newUrl;
        else
            window.location.replace(newUrl);
    }

/* Post Request */
function postRequestProdDetails(zipcode,toPageUrl) {
    var cordinateURL="/d/cordinates_finder.jsp?zip="+zipcode;
    var xmlHttp;
    
    if (window.XMLHttpRequest) { // Mozilla, Safari, ...
         var xmlHttp = new XMLHttpRequest();
    }else if (window.ActiveXObject) { // IE
         var xmlHttp = new ActiveXObject("Microsoft.XMLHTTP");
    }
    
    xmlHttp.open('POST', cordinateURL, true); 
    
    xmlHttp.onreadystatechange = function() {
    if (xmlHttp.readyState == 4) {
        var response = xmlHttp.responseText; 
       if (xmlHttp.status == 200) {
            if(response==null||response=="null" ||response==""){
                searchLatLng2(zipcode,null,null,null,toPageUrl);                                   

            } else {
             $.cookie("chkavl_zip", zipcode);
             $.cookie("customer_zip", zipcode, { expires:  expireDate, path: '/', domain: 'searsoutlet.com' });         
             toPageUrl= toPageUrl+"&amp;zip="+zipcode;
             $("#enter_zip").hide();
             window.location=toPageUrl;
            }
       }
    }
}
    xmlHttp.send(cordinateURL);
}
