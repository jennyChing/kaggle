    function setCallingPage() {
        var currentpage_url = document.location.href;

        isCalendarPage = false;
		isFollowLotPage = false;
        isWatchPage = false;
        isSaleBrowsePage = false;
        isSearchResultPage = false;
		isSearchResultNewPage = false; // Added  Case For SearchResultRedesign
        isSaleLandingUpcomingPage = false;
        isSaleLandingPastPage = false;
        isLotDetailsPage = false;
        isDepartmentsPage = false;
        isArtistPage = false;
        isArtistBioPage = false;

        isCalendarPage = (currentpage_url.toLowerCase().indexOf('/calendar/') != -1) ? true : false;
        isWatchPage = (currentpage_url.toLowerCase().indexOf('/watch') != -1) ? true : false;
        isSaleBrowsePage = (currentpage_url.toLowerCase().indexOf('/salebrowse') != -1) ? true : false;
        isSearchResultPage = (currentpage_url.toLowerCase().indexOf('/searchresult') != -1) ? true : false;
		isSearchResultNewPage = (currentpage_url.toLowerCase().indexOf('/searchresultsnew') != -1) ? true : false;  // Added  Case For SearchResultRedesign
		
        isSaleLandingPastPage = (document.getElementById("calendar-sale-salelanding") != null) ? true : false;
        isSaleLandingUpcomingPage = (document.getElementById("calendar-registerbid-upcomingsalelanding") != null) ? true : false;
        isLotDetailsPage = (document.getElementById("Lotdetails_placebid") != null) ? true : false;
        isDepartmentsPage = (currentpage_url.toLowerCase().indexOf('/departments/') != -1) ? true : false;
        isArtistPage = (currentpage_url.toLowerCase().indexOf('/artist') != -1) ? true : false;
        isArtistBioPage = (currentpage_url.toLowerCase().indexOf('/artistvio') != -1) ? true : false;
		isFollowLotPage = (currentpage_url.toLowerCase().indexOf('/my_lots.aspx') != -1) ? true : false;
		// MyChristies P3.0
		//isFollowLotPage = (currentpage_url.toLowerCase().indexOf('/upcomingactivity.aspx') != -1) ? true : false;
    }

    function getPageNameForOpenFrame() {
        setCallingPage();

        if (isCalendarPage == true) {
            return "calendar";
        }
        else if (isWatchPage == true) {
            return "";
        }
        else if (isSaleBrowsePage == true) {
            return "salebrowseplacebid";
        }
        else if (isSearchResultPage == true) {
			if (isSearchResultNewPage == true) // Added  Case For SearchResultRedesign
			{
			    return "searchresultnew"; // Added  Case For SearchResultRedesign
			}
			else{
            return "searchresult";
			}
        }
        else if (isSaleLandingUpcomingPage == true) {
            return "salelanding";
        }
        else if (isSaleLandingPastPage == true) {
            return "salelanding";
        }
        else if (isLotDetailsPage == true) {
            return "lotdetail";
        }
        else if (isDepartmentsPage == true) {
            return "department";
        }
        else if (isArtistPage == true) {
            return "artistpage";
        }
        else if (isArtistBioPage == true) {
            return "artistpage";
        }
        else {
            return "";
        }

    }
