function tab_onClick(which,opt) {
    var oldTab = $(".withTab" + opt, $(which).parent().parent())
    if (opt=='2') {
		if (which.id == 'clkRelated') {
	       $('#clkRecent').removeClass('selected');
        } else {
	       $('#clkRelated').removeClass('selected');
        }
    } else {
        $(oldTab).removeClass('selected');
    }
    $("ul", oldTab).hide();

    var newTab = $(which).parent();
    if (opt=='2') {
        $(which).addClass('selected');
    } else {
        $(newTab).addClass('selected');
    }
    $("ul", newTab).show();

    which.blur();                 // most browsers 
    which.hideFocus = true;       // internet explorer
    which.style.outline = 'none'; // mozilla
}
