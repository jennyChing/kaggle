			//Verticaly center playpause button
			var playpauseH = $codeart("#playpause"+id).height();

			var fgutterH = (playerH-gutterH)/2;

			//Verticaly adjust timer span to be above the seekbar
			var timerH = 27;
			var ftimerH = ((playerH-timerH)/2)-(gutterH);
			$codeart("#timer"+id).css("top",ftimerH);
			//Verticaly center volumebar

			$codeart("#volume"+id).css("top",fvolumeH);
			//Verticaly center mute button
			var muteH = $codeart("#muteButton"+id).height()+2;

     <h1>
      ACA Exchange Subsidies Headed to U.S. Supreme Court
     </h1>
     <div class="post-meta">
      <div class="posted-on">
       <a class="url fn" href="http://www.americanlegislator.org/author/jhauenschildalec-org/" rel="author" title="Posts by Jonathon Hauenschild">
        Jonathon Hauenschild
       </a>
       and
       <a class="url fn" href="http://www.americanlegislator.org/author/sriley/" rel="author" title="Posts by Sean Riley">
        Sean Riley
       </a>
       | November 7, 2014 |
       <a href="http://www.americanlegislator.org/aca-exchange-subsidies-headed-u-s-supreme-court/#respond">
        Add a comment
       </a>
      </div>
     </div>
     <p>
      The U.S. Supreme Court has agreed to hear arguments against the availability of exchange subsidies under the Affordable Care Act. Todays
      <a href="http://sblog.s3.amazonaws.com/wp-content/uploads/2014/11/orders-11-7-141.docx">
       order
      </a>
      from the Court granted review in
      <em>
       King v. Burwell
      </em>
      , one of
      <a href="http://www.americanlegislator.org/exchange-lawsuits-move-forward/">
       several cases
      </a>
      challenging IRS regulations extending tax-credit subsidies to the states with federal exchanges. Briefings will be submitted over the next few months, and a ruling is not expected until June.
     </p>
     <p>
      <img alt="health-care" class="aligncenter wp-image-2916" height="280" src="http://www.americanlegislator.org/wp-content/uploads/health-care.png" width="400"/>
     </p>
     <p>
      As
      <a href="http://www.scotusblog.com/2014/11/symposium-the-court-will-hear-king-thats-bad-news-for-the-aca/">
       commentators
      </a>
      quickly pointed out, the timing of the decision, coupled with the lack of a circuit split, does not bode well for the ACA. Granting review requires a vote from four of the nine justices, signaling the Court may be in the brink of striking down the subsidies. In all likelihood, that means that either Chief Justice Roberts or Justice Kennedy will again hold the key vote, according to Nicholas Bagley, Assistant Professor at Michigan Law.
     </p>
     <p>
      The three other cases challenging the IRS rule
      <em>
       Halbig v. Burwell
      </em>
      ,
      <em>
       Pruitt v. Burwell
      </em>
      , and
      <em>
       Indiana v. IRS
      </em>
      are likely to be placed on hold, or
      <a href="http://www.scotusblog.com/2014/11/court-to-rule-on-health-care-subsidies/">
       abeyance
      </a>
      , in anticipation of the Supreme Courts ruling, though
      <em>
       Indiana v. IRS
      </em>
      poses a slightly different legal question.

      <br/>
      Appellants Reply to Opposition  February 23, 2015
      <br/>
      Oral Arguments  TBD, likely March or later
     </p>

   $.fn.placeholder = function(value){
	// monitoring an input field for focus and blur
	// when you click on it, if it has the placeholder value, it will empty
	// when you click away, if it is empty, it will add the placeholder value
	var field = $(this);
	// return jQuery object to allow for chaining
	return field.on("focusin focusout", function(e){
		var $this = $(this);
		if(e.type == "focusin" &amp;&amp; this.value == value) {
				$this.val("");
		} 
		else if(e.type == "focusout" &amp;&amp; this.value == "") {
				$this.val(value);
		} 
	});
};
