        <p>
         MOSCOWFor about two hours on Thursday morning, the
         <a href="http://online.wsj.com/public/quotes/main.html?type=djn&amp;symbol=TWTR">
          Twitter
         </a>
         <span data-ticker-name="TWTR" data-widget="dj.ticker">
         </span>
         feed of tech-loving Russian Prime Minister Dmitry Medvedev blurted out a string of surprise announcements: He was quitting his post to become a freelance photographer and he didnt agree with his boss
         <a href="http://topics.wsj.com/person/P/vladimir,-putin/6409">
          Vladimir Putin
         </a>
         s policies.
        </p>
        <p>
         The flurry of tweets was quickly dismissed by a government spokesperson as the work of hackers and deleted, but not before amusing Twitter users with an ironic commentary on recent government decisions, including the annexation of Ukraines Crimea and a ban on some food products from Western countries that have sanctioned Russia.
        </p>
        <p>
         Im resigning. Im ashamed of the governments actions. Forgive me, a tweet via Mr.Medvedevs official Russian-language Twitteraccount @MedvedevRussia said.
        </p>
        <p>
         Ive wanted to say this for a long time: Vova youre wrong! another tweet said, referring to the president by thediminutiveform of his first name. Mr. Medvedev usually closely follows Mr. Putins line on almost all policy matters.
        </p>
        <p>
         Mr. Medvedev, Russias no. 1 technology and social media aficionado, has a following of 2.5 million people on his Russian account, which was created for him during hisvisit to Twitter headquarters in Silicon Valley in 2011.
        </p>
        <p>
         Mr. Medvedev did not publicly address the breach Thursday, and Twitter declined to comment on the situation citing its policy not to comment on individual accounts. A government official said told Russian news agencies that the prime ministers account had been hacked and that the information in the tweets was incorrect. No one claimed responsibility for the apparent hack.
        </p>
        <p>
         Crimea isnt ours. Please retweet, another message on the feed said ahead of the Thursday meeting of the Russian president with Russian legislators in Crimeato discuss the problem of the region.
        </p>
        <p>
         We could return to the 1980s. That is sad. If that is the objective of my colleagues in the Kremlin, theyll achieve that quickly, said another tweet criticizing the food ban.

     function attachListeners(fn) {
        
        if (window.addEventListener) {
        
        //window.addEventListener('DOMContentLoaded', fn, false);
        //window.addEventListener('load', fn, false);
        window.addEventListener('scroll', fn, false);
        window.addEventListener('resize', fn, false);

        window.attachEvent('DOMContentLoaded', fn);
        window.attachEvent('load', fn);
        window.attachEvent('onscroll', fn);
        window.attachEvent('onresize', fn);

 }

     function removeListeners(fn) {

       // window.removeEventListener('DOMContentLoaded', fn, false);

        window.detachEvent('load', fn);

// performs callback when element becomes visible, then removes listeners
function whenElementVisibleDo(el, cb) {
  var handler = function() {
    if (isElementInViewportAtAll(el)) {
      cb();
      removeListeners(handler);
    }
  }
  attachListeners(handler);
}

// similar to http://stackoverflow.com/a/7557433/458614 but returns true if any part of the element is in the viewport, not just the whole thing
// should work on: Chrome, Firefox, Safari, IE8+, iOS Safari, Android2+, Blackberry, Opera Mobile, IE Mobile
// doesn't handle z-index, being outside of viewable area of parent element, and possibly other edge cases
// we check if either the top OR the bottom of the element is within the viewport, while either the left right or the element is also in the viewport
function isElementInViewportAtAll (el) {
    var rect = el.getBoundingClientRect();
    var height = window.innerHeight || document.documentElement.clientHeight;
    var width = window.innerWidth || document.documentElement.clientWidth;
    return (
      (
        (rect.top &lt;= height &amp;&amp; rect.top &gt;= 0) ||
        (rect.bottom &lt;= height &amp;&amp; rect.bottom &gt;= 0)
      ) &amp;&amp;
      (
        (rect.left &lt;= width &amp;&amp; rect.left &gt;= 0) ||
        (rect.right &lt;= width &amp;&amp; rect.right &gt;= 0)
      ) 
    );
}

function attachListeners(fn) {
  if (window.addEventListener) {
    window.addEventListener('DOMContentLoaded', fn, false);
    window.addEventListener('load', fn, false);
    window.addEventListener('scroll', fn, false);
    window.addEventListener('resize', fn, false);
  } else if (window.attachEvent) { // IE fallback
    window.attachEvent('DOMContentLoaded', fn);
    window.attachEvent('load', fn);
    window.attachEvent('onscroll', fn);
    window.attachEvent('onresize', fn);
  }
}
function removeListeners(fn) {
  if (window.removeEventListener) {
    window.removeEventListener('DOMContentLoaded', fn, false);
    window.removeEventListener('load', fn, false);
    window.removeEventListener('scroll', fn, false);
    window.removeEventListener('resize', fn, false);
  } else if (window.attachEvent) { // IE fallback
    window.detachEvent('DOMContentLoaded', fn);
    window.detachEvent('load', fn);
    window.detachEvent('onscroll', fn);
    window.detachEvent('onresize', fn);
  }
}

// adds a class when top of element is off screen, removes class again when scrolling back up to that point
function addClassWhenOffTop(el, classname) {
  var scrollTopWhenAdded;
  attachListeners(function() {
    if (scrollTopWhenAdded === undefined) {
      // class has not been added , only add for ads with size less than 400
      if ( (el.getBoundingClientRect().top) &lt; 0 &amp;&amp; ($(el).children()[0].height &lt; 400 ) ) {
        // it's off the top of the screen
        el.className += ' ' + classname;
        scrollTopWhenAdded = document.documentElement.scrollTop || document.body.scrollTop;
      }
    }
    else {
      if ((document.documentElement.scrollTop || document.body.scrollTop) &lt;= scrollTopWhenAdded) {
        // we've gone back up past it
        el.className = el.className.replace(new RegExp('\\b'+classname+'\\b'), ''); // remove class. in PHP double-quoted string have to do 4 backslashes to get 2
        scrollTopWhenAdded = undefined;
      }
    }
  });
}
