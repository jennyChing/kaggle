          (function controller() {
  

    register_controller("RhsListCtrl", function (scope) {
        $('.fkhp-list-module',scope).each(function(){
            var expand=false;
            var $parent = $(this);
            var $list = $parent.find('.lm-list-wrap');
            var totalHeight = $list.css({height:'auto'}).height();
            var $liWrap = $list.find('.lm-li-wrap');
            var initialHeight = 45*Math.min(4,$liWrap.length);// 45 is the height of each li
            $list.css({height:initialHeight});
            $parent.find('.viewAll').on('click', function(){
                if(!expand){
                    $(this).removeClass('collapse').addClass('expand');
                    $list.animate({height:totalHeight},500);
                }
                else{
                    $(this).removeClass('expand').addClass('collapse');
                    $list.animate({height:initialHeight},500);
                }
                expand = !expand;
            });
        });
    });
