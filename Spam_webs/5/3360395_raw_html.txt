				var minimum = document.getElementById("bt_minimum_"+product_id).value;
				
				if(qty &lt; minimum){
					document.getElementById("bt_"+product_id).value = minimum;
				}
				else{
					document.getElementById("bt_"+product_id).value = qty;
				}
			}			
			function addProductToCart(product_id) {
				
				$.ajax({
					url: 'index.php?route=checkout/cart/add',
					type: 'post',
					data: 'product_id=' + product_id + '&amp;quantity=' + get_qty(product_id),
					dataType: 'json',
					success: function(json) {
						$('.success, .warning, .attention, information, .error').remove();
						
						if (json['error']) {
							if (json['error']['option']) {
								for (i in json['error']['option']) {
									$('#option-' + i).after('&lt;span class="error"&gt;' + json['error']['option'][i] + '&lt;/span&gt;');
								}
							}
						} else if (json['success']) {
														$('#cart_total').html(json['total']);
							$('.cart-module').html(json['output']);
							$.ajax({
								url: "index.php?route=product/ajax_notification", 
								type: "POST", 
								//data: "action=notif&amp;product_id="+2001, 
								data: "action=notif&amp;product_id="+product_id, 
								dataType: "xml", 
								success: notifShop,

							});
						}	
					}
