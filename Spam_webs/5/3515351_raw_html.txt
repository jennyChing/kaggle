function selectTSOAds(ads,max) {
	if (ads.length &lt;= max) {
		return ads;
	} else {
		var uniq = {};
		var found = 0;
		var selectedAds = [];
		while (found &lt; max) {
			var ad = ads[Math.floor(Math.random()*ads.length)];
			if (uniq[ad.id] != null) {
				continue;
			} else {
				uniq[ad.id] = true;
				found++;
				selectedAds.push(ad);
			}
		}
		return selectedAds;
	}
	return null;
