    PurchasingSearch.updateSort =function (sortBy) {
        GoogleAnalytics.trackEvent(PurchasingSearch.product_search_parameter_hash.selector_route, 'OrderResults', sortBy);
        PurchasingSearch.navReset = true;
        switch(sortBy){
            case 'best match':
                PSearch.sortBy(undefined);
                break;
            case 'avg_rating':
                PSearch.sortBy([sortBy,'-times_rated','-premium_level']);
                break;
            case '-avg_rating':
                PSearch.sortBy([sortBy,'-times_rated','-premium_level']);
                break;
            case '-times_rated':
                PSearch.sortBy([sortBy,'-premium_level']);
                break;
            case '-premium_level':
                PSearch.sortBy([sortBy,'-times_rated', '-avg_rating']);
                break;
            default:
                PSearch.sortBy(sortBy);
                break;
        }

        PurchasingSearch.updateSortLabel(sortBy);
        var sortMode = PurchasingSearch.getSortMode(sortBy);
        PurchasingSearch.queue_log_search(PurchasingSearch.ActionValues.UPDATED_ORDER,PSearch.state.filters,jQuery('#product_search_ipt').val(),sortMode);

        return false;

    };

PurchasingSearch.runUrlQuery=function () {
        var hidePending = true;
        if (PurchasingSearch.product_search_parameter_hash.sort &amp;&amp; PurchasingSearch.product_search_parameter_hash.sort.length &gt; 0) {
            // jQuery('.order-by-dropdown .sui-dropdown-toggle').val(PurchasingSearch.product_search_parameter_hash.sort);
            PurchasingSearch.updateSortLabel(PurchasingSearch.product_search_parameter_hash.sort[0]);
            hidePending = false;
        } else {
            PurchasingSearch.updateSortLabel(PurchasingSearch.product_search_parameter_hash.default_sort);
        }
        if (PurchasingSearch.product_search_parameter_hash.selector_query !== null &amp;&amp; PurchasingSearch.product_search_parameter_hash.selector_query != ''  ) {
            hidePending = false;
            PurchasingSearch.preloadNav = true;
            PurchasingSearch.refreshNav = true;

            PSearch.autoUpdate(false);

            var sortOption = jQuery('.order-by-dropdown .sui-dropdown-toggle').attr('data-sort-term');
            var sortBy = sortOption;
            switch(sortBy){
                case 'best match':
                    PSearch.sortBy(undefined);
                    break;
                case 'avg_rating':
                    PSearch.sortBy([sortBy,'-times_rated','-premium_level']);
                    break;
                case '-avg_rating':
                    PSearch.sortBy([sortBy,'-times_rated','-premium_level']);
                    break;

                    break;

                    PSearch.sortBy([sortBy,'-times_rated', '-avg_rating']);
                    break;
                default:
                    PSearch.sortBy(sortBy);
                    break;
            }

            PSearch.autoUpdate(true);

            var fields = {};

            if(PurchasingSearch.product_search_parameter_hash.selector_query==PurchasingSearch.product_search_parameter_hash.default_query){
                fields.keywords = PurchasingSearch.product_search_parameter_hash.default_query;
            }
            else{
                jQuery('#product_search_ipt').val(PurchasingSearch.product_search_parameter_hash.selector_query);
                fields.keywords = jQuery('#product_search_ipt').val().toLowerCase();
            }

            PSearch.options.callbackFunction = "PurchasingSearch.applyUrlFilters";
            PurchasingSearch.refreshNav = true;

            if(PurchasingSearch.product_search_parameter_hash.restrictions !== '' ){
                var restrictions = jQuery.parseJSON(''+PurchasingSearch.product_search_parameter_hash.restrictions);
                fields =Object.extend(restrictions,fields);
            }

            PSearch.search(fields);
        }
        if(hidePending){
            jQuery('.product_search_pending_search_state').css('display','none');
            jQuery('.product_search_start_state').css('display','block');

        PurchasingSearch.updateNavQuery();
    };
