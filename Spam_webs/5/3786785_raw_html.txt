    function populateVoices(languageTypeId) {
        var html = '';
        var gender = '';
        var first = true;

        var voiceCount = 0;

        for (var i = 0; i &lt; voices.length; i++) {
            if (voices[i].languageTypeId === languageTypeId) {
                voiceCount++;
                gender = voices[i].gender === 'M' ? 'male' : 'female';
                if (first) {
                    html += '&lt;li vid="' + voices[i].voiceId + '" demo="' + voices[i].ttsDemoText + '" class="selected"&gt;&lt;img src="/images/' + gender + '.png" /&gt;' + voices[i].name + ' (' + gender + ')&lt;/li&gt;';
                    first = false;
                } else {
                    html += '&lt;li vid="' + voices[i].voiceId + '" demo="' + voices[i].ttsDemoText + '" &gt;&lt;img src="/images/' + gender + '.png" /&gt;' + voices[i].name + ' (' + gender + ')&lt;/li&gt;';
                }
            }
        }
        $('#selected-voice').html('');
        $('.voices').html(html);
        $('.voices li.selected').clone().appendTo('#selected-voice');
        $('#tts-text').val($('.voices li.selected').attr('demo'));
        $('.voices').hide();
        if (voiceCount &lt; 2) {
            $('#voice-down').hide();
        } else {
            $('#voice-down').show();
        }
    }
