    function codeAddress() {
        var address = document.getElementById("txtSearch").value;
        geocoder.geocode({ 'address': address },
        function (results, status) {
            if (status == google.maps.GeocoderStatus.OK) {
                if (results.length &gt; 0) {
                    //Clear out the hidden pagination
                    $("#hdnPaginationIndex").val("0");

                    //If there's more than 10 rows, display the pagination buttons
                    $("#btnPreviousPage").attr("enabled", "disabled");
                    $("#btnNextPage").attr("enabled", "disabled");
                    if (results.length &gt; DEFAULT_RESULT_PER_PAGE)
                        $("#btnNextPage").attr("enabled", "enabled");

                    //Set the map to visible 
                    document.getElementById("map_canvas").style.display = "block";
                    var dLatitude = results[0].geometry.location.lat();
                    var dLongitude = results[0].geometry.location.lng();

                    //Call to get the locations by latitude/longitude
                    //alert("Lat: " + dLatitude + "/nLong:" + dLongitude);
                    GetLocationsByLatLong(dLatitude, dLongitude);
                    var myOptions = {
                        zoom: DEFAULT_ZOOM,
                        center: results[0].geometry.location,
                        mapTypeId: google.maps.MapTypeId.ROADMAP
                    }
                    map = new google.maps.Map(document.getElementById("map_canvas"), myOptions);
                    google.maps.event.addListener(map, 'dragend', function (event) { OnMapDragEnd(map.getCenter()); }); //OnMapDragEnd(map.getCenter());
                    google.maps.event.addListener(map, 'zoom_changed', function () { OnMapZoomChanged(); });
                }
            }
            else {
                alert("Geocode was not successful for the following reason: " + status);
            }
        });
    }

    //Gets the locations for the map
    function GetLocationsByLatLong(sLatitude, sLongitude) 
    {
        //alert("Number of meters:" + _nNumMeters);
        //alert("Latitude:" + sLatitude + "/nLongitude:" + sLongitude);
        $.post("/api/location/get", { latitude: sLatitude, longitude: sLongitude, postalCode: "", meters: _nNumMeters },
           function (data) {
               DisplayRetailers(data);
           }, "json");

    }

   //Displays the retailers on the list and the map
    function DisplayRetailers(data) 
    {
        //For each item in the result set, get the address, latitude and longitude
        $("#divLocations").empty();
        clearMarkers();

        _arrLocations = new Array();
        _arrMarkerContent = new Array();
        _arrMarkers = new Array();

        //alert("display retailers");
        if (data.length &gt; 0) 
        {
            //Add a table object to the divLocations tag
            var objTable = $(document.createElement("table"));
            objTable.attr("id", "tbLocations");
            for (var nCounter = 0; nCounter &lt; data.length; nCounter++) 
            {
                _arrLocations[nCounter] = data;
                var dLatitude = data[nCounter].Latitude;
                var dLongitude = data[nCounter].Longitude;
                var bPreferred = data[nCounter].Preferred == true;

                var sContentTable = "&lt;div onmouseover='BounceLocation("+ nCounter + ");' onmouseout='ResetMarkerAnimations();' style='cursor:pointer;'&gt;&lt;strong&gt;" + data[nCounter].Name + "&lt;/strong&gt;&lt;br/&gt;";
				if(!isBlank(data[nCounter].Address1))
                    sContentTable += ReplaceNull(data[nCounter].Address1) + ", " + ReplaceNull(data[nCounter].City) + ", " + ReplaceNull(data[nCounter].State);
                if(!isBlank(data[nCounter].Phone))
                    sContentTable += "&lt;br/&gt;" + "Telephone: " + ReplaceNull(data[nCounter].Phone);
				if(!isBlank(data[nCounter].Url))
                    sContentTable += "&lt;br/&gt;&lt;a href='http://" + data[nCounter].Url + "' target='_blank'&gt;" + data[nCounter].Url + "&lt;/a&gt;";
				if(!isBlank(data[nCounter].FBUrl))
                    sContentTable += "&nbsp;&nbsp;&lt;a target=_blank href='http://" + data[nCounter].FBUrl + "'&gt;&lt;img border=0 src='/assets/buttons/fb-retailers.png'/&gt;&lt;/a&gt;";
				if(!isBlank(data[nCounter].ShortDescription))
                    sContentTable += "&lt;br/&gt;Note: " + data[nCounter].ShortDescription;
                sContentTable += "&lt;/div&gt;";
                //add the data as a new row to the table
                objTable.append($("&lt;tr style='display:none;' class='locationRow'&gt;&lt;td&gt;" + sContentTable + "&lt;/td&gt;&lt;/tr&gt;"));

                //Add the marker content to the array.
                _arrMarkerContent[nCounter] = sContentTable;

                //alert("Marker content " + nCounter); 

                //Plot each item on the map as a marker
                var locPosition = new google.maps.LatLng(dLatitude, dLongitude);

                //Add an info window
                var infowindow = new google.maps.InfoWindow({ content: sContentTable });
                var marker = new google.maps.Marker({ position: locPosition, title: data[nCounter].Name });

                //If the item is preferred, mark it as blue. Otherwise, mark it as orange
                if (bPreferred) {
                    marker = new google.maps.Marker({ position: locPosition, title: data[nCounter].Name,
                        icon: "http://www.google.com/intl/en_us/mapfiles/ms/micons/blue-dot.png"});
                }

                google.maps.event.addListener(marker, 'click',
                    (function (marker, i) {
                        return function ()
                        { infowindow.setContent(_arrMarkerContent[i]); infowindow.open(map, marker); }
                    })
                    (marker, nCounter));

                //google.maps.event.addListener(marker, 'click', function () { infowindow.open(map, marker); });
                // To add the marker to the map, call setMap();   
                marker.setMap(map);
                _arrMarkers[nCounter] = marker;

                //Set the result count
                $("#divResultCount").text("We found " + _arrMarkerContent.length + " retailer(s) close to this location.");
                $("#divResultCount").css("display", "block");
                $("#divResultCount").attr("class", "orange16");
            }

            //Add the full table to the divlocations tag
            $("#divLocations").append(objTable);

            SetVisibleRows();
        }
        else {
            $("#tdPageOfResult").css("display", "none");
            $("#tdPageOfResult").text("");
             $("#divResultCount").text("");
             $("#divResultCount").css("display", "none");
             $("#btnPreviousPage").css("display", "none");
             $("#btnNextPage").css("display", "none");
            $("#divLocations").append($("&lt;div style='margin-top:20px;'&gt;&lt;strong&gt;Oops, we couldn't find a retailer in your area.&lt;/strong&gt;&lt;br&gt;If you'd like to shop online, &lt;a href='/online-retailers'&gt;click here&lt;/a&gt;. Or, we welcome you to &lt;a href='/contact-us'&gt;contact us&lt;/a&gt; for further assistance&lt;/div&gt;"));
        }
    }

    //Changes the visible rows of retailers for pagination effect.
    function SetVisibleRows() {
        var nIndex = parseInt($("#hdnPaginationIndex").val());
        //alert("nIndex:" + nIndex);
        var nStartRowIndex = nIndex;
        if (nStartRowIndex &gt; 0)
            nStartRowIndex = nStartRowIndex * DEFAULT_RESULT_PER_PAGE;
        var nEndRowIndex = nStartRowIndex + DEFAULT_RESULT_PER_PAGE;
        //reset all the rows for tbLocations to hidden.
        //alert("Start row index:" + nStartRowIndex + "/nEnd row index:" + nEndRowIndex);
        $('#tbLocations').find('tr.locationRow').each(function (nIndex) {
            var sDisplay = "none";
            if ((nIndex &gt;= nStartRowIndex) &amp;&amp; (nIndex &lt; nEndRowIndex)) 
            {
                //alert("adding item");
                sDisplay = "block";
            }

            //If the row index is between the first and last number to be displayed, show it
            $(this).css("display", sDisplay);

        })

        $("#btnPreviousPage").css("display", "none");
        $("#btnNextPage").css("display", "none");
        $("#tdPageOfResult").css("display", "none");

        //If the end row index is less than the number of rows, show the next button.
        if (nEndRowIndex &lt; $('#tbLocations').find('tr.locationRow').length)
            $("#btnNextPage").css("display", "block");
        //If the start row index is greater than 0, show the previous button.
        if (nStartRowIndex &gt;= DEFAULT_RESULT_PER_PAGE)
            $("#btnPreviousPage").css("display", "block");

        //Show the "Page X of Y" result
        var nPageCount = parseInt($('#tbLocations').find('tr.locationRow').length / DEFAULT_RESULT_PER_PAGE);
        if(($('#tbLocations').find('tr.locationRow').length % DEFAULT_RESULT_PER_PAGE) &gt; 0)
            nPageCount++;
        $("#tdPageOfResult").text("Page " + (nIndex + 1) + " of " + nPageCount);
        $("#tdPageOfResult").css("display", "block");
    }

    //Next button on click
    function OnNextPageClick() {
        //alert("Next page click");   //debug
        //Get the value of the hidden pagination field
        var nPageIndex = parseInt($("#hdnPaginationIndex").val());
        nPageIndex++;

        //increment the value.
        $("#hdnPaginationIndex").val(nPageIndex);
        SetVisibleRows();

        //Enabled the previous button
        $("#btnPreviousPage").attr("enabled", "enabled");

        //If there is a next group of rows after the current ones, enable the button.
        if ((nPageIndex + 1) * DEFAULT_RESULT_PER_PAGE &gt; $('#tbLocations').find('tr').length)
            $("#btnNextPage").attr("enabled", "enabled");
        else
            $("#btnNextPage").attr("enabled", "disabled");
        //If not, disable it.
    }

    //Next button on click
    function OnPreviousPageClick() {
        //alert("Previous page click"); //debug
        //Get the value of the hidden pagination field
        var nPageIndex = parseInt($("#hdnPaginationIndex").val());
        nPageIndex--;

        //increment the value.
        $("#hdnPaginationIndex").val(nPageIndex);
        SetVisibleRows();

        //If there is a next group of rows after the current ones, enable the button.
        if (nPageIndex &lt;= 0)
            $("#btnPreviousPage").attr("enabled", "enabled");
        else
            $("#btnPreviousPage").attr("enabled", "disabled");
        //If not, disable it.
    }

        //alert("bouncing location:" + nLocationIndex);   //debug
       
        //Set the selected marker to bounce
        _arrMarkers[nLocationIndex].setAnimation(google.maps.Animation.BOUNCE) 
    }

    {
        //Loop through the counters and set their animations to null

            _arrMarkers[nCounter].setAnimation(null);

        }
    }

    {
        if(objValue == null)
            return "";
        else
            return objValue;
    }

        if (_arrMarkers) {
            for (var i = 0; i &lt; _arrMarkers.length; i++) {
                _arrMarkers[i].setMap(null);
            }
        }
    }
