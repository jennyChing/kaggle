	function findPosX(obj) {
		var left = 0;
		if(obj.offsetParent){
			while(1) {
				left += obj.offsetLeft;
				if(!obj.offsetParent)
					break;
				obj = obj.offsetParent;
			}
		} else if(obj.x) {
			left += obj.x;
		}
		return left;
	}
	/**
	 function to use javascript refered in goToAnchor javascirpt method.
	 */
	function findPosY(obj) {
		var curtop = 0;
		if(obj.offsetParent)
			while(1){
				curtop += obj.offsetTop;
				if(!obj.offsetParent)
					break;
				obj = obj.offsetParent;
			}
		else if(obj.y)
			curtop += obj.y;
		return curtop;
	}

	/**
		function to use javascript for "Back To Top" button
		this is changed because if for top anchor href="#top" it changes the
		url with hash value = "#top". Need to avoid that because
		dojo.undo.browser uses # to keep track of states
	*/

	function goToAnchor(anchorName){
		var anchorToMoveTo = dojo.byId(anchorName)||document.anchors[anchorName];
		var moveToY = 0;
		if (anchorToMoveTo){
			if (anchorToMoveTo.posY){
				moveToY=anchorToMoveTo.posY;
			}else{
				moveToY=findPosY(anchorToMoveTo);
				anchorToMoveTo.posY=moveToY;
			}
		}
		window.scrollTo(0,moveToY);
	}

	// to implement back/forward functionality
	var useDojoUndoBrowser = true;
	var isLoadingContent = true;
	var selectedTabId = 0;
	/**
		TabState is the class which maintains the state of the given tabid.
		It have all the methods to fetch the tab information.
	*/
	dojo.declare("TabState", null, {
		constructor: function(tabId){
			this.tabId = tabId;
			this.changeUrl=null;
			var bookmarkValue = tabCls[tabId].bookMarkHash;
			if (bookmarkValue){
				this.changeUrl = bookmarkValue;
			}
			this.isInitialTab=false;
		},
		back: function(){
			/**
			 Displays the forward tab data,
			 If the tab is already selected then the data will fetch from cache.
			 */

			/**
			 fix for bug while using IE. Initial state if doesn't have hash identifier
			 the back button to intial state takes the value from previous visited tab
			 reported : http://trac.dojotoolkit.org/ticket/3418
			 */
			if (dojo.isIE &amp;&amp;
				this.isInitialTab   &amp;&amp;
				window.location.hash!=""){
				window.location.hash=this.changeUrl;
			}
			this.showStateData();
		},
		forward: function(){
			/**
			 Displays the forward tab data,
			 If the tab is already selected then the data will fetch from cache.
			 */
			this.showStateData();
		},
		showStateData: function(){
			/**
			 Displays the selected tab data,
			 If the tab is already selected then the data will fetch from cache.
			 */
			show(this.tabId);
		},
		addTabToHistory: function(){
			/**
			 Creates state for given tab and adds it to dojo history
			 */
			if (dojo.back) {
				dojo.back.addToHistory(this);
			}
		},
		addTabToBrowserHistory: function(){
			/**
			 Creates state for given tab and adds it to dojo history
			 */
			if (!useDojoUndoBrowser) return;
			if (this.tabId == selectedTabId) return;
			this.addTabToHistory();
		},
		getBookMarkHashId: function(){
			return location.hash;
		},
		getBookMarkIdFromUrl: function(){
			/**
			 Extracts book mark hash value from URL returns null if not found
			 */
			var bookmarkId = this.getBookMarkHashId();
			if (bookmarkId &amp;&amp; bookmarkId.length&gt;0){
				// drop # from value
				bookmarkId = bookmarkId.substring(1, bookmarkId.length);
				return bookmarkId;
			}
			return null;
		},
		changeTabForBookMarkHash: function(){
			/**
			 If the URL contains hash value that means the user bookmarked
			 a page other than overview page. This function simply changes
			 to that tab.
			 */
			if (useDojoUndoBrowser){
				var bookMark = this.getBookMarkIdFromUrl();
				if (bookMark != null){
					var indexOFBookMarkedTab = this.indexOfTabBookMarked(bookMark);
					if (indexOFBookMarkedTab != selectedTabId){
						isLoadingContent=false;
						show(indexOFBookMarkedTab);
					}
				}else{
					isLoadingContent=false;
				}
			}
		},
		indexOfTabBookMarked: function(bookMarkValue){
			/**
			 finds the tabId of the tab having the same given
			 book mark hash value
			*/
			for (var i = 0; i &lt; tabCls.length; i++) {
				if (bookMarkValue==tabCls[i].bookMarkHash){
					return i;
				}
			}
			// default to first tab if the hash mark cannot be identified
			return 0;
		},
		setTabAsInitialStateForBrowserHistory: function(){
			/**
			 This to initiate the Browser history state.
			 It clears the existing history and add the current tab state to browser.
			 */

			// creates state for given tab and makes it initial state
			if (!useDojoUndoBrowser) return;
			this.isInitialTab=true;
			this.setInitialStateForBrowserHistory();
		},
		setInitialStateForBrowserHistory: function(){
			/**
			 It is to add the current tab state to browser.
			 */
			if (dojo.back) {
				dojo.back.setInitialState(this);
			}
		}
	});

	/**
		This method to check the browser compatability.
	*/
	function isBrowserCompatible() {
		if (dojo.isOpera) {
			return true;
		} else if (dojo.isMozilla) {
			return true;
		} else if (dojo.isSafari) {
			return true;
		} else if (dojo.isIE) {
			return true;
		} else if (dojo.isChrome) {
			return true;
		}
		return false;
	}

		This method loads while the page is loading at all the time.
		It reset the data for the selected tab when you refresh the page and
		also initiate the browser history.

	dojo.addOnLoad(function() {
		var browserCompatible = isBrowserCompatible();
		useDojoUndoBrowser = useDojoUndoBrowser &amp;&amp; browserCompatible;

	});

	function invokeServerAction(url) {
		dojo.xhrGet({
				url: url,
				load: function(data,ioArgs) {}, //do nothing
				error: showError,
				sync: true
			});

	function closeKbSession() {
		//check whether kbSession is initialized

		//close session if it's initialized
		invokeServerAction('kbCloseSession.action?kbSessionId=' +kbSessionId+'&amp;kbSessionCk=' + kbSessionCk);
