	function isInSpoiler(){
		var textArea = $j('.write-comment textarea');
		var len = textArea.val().length;
		var start = textArea[0].selectionStart;
		var end = textArea[0].selectionEnd;
                var cursor = end;
                //set bounds for spoiler detection
                var insideStartIndex = textArea.val().lastIndexOf(openTag,start+openTag.length);
                var insideEndIndex = textArea.val().indexOf(closeTag,end);
                var outsideEndIndex = textArea.val().lastIndexOf(closeTag,start);
                var outsideStartIndex = textArea.val().indexOf(openTag,end);
                if(
			insideStartIndex != -1&amp;&amp;
                        insideEndIndex != -1&amp;&amp;(
                        (cursor &lt;= insideEndIndex+closeTag.length&amp;&amp;
                         insideEndIndex+closeTag.length &lt;= outsideStartIndex||
                         outsideStartIndex == -1
                         )&amp;&amp;
                        (insideStartIndex &gt;= outsideEndIndex||
                         outsideEndIndex != -1
                         ))
		){
			return true;
		}else{
			return false;
		
		}
	}
	$j.fn.selectRange = function(start, end) {
		if(!end) end = start; 
		return this.each(function() {
			if (this.setSelectionRange) {
				this.focus();
				this.setSelectionRange(start, end);
			} else if (this.createTextRange) {
				var range = this.createTextRange();
				range.collapse(true);
				range.moveEnd('character', end);
				range.moveStart('character', start);
				range.select();
			}
		});
	};
	function wrapInSpoilerTags(){
		//setup
        var textArea = $j('.write-comment textarea');
		var len = textArea.val().length;
		var start = textArea[0].selectionStart;
		var end = textArea[0].selectionEnd;
		// check to see if caret/selected text is currently in a spoiler block
                if(
		textArea.val().lastIndexOf(openTag,textArea[0].selectionStart) != -1&amp;&amp;
                textArea.val().indexOf(closeTag,textArea[0].selectionEnd) != -1
		){
			// spoiler found. Don't double up tags, strip it out
			textArea.val(
                            textArea.val().substring(0,textArea.val().lastIndexOf(openTag,textArea[0].selectionStart))+
                            textArea.val().substring(textArea.val().lastIndexOf(openTag,textArea[0].selectionStart)+openTag.length,textArea.val().indexOf(closeTag,textArea[0].selectionEnd))+
                            textArea.val().substring(textArea.val().indexOf(closeTag,textArea[0].selectionEnd)+closeTag.length,len)		
			)
		}else{
			// if we are actually selecting some text, wrap only that part with spoiler block
                        if(end-start&gt;0){
				var selectedText = textArea.val().substring(start, end);
				var replacement = openTag + selectedText + closeTag;
				textArea.val(textArea.val().substring(0, start) + replacement + textArea.val().substring(end, len));	
			// else wrap entire comment in spoiler block
                        }else{
				textArea.val(openTag+textArea.val()+closeTag);
			}
                        textArea.selectRange(textArea.val().lastIndexOf(openTag)+openTag.length);
			textArea[0].focus();
		}
	}
