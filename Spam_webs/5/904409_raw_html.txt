       <article class="article">
        <p>
         When the temperatures drop, I love making a big pot of this hearty chili. The dried chipotle adds a rich, bold flavor and just the right amount of spice.
        </p>
        <p>
         <strong>
          Vegan, Gluten-Free Recipe: 3-Bean, Kale &amp; Chipotle Chili
         </strong>
        </p>
        <p>
         Serves: 8
        </p>
        <p>
         <em>
          Ingredients:
         </em>
        </p>
        <ul>
         <li>
          1 dried chipotle pepper (or ancho)
         </li>
         <li>
          1 red, 1 yellow and 1 green bell pepper, diced
         </li>
         <li>
          1 yellow onion, diced
         </li>
         <li>
          1 bunch kale, stems &amp; leaves finely chopped (separately)
         </li>
         <li>
          2-4 cloves garlic, chopped
         </li>
         <li>
          28oz can fire roasted crushed tomatoes
         </li>
         <li>
          1 can (15oz) each: black beans, kidney beans, pinto beans
         </li>
         <li>
          2 Tbs chili powder (read ingredients and avoid blends with added sugar, salt, MSG, and other unnecessary fillers)
         </li>
        </ul>
        <p>
         <em>
          Directions:
         </em>
        </p>
        <p>
         Bring 3 cups water to a boil. Remove from heat and place dried chipotle pepper in hot water and allow to re-hydrate.
        </p>
        <p>
         Place a large soup pot on the stovetop over medium-high heat. Add 1 teaspoon olive oil, chopped bell peppers, onion, garlic, kale stems and leaves, and cook for 10 minutes stirring occasionally.
        </p>
        <p>
         Remove and discard seeds and stem from the rehydrated chipotle (option to reserve a few seeds as desired for added heat). Place chipotle in a blender with the boiling water and blend until smooth, adding seeds as desired (do so gradually and taste for heat before adding to chili  a few seeds can add a lot of heat!). Add pureed chipotle and water to sauted veggies.
        </p>
        <p>
         Add 28-ounce can of fire roasted diced tomatoes, 2 tablespoons chili powder and 3 cans of rinsed and drained beans. Bring to a boil then lower heat to low-medium, stirring occasionally and adding water or broth as needed. Continue to cook for 30 minutes, up to 1 hour.
        </p>
        <p>
         Season to taste with additional chili powder, salt and pepper. Enjoy! Note: Leftover chili freezes well.
        </p>

$('.twitter_sharequote_button').click(function(event){
    event.preventDefault();
    var by = '';
    
        by = ' by @PrepDish';
        
    
    var maxTweetLen = 97 - by.length; // 140(max tweet length) - 42 - by.length - 1 (a space for an ending ellipsis).
    var link = 'https://twitter.com/intent/tweet?url=' + encodeURIComponent(MBGArticle.shortUrl);
    var text = $(this).parent().attr("data-text");
    var via = 'mindbodygreen';
    
    if(text.length &gt; maxTweetLen){
    	try{
    		var trimmedText = '';
    		var words = text.split(' ');
    		var counter = 0;
    		for(var i in words){
    			var w = words[i];
    			if(counter + w.length &gt;= maxTweetLen){
    				if(trimmedText.substr(trimmedText.length-1) === ' '){
    					trimmedText = trimmedText.substr(0, trimmedText.length-1);
    				}
    				trimmedText += ''; // finish the trimmed quote with an ellipsis.
    				break;
    			}else{
    				trimmedText += (w + ' ');
    				counter += (w.length + 1);
    			}
    		}
    		text = trimmedText;
    	}catch(err){
    		console.log(err);
    	}
    }
    
    link += '&amp;via=' + via + '&amp;text=' + encodeURIComponent(text) + by;
    window.open(link);
    try{
        ga('send', 'event', 'Article', 'Twitter Share Clicked', '11223');
    }catch(err){}
});

$('#article_email_modal_send_button').click(function(event){
    event.preventDefault();
    $('#article_email_modal_response_target').addClass('hide').removeClass('error').addClass('success').html('');
    if($('#article_email_modal_from').val().trim()===''){
        $('#article_email_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! You need to provide your email address to send this article!');
        return false;
    }
    var to = [];
    var email = {};
    var emails = $('#article_email_modal_to').val().split(',');
    $.each(emails, function(i, v) {
        if(v.trim()!==''){
            to.push(v.trim());
        }
    });
    if(to.length===0){
        $('#article_email_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! You need to provide some email addresses to send this article to!');
        return false;
    }
    
    email.fromEmailAddress = $('#article_email_modal_from').val().trim();
    email.emailAddresses = to;
    email.type = 'article';
    email.postId = '11223';
    
    var inputString = JSON.stringify(email);
    $.ajax({
        type : 'POST',
        url : MBGGlobals.getBaseUrl() + 'emailafriend.action',
        data: {
            input : inputString
        }
    }).done(function(obj) {
        if(obj.success === true) {
            var successMessage = 'Your email has been sent!';
            $('#article_email_modal_response_target').removeClass('hide').removeClass('error').addClass('success').html(successMessage);
            setTimeout(function() {
                $('#article_email_modal').foundation('reveal','close');
                    $('#article_email_modal_response_target').addClass('hide').html('');
                }, 3500);
            try{
                ga('send', 'event', 'Article', 'Article Emailed', '11223');
            }catch(err){}
        } else {
            $('#article_email_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! Something went wrong and we could not send your email. If this error persists, please write us at support@mindbodygreen.com. ' + obj.message);
            console.log(obj.message);
        }
    }).fail(function(jqXHR, textStatus) {
        $('#article_email_modal_response_target').removeClass('hide').removeClass('success').addClass('alert').html('Whoops! Something went wrong and we could not save your article. If this error persists, please write us at support@mindbodygreen.com.');

$('.article_save_button').click(function(event){
    event.preventDefault();
    
    var getChannelsShowModal = function(response,error){
        if(error!==null){
            console.log(error);
            return;
        }
        if(response!==null){
            if(response===false){
                console.log('Login returned false');
                return;
            }else if(response===true){
                MBGUI.toggleUtilityMenuToMyMBG();                   
            }
        }
        $.ajax({
            url: MBGGlobals.getBaseUrl() + 'getchannels.action',
            traditional : true,
            cache : false
        }).done(function(data){
            if(data.success===true){
                if(data.loggedIn === true){
                    var html = "";
                    $(data.channels).each(function(index,item){
                        html += "&lt;input type='checkbox' name='channels' value='" + item.id + "' label='" + item.name + "'/&gt;&nbsp;&nbsp;" + item.name + "&lt;br&gt;";
                    });
                    if(html != ""){
                        $('#article_save_modal_channel_target').html(html);
                    } else {
                        $('#article_save_modal_channel_target').html('You have no saved channels');
                    }
                    $('#article_save_modal').foundation('reveal','open');
                }else{
                    // server thinks the user is not logged in but client does, so show log in screenset
                    MBGSocial.displayLoginModal(getChannelsShowModal);
                }
            }else{
                console.log('An error occcured getting channels: ' + data.message);
            }           
        }).fail(function(xhr,status,error){
            console.log('An error occurred getting channels ' + status + ', ' + error);
        });
                
    };
    
    if(MBGSocial.isUserLoggedIn()){
        try{
            ga('send', 'event', 'Article', 'Logged In Save Button', '11223');
        }catch(err){}
        getChannelsShowModal(null,null);
    }else{
        try{
            ga('send', 'event', 'Article', 'Logged Out Save Button', '11223');
        }catch(err){}
        // use register screenset
        MBGSocial.displayLoginModal(getChannelsShowModal,true);
    }
    $(this).children("img").attr("src", "//res.mindbodygreen.com/img/msc/article-save.svg");
});

        var newChannel = $('#article_save_modal_new_channel').val().trim();

            cache : false,

                subscription : inputString
            }
