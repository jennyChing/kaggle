        function form_events(forms) {
            forms.on("submit", function(e) {
                var this_form = $(this);

                this_form.addClass("disable-div");
                if (!_validation(this_form))
                {
                    this_form.removeClass("disable-div");
                    e.preventDefault();
                    return;
                }
                var request = $.ajax({
                    url: myAjax.ajaxurl,
                    type: "POST",
                    data: this_form.serialize() + this_form.data("params"),
                    dataType: "json"
                });

                request.done(function(msg) {
                    if (msg.type == 'success') {

                        // $("#subscribe_form_message").html(msg.data);
                        alert(msg.data);
                        this_form.removeClass("disable-div");
                        this_form.find('input:not(input[type="submit"],input[type="hidden"])').val('');
                    }
                    else
                    {
                        this_form.removeClass("disable-div");
                        alert(msg.data);
                    }
                });

                request.fail(function(jqXHR, textStatus) {
                    alert("Request failed: " + textStatus);
                    this_form.removeClass("disable-div");
                    this_form.find('input:not(input[type="submit"],input[type="hidden"])').val('');
                });
                e.preventDefault();
            });

        }
