				function findParentOfClass(classString, element){
					var $parent = $(element).parent();
					while (!$parent.hasClass(classString) &amp;&amp; parent){
						$parent = $parent.parent();
					}
					return $parent[0];

				function currentMenuLevel(level){
					var $menu = $('#mobile-header-menu');
					if (typeof(level) != "undefined"){
						$menu.data('menu-level',level);
					}
					else {
						level = $menu.data('menu-level');
					}
					return Number(level);
				}
				var timeout;
				function animateMenu(preAnimate, afterAnimate){
					$('#mobile-header-menu').addClass('animate');
					if (typeof(preAnimate) == "function") preAnimate();
					$('#mobile-header-menu').addClass('animating');
					if (timeout) clearTimeout(timeout);
					timeout = setTimeout(function(){
						if (typeof(afterAnimate) == "function") afterAnimate()
						$('#mobile-header-menu').removeClass('animating');
						$('#mobile-header-menu').removeClass('animate');
					}, 300);
				}
				$('#mobile-menu-toggle img').click(function(event) {
					var $menu = $('#mobile-header-menu');
					var wasOpen = $menu.hasClass('open');
					animateMenu(null, function(){
						if (wasOpen){
							var openMenus = $('.parent-menu.open');
							$(openMenus).each(function(index, el) {
								$(this).removeClass('open');
							});
							var level = Number($menu.data('menu-level'));
							for (var i=1;i&lt;=level;i++){
								$menu.removeClass('menu-level-'+i);
							}
							currentMenuLevel(0);
						}
					});
					if (wasOpen){
						$menu.height(0);
					}
					else{
						var height = $menu.children('ul').height();
						//if (height &gt; screen.availHeight) height = screen.availHeight;
						$menu.height(height);
						currentMenuLevel(0);
					}
					$menu.toggleClass('open');
					$(this).toggleClass('open');
				});
				$('body').on('click',function(event) {
					if ($('#mobile-header-menu').hasClass('open') &amp;&amp; event.originalEvent.srcElement != $('#mobile-menu-toggle img')[0]){
						if ($.contains($('#mobile-header-menu')[0],event.originalEvent.srcElement)){
							return;
						}
						else{
							$('#mobile-menu-toggle img').click();
							event.preventDefault();
							return true;
						}
					}
					return;
				});
				$('.parent-toggle').click(function(event) {
					var parent = findParentOfClass('parent',this);
					if (!parent) return false;
					event.preventDefault();
					var scrollMenu = $(parent).children('.scroll-menu')[0];
					var menu = $(scrollMenu).children('.parent-menu')[0];
					if (!$(menu).hasClass('open') )
					{
						$('.parent-menu.open').each(function(index, el) {
							if (!$.contains(this,menu))
								$(this).removeClass('open');
						});
						$('.scroll-menu.open').each(function(index, el) {
							if (!$.contains(this,menu))
								$(this).removeClass('open');
						});
						var curLevel = currentMenuLevel();
						var level = Number($(menu).data('menu-level'));
						var height = $(menu).height();
						var availHeight = $(window).height() - $('#mobile-header-top').height();
						//if (height &gt; availHeight) height = availHeight;
						$(scrollMenu).height(height);
						if (height &gt; $('#mobile-header-menu').height()) {
							animateMenu(function(){
								$('#mobile-header-menu').height(height);
							});
						}
						else {
							animateMenu(null, function(){
								$('#mobile-header-menu').height(height);
							});
						}
						if (level &gt; curLevel){
							$('#mobile-header-menu').addClass('menu-level-'+level);
							currentMenuLevel(level);
						}
						else {
							$('#mobile-header-menu').removeClass('menu-level-'+curLevel);
							currentMenuLevel(level);
						}
						$(menu).addClass('open');
						$(scrollMenu).addClass('open');
					}

						animateMenu();
						$(menu).removeClass('open');
						$(scrollMenu).removeClass('open');
					}

					var $menu = $('#mobile-header-menu');

					var $target = $($(this).data('target'));
