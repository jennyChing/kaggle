            var top_counts_div = top_bar.find('.big');
            var bottom_counts_div = bottom_bar.find('.big');
            var longtype = type == "fb" ? "facebook" : "twitter";

            if (count &gt; 0 &amp;&amp; count &gt; top_bar.data(longtype + "_count")) {

                counts[type] = count;
                if (type == 'fb') {

                }
