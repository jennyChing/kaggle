		var objButton = objSubForm.find('input[type="submit"]');
		objButton.click( function(e){
			e.preventDefault();

			if(objInput.val().match(regExValidateEmail)){ /* Validate Email Address */
				objInput.removeClass('error');
				$.ajax({
					url: "http://www.pawmygosh.com/wp-admin/admin-ajax.php", 
					type: 'POST',
					dataType: 'json',
					data: { action: "snp_popup_submit", popup_ID: "10765", email: objInput.val() }, 
					success: function(data){					
						if(data.Ok==true){
							disablemodal();

					}
				});
			}else{ objInput.addClass('error'); }
		});
