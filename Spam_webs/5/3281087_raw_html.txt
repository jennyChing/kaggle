     $(function(){
    $('.searchtext input').focusin(function(e){
        $(this).parent().addClass('focus');
    }).focusout(function(e){
        $(this).parent().removeClass('focus');
    });

    $('.cat0drop').on('click', 'a', function(e){
        e.preventDefault();
        var $that = $(this),
            $dd = $that.parents('.dropdown'),
            $trig = $dd.data('dropdown-trigger')
            $sbo = $trig.parents('.searchboxouter'),
            cat0 = $that.attr('cid');
        setcat0(cat0, true);
        if (cat0 != '0') {
            setTimeout( function(){ $('.cat1trig', $sbo).dropdown('show'); }, 1 );
        } else {
            setTimeout( function(){ $('.cat0trig', $sbo).dropdown('hide'); }, 1 );
        }
    });

    $('.cat1drop').on('click', 'a', function(e){
        e.preventDefault();
        var $that = $(this),
            $dd = $that.parents('.dropdown'),
            $trig = $dd.data('dropdown-trigger')
            $sbo = $trig.parents('.searchboxouter'),
            cat1 = $that.attr('cid');
        var cat1 = $that.attr('cid');
        setcat1(cat1, $that.text());
        $('td.searchtext input', $sbo).focus();
        setTimeout( function(){ $('.cat1trig', $sbo).dropdown('hide'); }, 1 );
    });

    $('.mobcat').on('change',function(event){
        var $option = $(this).find('option:selected');
        var cid = $option.val();
        var level = $option.attr('level');

        if( level == '0' ){
            event.preventDefault();
            setcat0(cid, false);
            setcat1(0, "");
            }else{
            setcat1(cid, $option.text());
        }
    });

    var dragon = false;
    $(document).on('touchmove',function(event){ dragon = true;  });
    $(document).on('touchend' ,function(event){ dragon = false; });

    $('.locdrop').on('click touchend', 'a', function(e){
        if( dragon ) {
            dragon = false;
            return;
        }
        var $that = $(this),
            $dd = $that.parents('.dropdown'),
            $trig = $dd.data('dropdown-trigger')
            $sbo = $trig.parents('.searchboxouter');
        if( $that.parent().hasClass('morelocs') ) return;
        e.preventDefault();
        var rid = $that.attr('rid');
        $('.rid').val(rid);
        $('.locs').text( $that.text().replace(/^[ 0-9]*/,'') );
        if( 'ontouchend' in window )
        setTimeout( function(){ $('.locdrop a').dropdown('hide'); window.scrollTo(0, 0); }, 350 );
        else
        setTimeout( function(){ $('.locdrop a').dropdown('hide'); $('td.searchbutton input', $sbo).focus(); }, 100 );
    });

    $('.cat0trig, .cat1trig, .loctrig').keydown(function(event){
        if( event.keyCode == 13 || event.keyCode == 32 || event.keyCode == 40 ){
            event.preventDefault();
            $(this).click();
            $("[tabindex="+$(this).attr('tabindex')+"]").not(this).first().focus();
        }
    });

    var keyclick = false;

    $('.dropdown-menu').on('keydown', 'a', function(event){
        switch( event.keyCode ){
            case 40: $(this).parent().next().children().focus(); break;
            case 38: $(this).parent().prev().children().focus(); break;
            case 32:
            case 13: keyclick = true; $(this).click(); break;
            default: return;
        }
        if( !$(this).parent().hasClass('morelocs') )
        event.preventDefault();
    });

    $('.locdrop').on('show',function(){
        $(this).find('input').focus();
    });

    $('.locdrop .input').click(function(event){
        $(this).find('input').focus();
    });

    $('.locdrop input').keydown(function(event){
        if( event.keyCode == 13 ){
            event.preventDefault();
            var $p = $(this).parents('.locdrop');
            ziplook();
            var whendone = function(){ $('a', $p).first().click(); };
            if( xhr ) xhr.done(whendone);
            else whendone();
            }else if( event.keyCode == 40 ){
            event.preventDefault();
            $('a', $p).first().focus();
        }
    });

    $('.dropdown').keydown(function(event){
        if( event.keyCode == 27 ){
            event.preventDefault();
            $('.dropdown-open').dropdown('hide').focus();
        }
    });

    var setcat0 = function(cat0, clicked){
        var $td = $('.cat1td');
        var $ul = $('.cat1drop ul');
        var $sel = $('.mobcat');

        $('.cat0in').val(cat0);

        var text = getcat0name(cat0);
        $('.cat0s').text(text);
        $('.cat01s').text( (cat0 == '0' ? '' : 'All ') + text );

        $ul.html('');
        $sel.html('');
        /* if( clicked ) */ $('.cid').val(cat0);
        if( cat0=='0' ){
            $td.hide();
            $ul = new $.fn.init; // select nothing, quickly!
            }else{
            $td.show();
            var cat0name = getcat0name(cat0);
            if( clicked ){
                $('.cat1s').text( "All "+cat0name );
                $('.cat01s').text( "All "+cat0name );
            }
        }

        var level = cat0 == '0' ? 0 : 1;
        $ul.append(function(i, h) {
            var ti = $(this).parents('.cat1drop').data('tabindex');
            return "&lt;li&gt;&lt;a href=# cid=0 tabindex=\"" + ti + "\"&gt;All "+cat0name+"&lt;/a&gt;&lt;/li&gt;";
        });
        $sel.append( "&lt;option level=0 value=0&gt;All Categories&lt;/option&gt;" );
        if( cat0name )
        $sel.append( "&lt;option level=0 value="+cat0+"&gt;All "+cat0name+"&lt;/option&gt;" );
        for( var c in cattree[cat0] ){
            var cid = cattree[cat0][c][0];
            var name = cattree[cat0][c][1];
            $ul.append(function(i, h) {
                var ti = $(this).parents('.cat1drop').data('tabindex');
                return "&lt;li&gt;&lt;a href=# cid="+cid+" tabindex=\"" + ti + "\"&gt;"+name+"&lt;/a&gt;&lt;/li&gt;";
            });
            $sel.append( "&lt;option level="+level+" value="+cid+"&gt;"+name+"&lt;/option&gt;" );
        }
        $sel.val([]); // select no option
    };

    var setcat1 = function(cat1, text){
        $('.cat1in').val(cat1);
        var cat0name = getcat0name($('.cat0in').val());
        if( cat1 != '0' ){
            $('.cid').val(cat1);
            $('.cat1s').text(text);
            $('.cat01s').text( cat0name + ': ' + text );
        } else {
            $('.cid').val($('.cat0in').val());
            $('.cat1s').text("All "+cat0name);
            $('.cat01s').text(cat0name);
        }
    };

    $('.cat1drop').on('show',function(){
        if( keyclick )
        $(this).find('a').first().focus();
        keyclick = false;
    });

    $('.searchform').on('submit',function(event){
        if( $('.moresearchflds').is(':visible') ){
            event.preventDefault();
            var qs = $('.searchform, #moreform').serialize();
            window.location = "/search.php?" + qs;
        }
    });

    $('.moresearchflds input, .moresearchflds select').keydown(function(event){
        if( event.keyCode == 13 ){
            $('#searchform').submit();
        }
    });

    var getcat0name = function(cat0){
        if( cat0 == '0' )
        return 'All Categories';

        for( var c in cattree[0] )
        if( cattree[0][c][0]==cat0 )
        return cattree[0][c][1];

        return "Unknown";
    };

    var to = null;
    var xhr = null;
    var lastval = "";
    var ziplook = function(){
        clearTimeout(to);
        if( xhr ) { xhr.abort(); xhr = null; }
        var $that = $(this),
            $zipin = $('.zipin');
        var $li = $zipin.parents('li.input');
        var val = $that.val();
        if( val.length &lt; 1 || val == lastval ) return;
        lastval = val;
        xhr = $.get("/guess_location.php?zip="+escape(val), function(data){
            data = $.parseJSON(data).reverse();
            for( d in data ) {
                $li.after(function(i, e) {
                    var tid = $(this).parents('.locdrop').data('tabindex');
                    return "&lt;li&gt;&lt;a href=# rid="+data[d].rid+" tabindex=" + tid + "&gt;"+data[d].name+"&lt;/a&gt;&lt;/li&gt;";
                });
            }
            if( data.length ) {
                $.each($li, function(i, v) {
                    var _that = $(this);
                    _that.siblings('li:gt('+(data.length-1)+')').not('.morelocs').remove();
                });
            }
        });
    };

    $('.zipin').on('keyup', function(){ clearTimeout(to); to = setTimeout(ziplook.bind(this),150); });
    $('.zipin').on('change', function(){ clearTimeout(to); to = setTimeout(ziplook.bind(this),150); });

    $('.minisearch').on( 'click', function(e) {
        var $that = $(this),
            $sf = $that.parents('.searchform');
        $('.searchboxouter, .searchbox2, .moreouter', $sf).removeClass('nomob');
        $('.minisearch', $sf).removeClass('mobonly').css('display', 'none');
        $('.searchtext input', $sf).focus();
        e.stopPropagation();
        return false;
    });

    setcat0(462, false);
    setcat1(83, 'Beauty &amp; Salon Services');
});
