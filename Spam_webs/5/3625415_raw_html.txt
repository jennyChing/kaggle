(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    1 cup lentils
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    1 cup whole grain elbow macaroni
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    Water, as needed
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    4 medium onions, halved and thinly
sliced
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    14 cup olive oil
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    4 garlic cloves, chopped
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    1 tsp ground cumin
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    12 tsp red pepper flakes
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    One 29-oz can tomato pure
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    Salt and pepper, to taste
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <br/>
                  </div>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="https://allfitnesse.blogspot.com/" target="_blank">
                    <img alt="Egyptian Kamut Kushari" border="0" height="400" src="http://4.bp.blogspot.com/-Jocp5A5bo5Y/U0QpqzsXl8I/AAAAAAAABHU/R33NvqCXiwE/s1600/Egyptian+Kamut+Kushari.jpg" title="Egyptian Kamut Kushari" width="330"/>
                   </a>
                  </div>
                  <br/>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <i>
                    <span lang="EN-GB" style='color: red; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                     Directions
                     <o:p>
                     </o:p>
                    </span>
                   </i>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    1. Soak kamut
                   </span>
                   <br/>
                   <a name="more">
                   </a>
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    berries in water
overnight. Drain and rinse berries.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    2. Place kamut berries and 3 cups of
water in a pot and bring to a boil. Cover and reduce heat to low. Simmer until
soft, about 1 hour. Drain off excess water.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    3. Rinse lentils and place in a pot with
3 cups of water. Cover and bring to a boil. Let boil for 2 to 3 minutes then
reduce heat to low and simmer until tender, about 25 to 30 minutes. Drain off excess
water.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    4. Cook macaroni in plenty of boiling
salted water until tender, about 6 to 7 minutes. Drain.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    5. While grains are cooking, prepare
onions and tomato sauce. Heat oil in a pan on medium-high heat and cook sliced
onions until brown and crispy, about 10 minutes.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    Remove from pan and drain on paper
towels. Season with salt.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    6. In the same pan, add garlic, cumin,
and red pepper flakes, and cook until fragrant, about 2 minutes. Add tomato
pure and cook until heated through. Season with salt and pepper.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                    7. To serve, combine cooked kamut,
lentils, and macaroni. Top with tomato sauce and crispy onions.
                    <o:p>
                    </o:p>
                   </span>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <b>
                    <i>
                     <span lang="EN-GB" style='color: #215868; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi; mso-themecolor: accent5; mso-themeshade: 128;'>
                      Nutrient
Analysis per serving
                      <o:p>
                      </o:p>
                     </span>
                    </i>
                   </b>
                  </div>
                  <div class="MsoNormal" style="margin-bottom: 0.0001pt;">
                   <b>
                    <i>
                     <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>
                     </span>
                    </i>
                   </b>
                   <span lang="EN-GB" style='color: #231f20; font-family: "Times New Roman","serif"; font-size: 12.0pt; mso-ansi-language: EN-GB; mso-ascii-theme-font: major-bidi; mso-bidi-theme-font: major-bidi; mso-hansi-theme-font: major-bidi;'>

g; Sat fat: 1 g, Sodium:
