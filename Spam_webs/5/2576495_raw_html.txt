var fbState = {userDetailsLoaded: false};
var bIntegrateDmTag = true;

/**
 * Routines to track events to Google Analytics
 * @param {String} sType Holds the type of the event (Category field on Google Analytics)
 * @param {String} sAction Action identifier
 * @param {String} sLabel Additional data or Action Label
 * @param {Function} fCallback Receive the callback function triggered after event track
 */
function sendGAEventFbss(sType, sAction, sLabel, fCallback) {
    if (_gaq &amp;&amp; sAction) {
        _gaq.push(['_set', 'hitCallback', null]); //cleaning callback
        if (fCallback &amp;&amp; typeof fCallback == 'function') {
            _gaq.push(['_set', 'hitCallback', fCallback]); //registering callback
        }
        _gaq.push(['_trackEvent', sType + ':' + tracking.cid, sAction, sLabel || 'no data']); //registering event
    } else {
        if (fCallback &amp;&amp; typeof fCallback == 'function') {
            fCallback(); //running callback anyway
        }
    }
}

$(document).facebookReady(function(){
    $(document).ready(function(){
        FB.getLoginStatus(function (response) {
            fbState.status = response.status;
            if (response.status == 'unknown') {
                sendGAEventFbss("FBSS", "NoFB");

                getFBPermissions();
                set_share_handlers();

});

function getFBPermissions(callback) {
    FB.api('/me/permissions', function (response) {
        fbState.canPost = (response.data &amp;&amp; response.data[0] &amp;&amp; response.data[0].publish_actions);
        if (!fbState.userDetailsLoaded) {
            return getUserDetails(callback);
        }
        if (callback &amp;&amp; typeof callback == 'function') {
            return callback();
        }
    });
}

function getUserDetails(callback) {
    FB.api('/me', function(response) {
        if (response &amp;&amp; !response.error) {

            //Enabling
            $('input[id^="fb_hidden_"]').removeProp('disabled');

            //Filling First and Last name fields with user data
            $('#fb_hidden_FirstName').val(response.first_name || '');
            $('#fb_hidden_LastName').val(response.last_name || '');

            //Updating or inserting DoB field
            if (response.birthday != null) { //separated tests to avoid error throwing "response.birthday is null"

                if (response.birthday.match(/^(0\d|1[012])\/(0\d|1\d|2\d|3[01])\/(19|20)\d\d$/)) {
                    $('#fb_hidden_BirthDateDay').val(response.birthday.substr(3,2) || '');
                    $('#fb_hidden_BirthDateMonth').val(response.birthday.substr(0,2) || '');
                    $('#fb_hidden_BirthDateYear').val(response.birthday.substr(6,4) || '');

                    //removing the additional field DoB if it's included
                    //$("#additional_form_fields select[name*='DoB']").parent().remove();
                    $("#additional_form_fields select[name*='DoB']").parent().hide();
                }
            }

            fbState.userDetailsLoaded = true;
        }

        if (callback &amp;&amp; typeof callback == 'function') {
            return callback();
        }
    });
}

function getHashFromUrl() {
    var aUrlParams = window.location.search.replace('?', '').split('&amp;');
    var sEntry = '';
    for (i in aUrlParams) {
        sEntry = aUrlParams[i];
        if (sEntry.indexOf('=') == -1 &amp;&amp; !/^\d+$/.test(sEntry) &amp;&amp; sEntry.length &gt;= 6) {
            return sEntry.substring(1);
        }
    }
    return false;
}

function validatePostPermissions(callback) {
    if (fbState.canPost) {
        callback();
    } else {
        FB.login(function(response) {
            getFBPermissions(callback);
        }, {scope: 'publish_actions,user_birthday'});
    }
}

function postToFacebook(msg, bSubmit) {
    var hash = getHashFromUrl();
    var post = {
        message: msg,
        description: "One country's failure to tackle the ivory trade could mean we lose African elephants forever. Find out how we can take a huge step to stop this tragedy in the next few hours.",
        name: "20,000 are killed every year...",
        link: "https://secure.avaaz.org/en/hours_to_save_elephants_usa/?" + (hash == '' ? 'fbss' : 's' + hash),
        picture: $('meta[property="og:image"]').attr('content').replace(/https?:/, document.location.protocol)
    };

    FB.api('/me/feed', 'post', post, function (response) {
        if (bSubmit) {
            $.ajax({
                url: '/act/facebook.php?action=save_share_counter&amp;cid=10072&amp;lang=en&amp;value=clicked',
                success: function () {
                    $(window).off("unload", submitPetitionFormInBackground);
                    $('#petition-form').submit();
                }
            });
        }
    });
}

function submitPetitionFormInBackground() {
  $.ajax({
      type: 'POST',
      async: false,
      url: $('#petition-form').attr('action'),

          if (window.location.href.indexOf('fbdm=22') &gt; -1) {

              var oSuggested = $('#suggestedMessageText');

              oSuggested.text(oSuggested.data('text'));
              $('#dialogTagFBFriendsLink').html(sUrl);
          }
      }
  });
}
