    var cookieName = 'CRGN';

    var compareTemplate = 'Compare ({0})';

    LoadCheckBox = function () {

        $(".csn-results .item").each(function () {
            var itemId = $(this).attr("id");
            if (comparisonManager.hasItem(cookieName, itemId)) {
                $(this).find(".check").toggleClass(checkedClass);
            }
        });

        UpdateCount();
    }

    ToggleCheckBox = function (sender, valueToAdd) {
        if (comparisonManager.hasItem(cookieName, valueToAdd)) {
            comparisonManager.removeItem(cookieName, valueToAdd);
        }
        else {
            comparisonManager.addItem(valueToAdd);
            $(sender).toggleClass(checkedClass);
        }

        UpdateCount();
    }

    RemoveCheckBox = function (valueToRemove) {
        $("." + checkedClass + "[data-id=" + valueToRemove + "]").toggleClass(checkedClass);

        UpdateCount();
    }
