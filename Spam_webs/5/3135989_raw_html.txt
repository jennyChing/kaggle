function handleSubmitOnEnter(e)
{
    
    if (window.ghKeyPressHandler != null)
    {
        if ( typeof(ghKeyPressHandler == "function") )
        {
            ghKeyPressHandler(e);
        }
    }
    
    var keyPressed = '';

    var isMSIE = navigator.userAgent.indexOf('MSIE') != -1;

    if (isMSIE)
        keyPressed = String.fromCharCode(window.event.keyCode);
    else
        keyPressed = String.fromCharCode(e.which);

    if (keyPressed == "\r" || keyPressed == "\n")
    {
        
        var authenticationFormToUse = null;

        if (document.authenticationEntryFormForHeader &amp;&amp; !document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryFormForHeader;
        }
        else if (!document.authenticationEntryFormForHeader &amp;&amp; document.authenticationEntryForm)
        {
            authenticationFormToUse = document.authenticationEntryForm;
        }

        if (authenticationFormToUse == null)
        {
            var headerFieldsFilledIn = 0;
            if (document.authenticationEntryFormForHeader)
            {
                if (typeof(document.authenticationEntryFormForHeader.screenName) != "undefined"
                    &amp;&amp; document.authenticationEntryFormForHeader.screenName.value.length &gt; 0)
                {
                    headerFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryFormForHeader.kclq) != "undefined"
                    &amp;&amp; document.authenticationEntryFormForHeader.kclq.value.length &gt; 0)
                {
                    headerFieldsFilledIn++;
                }
            }

            var bodyFieldsFilledIn = 0;
            if (document.authenticationEntryForm)
            {
                if (typeof(document.authenticationEntryForm.screenName) != "undefined"
                    &amp;&amp; document.authenticationEntryForm.screenName.value.length &gt; 0)
                {
                    bodyFieldsFilledIn++;
                }
                if (typeof(document.authenticationEntryForm.kclq) != "undefined"
                    &amp;&amp; document.authenticationEntryForm.kclq.value.length &gt; 0)
                {
                    bodyFieldsFilledIn++;
                }
            }

            if (headerFieldsFilledIn &gt; bodyFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryFormForHeader;
            }

            if (bodyFieldsFilledIn &gt; headerFieldsFilledIn)
            {
                authenticationFormToUse = document.authenticationEntryForm;
            }
        }

        // If no form or a search is being performed
        if ( authenticationFormToUse == null || (typeof searchCtlr !== "undefined" &amp;&amp; searchCtlr &amp;&amp; document.forms["hcSearchForm"].token.value != '') )
        {
            window.document.onkeypress = doNothing;
        }
        else
        {
            if (authenticationFormToUse.kclq.value == "")
            {   
                authenticationFormToUse.kclq.focus();
            }
            else
            {
                authenticationFormToUse.submitButton.click();
    
                window.document.onkeypress = doNothing;
            }
        }


        return false;
    }

    return true;
}

        window.document.captureEvents(Event.KEYPRESS);
    }
    window.document.onkeypress = handleSubmitOnEnter;
