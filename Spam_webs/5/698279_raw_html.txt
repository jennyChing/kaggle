function ContinueSSOLogin(siteUid,emailId,password,screenName){

	$.getJSON(url,function(result) {

		if(isLoggedIn == 'true'){

	});
}

	var loggedOut = result.loggedOut;

			createCookie("SSO_CHECKED", "true");

		}
	});
