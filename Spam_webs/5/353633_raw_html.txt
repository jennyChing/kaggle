function isEmailAddr(email)
{
  var result = false
  var theStr = new String(email)
  var index = theStr.indexOf("@");
  if (index &gt; 0)
  {
	var pindex = theStr.indexOf(".",index);
	if ((pindex &gt; index+1) &amp;&amp; (theStr.length &gt; pindex+1))
	result = true;
  }
  return result;
 }
