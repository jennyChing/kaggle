    function AddNewCityClick() {
        var State = document.getElementById("State");
        var CityNm = document.getElementById("NewCityName").value;
        

        if (State == null &amp;&amp; CityNm == "") {
            alert("Please Enter City Name");
        }
        else if (State == null &amp;&amp; CityNm != "") {
        //Add Only City
            $("#NewCity").slideDown("slow");
            $("#city").hide();
            $("#country").hide();
            var Country = document.getElementById("CountryName").innerHTML;
            $.ajax({
                cache: false,
                type: "GET",
                url: "/Home/AddNewCity/",
                data: { CountryName: Country, StateName: "", CityName: CityNm },

                success: function (data) {
                    alert(data);
                    var dataStr = data.toString();
                
                    var CountryAttribIndex = dataStr.indexOf("CountryID");
                    var CountryAttrib = data.substring(CountryAttribIndex, dataStr.length - 1).toString();
                    var CountryID = CountryAttrib.substring(CountryAttrib.lastIndexOf(":") + 1).toString();
                    //alert(CountryID);
                    if (CountryID != "0") {

                        LoadCities(CountryID);
                    }
                    else {
                        alert("Your City Request is Not Validated..! &lt;/br&gt; Request for Your City is taken but will be approved and listed Shortly..!")
                        $("#city").slideDown("slow");
                        $("#country").hide();
                        $("#NewCity").hide();
                    }
                }
            });
        }
        else if (State != null &amp;&amp; State.options[State.selectedIndex].text != "[SELECT]" &amp;&amp; CityNm != "") {
            var StName = State.options[State.selectedIndex].text;
            $("#NewCity").slideDown("slow");
            $("#city").hide();
            $("#country").hide();
            var Country = document.getElementById("CountryName").innerHTML;

            $.ajax({
                cache: false,
                type: "GET",
                url: "/Home/AddNewCity/",
                data: { CountryName: Country, StateName: StName, CityName: CityNm },

                success: function (data) {
                    var dataStr = data.toString();
                    var CountryAttribIndex = dataStr.indexOf("CountryID");
              
                    var CountryID = data.CountryID;
                    alert(CountryID);
                    if (CountryID != "0") {

                        LoadCities(CountryID);
                    }
                    else {
                        //  alert("Your City Request is Not Validated..! &lt;/br&gt; Request for Your City is taken but will be approved and listed Shortly..!")
                        var msgRequest = document.getElementById('msgRequestCity');
                        msgRequest.innerHTML = '&lt;font color=green&gt;Your City Request is Not Validated..! &lt;/br&gt; Request for Your City is taken but will be approved and listed Shortly..!&lt;/font&gt;';
                        $("#city").slideDown("slow");
                        $("#country").hide();
                        $("#NewCity").hide();
                    }
                }
            });
        }
        else { 
            alert("Please Select State &amp; Enter City Name...!");
        }
    }
