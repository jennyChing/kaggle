function voteComment(a, b, c, d) {
	var e = jQuery(a).parent().get();
	var d = 'moviehotties';
	
	if(jQuery(a).attr('title') == 'voted')
	{
		alert("You've already voted on this comment!")
	}
	else
	{
		jQuery.get("http://www.joblo.com/tb_vote.php", {
			talk_id: b,
			type: c,
			site: d
		}, function (a) {
			if (a &gt;= 0) {
				a = "+" + a
			}
			jQuery(e).find(".counter").html(a)
		});
		jQuery(e).find("a.up").attr({title: "voted"});
    	jQuery(e).find("a.down").attr({title: "voted"})
	}
