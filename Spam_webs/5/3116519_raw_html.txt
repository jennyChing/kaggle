   //swapna - for search box validation
   function valHeader(evt) {
      var theEvent = evt || window.event;
      var key = theEvent.keyCode || theEvent.which;

      if (key == 8) {
         a = $("#txtsearchBox").html();
         $("#txtsearchBox").html(a.substring(0, a.length - 1));
         currentCommand = currentCommand.substring(0, currentCommand.length - 1);

      }

      //ramos - AUR-3610 - the tab key is not working in Firefox, so I need to force it here. Added a header_btnSubmit id for the submit button
      //so I can set focus to it
      if (key == 9)
      {
         $("#header_btnSubmit").focus();
      }

      if (key != 13) {
         key = String.fromCharCode(key);
         var regex = /^[a-z0-9\\\-/.,&amp; ]$/i;
         if (!regex.test(key)) {
            theEvent.returnValue = false;
            if (theEvent.preventDefault) theEvent.preventDefault();
         }
      }
      else {
         valText(evt);
      }
   }
   //swapna - to restrict submit if special characters alone are entered in search box
   function valText(evt) {
      var theEvent = evt || window.event;
      var txt = document.getElementById('txtsearchBox').value.replace(/^\s+|\s+$/g, '');
      var regex = /[\\\-/.,&amp; ]$/i;
      if (txt.length &gt;= 1) {
         if (regex.test(txt.substring(0, 1))) {
            theEvent.returnValue = false;
            if (theEvent.preventDefault) theEvent.preventDefault();
            $("#searchBoxError").show();
         }
      }
   }

   //DG- checking to make sure we arent searching for placeholder text.
   $('#header_btnSubmit').on('focus', function () {
      $("#txtsearchBox").trigger("focus");
   });

   $('*[data-autocomplete-url]')
