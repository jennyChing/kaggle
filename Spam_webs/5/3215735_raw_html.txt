function checkDate(month, day, year) {
	var monthLength = new Array(31,28,31,30,31,30,31,31,30,31,30,31);
	if (!day || !month || !year) {
	return false;
	}
	if (year/4 == parseInt(year/4)) {
	monthLength[1] = 29;
	}
	if (month &lt; 1 || month &gt; 12) {
	return false;
	}
	if (day &gt; monthLength[month-1]) {
	return false;
	}
	return true;
}

        <p style="text-align: justify;">
         Gran parte de nuestras actividades las realizamos apoyados en una computadora tradicional. Ya sea para funciones laborales, por temas de estudios o por simples actividades ldicas, el uso del
         <strong>
          teclado
         </strong>
         es bsico para comenzar a utilizar un ordenador.
         <br/>
         <br/>
         Probablemente sea conocido por todos que Control + C para copiar y Ctrl + V para pegar. Sin embargo, existen diversos trucos y combinaciones de teclas que nos ayudarn a navegar por la web de forma rpida y prctica y, en ocasiones, dejando de lado al famoso mouse.
         <br/>
         <br/>
         Basta con memorizar la combinacin de algunas
         <strong>
          teclas
         </strong>
         para que podamos cerrar una ventana, minimizar una pantalla o cambiar todo un prrafo de minscula a mayscula o viceversa. Asimismo, al visualizar un video en
         <strong>
          YouTube
         </strong>
         tambin podemos recurrir a los botones de nuestro ordenador para detener o avanzar un clip.
         <br/>
         <br/>
         Conoce en la siguiente galera los mejores trucos y opciones que tiene el teclado de tu computadora.
