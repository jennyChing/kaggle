"Window Fairy"

INT. BY A WINDOW

    GIRL
What are you looking at?

    GUY
Outside.

    GIRL
What's outside?

    GUY
Everything.

    GIRL
You need to get out more.

    GUY
And the Window Fairy.

    GIRL
The Window Fairy?

    GUY
Yes, the Window Fairy.

    GUY
She comes every morning.

    GUY
She lands right there on that window seal.

    GIRL
It's not window seal, it's window pane.

    GUY
A pane? It can't be a pane. The pane is the glass. It has to be a seal.

    GIRL
How can it be a seal? What is it sealing?

    GUY
It's sealing the window to the bottom part there.

    GIRL
You need to learn how -

    GUY
Hold on, she's coming.

    GIRL
What does a Window Fairy do?

    GUY
Many different things.

    GUY
This one in particular, however, is an English Window Fairy. She gives me English advice.

    ENGLISH WINDOW FAIRY
Window sill. It's a window sill.

      $(document).ready(startRatings)
        var scoreSum = 4

        function doStars () {
          var avg = (totalRatings == 0) ? 0 : scoreSum / totalRatings
          avgRating = avg.toFixed(1)
          $('#score').html(avgRating)
          $('#totalRatings').html(totalRatings)
          var stars = Math.round(avgRating)
          for (s = 0;s &lt; 5;s++)
            if (s &lt; stars)
              $('#ratings-star'+s).css('background', 'url(http://popstrip.com/i/star1.png)')
            else
              $('#ratings-star'+s).css('background', 'url(http://popstrip.com/i/star0.png)')
        }
        function rateStrip (score) {
          $.getJSON("http://popstrip.com/ratings.php", {
                  "s": "window-fairy", 
                  "score": score,
                  "rate": 1
              }, rateStripCB)
        }
        function rateStripCB (r) {
          // if (typeof r.success == 'undefined') {
          if (r.error) {
            alert(r.msg)
            return
          }
          if (r.alreadyRated) {
            $('body').append('&lt;div id="rating-msg" style="position:absolute;z-index:10;color:green;font-weight:bold"&gt;Rating changed.&lt;/div&gt;')
            $('#rating-msg').center()
            $('#rating-msg').fadeOut(2000, removeRatingsMsg)
          }
          scoreSum = r.scoreSum
          totalRatings = r.totalRatings
          doStars()
        }
        function removeRatingsMsg () {
          $('#rating-msg').remove()
        }
