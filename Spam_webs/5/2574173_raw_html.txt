function caption_vote(ccId,cdId,id,type)
{
		$.ajax({
							dataType:"json",
							type:"POST",
							data:{ccId:ccId,cdId:cdId},
							url:'http://www.captionday.com/memegenerator/caption_vote',
							success:function(data)
											{
												if(data.success==1)
												{
													
													VoteCount=$('#count_'+type+'_'+ccId).text();
													VoteCount++;
													$('#count_'+type+'_'+ccId).text(VoteCount);
													$('#image_'+type+'_'+ccId).attr("src","/theme/images/like_counter-big-blue.png");;
													$('#'+id).attr('onclick','').unbind('click');
												}
												else
												{
													//location.reload();
												}
											}
													
					});	
}
