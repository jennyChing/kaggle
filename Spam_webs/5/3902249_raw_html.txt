        <section class="text_wrapper section intro">
         <p>
          Kawasaki motorcycles include its Ninja family of sportbikes such as the Ninja ZX-10R and Ninja ZX-6R, the entry-level Ninja 250R, the Z1000 streetbike, the Concours 14 sport touring motorcycle and Vulcan family cruisers such as the Vulcan 1700 Voyager. Kawasakis off-road motorcycles include KX450F motocross bike and KLX 140 enduro motorcycle.
         </p>
         <p>
          Kawasaki joined the motorcycle business in 1960 and is one of the major players from Japan. From the time it released the B8 125cc motorcycle in 1961 Kawasaki has seemingly never slowed down. Just five years later Kawasaki produced Japans biggest bike  the 650W1. The first Voyageur touring model, based on the liquid-cooled KZ1300, hit dealerships in 1978. In 1981 Kawasaki launched the hugely popular 600cc sportbike class with the GPz550 and just two years later the GPz900R Ninja is introduced to a stunned motorcycle press. The first ZXR-designated motorcycles reach the market in 1989 and are still going strong today. In addition to its vast lineup of street bikes, Kawasakis KX lineup of off-road motorcycles cannot be ignored, thanks largely to the racing success of James Bubba Stewart.
         </p>

         function get_specsNextDropdown(level, loaddefault){

	//selected={Make:$("#defaultManufacturer").val(),Category:$("#defaultCategory").val(),Year:$('#defaultYear').val(),Model:$('#defaultModel').val()};

    $.get("/specs/ajax.php",
               {
                ajax_action:"find_a_bike",
                level:level,
                MakeId:$("#MakeId").val(),
                ModelType:$("#ModelType").val(),
                year:$("#ModelYear").val(),
                //selected:selected,
                //parent_selected:parent_selected
            },
               function(j){
                  var options = "";
                  j=j.split(/,/);
                  var level=j[0];
                  var options=j[1];
                  if(level==0){
                      $("#MakeId").html(options.replace('Select a Manufacturer','Manufacturer'));
                      if (!jQuery.browser.msie)$("#MakeId option:first").attr("selected","selected");
                      $("#ModelType").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");
                      $("#ModelYear").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");
                      $("#TrimId").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");

                      // set default manufacturer by filename
                      if(loaddefault){
                          $("#MakeId option").each(
                             function (i){
                               if($('#defaultManufacturer').val().toUpperCase()==this.innerHTML.toUpperCase()){
                                     this.setAttribute('selected','selected');
                                     $("select#MakeId").val(this.value);
                               }
                             }
                          );

                          if($('#MakeId').val()){
                        	  get_specsNextDropdown(1, true);
                          }
                      }

                  }else if(level==1){
                      $("#ModelType").html(options.replace('--Choose a Model Type--','Category'));
                      if (!jQuery.browser.msie)$("#ModelType option:first").attr("selected","selected");
                      $("#ModelYear").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");
                      $("#TrimId").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");

                      if(loaddefault){
                          $("#ModelType option[value='"+$('#defaultCategory').val()+"']").attr("selected",true);
                          if($('#ModelType').val()){
                        	  get_specsNextDropdown(2, true);
                          }
                      }

                  }else if(level==2){
                      $("#ModelYear").html(options.replace('--Choose a Year--','Year'));
                      $("option:first", "#ModelYear").attr("selected","selected");
                      $("#TrimId").html("&lt;option value=''&gt;&nbsp;&lt;/option&gt;");

                      if(loaddefault){
                          $("#ModelYear option[value='"+$('#defaultYear').val()+"']").attr("selected",true);
                          if($('#ModelYear').val()){
                        	  get_specsNextDropdown(3, true);
                          }
                      }

                  }else if(level==3){
                      $("#TrimId").html(options.replace('--Choose a Motorcycle--','Model').replace(/&reg;/g,'').replace(/&trade;/g,''));
                      //$("#TrimId").html($("#TrimId").html().replace(/\u00AE/g,''));
                      $("option:first", "#TrimId").attr("selected","selected");

                      if(loaddefault){
                          $("#TrimId option").each(
                                  function (k,v){
                                    if($('#defaultModel').val()==$.trim($(this).html()) || $('#defaultModel').val()==$.trim($(this).val()) ){
                                          $(this).attr('selected','selected');
                                    }
                                  }
                               );
                      }

                  }
                  $.uniform.update();
            }
         );
}

	var date = new Date();
	date.setTime(date.getTime()+(days*24*60*60*1000));

else var expires = "";
document.cookie = name+"="+value+expires+"; domain=.motorcycle.com; path=/;";

var nameEQ = name + "=";
var ca = document.cookie.split(';');

	var c = ca[i];
