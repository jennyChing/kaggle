function validateEmail(email)
{

	var re_mail = /^([a-zA-Z0-9_\.\-])+\@(([a-zA-Z0-9\-])+\.)+([a-zA-Z])+$/;

	if (!re_mail.test(email.value))

	{

		return false;

	}

	return true;

}

	{

	});

}
