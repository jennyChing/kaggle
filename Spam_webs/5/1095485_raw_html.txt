   $(document).on({
	mouseenter: function() {
		var id = $(this).attr("id").replace(/blog_comment_/, "");
		$("#icon_edit_"+id).show();
		$("#icon_remove_"+id).show();
	},
	mouseleave: function() {
		var id = $(this).attr("id").replace(/blog_comment_/, "");
		$("#icon_edit_"+id).hide();
		$("#icon_remove_"+id).hide();
	}
}, "[id^=blog_comment]");


$(document).on({
	focus: function() {
		if ($("#comment_content").val() == "Write a comment...") {
			$("#comment_content").val("");
			$("#comment_content").css("color","#333333");
		}
	},
	blur: function() {
		if ($("#comment_content").val() == "") {
			$("#comment_content").css("color","#AAAAAA");
			$("#comment_content").val("Write a comment...");
		}
	}
}, "#comment_content");


$(document).on({
	click: function(event) {
		if ($("#comment_content").val() == "" || $("#comment_content").val() == "Write a comment...") {
			alert("Enter your comment!");
		} else {
			$("#comment_button_holder").html("&lt;a href=# onclick='return false;' class=button_inactive&gt;Comment&lt;/a&gt;");
			document.body.style.cursor = "wait";
			$.post("https://markethive.com/blogs_query.php", {
				action: "comment",
				post: "193045",
				comment: $("#comment_content").val()
			},
			function(response) {
				$("#new_comment").prepend(response);
				$("#comment_content").css("color","#AAAAAA");
				$("#comment_content").val("Write a comment...");
				$("#comment_button_holder").html("&lt;a id=comment_button href=# class=button_blue&gt;Comment&lt;/a&gt;");
				$("html, body").animate({ scrollTop: $(document).height() }, "slow");
				document.body.style.cursor = "default";
			});	
		}
		event.preventDefault();
	}
}, "#comment_button");

function FollowBlog(action) {

	console.log("FollowBlog("+action+")");

	if (action == "start") {
		$("#follow_blog_start").html("&lt;img src=https://markethive.com/images/working.gif&gt;");
	} else if (action == "stop") {
		$("#follow_blog_stop").html("&lt;img src=https://markethive.com/images/working.gif&gt;");
	}

	document.body.style.cursor = "wait";
	$.post("https://markethive.com/blogs_query.php", {
		action: "follow",
		type: action,
		blog: "marcoferrari_944"
	},
	function() {
		if (action == "start") {
			$("#follow_blog_start").hide();
			$("#follow_blog_stop").html("&lt;a href=# onclick=\"FollowBlog('stop'); return false;\"&gt;&lt;img src='https://markethive.com/images/follow_blog_stop.png' border=0 title='Stop following this blog'&gt;&lt;/a&gt;");
			$("#follow_blog_stop").show();
		} else if (action == "stop") {
			$("#follow_blog_stop").hide();
			$("#follow_blog_start").html("&lt;a href=# onclick=\"FollowBlog('start'); return false;\"&gt;&lt;img src='https://markethive.com/images/follow_blog.png' border=0 title='Follow this blog!'&gt;&lt;/a&gt;");
			$("#follow_blog_start").show();
		}
		document.body.style.cursor = "default";
	});

}


function SocialAction(id,social) {

	console.log("SocialAction("+id+","+social+")");

	$("#comment_social_"+id).html("&lt;span style='padding-right:10px'&gt;&lt;img src=https://markethive.com/images/working.gif&gt;&lt;/span&gt;");

	document.body.style.cursor = "wait";

	$.post("https://markethive.com/blogs_query.php", {
		action: "comment_social",
		id: id,
		social: social
	},
	function(response) {
		var json = $.parseJSON(response);
		$("#comment_social_"+id).html(json.social_links);
		if (social == "like" || social == "unlike") {
			$("#comment_likes_"+id).html(json.likes_total);
		} else if (social == "dislike" || social == "undislike") {
			$("#comment_dislikes_"+id).html(json.dislikes_total);
		}
		document.body.style.cursor = "default";
	});

}
