	function performLogin(inForm, modalIdPrefix)
	{
		//Expected that modalIdPrefix contains the '#' already in it
		//Caching these so that we don't have to calculate them every time
		var errorElement            = $(modalIdPrefix + '_error'),
		    closeElement            = $(modalIdPrefix + '_closeDialog'),
		    signUpTodayElementId    = modalIdPrefix + '_signUpToday',
		    createPasswordElementId = modalIdPrefix + '_createPassword',
		    errorText = '';
		
		errorElement.hide();
			
		if (! isValidEmail(inForm.emailAddress.value))
		{
			errorText += "Please enter a valid email address&lt;br&gt;";
		}
		
		if ( (isBlank(inForm.password.value)) || (inForm.password.value == 'Password') )
		{
			errorText += "Please enter a password&lt;br&gt;";
		}
		
		//Nonempty string will evaluate to truthy, i.e. they messed something up
		if (errorText)
		{
			// slap them around...
			// alert(errorText);
			errorElement.html(errorText);
			errorElement.show();
			return false;
		}
		
		// GoogleQueueTrackEvent("Engagement", "Login", "Header Submit" );
		var url = inForm.visitor_services_url.value;

		$.ajax({
		  type: 'POST',
		  url: url,
		  data: {
					'emailAddress'		: inForm.emailAddress.value,
					'password'			: inForm.password.value,
					'setSavedItemCount'	: "true"
				},
		  dataType: 'json',
		  error: function(xhr, error)
		  {
			errorElement.html('An error occurred while processing your request.');
			errorElement.show();
			
			// alert('performLogin: 5: statusCode: ' 	+ xhr.statusCode);
			// alert('performLogin: 5: status: ' 		+ xhr.status);
			// alert('performLogin: 5: statusText: ' 	+ xhr.statusText);
		  },
		  success: function(genericResultBean) 
			{
				if (genericResultBean.status == "SUCCESS")
				{
					
					loginRecentlyPerformed = true;
					
					
					
					
					
						jCookie('com.equinox.wedding_jack.gui.cookie.VisitorLoginCookie',      genericResultBean.resultObject.visitor.id+'~'+genericResultBean.resultObject.visitor.emailAddress, 	{ expires: 1460, path: '/' });
						jCookie('com.equinox.wedding_jack.gui.cookie.VisitorSavedItemsCookie', genericResultBean.resultObject.visitor.genericData05, 												{ expires: 1460, path: '/' });
					
					
					// close the modal login layer
					closeElement.click();

					// there is probably an "addEntryArticleToList" call that wants to be made now that we have logged in
					if(!(isBlank($('#signin_popup_post_login_method').val())))
					{ 
						// If there is a function that needs to be called on login (wowlist add)
						eval($('#signin_popup_post_login_method').val());
					}
					else 
					{
						document.location.reload(true);
					}
				}
				else
				{
					if (genericResultBean.resultObject.status == "email_address_not_found")
					{
						errorElement.html('The email address you entered is not subscribed to PureWow.com&lt;br&gt;&lt;br&gt;&lt;a class="popup_error_message lightbox" href="" onclick="$(\''+signUpTodayElementId+'\').click(); return false;"&gt;CLICK HERE TO SIGN-UP&lt;/a&gt;');
						errorElement.show();
						// alert('the email address you entered is not subscribed to PureWow.com');
					}
					if (genericResultBean.resultObject.status == "wrong_password")
					{
						errorElement.html('You entered an incorrect password');
						errorElement.show();
						// alert('you entered an incorrect password');
					}
					if (genericResultBean.resultObject.status == "successful_but_cookies_disabled")
					{
						errorElement.html('Cookies must be enabled to login to PureWow.com');
						errorElement.show();
						// alert('Cookies must be enabled to login to PureWow.com');
					}
					if (genericResultBean.resultObject.status == "inactive")
					{
						errorElement.html('The email address you entered is not subscribed to PureWow.com&lt;br&gt;&lt;br&gt;&lt;a class="popup_error_message lightbox" href="" onclick="$(\''+signUpTodayElementId+'\').click(); return false;"&gt;CLICK HERE TO SIGN-UP&lt;/a&gt;');
						errorElement.show();
						// alert('the email address you entered is not subscribed to PureWow.com');
					}
					if (genericResultBean.resultObject.status == "no_password")
					{
                        errorElement.html('You do not yet have a password for PureWow.com &lt;br&gt;&lt;br&gt;&lt;a href="#" onclick="$(\''+createPasswordElementId+'\').click();"&gt;CLICK HERE TO CREATE YOUR PASSWORD&lt;/a&gt;');
                        errorElement.show();
					}
					if (genericResultBean.resultObject.status == "failure")
					{
						errorElement.html('An error occurred during login');
						errorElement.show();
						// alert('An error occurred during login');
					}
				}
			}
		})
		
		return true;
	}

    
    function loginToFacebook(modalIdPrefix) 
    {
        FB.login(function(response) 
        {
            if (response.authResponse) 
            {
                
                loginRecentlyPerformed = true;
                        
                // close the modal login layer
                $(modalIdPrefix+'_closeDialog').click();
    
                // connected
                $("#login_button").hide();
                
                FB.api('/me', function(user) 
                {
                    facebookUserEmail 	= user.email;
                    facebookFirstName 	= user.first_name;
                    facebookLastName 	= user.last_name;
                    facebookGender 		= user.gender;
                    
                    if ($.cookie('com.equinox.wedding_jack.gui.cookie.FacebookLoginCookie') == null)
                    {
                        console.log( 'first time setting FB cookie' );
                    }

                    
                    $.cookie('com.equinox.wedding_jack.gui.cookie.FacebookLoginCookie', facebookUserEmail, { path: '/' });
                    
                    
                    var fbEmailStatusDeferred = CheckEmailStatusExtended('', facebookUserEmail );
                    fbEmailStatusDeferred.done(function(data) 
                    {
                        if(data.statusExtended == 'failure')
                        {
                            $('#signin_resolution_error').html('An error occurred. Please try again.');
                            $('#signin_resolution_error').show();
                            return;
                        }
                        
                        if(data.statusExtended == 'not_member')
                        {
                            
                            $('.popup_login_fb_email').html(facebookUserEmail); 
                            $('#popup-login').hide(); 
                            $('#popup_login_resolve_fb').show(); 
                        }
                        else
                        {
                            if(data.statusExtended == 'inactive_member' || data.statusExtended == 'inactive_member_by_fb_email')
                            {
                                ReactivateEmailAddress('', facebookUserEmail );
                            }
                            
                            if(data.statusExtended == 'active_member_by_fb_email' || data.statusExtended == 'inactive_member_by_fb_email')
                            {
                            }
                            
                            
                            var wowCountDeferred = getWowListCount('', facebookUserEmail );
                            wowCountDeferred.done(function(data) 
                            {
                                
                                    $.cookie('com.equinox.wedding_jack.gui.cookie.VisitorSavedItemsCookie', data.resultObject, { expires: 1460, path: '/' });
                                
                                
								// there is probably an "addEntryArticleToList" call that wants to be made now that we have logged in
								if(!(isBlank($('#signin_popup_post_login_method').val())))
								{ 
									// If there is a function that needs to be called on login (wowlist add)
									eval($('#signin_popup_post_login_method').val());
								}
                                else 
                                {
                                    // this will force all the FB login stuff to fire
                                    document.location.reload(true);
                                }
                            });
                            
                        }
                    });
                    fbEmailStatusDeferred.fail(function(data) 
                    {
                        $('#signin_resolution_error').html('An error occurred. Please try again');
                        $('#signin_resolution_error').show();
                        return;
                    });
                });
            } 
            else 
            {
                // cancelled
                $("#FB_Info").html('login cancelled');
            }
        }, {scope: 'email'});
    }
