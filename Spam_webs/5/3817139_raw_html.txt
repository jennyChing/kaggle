	function daysInFebruary( year ){
		var days = (((year % 4 == 0) &amp;&amp; ( (!(year % 100 == 0)) || (year % 400 == 0))) ? 29 : 28 );
		return days;
	}
	function daysInMonth( month, year ) {
		var days = 31;
		switch( true ) {
			case month == 2 :
				days = daysInFebruary( year );
				break;
			case month == 4 || month == 6 || month == 9 || month == 11 :
				days = 30;
				break;
		}
	   return days;
	}
	function checkDate( pf, idDate ) {
		var year = 0;
		var month = 0;
		var day = 0;
		if (document.getElementById('adminForm_'+pf).elements['year_'+idDate]) {
			year = document.getElementById('adminForm_'+pf).elements['year_'+idDate].value;
		}
		if (document.getElementById('adminForm_'+pf).elements['month_'+idDate]) {
			month = document.getElementById('adminForm_'+pf).elements['month_'+idDate].value;
		}
		if (document.getElementById('adminForm_'+pf).elements['day_'+idDate]) {
			day = document.getElementById('adminForm_'+pf).elements['day_'+idDate].value;
		}
		if (day &gt; 0 &amp;&amp; month &gt; 0 &amp;&amp; year &gt; 0) {
			var days = daysInMonth( month, year );
			if (day &gt; days) {
				day = days;
				document.getElementById('adminForm_'+pf).elements['day_'+idDate].value = days;
				var error = 'Only %days% days in the selected month ! Please set the month first.';
				alert( error.replace( '%days%', days ) );
			}
		}
		if (document.getElementById('adminForm_'+pf).elements[idDate]) {
			document.getElementById('adminForm_'+pf).elements[idDate].value = year+'-'+month+'-'+day;
		}
	}

	function startUploadFile(file_field, pf) {
		var r_id = document.getElementById('adminForm_'+pf).elements['r_id'].value;

	function endUploadFile(pf, file_field, attachment_name, attachment_id, error_type, error_message) {
		error_type = parseInt(error_type);
		hideWaitFileField(file_field, pf);
		switch( error_type ) {
			case 0 :
				$('adminForm_'+pf).elements[file_field+'_attachment_name'].value = attachment_name;
				$('adminForm_'+pf).elements[file_field+'_attachment_id'].value = attachment_id;
				showFileField(file_field, pf);
				break;
			case 1 :
				alert('This type of attachement is not allowed ! ( '+error_message+' ) ');
				cancelUploadFile(file_field, pf);
				break;
			case 2 :
				alert('File too big ! ( '+error_message+' ) ');
				cancelUploadFile(file_field, pf);
				break;
			case 3 :
				alert('Other error ! ( '+error_message+' ) ');
				cancelUploadFile(file_field, pf);
				break;
		}
		resetSendButtonTarget(pf);
	}

        <h1>
         .Net Web Application Development
        </h1>
        <p>
         Since our inception, we have been dedicated towards innovating the web-based application area. We have created a niche in the .NET sphere and strive to provide constant value addition through continuous innovation.
        </p>
        <p>
         The .NET platform enables us to create powerful, cost-effective, innovative business solutions that ensure a higher return on investment. .Net framework helps in developing flexible and scalable solutions. Using .NET framework we develop applications that connect people,information,business processes, devices and systems. We believe in working along with the clients team to understand the business inside out. After obtaining a comprehensive know-how and understanding of the objectives, we deliberate to develop the best-fit solution.
        </p>
        <h2>
         About .NET Framework
        </h2>
        <p>
         Our expertise in the .NET Framework helps provide a foundation for leadership in web application development. In-depth understanding of the technology has allowed us to successfully develop applications using WCF, Silverlight, Entity Framework, LINQ and other components.
        </p>
        <p>
         <strong>
          Windows Communication Foundation (WCF)
         </strong>
         is a framework for building service-oriented applications. Using WCF, sending data from one service endpoint to another becomes simpler. An endpoint can be a client that requests data from a service endpoint.
        </p>
        <p>
         <strong>
          Microsoft Silverlight
         </strong>
         is a platform that helps in creating interoperable, rich web applications. This allows creation of media-intensive websites.
        </p>
        <p>
         <strong>
          The Entity Framework
         </strong>
         is a set of technologies in ADO.NET that supports the development of data-oriented software applications. It helps in creation and maintenance of data-oriented applications with minimal coding compared to traditional applications.
        </p>
        <p>
         <strong>
          Language Integrated Query (LINQ)
         </strong>
         defines a set of method names (called standard query operators, or standard sequence operators), along with translation rules from so-called query expressions to expressions using these method names, lambda expressions and anonymous types.
        </p>
        <p>
         <strong>
          Reactive Extensions
         </strong>
         for .NET Framework is a managed library that provides APIs for writing reactive applications.
        </p>
        <p>
         <strong>
          Windows Identity Foundation (WIF)
         </strong>
         is a framework for building identity-aware applications. It helps inexternalizing identity logic from their application, improving developer productivity, enhancing application security, and enabling interoperability.
        </p>
        <h2>
         Advantages of .NET Framework
        </h2>
        <p>
         The .NET Framework is preferred due to its flexibility and interoperability. Some of it's advantages are:
        </p>
        <ul>
         <li>
          Consistent Programming Model
         </li>
         <li>
          Direct Support for Security: A web application created using the .NET technology is considered more reliable as the web server controls the pages on an ongoing basis.
         </li>
         <li>
          Simplified Development Efforts: .Net reduces the amount of code necessary for building applications. It ensures better performance through just-in-time compilation, early binding, caching services and native optimization.
         </li>
         <li>
          Easy Application Deployment and Maintenance: It provides web application developers with automatic deployment, WYSIWYG (What You See Is What You Get) editing and drag-and-drop server controls.
         </li>
         <li>
          The .NET technology is language independent, which lets users choose the language that best suits their application requirements.
         </li>
        </ul>
        <h2>
         Offering Service
        </h2>
        <p>
         ESS offers the following services using .NET technologies.
        </p>
        <ul>
         <li>
          <strong>
           Web application development:
          </strong>
          .NET technology offers a superior &amp; seamless web experience. Web pages created with .NET perform common tasks such as form submission and client authorization with much ease.
         </li>
         <li>
          <strong>
           Intranet portal:
          </strong>
          Portals help employees find information quickly and increases productivity..NET has been at the forefront of portal development due to it'ssuperior integration and data management capabilities.
         </li>
         <li>
          <strong>
           Enterprise and desktop applications:
          </strong>
          .NET Framework helps in meeting the demands of enterprise computing.This software model integrates the power of web-based solutions with the distributed application model of traditional multi-tier client/server design.
         </li>
         <li>
          <strong>
           Mobile applications for Windows Mobile 7:
          </strong>
          .NET Framework makes it easier to develop mobile applications. Microsoft has integrated the .NET Mobile Web SDK directly into the Visual Studio .NET environment.
         </li>
        </ul>
