   var httpobject = [];

	function newHttpReq()
	{
		var newRequest = null;
		
		try
		{
			newRequest = new XMLHttpRequest();
		}
		catch(trymicrosoft)
		{
			try
			{
				newRequest = new ActiveXObject("Msxml2.XMLHTTP");
			}
			catch(othermicrosoft)
			{
				try
				{
					newRequest = new ActiveXObject("Microsoft.XMLHTTP");
				}
				catch(failed)
				{
					newRequest = null;
				}			
			}
		}
		
		return newRequest;
	
	}	
	
	function detectsaveddiv()
	{
		if(document.getElementById('saveddiv'))
		{
			fadecounter = 100;
			setTimeout("fadeoutprogress()",2500);
		}
	}
	
	function fadeoutprogress()
	{
		
		var object = document.getElementById('saveddiv');	
	
		object.style.filter = "alpha(opacity = " + fadecounter*1 + ")";
		object.style.opacity = fadecounter*0.01;
		
		fadecounter -= 1;
		
		if (fadecounter &gt; 0)
		{
			t = setTimeout("fadeoutprogress()", 10);
		}
		else
		{
			object.parentNode.removeChild(object);
		}
	}
	
	function selectallimages()
	{

		var sourceobject = document.getElementById('checkboxselectall');

		var inputs = document.getElementsByTagName('input');
		for(i = 0; i &lt; inputs.length; i++)
		{

			if(inputs[i].name.indexOf('checkboximage') == 0)
			{
				inputs[i].checked = sourceobject.checked;
			}
			
		}

		if(sourceobject.checked)
		{
			if(document.getElementById('selectalldiv'))
			{
				document.getElementById('selectalldiv').style.height = '30px';
				document.getElementById('selectalldiv').style.visibility = 'visible';
			}
		}
		else
		{
			if(document.getElementById('selectalldiv'))
			{
				document.getElementById('selectalldiv').style.height = '1px';
				document.getElementById('selectalldiv').style.visibility = 'hidden';
				document.getElementById('checkboxselectallallpages').checked = false;
			}		
		}
		
		checkformultipleimages();
	
	}
	
	function checkformultipleimages()
	{

		var imagecount = 0;
	
		var inputs = document.getElementsByTagName('input');
		for(i = 0; i &lt; inputs.length; i++)
		{

			if(inputs[i].name.indexOf('checkboximage') == 0)
			{
				if(inputs[i].checked == true)
				{
					imagecount += 1;
				}
			}
			
		}

		if(document.getElementById('selectalldiv'))
		{		
			if(document.getElementById('checkboxselectallallpages').checked == true)
			{
				imagecount = document.getElementById('imagecount').value;
			}
		}
		
		if(imagecount &gt; 1)
		{

				//document.getElementById('editmultipleimagesdiv').style.height = '25px';
				//document.getElementById('editmultipleimagesdiv').style.paddingTop = '15px';
				//document.getElementById('editmultipleimagesdiv').style.visibility = 'visible';
				document.getElementById('displayimagecount').innerHTML = imagecount;
				document.getElementById('menudiv').style.visibility = 'visible';

		}
		else
		{

			//document.getElementById('editmultipleimagesdiv').style.height = '1px';
			//document.getElementById('editmultipleimagesdiv').style.paddingTop = '0px';
			//document.getElementById('editmultipleimagesdiv').style.visibility = 'hidden';
			document.getElementById('menudiv').style.visibility = 'hidden';
	
		}
	
	}	

	
	// Add Image to Group (Group Moderator)
	function addimagetogroup(id)
	{
		var localvalue = requestcount;
		var localimageid = id;
		var poststring = 'action=addimagetogroup&amp;groupid=10563&amp;imageid=' + id;
				
		httpobject[localvalue] = newHttpReq();			
		httpobject[localvalue].onreadystatechange = function(){
		
			try 
			{
				if(httpobject[localvalue].readyState == 4)
				{
				}
			
				if(httpobject[localvalue].status == 200)
				{
				}
			}
			catch (e)
			{
				return;
			}
			
			if(httpobject[localvalue].readyState == 4 &amp;&amp; httpobject[localvalue].status == 200)
			{
				document.getElementById('imagediv[' + localimageid + ']').innerHTML = httpobject[localvalue].responseText;
				detectsaveddiv();
			}
		
		
		};		
		
		document.getElementById('imagediv[' + localimageid + ']').innerHTML = "&lt;img src='../Searching.gif' style='float: left;'&gt;";
		
		var url = '../queries/querygroupimagemanagement.php';	

		httpobject[localvalue].open('post',url,true);
		httpobject[localvalue].setRequestHeader('content-type', 'application/x-www-form-urlencoded');
		httpobject[localvalue].send(poststring);		
		
		requestcount += 1;	
	
	}
