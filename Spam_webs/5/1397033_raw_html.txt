      <h1>
       Jobs Rated 2013: Ranking 200 Jobs From Best To Worst
      </h1>
      <p>
       <img align="left" alt="The Many Faces of the Workplace" class="ccPrimaryPhoto ccImageStyle" src="/sites/default/files/workplace-snapshot.jpg" title=""/>
       <span>
        By CareerCast.com
       </span>
       <br/>
      </p>
      <p>
       THIS IS THE 2013 VERSION OF OUR REPORT. CLICK HERE TO READ THE
       <strong>
        <a href="http://www.careercast.com/jobs-rated/jobs-rated-2014-ranking-200-jobs-best-worst">
         2014 JOBS RATED REPORT
        </a>
       </strong>
       .
      </p>
      <p>
       Our annual Jobs Rated report celebrates its Silver Anniversary this year, 25 years removed from its inception in 1988.
      </p>
      <p>
       From the outset, the Jobs Rated report has generated a lot of attention primarily because most Americans spend more time on the job than they do on anything else, and knowing how their job ranks against other jobs is compelling information. The U.S. Bureau of Labor Statistics (BLS) estimates the average employee spends more than two-thirds of his or her day at work or on work-related activities. Thats more time than we spend sleeping or raising our children.
      </p>
      <p>
       The findings we present here are more than just curiosities. Unlike many other facets of life, the jobs we work at are choices we make; ones we can stick with or change. Therefore the measure of a job can tell us a lot about the quality of our lives.
      </p>

(function() {
    var win = aiJQuery(window);

    aiJQuery.fn.isOnScreen = function(){
        var bounds = this.offset(),
            viewport = {
                top: win.scrollTop(),
                left: win.scrollLeft()
            };

        if(bounds === null) {
            return false;
        }

        viewport.right = viewport.left + win.width();
        viewport.bottom = viewport.top + win.height();

        bounds.right = bounds.left + this.outerWidth();
        bounds.bottom = bounds.top + this.outerHeight();

        return (!(viewport.right &lt; bounds.left || viewport.left &gt; bounds.right || viewport.bottom &lt; bounds.top || viewport.top &gt; bounds.bottom));
    };

    var relatedArticleHook = aiJQuery('#related_content_hook');

    var checkShowRelatedArticles = function() {
        if(relatedArticleHook.isOnScreen()) {
            if(careercastVariables.relatedArticlesBlockId !== undefined) {
                var relatedArticlesBlock = aiJQuery('#' + careercastVariables.relatedArticlesBlockId).detach();

                relatedArticlesBlock.appendTo(document.body);

                relatedArticlesBlock.show();
                var blockHeight = relatedArticlesBlock.outerHeight(),
                    blockTop = 0;

                relatedArticlesBlock.css({position: 'absolute', "z-index": 100, width: '100%'});

                var inFullView = false,
                    shortPage = false,
                    scrollDir = 'down',
                    scrollMod = 0.5,
                    winHeight = win.height(),
                    winScrollTop = win.scrollTop(),
                    bodyHeight = aiJQuery(document.body).height(),
                    hookTop = relatedArticleHook.offset().top;

                if((hookTop &lt;= winScrollTop + winHeight - (blockHeight * 0.5)) || hookTop &lt; bodyHeight * scrollMod) {
                    relatedArticlesBlock.css({top: winHeight - blockHeight, position: 'fixed'});
                    inFullView = true;
                    blockTop = winScrollTop + winHeight - blockHeight;

                    if(hookTop &lt; bodyHeight * scrollMod) {
                        shortPage = true;
                    }
                } else {
                    blockTop = winScrollTop + winHeight;
                    relatedArticlesBlock.css({top: winScrollTop + winHeight, position: 'absolute'});
                }


                var handleRelatedArticlesBlockActive = function(ev) {
                    var evType = ev.type.toLowerCase(),
                        newScrollTop = win.scrollTop();

                    scrollDir = newScrollTop &gt; winScrollTop ? 'down' : 'up';

                    winScrollTop = newScrollTop; 

                    if(evType === 'resize' || shortPage || (scrollDir === 'down' &amp;&amp; !inFullView)) {
                        if(evType === 'resize') {
                            winHeight = win.height();
                            scrollMod = winScrollTop / bodyHeight;
                        }

                        if((evType === 'resize' &amp;&amp; inFullView) || (!inFullView &amp;&amp; blockTop + blockHeight &lt;= winScrollTop + winHeight)) {
                            relatedArticlesBlock.css({top: winHeight - blockHeight, position: 'fixed'});
                            inFullView = true;
                        }
                    } else if(inFullView &amp;&amp; scrollDir === 'up' &amp;&amp; winScrollTop + winHeight &lt; bodyHeight * scrollMod) {
                        blockTop = winScrollTop + winHeight - blockHeight;
                        relatedArticlesBlock.css({top: blockTop, position: 'absolute'});
                        inFullView = false;
                    }
                }

                win.bind('resize.relatedArticlesActive', handleRelatedArticlesBlockActive)
                win.bind('scroll.relatedArticlesActive', handleRelatedArticlesBlockActive);
            }

            relatedArticleHook.remove();
            win.unbind('load.relatedArticlesHook');
            win.unbind('resize.relatedArticlesHook');
            win.unbind('scroll.relatedArticlesHook');
            aiJQuery(document).unbind('ready.relatedArticlesHook');
        }
    }

    aiJQuery(document).bind('ready.relatedArticlesHook', checkShowRelatedArticles);
    win.bind('load.relatedArticlesHook', checkShowRelatedArticles);
    win.bind('resize.relatedArticlesHook', checkShowRelatedArticles);
    win.bind('scroll.relatedArticlesHook', checkShowRelatedArticles);
})();
