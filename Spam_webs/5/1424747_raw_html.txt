function submit_login_form()
{
	var url = "ajax_login.php";
	var params = "";

	var login_action = document.getElementById("login_action").value ;
	var login_email = document.getElementById("login_email").value ;
	var login_password = document.getElementById("login_password").value ;

	params="login_email="+login_email+"&amp;login_password="+login_password+"&amp;login_action="+login_action;

	if(window.XMLHttpRequest)
	{
		httpDealClose=new XMLHttpRequest();
	}
	else if(window.ActiveXObject)
	{
		httpDealClose=new ActiveXObject('msxml2.XMLHTTP');
	}
	httpDealClose.open("POST", url, true);

	//Send the proper header information along with the request
	httpDealClose.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
	httpDealClose.setRequestHeader("Content-length", params.length);
	httpDealClose.setRequestHeader("Connection", "close");

	httpDealClose.onreadystatechange = function() 
	{	//Call a function when the state changes.
		if(httpDealClose.readyState == 4 &amp;&amp; httpDealClose.status == 200)
		{			
						
			if(httpDealClose.responseText=='SUCCESS')
			{
				window.location.href="http://www.dentists4u.co.za/member-area";

				
				
			}else if(httpDealClose.responseText=='SUCCESS_FORGOTPASSWORD'){

				window.location.href="http://www.dentists4u.co.za/upgrade-password";
			
			}else{

				document.getElementById("login_msg_div_id").innerHTML=httpDealClose.responseText;		
			}
					
		}
	}
	httpDealClose.send(params);

}

	function LTrim( value ) {
		
		var re = /\s*((\S+\s*)*)/;
		return value.replace(re, "$1");
		
	}

	// Removes ending whitespaces
	function RTrim( value ) {
		
		var re = /((\s*\S+)*)\s*/;
		return value.replace(re, "$1");
		
	}

	// Removes leading and ending whitespaces
	function trim( value ) {
		
		return LTrim(RTrim(value));
		
	}
