  <script type="text/javascript">
   $('.inv_login_logout a.login').on('click', function(e)
	{
		var dialog, template, form;

		template = $('#login-template');
		form = _.template(template.html(), {});
		dialog = '&lt;div id="defaultModal" class="modal fade modal-gray" tabindex="-1"';
		dialog+= 'data-keyboard="true" style="display:none;" data-focus-on="input:first"&gt;'
		dialog+= '&lt;div class="modal-dialog"&gt;&lt;div class="modal-content"&gt;&lt;div class="modal-header"&gt;';
		dialog+= '&lt;button type="button" class="close" data-dismiss="modal" aria-label="Close"&gt;';
		dialog+= '&lt;span aria-hidden="true"&gt;&times;&lt;/span&gt;&lt;/button&gt;';
		dialog+= '&lt;h3 class="modal-title" id="myModalLabel"&gt;Sign in to your account&lt;/h3&gt;';
		dialog+= '&lt;p class="modal-title-tagline" style="margin:5px 0 0 0;"&gt;';
		dialog+= 'Please enter your email and password to log in.';
		dialog+= '&lt;span class="loginError text-danger pull-right"&gt;&nbsp;&lt;/span&gt;&lt;/p&gt;&lt;/div&gt;';
		dialog+= '&lt;div class="modal-body"&gt;&lt;/div&gt;&lt;/div&gt;&lt;/div&gt;&lt;/div&gt;';
		
		jqDialog = $(dialog);

		jqDialog.find('.modal-body').html(form);

		$('body').append(jqDialog);

		jqDialog
		.modal('show')
		.off('shown.bs.modal')
		.on('shown.bs.modal', function()
		{
			centerModals();

			var loginForm = $(this).find('form');

			loginForm.on('submit', function(e)
			{
				e.preventDefault();

				var formData = loginForm.serialize();
				
				var btnSubmit = loginForm.find('button[type="submit"]');
				
				btnSubmit.html('&lt;i class="fa fa-spinner fa-spin"&gt;&lt;/i&gt; ' + btnSubmit.html());

				$.post(loginForm.attr('action'), formData, function(r)
				{
					$('.error, .loginError').html('&nbsp;');

					if(r.success)
					{
						btnSubmit
						.html('&lt;i class="fa fa-check"&gt;&lt;/i&gt; Please wait, redirecting...');

						return location.href = r.redirect;
					}
					
					if(!r.success)
					{
						$('.loginError').html('&lt;i class="fa fa-warning"&gt;&lt;/i&gt; '+r.login);
					}
				}, 'json')
				.error(function(r)
				{
					$('.error, .loginError').html('&nbsp;');

					$.each(r.responseJSON, function(field, error)
					{
						$('[name='+field+']').prev('.error').html(error);
					});
				})
				.always(function(e)
				{
					btnSubmit.find('i.fa-spin').remove();
				});
			});

			loginForm.find('a.register').on('click', function(e)
			{
				var template, form;
				template = $('#register-template');
				form = _.template(template.html(), {});
				jqDialog.find('.modal-body').html(form);
				var regForm = jqDialog.find('form');
				regForm.off('submit')
				.on('submit', function(e)
				{
					e.preventDefault();

					var formData = regForm.serialize();
					
					var btnSubmit = regForm.find('button[type="submit"]');
					
					btnSubmit.html('&lt;i class="fa fa-spinner fa-spin"&gt;&lt;/i&gt; ' + btnSubmit.html());

					$.post(regForm.attr('action'), formData, function(r)
					{
						$('.error, .loginError').html('&nbsp;');

						if(r.success)
						{
							$.get(r.redirect, function(r)
							{
								var iconCheck = '&lt;i class="fa fa-check fa-2x text-success"&gt;&lt;/i&gt; ';
								jqDialog.find('.modal-title')
								.fadeOut()
								.html(iconCheck+$(r).find('.main-register').find('h2:first').html())
								.fadeIn();
								
								var html = $(r)
								.find('.main-register')
								.find('h2:first')
								.hide().end().find('hr').hide().end().html();

								jqDialog.find('.modal-body')
								.fadeOut()
								.html(html)
								.append('&lt;div class="clearfix"&gt;&lt;/div&gt;')
								.fadeIn();
							});
						}
						
						if(!r.success)
						{
							$('.loginError').html('&lt;i class="fa fa-warning"&gt;&lt;/i&gt; '+r.login);
						}
					}, 'json')
					.error(function(r)
					{
						$('.error, .loginError').html('&nbsp;');

						$.each(r.responseJSON, function(field, error)
						{
							$('[name='+field+']').prev('.error').html(error);
						});
					})
					.always(function(e)
					{
						btnSubmit.find('i.fa-spin').remove();
					});
				});
				
				e.preventDefault();
			});
		});
		
		e.preventDefault();
	});

	function centerModals()
	{
		$('.modal').each(function(i)
        {
            var $clone = $(this).clone().css('display', 'block').appendTo('body');
            var top = Math.round(($clone.height() - $clone.find('.modal-content').height()) / 2);
            top = top &gt; 0 ? top : 0;
            $clone.remove();
            $(this).find('.modal-content').animate({"margin-top":top}, 700);
        });
	}
