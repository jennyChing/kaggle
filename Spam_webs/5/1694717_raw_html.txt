function perez_set_category_retargeting_cookie(post_categories, site_version) {
    if (typeof (post_categories) == "undefined") {
        return;
    }

    var cookie_name = ph_retargeting_cookie_name;
    var cookie_value = "";
    var cookie_value_array;
    var cookie_value_current = $.cookie(cookie_name);
    var cookie_value_max_length = 70;
    var needed_categories = ph_retargeting_needed_categories;

    for (var i = 0; i &lt; perez_object_length(needed_categories); i++) {
        if ($.inArray(needed_categories[i]["cat_id"], post_categories) &gt;= 0 &amp;&amp;
                (!cookie_value_current || cookie_value_current.toLowerCase().indexOf("|" + needed_categories[i]["cat_id"] + ",") &lt; 0)) {
            cookie_value += "|" + needed_categories[i]["cat_id"] + ",";

            if (typeof (perez_statLog) == "function") {
                perez_statLog({
                    app: 'perez_category_retargeting',
                    options: 'd',
                    category: perez_retargeting_replace_cat_name(needed_categories[i]["name"]) + "-" + site_version
                });
            }

            perez_retargeting_call_tradedesk(needed_categories[i]["td_id"]);
        }
    }

    if (cookie_value !== "") {
        if (cookie_value_current) {
            cookie_value = cookie_value + cookie_value_current;
        }

        cookie_value_array = cookie_value.split(",");

        if (typeof (cookie_value_array) == "object" &amp;&amp; cookie_value_array.length &gt; cookie_value_max_length) {
            cookie_value_array = cookie_value_array.slice(0, cookie_value_max_length);
            cookie_value = cookie_value_array.join();
            cookie_value = cookie_value + ",";
        }

        perez_setCookie(cookie_name, cookie_value, 730);
    }
}

function perez_retargeting_replace_cat_name(cat_name) {
    cat_name = cat_name.replace(/ /gi, '_');
    cat_name = cat_name.replace(/\./gi, '_');
    cat_name = cat_name.replace(/\'/gi, '_');
    cat_name = cat_name.replace(/\"/gi, '_');

    return cat_name;
}
