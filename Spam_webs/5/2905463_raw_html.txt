		// slideshow callback before transition
		function onSlideShowBefore(curr,next,opts) {
		    var $slide = $(next);
		    var w = $slide.outerWidth();
		    var h = $slide.outerHeight();
		    $slide.css({
		        marginLeft: (280 - w) / 2,
		        marginTop: (150 - h) / 2
		    });
		 };

		// Activate tabbed panels
