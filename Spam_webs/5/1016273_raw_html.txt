	var tmpCalculateHeadline = function(headline) {
		
		jQuery(headline).hide();
		var thisWidth = jQuery(headline.parentNode).width();
		jQuery(headline).show();
		
		// Noch die Hoehe der Ueberschrift - bzw. dem Container - merken, damit, falls die Ueberschrift
		// geschrumpft wird, die Hoehe nicht kleiner wird, als sie mal war...
		var originalHeight = jQuery(headline.parentNode).height();
				
		var fontSize = currentSize = 18;
		var fontSizeLimit = 14;
		var headline = headline;
		
		var words = [];
		
		// Test-Element unsichtbar erstellen und darin immer die Groesse der Worter
		// messen ...
		var testEl = document.createElement(headline.nodeName);
		jQuery(testEl).css({ 'position':'absolute', 'visibility':'hidden' });
		jQuery(headline.parentNode).append(testEl);
		
		var checkWords = function() {
			
			words = headline.innerHTML.split(' ');
			
			var longestWord = false;
			var longestWordLetters = 0;
			
			for(var i = 0; i &lt; words.length; i++) {
	
				var tmpFontSize = Math.floor((thisWidth / jQuery(testEl).width()) * fontSize);
							
				if(tmpFontSize &lt; currentSize) currentSize = tmpFontSize;
				
				// Hier vorab schonmal merken, welches das laengste verfuegbare Wort
				// war ...
				if(words[i].length &gt; longestWordLetters) {
				
					longestWordLetters = words[i].length;
					longestWord = i;
					
					testEl.innerHTML = words[i];
				}
			};
			
			return longestWord;
			
		};
		
		var longestWord = checkWords();
		
		// Wenn die neue Fontgroesse das Limit nicht unterbietet, dann die Fontgroesse anpassen,
		// andernfalls die Worter hart in der Mitte trennen.
		if(fontSizeLimit &lt;= currentSize) {
			
			if(fontSize != currentSize) jQuery([headline, testEl]).css({ 'font-size' : currentSize + 'pt' });
			
		} else {
			
			// auf das Limit gesetzt (auch fuer den Testcontainer...).
			jQuery([headline, testEl]).css({ 'font-size' : fontSizeLimit + 'pt' });
			
			do {
				
				// Da nun das laengste Wort nicht in die Box passt, wird die Schriftgroesse
				var tmpWord = words[longestWord];
				
				for(var i = tmpWord.length; i &gt; 0; i-=1) {
				
					var shortWord = testEl.innerHTML = tmpWord.substr(0, i);
					
					if(jQuery(testEl).width() &lt;= thisWidth) break;
				};
				
				var firstWord = tmpWord.substr(0, shortWord.length) + ' ';
				var secondWord = tmpWord.substr(shortWord.length);
				
				words[longestWord] = firstWord + secondWord;
				
				headline.innerHTML = words.join(' ');
				
				currentSize = fontSize = fontSizeLimit;
				var longestWord = checkWords();
				
			} while(currentSize &lt; fontSizeLimit);
		}
		
		// Das &lt;h2&gt; per CSS-Klasse als modifiziert markieren, dann kann der Style noch etwas
		// angepasst werden, wenn es mehrzeilig wird.
		if(jQuery(headline).height() &gt; jQuery(testEl).height()) jQuery(headline).addClass("modified");
		
		// Nach der Messung das Testelement wieder entfernen.
		jQuery(testEl).remove();
		
		// Abschliessen noch ueberpruefen, ob die Hoehe des Containers nun geringer ist als urspruenglich,
		// wenn ja, dann wieder auf die Hoehe setzen!
		var tmpDiff = originalHeight - jQuery(headline.parentNode).height();
		
		if(tmpDiff &gt; 0) {
		
			var middle = Math.ceil(tmpDiff / 2);
			var rest = tmpDiff - middle;
			
			jQuery(headline).css({ 'padding-top':rest + 'px', 'padding-bottom':middle + 'px' });
		}
	};
	
	jQuery('#artistPageArtistHeadline').each(function() {
		
		tmpCalculateHeadline(this);
	});
