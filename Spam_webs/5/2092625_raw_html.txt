                });

                    $('#cid').val($('#CityId').val());
                });
                $('#ddlSociety').change(function() {
                    $('#Society').val($('#ddlSociety :selected').text());
                    $('#sid').val($('#ddlSociety').val());
                });
                
                $('#ddlBlock').change(function () {
                    $('#Block').val($('#ddlBlock :selected').text());
                    $('#bid').val($('#ddlBlock').val());
                });

                $('#ddlSociety').change(function() {
                    var Id = $(this).val();
                    if (Id != -1) {
                        $.ajax({
                            type: "POST",
                            url: "/Home/GetBlocks",
                            data: "{Parent:'" + Id + "'}",
                            contentType: "application/json; charset=utf-8",
                            dataType: "json",
                            success: function(data) {

                                if (data == null) {
                                    $('#ddlBlock').hide();
                                } else {
                                    BindBlock(data);
                                    $('#ddlBlock').show();
                                }
                            }
                        });
                    } else {
                        $('#ddlBlock').hide();
                    }
                });                

            });
            
            
            function BindSociety(msg) {
                $("#ddlSociety").empty();
                $.each(msg, function () {
                    $("#ddlSociety").append($("&lt;option&gt;&lt;/option&gt;").val(this['Id']).html(this['Description']));
                });
            }
            function BindBlock(msg) {
                $("#ddlBlock").empty();
                $.each(msg, function () {
                    $("#ddlBlock").append($("&lt;option&gt;&lt;/option&gt;").val(this['Id']).html(this['Description']));
                });
