			function rotate() {
				var current, next, sibs;

				current = $('.rotator ul li.show') ? $('.rotator ul li.show') : $('.rotator ul li:first');

				if (current.length == 0)
					current = $('.rotator ul li:first');

				if (current.next().length) {
					next = current.next().hasClass('show') ? $('.rotator ul li:first') : current.next();
				} else {
					next = $('.rotator ul li:first');
				}

				sibs = current.siblings();

				// Set the fade in effect for the next image, the show class has higher z-index
				next
					.css({opacity: 0.0})
					.addClass('show')
					.animate({opacity: 1.0}, 1000);

				// Hide the current image
				current
					.animate({opacity: 0.0}, 1000)
					.removeClass('show');
			};

				var el = $(this);
				var href = $(this).attr('href');
				if(typeof href != 'undefined') {
					var to = el.data('loading-timeout');
					to = 1*to;
					if(!(to&gt;0)) { to = 20; }
					var html = el.html();
					var d = 'disabled';
					//This is in a setTimeout so the click event processes while there's still a href
					setTimeout(function() {
						el.addClass(d).removeAttr('href').html(el.data('loading-text'));
					});
					setTimeout(function() {
						el.removeClass(d).attr('href', href).html(html);
					}, 1000*to);
				}
				return true;
			});

			$(document).on('click', 'input\[data-loading-text\]', function(ev) {
				var el = $(this);
				var to = el.data('loading-timeout');
				to = 1*to;
				if(!(to&gt;0)) { to = 20; }
				var val = el.val();
				var d = 'disabled';
				// This is in a setTimeout so the click event processes while the button has its original value
				setTimeout(function() {
					el.addClass(d).attr(d, d).val(el.data('loading-text'));
				});
				setTimeout(function() {
					el.removeClass(d).removeClass(d).val(val);
				}, 1000*to);
				return true;
			});

			// bootstrap handles these
			$(document).on('click', 'button\[data-loading-text\]', function(ev) {
				var el = $(this);
				var to = el.data('loading-timeout');
				to = 1*to;
				if(!(to&gt;0)) { to = 20; }
				setTimeout(function() {
					el.button('loading');
				},0);
				setTimeout(function() {
					el.button('reset');
				}, 1000*to);
				return true;
			});

			// Debounce duplicate form submissions 
			$('form').on('submit', function(ev) {
				var $form = $(this);

				if ($form.data('submitted') === true) {
					// Already submitted
					ev.preventDefault();
				} else {
					// Mark the form as submitted so multiple submissions
					// from double clicks are ignored
					$form.data('submitted', true);
					setTimeout(function() {
						$form.data('submitted', false);
					}, 500);
				}
			});
