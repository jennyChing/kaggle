            (e.keyCode == 65 &amp;&amp; e.ctrlKey === true) ||
             // Allow: home, end, left, right
            (e.keyCode &gt;= 35 &amp;&amp; e.keyCode &lt;= 39)) {
                 // let it happen, don't do anything
                 return;
        }
        // Ensure that it is a number and stop the keypress
        if ((e.shiftKey || (e.keyCode &lt; 49 || e.keyCode &gt; 57)) &amp;&amp; (e.keyCode &lt; 96 || e.keyCode &gt; 105)) {
            e.preventDefault();
            return;
        }
    });

function scrollQty(scroll) {
	if (jQuery("#HDP_QB_QTY")[0].value == '') {
		jQuery("#HDP_QB_QTY")[0].value = 1;
	}
	var current = parseInt(jQuery("#HDP_QB_QTY")[0].value);
	switch (scroll) {
	   case -1:
		   current -= 1;
		   if (current &lt; 1) {
			   return;
		   }
		   break;
	   case 1:
		   current += 1;
		   if (current &gt; 9) {
			   return;
		   }
		   break;
	   default:
		   break;
	}
	skuSelect.setQty(current);
	jQuery("#HDP_QB_QTY")[0].value = current;
}
