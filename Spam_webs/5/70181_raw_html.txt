       <div class="article-text clearfix">
        <span>
         <p>
          It seems as though everywhere we turn, Brussels sprouts are appearing alongside their ubiquitous counterpart, bacon. George Sabatino of
          <a href="http://statesidephilly.com" target="_blank" title="Stateside">
           Stateside
          </a>
          in Philadelphia insists that the hearty green vegetable, though, can stand tall without the bacon boost. In Sabatino's dish, oyster mushrooms and roasted chestnuts offer texture and earthy underpinnings, while red pepper flakes, honey and sherry vinegar have the tiny cabbages popping with zippy flavor. It's a rich, handsome side, with nary a bacon piece in sight.
         </p>
        </span>

       <div class="recipe-title">
        <h2>
         <span class="" id="name" itemprop="name">
          Brussels Sprouts with Chestnut Brown Butter
         </span>
        </h2>
        <p class="author">
         <span itemprop="author">
          Recipe adapted from George Sabatino, Stateside, Philadelphia, PA
         </span>
        </p>
       </div>
       <!-- INGREDIENT COLUMN -->
       <div class="yield-and-ingredients" id="yield-and-ingredients">
        <p class="ing-title">
         <strong>
          Yield:
         </strong>
         <span itemprop="recipeYield">
          Serves 4
         </span>
        </p>
        <p class="ing-title second-block">
         <br/>
         <strong>
          Cook Time:
         </strong>
         <span content="PT20M" itemprop="cookTime">
          20 minutes
         </span>
        </p>
        <ul id="ingreed" style="margin-left:0px;">
         <li id="ing-title">
          <strong>
           INGREDIENTS
          </strong>
         </li>
         <p itemprop="ingredients">
          2 pounds Brussels sprouts, trimmed and ends scored with an "X"
         </p>
         <p itemprop="ingredients">
          Salt and freshly ground black pepper
         </p>
         <p itemprop="ingredients">
          1 tablespoon plus 1 teaspoon extra-virgin olive oil, divided
         </p>
         <p itemprop="ingredients">
           pound oyster mushrooms, stemmed and sliced
         </p>
         <p>
          2 tablespoons
          <a href="javascript:showGlossaryTerm(11727,1);">
           sherry vinegar
          </a>
          , divided
         </p>
         <p itemprop="ingredients">
          2 tablespoons unsalted butter, at room temperature
         </p>
         <p itemprop="ingredients">
          2 garlic cloves, thinly sliced
         </p>
         <p itemprop="ingredients">
          1 tablespoon red pepper flakes
         </p>
         <p itemprop="ingredients">
          1 teaspoon honey
         </p>
         <p itemprop="ingredients">
          4 jarred roasted chestnuts, coarsely chopped
         </p>
         <p itemprop="ingredients">
          1 tablespoon water
         </p>
         <p itemprop="ingredients">
          Flaky sea salt
         </p>
        </ul>
       </div>
       <!-- END INGREDIENT COLUMN -->
       <!-- DIRECTIONS -->
       <div class="directions-and-share" id="directions-and-share">
        <span class="dir">
         DIRECTIONS
        </span>
        <span id="instructions" itemprop="recipeInstructions">
         <p>
          1. Fill a large bowl with ice and water and set aside. Cook the Brussels sprouts in a large pot of liberally salted boiling water over high heat until the sprouts are light green, 3 to 4 minutes. Using a slotted spoon, transfer the sprouts to the ice-water bath and cool for 10 minutes. Drain and halve lengthwise.
          <br/>
          <br/>
          2. In a large skillet set over high heat, heat 1 tablespoon of the olive oil until smoking, about 3 minutes. Add the mushrooms and cook, without stirring, for 1 minute, then shake the pan and continue to brown the mushrooms for 2 minutes more. Season with salt and pepper; add 1 tablespoon of the sherry vinegar and stir with a wooden spoon, scraping up the browned bits from the bottom of the pan. Remove the mushrooms from the pan and set aside to cool on a baking sheet lined with a paper towel.
         </p>
         <p>
          3. Set the skillet back over high heat and add the remaining teaspoon of olive oil. Place the Brussels sprouts in the pan, cut side down; season with salt and pepper and cook until brown, 5 to 7 minutes. Add the butter, garlic, red pepper flakes and the reserved mushrooms and cook until the butter is browned, about 1 minute. Stir in the honey and chestnuts, season with salt and pepper add the remaining 1 tablespoon of sherry vinegar and the water and stir to combine. Transfer to a serving dish, finish with sea salt and serve hot.
         </p>
        </span>
       </div>

// returns true if a domain DISALLOW.
// TT starting out allowing certain domains, but we had so many
// folks in Europe that wanted to subscribe, we decided to perform
// the opposite check and only disallow a handful of domains
//////////////////////////////////////////////////////////////////
function endsWithBadDomain(s) 
{
	var i = 0;
	if (s.length == 0)
	{
		return false;
	}
	
	// IN-valid domains:
	// 	see this URL for details: http://www.iana.org/domains/root/db/
	var domainArray = new Array(
							".cn",
							".conm",
							".con",
							".comm",
							".cn"
							);

	for (i=0; i&lt;domainArray.length; i++)
	{
		if (s.toLowerCase().match(domainArray[i]+"$")==domainArray[i])
		{
			return true;
		}
	}
	
	return false;
}

//////////////////////////////////////////////////////////////////
// Determines if the input string represents a syntactically correct
// email address.
//////////////////////////////////////////////////////////////////
function isValidEmail(addr) 
{
// return isValidEmailServer(addr);

	// make sure we just have 1 '@'
	var tokens = addr.split('@');
	
   if ((addr.indexOf("@")  == -1) 
   ||  (addr.indexOf(".")  == -1) 
   ||  (addr.indexOf(" ")  != -1) 
   ||  (addr.indexOf(",")  != -1) 
   ||  (addr.indexOf(";")  != -1) 
   ||  (addr.indexOf(":")  != -1) 
   ||  (addr.indexOf("/")  != -1) 
   ||  (addr.indexOf("\\") != -1) 
   ||  (addr.indexOf("!")  != -1) 
   ||  (addr.indexOf("@@") != -1) 
   ||  (addr.indexOf("..") != -1) 
   ||  (addr.indexOf("@.") != -1) 
   ||  (addr.indexOf(".@") != -1) 
   ||  (addr.indexOf(")")  != -1) 
   ||  (addr.indexOf("(")  != -1) 
   ||  (addr.indexOf("#")  != -1) 
   ||  (addr.indexOf("[")  != -1) 
   ||  (addr.indexOf("]")  != -1) 
   ||  (addr.indexOf("{")  != -1) 
   ||  (addr.indexOf("}")  != -1) 
   ||  (addr.indexOf("$")  != -1) 
   ||  (addr.indexOf("+")  != -1) 
   ||  (endsWithBadDomain(addr)  == true) 
   ||  (endsWithCommonMispelling(addr)  == true) 
   ||  (tokens.length != 2)
      ) 
   {
      return false;
   } 
   else 
   {
		return true;
   }
}

//////////////////////////////////////////////////////////////////
// Determines if the input string contains only whitespace.
//////////////////////////////////////////////////////////////////
function isBlank(s) {
   var i = 0;
   if (s == "") {return true;}
   for (i = 0; i &lt; s.length; i++) {
      var c = s.charAt(i);
      if ((c != " ") &amp;&amp; (c != '\n') &amp;&amp; (c != '\t')) {return false;}
   }
   return true; 
}

///////////////////////////////////////////////////////////////////////
// Trims leading/trailing spaces
//////////////////////////////////////////////////////////////////////
function trim(str) 
{ 
  str = str.replace( /^\s+/g, "" ); // strip leading 
  str = str.replace( /\s+$/g, "" ); // strip trailing 

  return str; 
} 

///////////////////////////////////////////////////////////////////////
// Validate the Ask-A-Question Form
//////////////////////////////////////////////////////////////////////
function validateSubmitAskAQuestion(inForm)
{ 
	var valid = true;
		
	errorText = "";
	
	inForm.emailAddress.value = trim(inForm.emailAddress.value);
	if (!isValidEmail(inForm.emailAddress.value))
	{
		errorText += 'a valid email address is required.';
		return;
	}	

	if ( isBlank(inForm.genericData03.value) )
	{
		errorText += 'please supply the city you are in';
		return;
	}	
	
	if ( isBlank(inForm.messageBody.value) )
	{
		errorText += 'a question or description of an issue is required ';
		return;
	}	
	
	if (errorText != '')
	{
		valid = false;
	}

	return valid;
}

// custom file input module
function customFileUpload(obj, opt) {
    if(obj) {
        this.options = {
            jsActiveClass:'file-input-js-active',
            fakeClass:'file-input-value',
            hoverClass:'hover'
        }
        this.fileInput = obj;
        this.fileInput.custClass = this;
        this.init();
    }
}
customFileUpload.prototype = {
    init: function() {
        this.getElements();
        this.setStyles();
        this.addEvents();
    },
    getElements: function() {
        this.fileInputParent = this.fileInput.parentNode;
        this.fileInputParent.className += ' ' + this.options.jsActiveClass;
        var tmpInputs = this.fileInput.parentNode.getElementsByTagName('input');
        for(var i = 0; i &lt; tmpInputs.length; i++) {
            if(tmpInputs[i].className.indexOf(this.options.fakeClass) != -1) {
                this.fakeInput = tmpInputs[i];
                this.fakeInput.readOnly = true;
                break;
            }
        }
    },
    getFileName: function(){
        return this.fileInput.value.replace(/^[\s\S]*(?:\\|\/)([\s\S^\\\/]*)$/g, "$1");
    },
    setStyles: function() {
        // IE styling fix
        if((/(MSIE)/gi).test(navigator.userAgent)) {
            this.tmpNode = document.createElement('span');
            this.fileInputParent.insertBefore(this.tmpNode,this.fileInput);
            this.fileInputParent.insertBefore(this.fileInput,this.tmpNode);
            this.fileInputParent.removeChild(this.tmpNode);
        }
        //this.fileInput.style.opacity = 0;
        //this.fileInput.style.filter = 'alpha(opacity=0)';
    },
    addEvents: function() {
        this.fileInput.onchange = this.bind(this.updateTitle,this);
        this.fileInput.onmouseover = this.bind(function(){
            this.fileInputParent.className += ' ' + this.options.hoverClass;
        },this);
        this.fileInput.onmouseout = this.bind(function(){
            this.fileInputParent.className = this.fileInputParent.className.replace(' '+this.options.hoverClass,'');
        },this);
    },
    updateTitle: function() {
        if(this.fakeInput) {
            this.fakeInput.value = this.getFileName();
        }
    },
    bind: function(func, scope) {
        return function() {
            return func.apply(scope, arguments);
        }
    }
}

else if (window.attachEvent) window.attachEvent("onload", initCustomFile);

    $(document).ready(function()

	function showGlossaryTerm(id, entry_type_id)
	{
		var glossarySelector = ' ';
		var url = '/entry_detail/national/' + id;
		
		// grab the populated div layer from the server
		$.ajax({
		  url:       url,
		  type:     'GET',
		  data:		{forcedViewName:	'entry_detail_style_14'},
		  dataType: 'html',
		  success: function(divHtml) 
					{
						$(glossarySelector + "#glossary_holder").html(divHtml);
						$(glossarySelector + "#glossary_closeX").click(function(){
							$(glossarySelector + '#glossary_holder').hide();
							$(glossarySelector + '#glossary_holder').removeClass('cardSlideDown');
						});
					}
		});	
		return false;
	
	}
	
	var glossarySelector = ' ';
	$(glossarySelector + 'a[href^="javascript:showGlossaryTerm"]').each(function()
	{
		// save the original href so we can use it as a hover action
		var oldHref = $(this).attr('href');
		
		// we don't need the "javascript:" since we are in JS already
		oldHref = oldHref.replace("javascript:","");
		
		var parenStart 	= oldHref.indexOf("(");
		var parenEnd 	= oldHref.indexOf(")");
		var paramLength	= (parenEnd-parenStart);
		var params		= oldHref.substr(parenStart+1, paramLength-1);
		var paramArray = params.split(',');
		var glossaryTermId 		= paramArray[0];
		var glossaryTermType 	= paramArray[1];
		
		// add an ID we can use to find this term for cases when we get here from a term click in an email
		// and need to position the pop-up on the term
        $(this).attr('id', 'hoverId_'+glossaryTermId);
		
		// remove the original href event
        $(this).attr('href', 'javascript:void(0);');
		
		
			// use the href value to create mouseIn, mouseOut (null) functions
			$(this).hover(
			  function (e) 
			  {
				// need to "eval" it because the function and paramaters are stored as a string
				eval(oldHref);
				$("#glossary_holder").show();
				var leftPos = e.pageX;
				var topPos  = e.pageY;
				$('#glossary_holder').css({position:'absolute', left:leftPos, top:topPos});
			  },
			  function () 
			  {
					// don't want any mouseOut action
			  }
			);
		
		
		
		//Style the link with a dotted line
		$(this).css('text-decoration', 'none');
		$(this).css('border-bottom', '1px dotted');		
		if (glossaryTermType == '2')
		{
			$(this).css('background', '#f2f2f2');
		}
    });
	
	
	
	
});
