function updateGridButton(iGridNumber, sState){
    if (sState == "REMOVE"){
        $(".frmProductRemove").eq(iGridNumber-1).hide();
        $(".frmProductGrid input[name='btnUpdateCart']").eq(iGridNumber-1).hide();
        $(".frmProductGrid input[name='frmProductRemove']").eq(iGridNumber-1).hide();
        $(".frmProductGrid input[name='btnAddToCart']").eq(iGridNumber-1).fadeTo('slow',1).removeAttr('disabled');
    } else if (sState == "ADD") {
        $(".frmProductGrid input[name='btnAddToCart']").eq(iGridNumber-1).hide();
        $(".frmProductRemove").eq(iGridNumber-1).fadeTo('slow',1);
        $(".frmProductGrid input[name='btnUpdateCart']").eq(iGridNumber-1).fadeTo('slow',1).removeAttr('disabled');
        $(".frmProductGrid input[name='frmProductRemove']").eq(iGridNumber-1).fadeTo('slow',1).removeAttr('disabled');
    } else if (sState == "ADDING") {
        if ($(".frmProductGrid input[name='btnAddToCart']").eq(iGridNumber-1).css("display") != 'none') {
            $(".frmProductGrid input[name='btnAddToCart']").eq(iGridNumber-1).attr('disabled', 'disabled').fadeTo('slow',0.3);
        } else {
            $(".frmProductGrid input[name='btnUpdateCart']").eq(iGridNumber-1).attr('disabled', 'disabled').fadeTo('slow',0.3);
        }
    } else if (sState == "REMOVING"){
        $(".frmProductGrid input[name='btnUpdateCart']").eq(iGridNumber-1).fadeTo('slow',0.3);
        $(".frmProductGrid input[name='frmProductRemove']").eq(iGridNumber-1).fadeTo('slow',0.3);
    }
}
