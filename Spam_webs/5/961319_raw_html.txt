			$("#userLogin-form").validate(
			{
				debug: true,
				rules: {
					"password": {
						required: true,
						notDefault: true
					},
					"username": {
						required: true,
						notDefault: true
					}
				},
				invalidHandler: function(form, validator) {
				    $("#userLogin-form .error-message").show();
				    $("#userLogin-form input").addClass("error");
				},
				errorPlacement: function(error, element) {},
				messages:{
					"password": "Please enter your password",
					"username" : "Please enter your username/email address"
				},
				submitHandler: function(form) {
					// do other stuff for a valid form
					$.post("application/views/ajax//userLogin.php", $("#userLogin-form").serialize(), function(data) 
					{
						if(data == "badLogin" || data == "emptyForm" || data == "emptyFields")
						{
							$("#userLogin-form .error-message").show();
							$("#userLogin-form input").addClass("error");
						}
						else if(data == "success")
						{
							$("#userLogin-form .error-message").hide();
							window.location.replace("application/views/ajax/redirectLogin.php?view=club&amp;q=&amp;id=5466");
						}
						else
						{
							alert("Sign In Error - Please contact the webmaster. Response: "+data);
						}
					});
				}
			});
			
			///// PASSWORD RESET SUBMISSION /////
			$("#forgotPassword-form").validate({
				debug: true,
				rules: {
					"email": {
						required: true,
						email: true
					}
				},
				messages:{
					"email" : "&lt;i class=\"icon-attention\"&gt;&lt;/i&gt;Please enter your email address"
				},
				errorPlacement: function(error, element) {},
				submitHandler: function(form) 
					{
						var submitTarget = "application/views/ajax/processForgottenPassword.php";
						// do other stuff for a valid form
						$.post(submitTarget, $("#forgotPassword-form").serialize(), function(data)
						{
							if(data == "success")
							{
								$("#email-error-label").fadeOut(100);
								$("#email-password-button").hide();
								$("#forgotPassword-form input").hide();
								$("#forgotPassword-form #description").css("width", "300px");
								$("#forgotPassword-form #description").html("Thank you. Please check your email inbox for password reset instructions.");
								$("#forgotPassword-form .cancel.button .text").html("Back");
							}
							else
							{
								alert(data);
								$("#email-error-label").fadeIn(100);
								$("#userLogin-form input").addClass("input-error");
							}
						});
					}
				});
		});

      <div class="description f3 cf4">
       <p>
        <strong>
         About Amaze (Formerly Autism Victoria)
        </strong>
       </p>
       <p>
        Amaze is a member-based not-for-profit organisation and is the peak body for Autism Spectrum Disorder (ASD) in the state of Victoria.
       </p>
       <p>
        Amaze represents around 55,000 Victorians who have ASD and works for the benefit of all individuals and their families and to promote better understanding of ASD in the general community.
       </p>
       <p>
        <strong>
         Our Aim
        </strong>
       </p>
       <p>
        Our aim is to improve the quality of life for people affected by Autism Spectrum Disorder, their family and carers, and to provide a range of services to individuals with ASD and the agencies that work with them.
       </p>
       <p>
        <strong>
         Our History
        </strong>
       </p>
       <p>
        In 1967, a group of parents who were struggling to find ASD resources, set up the Victorian Autistic Childrens and Adults Association. It provided a central information source and supported individuals and their families. Since then, the organisation has undergone much change, becoming Autism Victoria Inc in 1999.
       </p>
       <p>
        In September 2011, Autism Victoria began to trade as Amaze:
       </p>
       <ul>
        <li>
         sometimes life with ASD can seem like a maze and
        </li>
        <li>
         sometimes life with ASD can amaze!
        </li>
       </ul>
       <p>
        At the AGM in 2013, members voted for the organisation to change its name formally to Amaze Inc.
       </p>

        function preloadListing()
		{
			var preloadOffset = parseInt(sessionStorage.getItem("offset"));
			var listSize = preloadOffset;
			var currentSource = sessionStorage.getItem("source");
			
			if(preloadOffset &gt; 0 &amp;&amp; currentSource == "club")
			{
				$("input#scrollLoading").val("true");
				
				var get_id = "5466";
				var get_source = "club";
				var get_sort = "auctionName";
				var get_sortOrder = "ASC";
			
				var targetURL = "application/views/ajax/auctionListingSegment.php?listSource=" + get_source + "&amp;listSize=" + listSize + "&amp;listOffset=0&amp;&amp;sort=" + get_sort + "&amp;sortOrder=" + get_sortOrder + "&amp;id=" + get_id;
				
				$.ajax({
					url: targetURL,
					type: "GET",
					success: function(data)
						{
							if(data != "endOfListing")
							{
								$("table#auction-table").find("tr:gt(0)").remove();
								$("table#auction-table tr:first").after(data);
								var newOffset = preloadOffset + 10;
								
								$("input#scrollLoading").val("false");
								sessionStorage.setItem("offset", newOffset);
							}
						}
					});
			}
			else
			{
				try
				{
					sessionStorage.setItem("offset", 10);
				}
				catch(e)
				{
					alert("Auction listings require that private browsing mode be disabled. Please disable it in your browser settings and try again.");
				}
			}
			
			sessionStorage.setItem("source", "club");
		}

				$("#listing-ajax").load(target, function(response, status, xhr)

				});

		function actionScroll()
		{
			var scrollLoading = $("input#scrollLoading").val();

			var get_id = "5466";
			var get_source = "club";
			var get_listSize = 10;
			var get_offset = sessionStorage.getItem("offset");
			var get_sort = "auctionName";
			var get_sortOrder = "ASC";
			var newOffset = parseInt(get_offset);
			
			var y_scroll_pos = window.pageYOffset;
			var scroll_pos_test = Math.round($("#auction-table tbody").height() * 0.78);
			
			if(isScrolledIntoView("#auction-table&gt;tbody&gt;tr:last-child") &amp;&amp; $("input#scrollLoading").val() == "false")
			{
				var targetURL = "application/views/ajax/auctionListingSegment.php?listSource=" + get_source + "&amp;listSize=" + get_listSize + "&amp;listOffset=" + newOffset + "&amp;sort=" + get_sort + "&amp;sortOrder=" + get_sortOrder + "&amp;id=" + get_id;

				$("input#scrollLoading").val("true");
				$("#listing-loading").show();
				$("#listing-loading").removeClass("hidden");
				
				$.ajax({
				url: targetURL,
				type: "GET",
				success: function(data)
					{						
						if(data != "endOfListing")
						{
							$("table#auction-table tbody").append(data);							
							get_offset = newOffset;
							newOffset = parseInt(get_offset) + 10;
							$("input#scrollLoading").val("false");
							
							sessionStorage.setItem("offset", newOffset);
						}

							$("#listing-end").removeClass("hidden");
						}

						$("#listing-loading").fadeOut(2000, function()
						{
							$("#listing-loading").addClass("hidden");
						});
					}
				});
			}
		}
