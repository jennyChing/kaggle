	function truncate_string(string, length) {
		var k;
		var _string = "";
		if (string.length &gt; length) {
			for (k=0; k&lt;length; k++) {
				_string += string.charAt(k);	
			}
			return _string + "...";
		}
		else { 
			return string; 
		}
