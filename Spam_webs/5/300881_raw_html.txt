	onFadeFinish : function() {
	
	  // get current module
    var el = this.getEl();        
    var container = D.getAncestorByClassName(el, 'flickr-container');
    var stringLength = "flickr-container-";
    var mod = container.id.substr(stringLength.length); 
    var that = Squarespace.Social.Flickr;

		if (that.nextSlideIndex[mod] == 0) {
		
			// jump to end
			that.nextSlideIndex[mod] = that.loadCounts[mod];
			that.activeSlideIndex[mod] = that.loadCounts[mod] + 1;
		
		} else if (that.nextSlideIndex[mod] == (that.loadCounts[mod] + 1)) {
			//jump to beginning
			that.nextSlideIndex[mod] =  1;
			that.activeSlideIndex[mod] = 0;
		}		
    
		// fade in next slide
		var slide = D.get('flickr-container-' + mod + '-' + that.nextSlideIndex[mod]);
		D.setStyle(slide, 'opacity', '0');
		var anim = new YAHOO.util.Anim(slide, { 
			opacity: { from: 0, to: 1 }
		}, that.FADE_SPEED);
		anim.onStart.subscribe(that.selectSlide);
		anim.animate();  
	},		
  
	animateSwipe : function(mod) {
	  // swipe out active slide
	 	var anim = new YAHOO.util.Motion(this.containerElements[mod], {
		    points: {
		        to: [this.containerXPositions[mod], D.getY(this.containerElements[mod])]
		    } 
		}, this.SWIPE_SPEED, YAHOO.util.Easing.easeOutStrong);
		anim.onComplete.subscribe(this.onSwipeFinish);
		anim.animate(); 
	},
	 
	onSwipeFinish : function() {
	
	  // get current module
	  var el = this.getEl();	  
	  var stringLength = "flickr-container-";
	  var mod = el.id.substr(stringLength.length);
	  var that = Squarespace.Social.Flickr;
    
    if (that.activeSlideIndex[mod] == 0) {

      // jump to post slide
      that.containerXPositions[mod] = D.getX(that.containerElements[mod]) - ( that.loadCounts[mod] * that.sidebarWidths[mod]);
      D.setX(that.containerElements[mod], that.containerXPositions[mod]);
      that.activeSlideIndex[mod] = that.loadCounts[mod];      
     
    } else if (that.activeSlideIndex[mod] == (that.loadCounts[mod] + 1)) {

      // jump to pre slide    
      D.setX(that.containerElements[mod], that.containerXOffsets[mod] - that.sidebarWidths[mod]);
      that.containerXPositions[mod] = D.getX(that.containerElements[mod]);
      that.activeSlideIndex[mod] = 1;

    }
    
	},
	
	doAutoPlay : function(mod) {

	  if (this.showSlideTransition[mod]) {
	  	  
  		if (this.isSlideTransitionFade[mod]) {
  			this.nextSlideIndex[mod] = this.activeSlideIndex[mod] + 1;
  			this.animateFade(mod);					
  			this.activeSlideIndex[mod] += 1;	
  		
  		} else if (this.isSlideTransitionSwipe[mod]) {
  			this.containerXPositions[mod] = this.containerXPositions[mod] - this.sidebarWidths[mod];
  			this.animateSwipe(mod);					
  			this.activeSlideIndex[mod] += 1;	
  		}  
  		
    } else {
    
        // no animation
        this.nextSlideIndex[mod] = this.activeSlideIndex[mod] + 1;
        this.animateNoTransition(mod);          
        this.activeSlideIndex[mod] += 1;
    }	
 	},
 	 
 	swapPausePlayButton : function(mod) {
		var buttonElement = D.get('flickr-nav-playPause-' + mod);

			buttonElement.src = "/universal/images/slideshow/pause-button.png";

			buttonElement.src = "/universal/images/slideshow/play-button.png";
		} 	 
 	 }
