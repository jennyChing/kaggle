function autoLightsoff(){
    idleTimer = null;
    idleState = false;
    idleWait  = 20000;
    $(document).bind('mousemove keydown scroll', function(){
        clearTimeout(idleTimer); //    
        if(idleState == true){
            $('#lightsoff-background')
            .fadeOut(function() {
                $('.lightsoff-background').remove();
                $('#player_code').css({'z-index' : 0 });
            });
        }
        idleState = false;
        idleTimer = setTimeout(function(){
            $('#player_code').css({ 'visibility' : 'visible', 'position' : 'relative', 'z-index' : 1999 });
            $('body').prepend('&lt;div class="lightsoff-background" id="lightsoff-background" title="'+LANG_CLICK_LIGHT_ON+'"&gt;&lt;/div&gt;');
            $('#lightsoff-background').css({ height: $(document).height() }).fadeIn();
            $('#lightsoff-background').show();
          idleState = true;
        }, idleWait);
    });
    $("body").trigger("mousemove");
}
