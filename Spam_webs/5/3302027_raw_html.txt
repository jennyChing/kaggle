    function ValidatePhoneNumber(phoneNumber) {
        if (phoneNumber.length &gt; 10) {
            return false;
        }
        var regEx = new RegExp(/^(\([2-9]|[2-9])(\d{2}|\d{2}\))(-|.|\s)?\d{3}(-|.|\s)?\d{4}$/i);
        return regEx.test(phoneNumber);
    }
    function ValidateEmail(emailAddress) {

    function GetFormCookieValue() {

        var formCookieStart = cookies.indexOf("FormCookie=");

            cookies = null;
            return "";
        }
        else {
            formCookieStart = cookies.indexOf("=", formCookieStart) + 1;
            var end = cookies.indexOf(";", formCookieStart);
            if (end == -1) {
                end = cookies.length;
            }

                }
            }
            return filterString;
        }

    function addPushPin(location, title, description,currentCommunity,pushpinmapping) {
        var pushpinOptions;
        if (currentCommunity) {
            pushpinOptions = {icon: "/~/media/DRHorton/Sites/Com/Map Pins/default.ashx", width: 20, height: 30 };  
        } else {
            pushpinOptions = { icon: pushpinmapping.url, width: pushpinmapping.width, height: pushpinmapping.height, draggable: false }
        }

        var pin1 = new window.Microsoft.Maps.Pushpin(location, pushpinOptions);
        window.Microsoft.Maps.Events.addHandler(pin1, 'click', displayEventInfo);  
        nearbyCommunitiesMap.entities.push(pin1);
        var infoboxOptions = { title: title, description: description, pushpin: pin1, width: '235px' };

    function displayEventInfo(e) {
        if (e.targetType == 'pushpin') {
            var location = e.target.getLocation();
            var pixels = nearbyCommunitiesMap.tryLocationToPixel(location);
            pixels.x = pixels.x + 90;
            pixels.y = pixels.y - 60;
           
            var newLocation = nearbyCommunitiesMap.tryPixelToLocation(pixels);
            var zoomLevel = nearbyCommunitiesMap.getZoom();
            nearbyCommunitiesMap.setView({ zoom: zoomLevel, center: newLocation });
            setTimeout( "jQuery('.Infobox2').each(function(){this.style.width = '235px';});", 500);
        }

    function calculateView() {
        var kilometersToMiles = 1.60934 * 30;
        var defaultScales = new Array(0, 9000, 4000, 2000, 1000, 500, 250, 100, 60, 30, 15, 8, 4, 2, 1, 0.5, 0.25, 0.1, 0.06, 0.03);
        var mapWidth = document.getElementById("nearbyCommunitiesMap").offsetWidth;
        var mapHeight = document.getElementById("nearbyCommunitiesMap").offsetHeight;
        var scalebarsize = 141;

        //calculate the ratio between the mapWidth and scalebarWidth
        var ratioX = mapWidth / scalebarsize;
        var ratioY = mapHeight / scalebarsize;
        //calculates the x and y scales
        var meanScaleValueX = kilometersToMiles / ratioX;
        var meanScaleValueY = kilometersToMiles / ratioY;

        var meanScale;

        //gets the largest scale value to work with
        if (meanScaleValueX &gt; meanScaleValueY)
            meanScale = meanScaleValueX;
        else
            meanScale = meanScaleValueY;

        //intialize zoom level variable
        var zoom = 1;

        //calculate zoom level
        for (var i = 1; i &lt; 20; i++) {
            if (meanScale &gt;= defaultScales[i]) {
                zoom = i;
                break;
            }
        }
        return zoom;

    }

    function RfiSession(currentAction) {
        this.currentAction = currentAction;
        this.actions = new RfiActions();
        this.sessionTimer = null;

        this.endTimeout = function() {
            clearTimeout(this.sessionTimer);
            this.sessionTimer = null;
        };


        this.trackEvent = function(action) {
            this.endTimeout();
            if (action != this.actions.timeout &amp;&amp; action != this.actions.abandon &amp;&amp; action != this.actions.success) {
                this.sessionTimer = this.startSessionTimeout();
            }
            if (action == this.actions.abandon &amp;&amp; this.actions.success == this.currentAction) {
                return;
            } else {
                this.currentAction = action;
                ga('send', 'event', 'RFI', action, '', 0, { 'nonInteraction': 1 });
            }
        };
    }

    RfiSession.prototype.startSessionTimeout = function() { 
        var rst = function() {
       
            var ac = new RfiActions();
          
            g_rfiUserSession.sessionCurrentAction = ac.timeout;
            g_rfiUserSession.trackEvent(ac.timeout); 
            jQuery("#cityRequest").dialog("close");

        };
        return   setTimeout( function () { rst(); }, 180000);
    };
 
    RfiSession.prototype.sessionCurrentAction = "initialize";

    function RfiActions() {
        this.open = "open";
        this.submit = "submit";
        this.success = "success";
        this.abandon  = "abandon";
        this.required = "required";
        this.timeout = "timeout";
    }

    RfiActions.prototype.open = function() {
        return this.open;
    };
    
    RfiActions.prototype.submit = function() {
        return this.submit;
    };
    
    RfiActions.prototype.success = function() {
        return this.success;

    RfiActions.prototype.abandon = function() {
        return this.abandon;

    RfiActions.prototype.required = function() {
        return this.required;
