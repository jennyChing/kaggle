(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};

            <h3 class="post-title entry-title">
             <a href="http://www.polarbelle.com/2011/06/not-sales-pitch-testimonial-you-might.html">
              Not a Sales Pitch - A Testimonial You Might Find Interesting
             </a>
            </h3>
            <div class="post-header-line-1">
            </div>
            <div class="postmeta-primary">
             <span class="meta_date">
              Wednesday, June 29, 2011
             </span>
            </div>
            <div class="post-body entry-content">
             <p>
              If you know me, you know I am not a pushy sales person. If you don't know me personally, well now you know that about me. If you know me, you know my husband has Cystic Fibrosis, considered to be a terminal childhood illness. If you know my husband, you'll know he's WELL past childhood. If you've known us for long, you'll know that the past six years, every June his allergies have set off his CF and he's gotten critically ill...every June....six years...every time we thought he might die. I am not embellishing nor making it a bigger story than it is.
              <br/>
              <br/>
              This June, we tried something new. I have my cousins to thank for introducing me to Cie Aura. I thought it was all heebie jeebie wierd stuff, which I'm not totally adverse to necessarily, but hearing about it, I thought uh huh, sure. But a cousin that I really trust told me she tried it and it worked. So I tried a couple things based on her testimony.
              <br/>
              <br/>
              They have an allergy/sinus chip. Everyone in Alaska has sinus problems. Ok, that might be an overstatement, but most people I know do. I could instantly tell the difference and when the chip was done after it's three day time period, nobody had to tell me, I KNEW it. I thought if this worked for my husband, maybe he wouldn't get so sick in June. So I forced him to try it, like any good wife would.
              <br/>
              <br/>
              It's just a doggone miracle. He has survived his June allergie season with NO occurance. He's not had a Cystic Fibrosis episode; no trips to the hospital and I am thanking God for finding these chips and thanking my cousins for telling me. I really am so thankful that I thought I should tell people. It really is a true miracle that he hasn't gotten sick this year in June.
              <br/>
              <br/>
              If you'd like to find out more about it, you can visit my website
              <a href="http://polarbelle.cieaura.com/index.html">
               here
              </a>
              .
              <br/>
              <br/>
              I know this isn't my usual kind of post, but I feel strongly enough about what it's done for my sinuses and for my husband's illness, that I wanted to share.  So, want a pretty picture?  Here's just about half of my Maybelline quads.  They ROCK!

               <img alt="Maybelline Eye Studio Quads" border="0" src="http://i257.photobucket.com/albums/hh215/ragamu/IMG_2407-1.jpg"/>
