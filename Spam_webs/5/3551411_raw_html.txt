   function swapImg(a, image)
	{
		var the_path = 'http://lucydavey.co.uk/files/gimgs/' + image;

		show.src = the_path;
		$('body').css({ backgroundImage: 'url(' + show.src + ')', backgroundPosition: '215px 0'  });

		var title = $('#img' + a).attr('title');
		var caption = $('#img' + a).attr('alt');

			$('#backgrounded-text').html('&lt;span style="background: white; line-height: 24px;"&gt;' + title + caption + '&lt;/span&gt;');
		}
		else
		{
			$('#backgrounded-text').html('');
		}
	}
