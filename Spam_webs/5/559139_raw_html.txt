function ddtabcontent(tabinterfaceid){
this.tabinterfaceid=tabinterfaceid //ID of Tab Menu main container
this.tabs=document.getElementById(tabinterfaceid).getElementsByTagName("a") //Get all tab links within container
this.enabletabpersistence=true
this.hottabspositions=[] //Array to store position of tabs that have a "rel" attr defined, relative to all tab links, within container

this.revcontentids=[] //Array to store ids of arbitrary contents to expand/contact as well ("rev" attr values)
this.selectedClassTarget="link" //keyword to indicate which target element to assign "selected" CSS class ("linkparent" or "link")
}

ddtabcontent.getCookie=function(Name){
var re=new RegExp(Name+"=[^;]+", "i"); //construct RE to search for target name/value pair
if (document.cookie.match(re)) //if cookie found
return document.cookie.match(re)[0].split("=")[1] //return its value
return ""
}

ddtabcontent.setCookie=function(name, value){
document.cookie = name+"="+value+";path=/" //cookie value is domain wide (path=/)
}

              <b>
               ICC Cricket World Cup 2015 Indian Team Final 15 Squad
              </b>
              is to be announced on 6th Jan, 2015. Which means, less than 12 hours are left after which the dies hard Indian cricket fans will come to know the final 15 members of Indian Cricket Team for ICC Cricket World Cup 2015. The BCCI had earlier revealed the list of 30 names which included the probables for the ICC Cricket World Cup 2015 Indian squad. Now, the day has come when the Indian Team Final 15 Squad will be announced for the
              <b>
               ICC Cricket World Cup 2015
              </b>
              .
              <br/>
              <br/>
              <div class="separator" style="clear: both; text-align: center;">
               <a href="http://1.bp.blogspot.com/-RspEiM5Wqg8/VKrITTag1BI/AAAAAAAAEWU/Cs-3-XneIVY/s1600/30-man-squad-for-icc-cricket-world-cup-2015.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">

               </a>
              </div>
              <br/>
              We here, will publish the list of all the final 15 members of the ICC Cricket World Cup 2015 Indian Squad as soon as the same is revealed by BCCI. The cricket world cup 2015 will be hosted by Australia &amp; New Zealand jointly. The main event matches will strat from 14th Feb, 2015 that is Valentine's Day. Well, if you don't know, the warm up matches will be played by all the teams between 8th Feb to 13th Feb, and the very first match will be India v/s Australia. Yes, a big clash in the very first match.
              <br/>
              <br/>
              However, we here want to discuss more about the ICC Cricket World Cup 2015 Indian Team Final 15 Squad. So, below is the list which includes the names of all players who will be representing India in the ICC Cricket World Cup 2015. The list of final 15 member Indian Quad for World Cup 2015 is presented below.
