      function leaderTabs(tName) {
        if(!$('#'+tName).hasClass('current')) {
            $('.times-tab li').removeClass('current');
            $('#'+tName).addClass('current');
            $('#timestab1-content, #timestab2-content, #timestab3-content').hide();
            $('#'+tName+'-content').show();
        }
    }

    function flipLeaderTab(clsName, tabId) {
        if(!$('#'+tabId).hasClass('weekly_monthly_actbtn')) {
            $('.'+clsName).removeClass('weekly_monthly_actbtn');
            $('#'+tabId).addClass('weekly_monthly_actbtn');
            $('.'+clsName+'-content').hide();
            $('.loader').css('display', 'block');
            setTimeout(function(){
                $('.loader').css('display', 'none');
                $('.'+tabId+'-content').show();
            },500);
        }
    }

    // Load this script once the document is ready
    $(document).ready(function () {
        leaderTabs('timestab2');
        // Get all the thumbnail
        var popdiv = $("div#badgepopup");
        $('div.badgesIMG, div.myprofile-badgeBLK').mouseenter(function(e) {
            $(this).css('z-index','15');
            var cntt = $(this).children("div.tooltip").html();
            popdiv.css({'display': "none"});
            cntt = '&lt;div style="height:10px"&gt; &lt;/div&gt;' + cntt;
            popdiv.html(cntt);
            x = $(this).offset().left + $(this).width()/2 - popdiv.width()/2;
            y = $(this).offset().top + $(this).height() -10;
            popdiv.css({'top': y,'left': x })
            .show();

        }).mouseleave(function() {
            popdiv.hide();
        });
        popdiv.mouseleave(function() {
           popdiv.hide();
        });
        popdiv.mouseenter(function() {
           popdiv.show();
        });
    });
