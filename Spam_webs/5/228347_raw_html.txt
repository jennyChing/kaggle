 * Allow GA events via `ga-event` attributes on elements, both those
 * static on the page and those added in dynamically.
 * Elements can have two ga-* related attributes:
 *
 *  - ga-event - ',' separated list of GA _trackEvent parameters
 *  - ga-event-type - {optional} the action that should trigger the event (default is 'click')
 *
 * Elements that have been processed on the page already will have a 'ga-event-added' attribute
 *
 */
$(document).ready(function() {

    // Tracks each ga-event handler, one per event type
    var _handlers = {};


    // Track a 'click' (or other) event on elements with a 'ga-event' attribute
    // The comma separated value of the ga-event attribute follows the params for the _trackEvent() 
    // call in google analytics (in same order):
    //
    //  category (required) - name of group wanting to track
    //  action (required) - type of interaction
    //  label (optional) - optional string for identification
    //  value (optional) - numerical data about the event
    //  non-interactive (optional) - true for non-interactive events (ie. it shouldn't be used in bounce-rate calcs)
    //
    // var ga_event_handler = function(ev) {
    var ga_event_handler = function(ev, el) {
        var args = $(el).attr('ga-event').split(/\s*,\s*/),
            len = args.length,
            event = ['_trackEvent'];

        if (length &lt; 2) {
            // throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", this);
            throw new Error("[error] [ga-event] attribute tracking missing required arguments on element: %s", ev.target);
        }

        // build event arguments and push to google analytics
        event = event.concat(args);
        _gaq.push(event);

        // Special handling for 'click's on links
        var href = $(el).attr('href');

        if (ev.type === 'click' &amp;&amp; href !== undefined &amp;&amp; !/^#/.test(href)) {
            // outbound link, allow ga time to process click event 
            ev.preventDefault();
            window.setTimeout(function() {
                window.location.href = href;
            }, 100);
        }
    };


    function poll_ga_attr_events() {
        $('[ga-event]:not([ga-event-added])').each(function() {
            var type = $(this).attr('[ga-event-type]') || 'click',
                selector = '[ga-event][ga-event-added]';
                
            // Only add a delegated handler once for a given even `type'
            if (!_handlers[type]) {
                // Delegate to handle dynamic elements, track element ref via closure 
                $(document).on(type, selector, (function() {
                    return function(ev) {
                        ga_event_handler.apply(null, [ev, ev.currentTarget]);
                    };
                })());
                _handlers[type] = true;
            }

            // mark element as processed
            $(this).attr('ga-event-added', 'true');
        });

        // wait for 100ms and poll again (catch dynmically added elements)
        window.setTimeout(poll_ga_attr_events, 150);
    }


    // let's go, start processing ...
    poll_ga_attr_events();

});

        <img alt="Is Ferris a Cheater?" class="image" src="//file3.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/2/1/8/7/0/2187045582/b355ced3f4b861e464c81d96fa1c08d2ad8077a1.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         Is Ferris a Cheater?
        </span>
       </h2>
       <div class="section_body">
        <p>
         Shortly after Ferris encounters the sunbathing beauties, he shakes their hands and introduces himself, one can only assume he finds them attractive and is interested in pursuing something further. Sadly, this happens right after he kisses his girlfriend, Sloane. He also tells her he loves her, yet minutes later he's flirting. Bad Ferris.
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="7">
        <img alt="Ferris and His Wardrobe" class="image" src="//file3.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/2/3/3/7/2/2337255503/75d3b9b7fd572dc3fa637c9c04436fc9f366bee6.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         Ferris and His Wardrobe
        </span>
       </h2>
       <div class="section_body">
        <p>
         For most of the movie, Ferris is wearing typical teenage guy clothing. Nothing wrong with that, but at one point, he puts on a suit so he can pretend to be Sloane's father when he picks her up at school. During this scene, he's wearing a huge coat over the suit, so it's never even really visible. Why did Ferris put on this suit? Does he enjoy running up a dry cleaning bill?
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="8">
        <img alt="Sloane Kissing Her Father" class="image" src="//file2.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1423068956/t2seixuipfpwugrqbk48.png"/>
       </span>
       <h2 class="section_title">
        <span>
         Sloane Kissing Her Father
        </span>
       </h2>
       <div class="section_body">
        <p>
         And speaking of the scene where Ferris goes to the school to pick up Sloane, pretending to be her father, what is up with the principal? He sees Sloane making out with this guy who is supposed to be her father, and all the principal says is, "So, that's how it is in their family." Why isn't he reporting this to child services?
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="9">
        <img alt="Sloane Flirting with Ferris's Father" class="image" src="//file3.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/2/1/9/1/1/2191145913/4a9fe53b8e754379b78fc6a44dc77a16964298af.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         Sloane Flirting with Ferris's Father
        </span>
       </h2>
       <div class="section_body">
        <p>
         Of course, after she makes out with who the principal thinks is her dad, Sloane flirts with Ferris's dad. Who is this girl? She and Ferris are riding in a cab when Mr. Bueller pulls up beside them. Sloane, who is wearing her sunglasses, begins pouting and blowing kisses. Isn't that kind of creepy to do to the guy who may one day be your father-in-law? What if he remembers you?
        </p>
       </div>
      </div>
      <div class="section listitem slideshow_flagged ">
       <span class="section_image" data-index="10">
        <img alt="Ferris Exhibits Symptoms of a Sociopath" class="image" src="//file3.answcdn.com/answ-cld/image/upload/f_jpg,w_672,c_fill,g_faces:center,q_70/v1/tk/view/cew/1/0/1/7/4/1017421127/32cb7f923159cee76fc9e5b55712a3c501ef82f1.jpeg"/>
       </span>
       <h2 class="section_title">
        <span>
         Ferris Exhibits Symptoms of a Sociopath
        </span>
       </h2>
       <div class="section_body">
        <p>
         According to the Mayo Clinic, a sociopath exhibits the following signs and symptoms: disregard for right and wrong, persistent lying to deceive others, using charm or wit to manipulate others, unnecessary risk-taking, and irresponsible work behavior. If you have ever seen the movie, you have to think Ferris fits the description perfectly.
        </p>
       </div>

           <h1 class="title " ga-event="Feed_Click,index/article/15514,f_g_574654">
            <a class="title_link black" href="http://africa.answers.com/travel/online-ferry-listings-from-spain-to-morocco">
             Online Ferry Listings From Spain to Morocco
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_574654">
            One of the most popular ways to travel from Spain to Morocco is by ferry. Spain Morocco ferries leave multiple times a day from several Spanish ports. Some trips are very fast
            <span class="hellip">
             
            </span>
            <span class="hidden">
             , while others take up to a day.Direct Ferries is a booking company based in the United Kingdom. Its database has a listing of all ferries between Spain and Morocco. When you click on the city-to-city listing, you receive details about the frequency of the ferries and the prices. Once you have made your decision, it is easy to book online.[Direct Ferries](http://www.directferries.co.uk/ferries_from_spain_to_morocco.htm)This company is based in Italy. Its listings have departures to Morocco from both Italy and Spain.  Just click on the route you are interested in and all the timetables appear.[Cemar Agency](http://www.cemar.it/dest/ferries_morocco.htm)FRS is a company that operates high-speed ferries between various ports in Spain and Morocco.  Book tickets and get helpful information about the transition to Morocco from this site. FRS ferries are fast, comfortable, and have cafe-bars and duty-free facilities onboard.[FRS](http://www.frs.es/en/ferrys-to-spain-morocco.html)Ferries Spain is a site managed by a Danish woman who has been living in Spain for over a decade. You do not book directly from this site, but after perusing the possibilities and choosing your route, you link to the company you are interested in and buy the tickets from them. Ferries Spain is an easy-to-use resource that points you in the right direction.[Ferries Spain](http://www.ferriesspain.com/ferry-to-morocco.html)Net Ferry is a booking center that books ferry passage all over the Mediterranean. Its Spain to Morocco page has a listing for the Barcelona to Tangier ferry, a once-a-week service that takes over a day. If you want to relax and enjoy a cruise on the Mediterranean Sea on your way to Africa, this is the ferry for you.[Net Ferry](http://ferriesmorocco.com/ferries-spain-morocco/)From the 30 minute dash across the Strait of Gibraltar, to the leisurely cruise from Barcelona to Tangier, there are many options to consider when planning your Spain Morocco ferry. These sites give you all the information you need in order to make the best decision. Take the time to peruse the listings to make the most of this leg of your African adventure.As it's only a short trip from Europe across the Mediterranean, Morocco's popularity as a tourist destination draws in nearly 10 million visitors each year.

            <a class="title_link black" href="http://www.answers.com/article/1176907/world-s-10-largest-ferris-wheels">
             World's 10 Largest Ferris Wheels
            </a>
           </h1>
           <div class="content_text" ga-event="Feed_Click,index/article/15514,f_g_1176907">
            Currently, the world's tallest Ferris wheel is located in Las Vegas. However, this could change at any time as architects try to attract travelers who like amusement rides. Th
            <span class="hellip">
             
            </span>
            <span class="hidden">
             e Vega Ferris wheel, the Las Vegas High Roller, stands a whopping 550 feet tall. Riders can see the entire Las Vegas Strip down to Mandalay Bay. The cabins hold up to 40 people and the ride takes about 30 minutes.The SkyVue Observation Wheel stands 500 feet tall, just short of the larger High Roller Ferris wheel. Riders can observe scenes from Times Square and the London Eye on a 100,000 square foot high LED screen. This screen will also serve as a huge advertisement platform. SkyVue features two floors of retail shops and dining for visitors waiting to ride.The London Eye, located in London, UK, stands 443 feet high. From the top of the London Eye, visitors can see 25 miles on a clear day, making it possible to see Windsor Castle from high in the sky. It takes about 30 minutes for each trip around the Ferris wheel and holds 800 people, 25 per each of the 32 capsules. Riders are free to get up and walk around inside the capsules.The Star of Nanchang is a 525 feet high Ferris wheel located in Nanchang, China. Built in 2008, it was the tallest in the world at that time. The Ferris wheel has 60 enclosed gondolas that carry eight people in each. There are more than 21,000 fluorescent lights on the wheel to light up the sky at night. Like most of the other Ferris wheels, the rotation takes approximately half an hour.The Singapore Flyer is located in Marina Centre Singapore and stands a whopping 541 feet tall. The Ferris wheel opened in 2008. The strong belief in feng shui caused the operators to reverse the spinning of the wheel on the advice of feng shui masters. Supposedly, doing so would resolve wealth back into the financial district. The wheel has 28 capsules that hold up to 28 passengers each.The Dubai Eye has not yet opened; the opening is set for New Year's Eve 2015 in Bluewaters Island, UAE. It will stand 688 feet tall, making it the tallest Ferris wheel in the world. Construction is on a new island, the Blue Waters Island. The Island will feature shops and restaurants for visitors.On New Year's Eve 2015, New Yorkers will welcome the opening of the New York Wheel on Staten Island. It will stand an enormous 630 feet tall. It will hold up to 1,440 people per ride. The ride should bring great economic impact to New York and the Staten Island area. A water taxi is under development to make access to the New York Wheel easier and quicker.The Tianjin Eye Ferris wheel stands 394 feet tall and is located in Tianjin, China. It is built above the Yongle Bridge and is supposedly the only Ferris wheel built over a bridge. The wheel carries up to 768 passengers every 30 minutes on a breathtaking view of the Tianjin area. Although there are bigger Ferris wheels today, it is still among the largest.The Southern Star Ferris wheel stands 394 feet tall and is located in Waterfront City Melbourne Australia. The entire cabins are floor to ceiling glass, which gives visitors complete and unimpeded views of the city from all angles. Each cabin holds up to 20 people with room for them to walk around comfortably. Viewers can see Melbourne, Port Phillip Bay, and the surrounding area.Located in Yokohama, Japan, the Cosmo Clock 21 Ferris wheel stands 369 feet tall. It was originally only 353 feet but was dismantled in 1997 and moved to a higher base to increase the height. It is located inside the Cosmo World Amusement park in Yokohama, Japan. The Ferris wheel is also the world's largest clock. The Ferris wheel ride only takes 15 minutes and can accommodate 480 people per ride.
