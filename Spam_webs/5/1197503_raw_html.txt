function trim(str) 

{ 

    if (str != null) {

        var i; 

        for (i=0;i&lt;str.length;i++) {

            if (str.charAt(i)!=" ") {

                str=str.substring(i,str.length); 

                break;

            } 

        } 

    

        for (i=str.length-1;i&gt;=0;i--) {

            if (str.charAt(i)!=" ") {

                str=str.substring(0,i+1); 

                break;

            } 

        } 

        

        if (str.charAt(0)==" ") {

            return ""; 

        } else {

            return str; 

        }

    }

}
