function riposizionaColonnaFloat()
{ 
	var scroll_bottom =  parseInt(jQuery(window).scrollTop() + (jQuery(window).height() )) ;
	var scroll_top = parseInt(jQuery(window).scrollTop());
	var scroll_left = parseInt(jQuery(window).scrollLeft());
	var zona_float_container = jQuery(".colonna-float").parent();
	var zona_float_container_top = parseInt(zona_float_container.offset().top);
	var zona_float = jQuery(".colonna-float");
	var zona_float_bottom =  parseInt(zona_float.offset().top + zona_float.height());
	var zona_float_top = parseInt(zona_float.offset().top);
	var footer = jQuery("#footer");
	var footer_top_offset = scroll_bottom - parseInt(footer.offset().top);
	var zona_float_top_offset = zona_float.height()-jQuery(window).height();
//----- INIZIO
	var barra_menu_height = 0; 
	if(jQuery("#barra-menu").length &gt; 0 &amp;&amp; jQuery("#barra-menu").hasClass("fixed"))
	{
		barra_menu_height = jQuery("#barra-menu").height();
		zona_float_top -= barra_menu_height;
		zona_float_bottom -= barra_menu_height;
		// console.log("aggiunge barramenu = " + jQuery("#barra-menu").height());
	}
	if(external_top_height)
	{
		zona_float_top -= external_top_height;
		zona_float_bottom -= external_top_height;
	}
	
	var fixed_top =0;
	if(zona_float_top_offset &gt; 0)
	{
		console.log("zona_float_top_offset = " + zona_float_top_offset);
		fixed_top = -zona_float_top_offset;
		// console.log("fixed_top DOPO = " + fixed_top);
	}
	if(barra_menu_height)
		fixed_top += barra_menu_height;
	if(external_top_height)
		fixed_top += external_top_height;	

// ----- FINE 
	if(!jQuery(".colonna-float").hasClass("fixed") &amp;&amp; 
			(scroll_bottom &gt;= zona_float_bottom) &amp;&amp; scroll_top &gt; zona_float_top )
	{
		jQuery(".colonna-float").addClass("fixed");
		jQuery(".colonna-float").animate({'top': fixed_top});
		
	} else {
		if(scroll_top &lt; zona_float_container_top)
		{
			jQuery(".colonna-float").removeClass("fixed");
		} else {
			if(footer_top_offset &gt; 0)
			{
				jQuery(".colonna-float").animate({'top': fixed_top - footer_top_offset });
			} else {
				jQuery(".colonna-float").animate({'top': fixed_top});
			}

			var new_left = (zona_float.parent().offset().left -  scroll_left);
			console.log("PARENT LEFT " + new_left);
			zona_float.css('left', new_left);
		}
	}
}

		var offset = Math.abs(position.left) + jQuery(".gallery_thumbnails_mask").width();

			jQuery("#ico_video_arrow_left").fadeTo('50', '0.3');
			jQuery("#ico_video_arrow_right").fadeTo('50', '1.0');

		{
			// disabilita freccia destra
			jQuery("#ico_video_arrow_right").fadeTo('50', '0.3');
			jQuery("#ico_video_arrow_left").fadeTo('50', '1.0');
		} else {
			jQuery("#ico_video_arrow_right").fadeTo('50', '1.0');
			jQuery("#ico_video_arrow_left").fadeTo('50', '1.0');
		}  
		thumbnails_animation_start = false;
	}
	
}

    toggleCalendar = function() {

	jQuery("#brandlist").hide(600);
	if(jQuery("#pulsante_bottom_calendario").hasClass("titolo-pulsante-bottom-sel"))
	{
		jQuery("#pulsante_bottom_calendario").removeClass("titolo-pulsante-bottom-sel");
	} else {
		jQuery(".titolo-pulsante-bottom").removeClass("titolo-pulsante-bottom-sel");
		jQuery("#pulsante_bottom_calendario").addClass("titolo-pulsante-bottom-sel");
	}
	
	loadCalendarUrl("/funweek-frontend/calendario", "calendarioPlaceholder");
	
	jQuery('#calendario').slideToggle(600);
}

loadCalendarUrl = function(url, target)
{
	var loaderDiv = jQuery("&lt;div id='loaderDiv'&gt;&lt;img id='spin' style='position:absolute; top:245px; left: 440px;' src='http://static.funweek.it/static/html/img/spinner.gif'&gt;&lt;div&gt;");
	loaderDiv.css("padding", "10px");
	loaderDiv.css("text-transform", "none");
	jQuery("#" + target).html(loaderDiv);
	jQuery.ajax({
	      url: url, // ,
	      dataType: 'html',
	      cache: false,
	      success: function(data)
	      {
	    	  jQuery("#" + target).html(data);
	      },
	      error: function (request, status, error) {
			console.log("errore: " + error);
	      }
	    });
}

}
