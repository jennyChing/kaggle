function autosize()
{
 	var body = document.getElementById("main");
 	var iframe = document.getElementById("framecenter");
 	var leftheight = height = document.getElementById("leftcontent").clientHeight + 20;
 	var content = iframe.contentWindow.document.getElementById("content"); 
   	if (content != undefined)
	{
		content.style.height = '';
		var clientHeight = content.clientHeight &gt; 0 ? content.clientHeight : content.offsetHeight;
		if (clientHeight &lt; (leftheight - 35)) content.style.height = leftheight - 35 + "px";
		height = (content.style.height != '') ? leftheight + 35 : clientHeight + 35; 
  	}
	else height = iframe.contentWindow.document.documentElement.scrollHeight + 20;
   iframe.style.height = height + "px";
	iframe.style.width = document.documentElement.clientWidth - 50 + "px";
}

{

	window.attachEvent('onresize',autosize);
}
