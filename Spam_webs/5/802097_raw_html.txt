function EncodeURI(name)
{
var newName = ""

{
ch = name.charAt(i);
c  = name.charCodeAt(i);
if (('A' &lt;= ch &amp;&amp; ch &lt;= 'Z') || ('a' &lt;= ch &amp;&amp; ch &lt;= 'z') || ('0' &lt;= ch &amp;&amp; ch &lt;= '9'))
{
newName += ch;
}
else if (ch == ' ') 
{
newName += '+';
}
else if (ch == '-' || ch == '_'
|| ch == '.' || ch == '!'
|| ch == '~' || ch == '*'
|| ch == '\'' || ch == '('
|| ch == ')') 
{
newName += ch;
} 
else if (c &lt;= 0x007f) 
{
newName += gHexArray[c];
}
else if (c &lt;= 0x07FF) 
{
newName += gHexArray[0xc0 | (c &gt;&gt; 6)];
newName += gHexArray[0x80 | (c &amp; 0x3F)];
} 
else 
{
newName += gHexArray[0xe0 | (c &gt;&gt; 12)];
newName += gHexArray[0x80 | ((c &gt;&gt; 6) &amp; 0x3F)];
newName += gHexArray[0x80 | (c &amp; 0x3F)];
}
}
return newName;
}
