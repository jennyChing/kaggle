	function displayAllPlaces(){
		var placesText = "&lt;div id='placesLocations'&gt;";
		for(var key in places){
			var place = places[key];
			if(place){
				placesText=placesText+placeSideBarHtml(place);
			}
		}
		placesText=placesText+"&lt;/div&gt;";
		placesText=placesText+addPlaceLink();
		jq('#placesSideList').html(placesText);
		for(var key in places){
			var place = places[key];
			if(place){
				getTimeAndDistanceForAllModes(place, currentPoint);
			}
		}
		rebindPlacesAddLinks();
	}

	function addPlace(address, lat, lng, name, id){
		if(name == null){
			name = jq('#placesName').val();
		}
		if(id == null){
			var now = new Date();
			id=""+now.getTime();
		}
		//Add to data store
		var place = new Object();
		place.address = address;
		place.lat = lat
		place.lng = lng
		place.name = name;
		place.id = id;
		places[place.id]=place;
		displayPlace(place);
		if(openerIsOpen()){
			window.opener.addPlace(address, lat, lng, name, id);
		}else{
			savePlaceInBackend(place);
		}

		//Pull times, draw polyline
		if(currentPoint){
			getTimeAndDistanceForAllModes(place, currentPoint);
		}
	}

	function deletePlace(id){
		place = places[id];
		if(typeof map!="undefined"){
			removeMarkerFromMap(place.marker);
			//map.removeOverlay(place.marker);
		}
		deletePlaceInBackend(place);
		places[id]=null;
		jq('.place'+id).remove();
		if(openerIsOpen()){
			window.opener.deletePlace(id);
		}
	}
