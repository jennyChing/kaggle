     if (typeof String.prototype.startsWith != 'function') {
        String.prototype.startsWith = function (str) {
            return this.slice(0, str.length) == str;
        };
    }
    $(document).ready(function () {
        var pathname = location.href;
        var sIndex = pathname.lastIndexOf('/');
        pathname = pathname.substring(sIndex + 1);
        var matchingMenuCount = 0;
        $("#cssmenu &gt; ul &gt; li &gt; a").each(function () {
            if (pathname.startsWith(this.getAttribute('href'))) {
                if (matchingMenuCount === 0) {
                    $(this).parent().addClass('active');
                }
                matchingMenuCount++;
            } else {
                $("ul li a", $(this).parent()).each(function () {
                    if (pathname.startsWith(this.getAttribute('href'))) {
                        if (matchingMenuCount === 0) {
                            var parentUl = $(this).parent().parent();
                            $(this).addClass('active');
                            parentUl.show();
                            parentUl.parent().addClass('active');
                        }
                        matchingMenuCount++;
                    }
                });
            }
        });
//        if (matchingMenuCount !== 1) {
//            if (matchingMenuCount &gt; 1) {
//                console.log("more than 1 left menu matched with the current url");
//            } else {
//                console.log("no left menu matched with the current url");
//            }
//        }
        $('#cssmenu &gt; ul &gt; li:has(ul)').addClass("has-sub");
        $('#cssmenu &gt; ul &gt; li &gt; a').click(function () {
            var checkElement = $(this).next();
            $('#cssmenu li').removeClass('active');
            $(this).closest('li').addClass('active');



            if ((checkElement.is('ul')) &amp;&amp; (!checkElement.is(':visible'))) {
                $('#cssmenu ul ul:visible').slideUp('normal');
                //checkElement.slideDown('normal');
            }

            if (checkElement.is('ul')) {
                return false;
            } else {
                return true;
            }
        });
    });
