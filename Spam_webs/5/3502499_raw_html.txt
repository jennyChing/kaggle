function sec () {
  var now = new Date();
  return (now.getHours() * 60 + now.getMinutes()) * 60 + now.getSeconds();
}
function tick () {
  now = new Date();
  nowSec = (now.getHours() * 60 + now.getMinutes()) * 60 + now.getSeconds();
  document.Info.Showtime.value = dateStr(now) + "  " + nowSec + " s";
  delta  = nowSec - document.Info.Startsec.value;
  //   document.Info.Gossip.value = nowSec + " " + delta + " " +  document.Info.Startsec.value;
  if(delta &gt;= 0 &amp;&amp; nowSec &lt;= document.Info.beyondSec.value) {
    n = Math.floor( delta / 10 );
    if(delta % 10 == 0 &amp;&amp; n != 0) { n = n - 1 }
    d = delta - (n * 10);
    eval("document.images.clk" + (n + 1) + ".src = \"g10x1_" + d + ".gif\"");
  } else {
    self.clearInterval(tick_interval);
    document.Info.Showtime.value = "paused.. hit Reload";
    window.location.replace("http://" + window.location.hostname + window.location.pathname  + "?do=" + dateStr(now) + "#topofclock");
  }
}
tick_interval = self.setInterval('tick()',1000);
