// A jQuery based placeholder polyfill
$(document).ready(function(){
function add() {
if($(this).val() === ''){
$(this).val($(this).attr('data-placeholder')).addClass('data-placeholder');
}
}

function remove() {
if($(this).val() === $(this).attr('data-placeholder')){
$(this).val('').removeClass('data-placeholder');
}
}

// Create a dummy element for feature detection
if (!('data-placeholder' in $('&lt;input&gt;')[0])) {

// Select the elements that have a placeholder attribute
$('input[data-placeholder], textarea[data-placeholder]').blur(add).focus(remove).each(add);

// Remove the placeholder text before the form is submitted
$('form').submit(function(){
$(this).find('input[data-placeholder], textarea[data-placeholder]').each(remove);
});
}
});
