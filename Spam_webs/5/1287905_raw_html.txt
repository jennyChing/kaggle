         <p>

          </span>
         </p>
         <span class="recipe-author" content="" itemprop="author">
         </span>
         <meta content="http://simplegreensmoothies.com/Recipes/cilantro-mango-detox" itemprop="url"/>
         <span class="recipe-name" itemprop="name">
          Cilantro-Mango Detox
         </span>
         <p>
          <em>
           <span itemprop="recipeYield">
            Serves 2
           </span>
          </em>
         </p>
         <hr class="sep"/>
         <h4>
          Ingredients
         </h4>
         <ul class="ingredients">
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            1
            <sup>
             1
            </sup>
            
            <sub>
             2
            </sub>
            cups spinach, fresh
           </span>
          </li>
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            <sup>
             1
            </sup>
            
            <sub>
             2
            </sub>
            cup cilantro, fresh
           </span>
          </li>
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            1
            <sup>
             1
            </sup>
            
            <sub>
             2
            </sub>
            cups mango
           </span>
          </li>
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            2 cups water
           </span>
          </li>
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            1 cup pineapple
           </span>
          </li>
          <li>
           <span class="recipe-ingredient" itemprop="ingredients">
            <sup>
             1
            </sup>
            
            <sub>
             2
            </sub>
            avocado
           </span>
          </li>
         </ul>
         <h4>
          Instructions
         </h4>
         <p>
          <span itemprop="recipeInstructions">
