var maxUserNameLength = 42; // 0: don't cut, &gt;4: cut usernames

var txtTopLine = '&lt;b&gt;[#].&lt;/b&gt; [image] [user] ([count])';
var txtNoTopCommenters = 'No top commentators at this time.';
var txtAnonymous = '';      // empty, or Anonymous user name localized if you want to localize

var sizeAvatar = 56;
var cropAvatar = true;
//
var urlNoAvatar = "http://lh4.googleusercontent.com/-069mnq7DV_g/TvgRrBI_JaI/AAAAAAAAAic/Iot55vywnYw/s"+sizeAvatar+"/avatar_blue_m_96.png"; // http://www.blogger.com/img/avatar_blue_m_96.png resizeable
var urlAnoAvatar = 'http://www.gravatar.com/avatar/00000000000000000000000000000000?d=mm&amp;s=' + sizeAvatar;

function replaceTopCmtVars(text, item, position)
{
  if(!item || !item.author) return text;
  var author = item.author;

  var authorUri = "";
  if(author.uri &amp;&amp; author.uri.$t != "")
    authorUri = author.uri.$t;

  var avaimg = urlAnoAvatar;
  var bloggerprofile = "http://www.blogger.com/profile/";
  if(author.gd$image &amp;&amp; author.gd$image.src &amp;&amp; authorUri.substr(0,bloggerprofile.length) == bloggerprofile)
    avaimg = author.gd$image.src;
  else {
    var parseurl = document.createElement('a');
    if(authorUri != "") {
      parseurl.href = authorUri;
      avaimg = 'http://www.google.com/s2/favicons?domain=' + parseurl.hostname;
    }
  }
  if(urlMyProfile != "" &amp;&amp; authorUri == urlMyProfile &amp;&amp; urlMyAvatar != "")
    avaimg = urlMyAvatar;
  if(avaimg == "http://img2.blogblog.com/img/b16-rounded.gif" &amp;&amp; urlNoAvatar != "")
    avaimg = urlNoAvatar;
  var newsize="s"+sizeAvatar;
  avaimg = avaimg.replace(/\/s\d\d+-c\//, "/"+newsize+"-c/");
  if(cropAvatar) newsize+="-c";
  avaimg = avaimg.replace(/\/s\d\d+(-c){0,1}\//, "/"+newsize+"/");

  var authorName = author.name.$t;
  if(authorName == 'Anonymous' &amp;&amp; txtAnonymous != '' &amp;&amp; avaimg == urlAnoAvatar)
    authorName = txtAnonymous;
  var imgcode = '&lt;img class="top-commenter-avatar" height="'+sizeAvatar+'" width="'+sizeAvatar+'" title="'+authorName+'" src="'+avaimg+'" /&gt;';
  if(authorUri!="") imgcode = '&lt;a href="'+authorUri+'"&gt;'+imgcode+'&lt;/a&gt;';

  if(maxUserNameLength &gt; 3 &amp;&amp; authorName.length &gt; maxUserNameLength)
    authorName = authorName.substr(0, maxUserNameLength-3) + "...";
  var authorcode = authorName;
  if(authorUri!="") authorcode = '&lt;a class="profile-name-link" href="'+authorUri+'"&gt;'+authorcode+'&lt;/a&gt;';

  text = text.replace('[user]', authorcode);
  text = text.replace('[image]', imgcode);
  text = text.replace('[#]', position);
  text = text.replace('[count]', item.count);
  return text;
}

var topcommenters = {};
var ndxbase = 1;
function showTopCommenters(json) {
  var one_day=1000*60*60*24;
  var today = new Date();

  if(urlMyProfile == "") {
    var elements = document.getElementsByTagName("*");
    var expr = /(^| )profile-link( |$)/;
    for(var i=0 ; i&lt;elements.length ; i++)
      if(expr.test(elements[i].className)) {
        urlMyProfile = elements[i].href;
        break;
      }
  }

  for(var i = 0 ; i &lt; json.feed.entry.length ; i++ ) {
    var entry = json.feed.entry[i];
    if(numDays &gt; 0) {
      var datePart = entry.published.$t.match(/\d+/g); // assume ISO 8601
      var cmtDate = new Date(datePart[0],datePart[1]-1,datePart[2],datePart[3],datePart[4],datePart[5]);

      //Calculate difference btw the two dates, and convert to days
      var days = Math.ceil((today.getTime()-cmtDate.getTime())/(one_day));
      if(days &gt; numDays) break;
    }
    var authorUri = "";
    if(entry.author[0].uri &amp;&amp; entry.author[0].uri.$t != "")
      authorUri = entry.author[0].uri.$t;

    if(excludeMe &amp;&amp; authorUri != "" &amp;&amp; authorUri == urlMyProfile)
      continue;
    var authorName = entry.author[0].name.$t;
    if(excludeUsers.indexOf(authorName) != -1)
      continue;

    var hash=entry.author[0].name.$t + "-" + authorUri;
    if(topcommenters[hash])
      topcommenters[hash].count++;
    else {
      var commenter = new Object();
      commenter.author = entry.author[0];
      commenter.count = 1;
      topcommenters[hash] = commenter;
    }
  }
  if(json.feed.entry.length == 200) {
    ndxbase += 200;
    document.write('&lt;script type="text/javascript" src="http://'+window.location.hostname+'/feeds/comments/default?redirect=false&amp;max-results=200&amp;start-index='+ndxbase+'&amp;alt=json-in-script&amp;callback=showTopCommenters"&gt;&lt;/'+'script&gt;');
    return;
  }

  // convert object to array of tuples
  var tuplear = [];
  for(var key in topcommenters) tuplear.push([key, topcommenters[key]]);

  tuplear.sort(function(a, b) {
    if(b[1].count-a[1].count)
        return b[1].count-a[1].count;
    return (a[1].author.name.$t.toLowerCase() &lt; b[1].author.name.$t.toLowerCase()) ? -1 : 1;
  });

  // list top topcommenters:
  var realcount = 0;
  for(var i = 0; i &lt; maxTopCommenters &amp;&amp; i &lt; tuplear.length ; i++) {
    var item = tuplear[i][1];
    if(item.count &lt; minComments)
        break;
    document.write('&lt;di'+'v class="top-commenter-line"&gt;');
    document.write(replaceTopCmtVars(txtTopLine, item, realcount+1));
    document.write('&lt;/d'+'iv&gt;');
    realcount++;
  }
  if(!realcount)
    document.write(txtNoTopCommenters);
}
