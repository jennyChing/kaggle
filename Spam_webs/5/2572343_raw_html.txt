  function bindErrorEvent(id)
  {
    if ($('#'+id).prop('tagName').toLowerCase() == 'input')
    {
      $('#'+id).addClass('error_bg');
      $('#'+id).live('keyup blur', function() {
        if ($(this).val().length)
          $(this).removeClass('error_bg');
      });
    }
    else if ($('#'+id).prop('tagName').toLowerCase() == 'select')
    {
      $('#b-custom-select_'+id).find('.b-custom-select__title__text').addClass('error_bg');
      $('#'+id).live('change', function() {
        if ($(this).val())
          $('#b-custom-select_'+id).find('.b-custom-select__title__text').removeClass('error_bg');
      });
    }
  }

  function validateInput(id, isAfterPost)
  {
	  var errorMsg = 'This is required.';
	  var inputType = $('#'+id).prop('tagName').toLowerCase();
	  var inputValueLength = $('#'+id).val().length;

	  if (typeof $('#'+id).attr('required_message') != "undefined")
	  	errorMsg = $('#'+id).attr('required_message');

		var errorHTML = "&lt;ul class='error_list'&gt;&lt;li&gt;"+errorMsg+"&lt;/li&gt;&lt;/ul&gt;";
		
	  if (inputValueLength &gt; 0)
	  {
		  if (inputType == 'input' &amp;&amp; isAfterPost == false)
		  	$('#'+id).removeClass('error_bg');
		  else if (isAfterPost == false)
			  $('#b-custom-select_'+id).find('.b-custom-select__title__text').removeClass('error_bg');

		  if (isAfterPost == false)
		  	$('#'+id).closest("div").find('.form_error').html('');
	  }
	  else
	  {
		  if (inputType == 'input')
		  	$('#'+id).addClass('error_bg');
		  else
			  $('#b-custom-select_'+id).find('.b-custom-select__title__text').addClass('error_bg');

		  $('#'+id).closest("div").find('.form_error').html(errorHTML);
  	}
  }

  function enableDisableSubmitButton(formId, buttonId, requiredClassName)
  {
	  var empty = false;

		$('#'+formId).find(':input').each(function(){
      	if ($(this).hasClass(requiredClassName) &amp;&amp; $(this).val() == '')
		  		empty = true;
		});

      if (empty) 
	    {
    		$('#'+buttonId).attr('disabled', 'disabled'); // updated according to http://stackoverflow.com/questions/7637790/how-to-remove-disabled-attribute-with-jquery-ie
        
        	if ($('#'+buttonId).hasClass('active'))
        	{
        		$('#'+buttonId).removeClass('active');
        		$('#'+buttonId).addClass('btn_disable');
        	}
    	} 
    	else 
	    {
    		$('#'+buttonId).removeAttr('disabled'); // updated according to http://stackoverflow.com/questions/7637790/how-to-remove-disabled-attribute-with-jquery-ie
        
        	if ($('#'+buttonId).hasClass('active') == false)
        	{
        		$('#'+buttonId).addClass('active');
        		$('#'+buttonId).removeClass('btn_disable');
        	}
    	}
  }

  function doLogin(url)
  {
    $('#divLoginRequestError').html('');
    $('#indicator').show();

    var formData = $('#frmLoginForm').serialize();
    $.post(url, formData, function(data) {
      $('#indicator').hide();

function getWideSearchCounter()
{
  data = Url.decode(unescape($('#wide_search').serialize()));
  $('#suggestionHtml').hide().html(null);
  $.ajax({
    url: "http://www.classicandperformancecar.com/s_counter",
    data: data,
    async: true,
    success: function(counter) {
      var retObj = $.parseJSON(counter);
      $('#wide_search_count, #wide_search_count_quick, #wide_search_count_adv').html(retObj.countHtml);
      if (retObj.suggestionHtml)
      {
        $('#suggestionHtml').html(retObj.suggestionHtml).show();
        searchUrl = retObj.searchUrl;
      }
      else
      {
        searchUrl = '';
      }
    }
  });
}

function cleanQS()
{

  {
     window.location.href = searchUrl;
     return false;
  }

  if ($('#qstType').val() == 'qs')
    $('#qstType').remove();

    $('#qsearch_keywords').attr('disabled', true);

    $('#qsearch_year_from').remove();

  if ($('#qsearch_year_to').val() == '')
    $('#qsearch_year_to').remove();

  if ($('#qsearch_price_from').val() == '')
    $('#qsearch_price_from').remove();

  if ($('#qsearch_price_to').val() == '')
    $('#qsearch_price_to').remove();

  if ($('#qsearch_agent_name').val() == '')
    $('#qsearch_agent_name').remove();

  if ($('#qsearch_currency').val() == '')
    $('#qsearch_currency').remove();

  if ($('#qsearch_country_code').val() == '')
    $('#qsearch_country_code').remove();

  if ($('#qsearch_color_id').val() == '')
    $('#qsearch_color_id').remove();

  if ($('#qsearch_vehicle_make_id').val() == '')
    $('#qsearch_vehicle_make_id').remove();

  if ($('#qsearch_vehicle_model_id').val() == '')
    $('#qsearch_vehicle_model_id').remove();

  if (!$('#qsearch_location').val())
  {
    $('#qsearch_location').attr('disabled', true);
    $('#qsearch_location').remove();
  }

  if ($('#qsearch_mile').val() == '0')
  {
    $('#qsearch_mile').attr('disabled', true);
    $('#qsearch_mile').remove();
  }

  if ($('#qqst').val() == '')
    $('#qqst').remove();

  $('#qsearch_agent_notification, #qsearch_allow_view_agent, #qsearch_agent_name').remove();

  event = ($('#qstType').val() != 'as' || $('#qstType').val() == 'undefined') ? 'Quick search' : 'Advanced search';
  ga('send', 'event', 'Home', 'Search Ad', event);
  ctr =  document.getElementById('wide_search').elements.length;

  if(ctr &gt; 2)
  {
    return true;
  }
  else
  {
    window.location.href = $('#wide_search').attr('action');
    return false;
  }
}
