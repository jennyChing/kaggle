(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) &lt; cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p &amp;&amp; p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};

                 <h3 class="post-title entry-title" itemprop="name">
                  Minuman Berenergi Aman Tidak Berbahaya
                 </h3>
                 <div class="post-header">
                  <div class="post-header-line-1">
                  </div>
                 </div>
                 <div class="post-body entry-content" id="post-body-8972145412697694487" itemprop="articleBody">
                  <div class="subtitle">
                   Kratingdaeng adalah minuman tambahan pelengkap berenergi
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Kratingdaeng jenis minuman berenergi yang aman atau tidak berbahaya serta praktis untuk dikonsumsi agar stamina tetap terjaga dalam melakukan aktivitas. Kratingdaeng adalah minuman suplemen penambah energi yang telah diakui dan dikonsumsi oleh berbagai lapisan masyarakat di Indonesia, sehingga wajar jika dikatakan Kratingdaeng minuman energi Indonesia. Minuman energi Kratingdaeng tergolong minuman berenergi yang tidak berbahaya karena Kratingdaeng merupakan minuman berenergi yang mengandung kafein kurang dari 1 cangkir kopi, dan juga merupakan minuman berenergi gula alami. Dalam memilih produk minuman berenergi terbaik untuk dikonsumsi, ketahuilah terlebih dahulu apakah produk tersebut memiliki izin resmi dari BPOM RI. Pastikan juga untuk mengetahui kandungan dan manfaat minuman berenergi tersebut agar mendapatkan hasil yang maksimal. Misalnya, minuman berenergi untuk olah raga sebaiknya memilih jenis minuman energi gula alami seperti Kratingdaeng.
                     <a href="http://ferry-bachtiar.blogspot.com/2015/04/minuman-berenergi-aman-tidak-berbahaya.html">
                      Kratingdaeng adalah minuman tambahan pelengkap berenergi
                     </a>
                     , dan oleh BPOM RI telah menyatakan bahwa
                     <i>
                      Kratingdaeng
                     </i>
                     daftar minuman berenergi yang tidak berbahaya.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="http://1.bp.blogspot.com/-WCy_QlO4pbo/VT0nvXRc8hI/AAAAAAAAEG4/bf_LekBGgk8/s1600/minuman-berenegi.png" style="margin-left: 1em; margin-right: 1em;">
                    <img alt="minuman berenergi yang tergolong aman atau tidak berbahaya untuk dikonsumsi adalah kratingdaeng karena mengandung gula alami." height="230" src="http://1.bp.blogspot.com/-WCy_QlO4pbo/VT0nvXRc8hI/AAAAAAAAEG4/bf_LekBGgk8/s1600/minuman-berenegi.png" title="Minuman Berenergi Kratingdaeng" width="320"/>
                   </a>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Berbahayakah minuman berenergi itu?
                    </span>
                   </span>
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     <i>
                      Minuman berenergi
                     </i>
                     bisa saja berbahaya untuk dikonsumsi jika minuman berenergi tersebut diproduksi dari bahan yang komposisinya tidak memenuhi standar. Faktor lain yang bisa menyebabkan berbahaynya minuman berenergi bagi kesehatan adalah apabila dikonsumsi secara berlebihan atau dengan kata lain tidak sesuai aturan minumnya. Selain itu, pastikan jika minuman berenergi yang dikonsumsi tersebut masih layak minum (Belum Kadaluarsa / Expired). Mengonsumsi minuman berenergi agar tidak menimbulkan dampak buruk bagi kesehatan, sebaiknya dilakukan seteliti mungkin. Terlebih sekarang Banyak minuman berenergi yang telah beredar di pasaran. Jadi, pilihlah minuman berenergi terbaik untuk dikonsumsi seperti Kratingdaeng. Kratingdaeng diproduksi menggunakan bahan gula alami dimana kandungan kafeinnya kurang dari 1 cangkir kopi. Selain itu, Kratingdaeng adalah minuman terkenal sebagai suplemen penambah energi untuk tubuh yang telah memenuhi standar BPOM RI serta telah meraih penghargaan terbanyak dan paling bergengsi untuk kategori minuman energi.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="http://4.bp.blogspot.com/-yYldT8JBpGk/VT0oZrhZ5oI/AAAAAAAAEHA/wNld1AcevPM/s1600/kratingdaeng-minuman-berenergi-yang-aman.png" style="margin-left: 1em; margin-right: 1em;">
                    <img alt="Minuman berenergi Kratingdaeng tergolong minuman berenergi yang aman karena mengandung kafein kurang dari 1 cangkir kopi." height="230" src="http://4.bp.blogspot.com/-yYldT8JBpGk/VT0oZrhZ5oI/AAAAAAAAEHA/wNld1AcevPM/s1600/kratingdaeng-minuman-berenergi-yang-aman.png" title="Kratingdaeng Minuman Berenergi Yang Aman" width="320"/>
                   </a>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Minuman berenergi Kratingdaeng peraih penghargaan selama 15 tahun berturut-turut
                    </span>
                   </span>
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Kratingdaeng untuk pertama kalinya diformulasikan di negara Thailand pada tahun 1962 yang dinaungi oleh TC Pharmaceutical. Dengan menggunakan merek dagang RedBull, Kratingdaeng dipasarkan ke seluruh penjuru dunia dengan standar kualitas dan keamanan internasional. Minuman berenergi Kratingdaeng adalah minuman terbaik Indonesia yang berkelas dunia dengan standar Internasional. Rasa maupun aroma lokal telah disesuaikan dengan selera orang Indonesia. Kratingdaeng merupakan  Minuman Energi Terpercaya Pilihan Indonesia yang meraih Indonesian Customer Satifaction Award (ICSA) selama 15 tahun secara berturut-turut. Suatu kebanggan tersendiri yang dimiliki Kratingdaeng, karena hanya Kratingdaeng yang berhasil meraih Indonesian Customer Satifaction Award (ICSA) sejak pertama diluncurkan. Selain keberhasilan Kratingdaeng dalam meraih Indonesian Customer Satifaction Award (ICSA), Kratingdaeng juga berhasil meraih penghargaan terbanyak dan paling bergengsi dalam kategori minuman berenergi. Penghargaan-penghargaan tersebut diantaranya adalah:
                    </span>
                   </span>
                  </div>
                  <ul>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Indonesian Best Brand Award
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Top Brand Award
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Indonesia Most Favorite Youth Brand
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Wow Brand Champion
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Social Media Award
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Super Brand Award
                     </span>
                    </span>
                   </li>
                  </ul>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Minuman berenergi Kratingdaeng diproduksi dengan menggunakan gula murni
                    </span>
                   </span>
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Minuman berenergi Kratingdaeng yang juga banyak digunakan sebagai minuman olahraga, diproduksi dengan menggunakan gula murni dan dikemas secara praktis sehingga bisa langsung diminum. Selain praktis jika ingin diminum, kandungan Kratingdaeng juga mudah dan cepat diserap didalam tubuh untuk mendapatkan energi. Sesuai dengan SNI 01-668-2002, telah mendefinisikan bahwa minuman berenergi adalah minuman mudah dan cepat diserap oleh tubuh untuk menghasilkan energi dengan atau tanpa bahan tambahan makanan, dengan persyaratan total energy minimal 100 Kkal/sajian. Dan BPOM sendiri mendefinisikan minuman berenergi adalah minuman yang jika diminum dalam jumlah yang wajar sesuai aturan minum perhari, maka dapat menghasilkan energi tidak kurang dari 300Kkal (sesuai dengan KepDirjen POM RI No 02240/B/SK/VII/91), pangan yang dapat memberikan energi minimal 300 Kkal perhari (sesuai dengan BPOM RI No HK 00.05.52.4321). Cakupan dari definisi-definisi mengenai minumen berenergi tersebut di atas,bisa ditemukan pada produk minuman energi Kratingdaeng, dan tentunya aman atau tidak berbahaya untuk dikonsumsi karena diproduksi dengan menggunakan bahan gula murni.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="http://1.bp.blogspot.com/-pYhkCMCnTPk/VT0o-y7Kl5I/AAAAAAAAEHI/Z4gZ1NQpsME/s1600/kratingdaeng-minuman-berenergi-yang-tidak-berbahaya.png" style="margin-left: 1em; margin-right: 1em;">
                    <img alt="Minuman berenergi Kratingdaeng tergolong minuman berenergi yang tidak berbahaya karena diproduksi dengan menggunakan bahan gula murni." height="230" src="http://1.bp.blogspot.com/-pYhkCMCnTPk/VT0o-y7Kl5I/AAAAAAAAEHI/Z4gZ1NQpsME/s1600/kratingdaeng-minuman-berenergi-yang-tidak-berbahaya.png" title="Kratingdaeng Minuman Berenergi Yang Tidak Berbahaya" width="320"/>
                   </a>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Minuman Berenergi Kratingdaeng terdaftar di BPOM dan telah lolos uji keamanan pangan HACCP
                    </span>
                   </span>
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Semua kandungan yang terkandung dalam produk minuman berenergi Kratingdaeng dinyatakan sudah sesuai dengan batas keamanan yang ditetapkan oleh BPOM sehingga Kratingdaeng menjadi minuman energi terbaik yang aman atau tidak berbahaya untuk dikonsumsi 3 botol sehari. Minuman berenergi Kratingdaeng diproduksi melalui pabriknya di Sukabumi, dimana pabrik tersebut sudah memperoleh sertifikasi ISO 9002 dan dinyatakan telah lulus uji keamanan pangan. Setiap produk yang diproduksinya telah melewati proses Hazard Analysis &amp; Critical Control Points (HACCP) yang ketat. Oleh karena itu, minuman berenergi Kratingdaeng merupakan produk minuman berenergi yang tidak berbahaya dikonsumsi dan tidak memiliki efek samping bahkan tidak akan menimbulkan efek ketagihan.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Apa benar minuman berenergi bisa menyebabkan kematian?
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="http://3.bp.blogspot.com/-mgSrGI3UZlo/VT0pmI20HkI/AAAAAAAAEHQ/WcrwAI7wUIY/s1600/kratingdaeng-minuman-berenergi-kelas-dunia.png" style="margin-left: 1em; margin-right: 1em;">
                    <img alt="Minuman berenergi kratingdaeng adalah minuman energi kelas dunia dengan standar Internasional dan dipasarkan di seluruh dunia." height="230" src="http://3.bp.blogspot.com/-mgSrGI3UZlo/VT0pmI20HkI/AAAAAAAAEHQ/WcrwAI7wUIY/s1600/kratingdaeng-minuman-berenergi-kelas-dunia.png" title="Kratingdaeng Minuman Berenergi Kelas Dunia" width="320"/>
                   </a>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Kebanyakan kasus kematian yang ditimbulkan oleh minuman berenergi karena akibat dari over dosis kafein yang terkandung di dalam minuman energi tersebut. Dari beberapa informasi,
                     <b>
                      minuman berenergi
                     </b>
                     yang beredar sekarang utamanya yang diimpor dari luar negeri mengandung kafein yang tinggi kandungannya sampai 500ml/sajian atau 10x lebih tinggi dari kopi. Tingginya kandungan kafein tersebut tentunya akan menyebabkan efek buruk, misalnya jantung akan berdebar diluar normal.
                     <i>
                      <u>
                       Minuman berenergi yang berasal dari luar negeri juga tidak mengantongi izin resmi dari BPOM sehingga dari faktor keamanannya tidak terjamin untuk dikonsumsi
                      </u>
                     </i>
                     . Bahkan juga ada informasi yang menyatakan bahwa produk Kratingdaeng yang diimpor secara illegal dilarang beredar di Indonesia. Produk Kratingdaeng yang diimpor secara illegal tersebut sudah pasti tidak mengantongi izin dari BPOM. Oleh karena itu, agar terhindar dari minuman energi penambah stamina yang dapat menimbulkan kematian, disarankan hanya mengonsumsi produk minuman berenergi yang mendapat dan mencantumkan izin dari BPOM. Minuman berenergi yang aman tidak berbahaya bagi kesehatan terdapat pada produk minuman bernergi Kratingdaeng yang diproduksi di Indonesia.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Minuman berenergi Kratingdaeng mengandung kafein sama dengan satu cangkir kopi
                    </span>
                   </span>
                  </div>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Kafein yang terkandung dalam produk mineman berenergi Kratingdaeng adalah 50ml untuk setiap botolnya. Kandungan kafein sebanyak 50 ml dalam satu botolnya tersebut, sama dengan satu cangkir kopi. Sehingga dengan demikian, produk minuman berenergi Kratingdaeng aman atau tidak berbahaya untuk dikonsumsi setiap hari karena kandungan kafein tersebut sudah memenuhi standar Badan Pengawasan Obat-Obatan dan Makanan (BPOM). Fakta menyangkut kafein berdasarkan data yang tercatat oleh Food and Drug Administration (FDA) Amerika Serikat menyatakan bahwa kandungan yang terdapat pada minuman berenergi Indonesia masih jauh di bawah dari kandungan yang terdapat dalam secangkir kopi, sehingga sangat aman atau tidak berbahaya untuk dikonsumsi setiap harinya.
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Manfaat kandungan yang terdapat dalam minuman berenergi Kratingdaeng
                    </span>
                   </span>
                  </div>

                   </a>

                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Kandungan minuman berenergi Kratingdaeng memiliki banyak manfaat. Vitamin dan komposisi lainnya yang terdapat di dalam Kratingdaeng bermanfaat untuk menambah energi. Kandungan gula yang terdapat di dalam Kratingdaeng merupakan sumber utama dari energi yang dihasilkan. Demikian juga dengan vitamin B kompleks pada minuman berenergi ini, dapat membantu untuk mengkonversi karbohidrat menjadi energi. Kafein akan menstimulasi sistem pada metabolik dan syaraf pusat. Bahkan dari beberapa penelitian, telah diketahui bahwa kafein itu memiliki fungsi yang dapat meningkatkan konsentrasi, kemampuan kognitif, serta dapat memperbaiki mood.
                     <u>
                      Manfaat kandungan lainnya yang terdapat dalam minuman berenergi Kratingdaeng adalah sebagai berikut
                     </u>
                     :
                    </span>
                   </span>
                  </div>
                  <ol>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Glukuronolakton untuk meningkatkan konsentrasi dan daya ingat, serta melindungi liver pada saat bekerja keras.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Lisin (Lysine) untuk membantu proses produksi energy, menurunkan kadar kolesterol membantu proses pertumbuhan tulang, kulit dan pembentukan otot.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Kolin (Choline) untuk melindungi hati dan meningkatkan proses metabolism lemak.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Vitamin B12 (Cyanocobalamine) berperan dalam pembentukan sel darah merah, meningkatkan kerja sistem saraf, membantu dalam proses pembentukan jaringan yang melindungi serabut saraf.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Vitamin B6 (Pyridoxine HCl) membantu metabolism asam amino, lemak dan berperan dalam pembentukan sel darah merah.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Provitamin B5 (Dexpanthenol) adalah komponen koenzim yang berperan penting dalam metabolism karbohidrat, lemak, dan protein.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Vitamin B3 (Niacinamide) adalah komponen koenzim di dalam tubuh yang membantu mengkonversi karbohidrat dan lemak menjadi energi.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Inositol untuk meningkatkan proses metabolism lemak.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Kafein merupakan antioksidan yang banyak ditemukan dalam kopi dan cokelat yang bermanfaat untuk menstimulasi saraf otak untuk meningkatkan proses metabolism dan meremajakan sel-sel dalam tubuh dan meeningkatkan konsentrasi serta kewaspadaan.
                     </span>
                    </span>
                   </li>
                   <li style="text-align: justify;">
                    <span style="font-family: Arial,Helvetica,sans-serif;">
                     <span style="font-size: small;">
                      Manfaat Taurin (Taurine) merupakan asam amino yang banyak terkandung dalam ikan dan laktosa (ASI) dengan beberapa manfaat memperbaiki fungsi kerja jantung, memaksimalkan fungsi otak dan membantu proses sekresi enzim untuk metabolism lemak dalam tubuh.
                     </span>
                    </span>
                   </li>
                  </ol>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: large;">
                     Waktu yang ideal mengonsumsi minuman berenergi Kratingdaeng
                    </span>
                   </span>
                  </div>
                  <br/>
                  <div class="separator" style="clear: both; text-align: center;">
                   <a href="http://2.bp.blogspot.com/-_DUTQsCTSCY/VT0rMu1d_uI/AAAAAAAAEHg/Mn-d1OFOaE4/s1600/minuman-bernergi-untuk-olahraga.png" style="margin-left: 1em; margin-right: 1em;">
                    <img alt="Minuman berenergi kratingdaeng sebagai minuman olahraga yang aman untuk dikonsumsi sebelum dan sesudah melakukan aktivitas olahraga." height="230" src="http://2.bp.blogspot.com/-_DUTQsCTSCY/VT0rMu1d_uI/AAAAAAAAEHg/Mn-d1OFOaE4/s1600/minuman-bernergi-untuk-olahraga.png" title="Kratingdaeng Minuman Berenergi Untuk Olahraga" width="320"/>
                   </a>
                  </div>
                  <br/>
                  <div style="text-align: justify;">
                   <span style="font-family: Arial,Helvetica,sans-serif;">
                    <span style="font-size: small;">
                     Minuman berenergi Kratingdaeng bisa dikonsumsi sebelum dan setelah beraktivitas; Apalagi aktivitas-aktivitas yang memerlukan ketahanan fisik maupun mental yang tinggi. Misalnya, para pelajar sekolah menengah serta universitas sebelum dan setelah melakukan aktivitas rutin mereka di sekolah atau di kampus; para professional sebelum atau setelah melakukan pekerjaan mereka; Para olahragawan sebelum dan setelah melakukan aktivitas olahraga; sebelum menyetir kendaraan terutama untuk jarak jauh; serta pada saat sedang berkumpul atau hang out bersama dengan teman-teman. Kratingdaeng dapat menemani segala macam aktivitas hidup, apapun itu bentuknya. Agar hasilnya lebih maksimal, maka disarankan meminum minuman berenergi Kratingdaeng sekitar 30-60 menit sebelum melakukan aktivitas fisik juga mental. Konsumsi untuk sehari-hari kategori aktivitas ringan cukup mengonsumsi minuman berenergi Kratingdaeng 1 botol sehari secara rutin. Sedangkan untuk aktivitas berat atau aktivitas yang memerlukan energi yang banyak, sebaiknya mengonsumsi minuman berenergi Kratingdaeng sebanyak 3 botol setiap harinya. Pastikan untuk memilih minuman berenergi yang tidak berbahaya untuk dikonsumsi seperti Minuman Berenergi Kratingdaeng yang diproduksi dengan menggunakan gula murni dengan kandungan kafein yang kurang dari 1 cangkir kopi. Terlebih lagi, produk minuman berenergi Kratingdaeng telah mengantongi izin BPOM RI, yang tentunya minuman berenergi Kratingdaeng memenuhi standar untuk dikonsumsi.
                    </span>
                   </span>
                  </div>

                      </i>
                     </b>
                     ", dan semoga informasi ini bermanfaat untuk anda semuanya.
