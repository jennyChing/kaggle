	$j('#ecnPlayer').css('min-height','560px');

	var unix_timestamp = Math.floor(Date.now() / 1000);
	var seconds_week = 60*60*24*7; // 604,800
	
	if ( Math.abs(json_base.cam[_cam].datecreated - unix_timestamp) &lt;= (seconds_week))
		js_forceplaymobile = true;

	// special cases

	{

		{
			sWidth			= "960";
			sHeight			= "540";
			basic_html	= '&lt;div id="ecvrDIV" style="width:'+sWidth+'; height:'+sHeight+'px;"&gt;&lt;/div&gt;';

			flashContainer.html(loading);
			flashContainer.html(basic_html);
			ecvr.init();
			return(1);
