   $("#subscribe_to_trumpet").click( function() {

        // validate before submitting
        var firstname = $("#signup-trumpet input#firstname").val();
        var lastname = $("#signup-trumpet input#lastname").val();
        var email = $("#signup-trumpet input#email").val();
        var city = $("#signup-trumpet input#city").val();
        var state = $("#signup-trumpet input#state").val();
        var country = $("#signup-trumpet input#country").val();

        if((firstname=="") || (lastname=="") || (email=="") || (city=="") || (state=="") || (country=="")) {
            // do nothing
        } else {
            // submit form
            sendMessage("signup-trumpet");
        }
    });
    
    function sendMessage(theForm) {

        event.preventDefault();

        if(theForm == 'signup')
        {
            var firstname = $("form:eq( 1 ) input#firstname").val();
            var lastname = $("form:eq( 1 ) input#lastname").val();
            var email = $("form:eq( 1 ) input#email").val();
            var service = $("form:eq( 1 ) #service").find(":selected").text();
            var address = $("form:eq( 1 ) #address").val();
            var signup = "1";
        }

        if(firstname==undefined) {
            var firstname = $("#signup-trumpet input#fullnames").val();
            var email = $("#signup-trumpet input#email").val();
            var lastname = "";
            var service = "";
            var address = "";
            var signup = "2";
        }

        if( (firstname=='') || (email=='') )
        {
            $("#signup input").css({ "background-color": "#FFD3D3" });
            $("#signup-trumpet input").css({ "background-color": "#FFD3D3" });

        } else {

            var postdata="name="+firstname+"&amp;lastname="+lastname+"&amp;email="+email+"&amp;service="+service+"&amp;address="+address+"&amp;subscribe=1&amp;signup="+signup;

            $("input").val("");
            $("textarea").val("");
            $(".subscribe-button").html("Succesfully Subscribed");

            // $("#submitButton").html("SUBSCRIBING");

              ////////////////POST DATA TO THE SERVER/////////////////////
            $.ajax({
                type: "POST",
                url: "http://savetheelephants.org/sendemail.php",
                data: postdata,
                success: function(response){

                    $("input").val("");
                    $("textarea").val("");

                    // $("button").attr("style", "background:#F2F0E1 !important;color: #921F2B !important;");
                    $(".subscribe-button").html("Succesfully Subscribed");

                    // clear formatting
                    $("input").css({ "border": "1px solid CCC","border-bottom": "2px solid CCC","background-color": "#FFF" });
                    $("textarea").css({ "border": "1px solid CCC","border-bottom": "2px solid CCC","background-color": "#FFF" });

                    if(signup == "2") {
                        $("#"+theForm).submit();
                    }

                }
            });//end the ajax post

            var close_modal = document.getElementsByClassName("modal_close")[0];
            close_modal.click();
        }
    }
  </script>

The programme is best suited to university students or graduates pursuing a career in conservation, with skills in scientific data collection and analysis.
