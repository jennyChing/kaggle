    // CREATE GLOBAL VARIABLE TO HOLD DOCUMENT READY
    $.headerReady = function () {

       


        // BUYING / SELLING GUIDE        

        $('#BuyingAndSellingGuideLnk').click(function (e) {
            //$('.BuyingAndSellingGuideCon').stop();

            if (!$('.m_hdr_glb_nav_drpdwn_pnl').is(":visible")) {
                $('.m_hdr_glb_nav_drpdwn_pnl').show();
                return true;
            } else {
                $('.m_hdr_glb_nav_drpdwn_pnl').hide();
                return true;
            }

        });

        $(document).click(function (e) {
            var $target = $(e.target);

         if ($target.is('#BuyingAndSellingGuideCon *')){
                return true;
            } else {
                $('.m_hdr_glb_nav_drpdwn_pnl').hide();
            }
        });

       
        $('#SellingGuideLnk').blur(function (e) {
            $('.m_hdr_glb_nav_drpdwn_pnl').stop();
            $('.m_hdr_glb_nav_drpdwn_pnl').slideUp();
        });

        // SEARCH BAR
        $('#m_hdr_glb_search_magnify').click(function (e) {
            e.preventDefault();
            startSearch();
        });

        $('#m_hdr_glb_search_input').keypress(function (e) {
            if (e.which == 13) {
                e.preventDefault();
                startSearch();
            }
        });

        // PROPERTY COMPARE
        $('#m_hdr_glb_actions_compare').click(function (e) {
            document.location.href = '/PropertyCompare.aspx';
        });

       

        function startSearch(){

            // Invalidate if the user did not input text
            if( $("#m_hdr_glb_search_input").pVal() === '' )
            {
                //redirectToDisclaimer('/Map.aspx');
                getGeoLocated_EmptySearch();
                return false;
        
            }

            if (location.href.toLowerCase().indexOf('map.aspx') &gt; -1) {
                HashSearch.remove('Area');
                HashSearch.set('Area', $('#m_hdr_glb_search_input').val());
                $('#m_hdr_glb_search_input').val('');
            } else {

                var areaSearch = new SubAreaSearch();
                areaSearch.input = $('#m_hdr_glb_search_input').val(); // INPUT TEXTBOX
                areaSearch.searchType = 'BASIC'; // BASIC || FILTERS
                areaSearch.searchCriteriaCommon += 'CultureId=' + ApplicationSettings.CultureID;
                areaSearch.searchCriteriaCommon += '&amp;ApplicationId=' + ApplicationSettings.ApplicationId;
                areaSearch.searchCriteria = '&amp;Area=' + encodeURIComponent($('#m_hdr_glb_search_input').val());
                areaSearch.Search();
                $('#m_hdr_glb_search_input').val('');
            }

            //$('#m_hdr_glb_search_input').focus();

            return false;

        }

    };

    function GetPreiouslyViewedIds() {

        var previouslyViewed = CookieHandler.getCookieValueAsArray(GlobalData.PRE_VIEWED_MLS_COOKIE);
        var ids = '';
        for (var index = previouslyViewed.length; index &gt;= 0; index--) {
            if (previouslyViewed[index] &amp;&amp; previouslyViewed[index] != '')
                ids += previouslyViewed[index] + ",";
        }

        if (ids.length &gt; 0) {
            ids = ids.substring(0, ids.length - 1);
        }
        
        return ids;
    }
        

    function makeAjaxCall() {

        var theQuery = "ApplicationId=" + ApplicationSettings.ApplicationId + "&amp;cultureid=" + ApplicationSettings.CultureID + "&amp;RecordsPerPage=" + GlobalData.MAX_PRE_MLS + "&amp;Listingids=" + GetPreiouslyViewedIds();

        if (theQuery) {
            AjaxEngine.getResults(ApplicationSettings.ApiPath + "Listing.svc/PropertySearch_Post", theQuery, loadResults, 1, 'POST');
        }
    }

    function loadResults(results) {

        allServerRsults = results;

        LoadPreviouslyViewed();
    }

    function LoadPreviouslyViewed() {

        pageStr = '';
        if (!allServerRsults)
            return;

        var page = Number(HashSearch.get('Page', 1));

        favCookieEndIndex = (GlobalData.MAX_PRE_MLS_PER_PAGE * page);
        favCookieStartIndex = favCookieEndIndex - (GlobalData.MAX_PRE_MLS_PER_PAGE);

        RenderResults(allServerRsults);
                
        event.preventDefault ? event.preventDefault() : event.returnValue = false;
        return true;      
    }

    function RenderResults(results) {
        
        var resultMatched = false;
        var totalFavListLength = null;
        var resultsHTML = '';
        var IDArray = GetPreiouslyViewedIds().split(",");

        activatePager(results.Results.length);
        
        var count = 0;
        var actualResults = [];
        for (var index = 0; index &lt; IDArray.length; index++) {

            var id = IDArray[index].split('_')[0];
            var result = null;

            for (var rindex = 0; rindex &lt; results.Results.length; rindex++) {

                if (results.Results[rindex].Id == id) {
                    actualResults.push(results.Results[rindex]);                  
                    break;
                }
            }          
        }

        for (var index = favCookieStartIndex; index &lt; favCookieEndIndex; index++) {

            if (actualResults[index]) {
                resultsHTML += RenderListing(actualResults[index], index % 2, index);
            }
        }

   

        $('#recentlyViewedListings').html('');
        $('#recentlyViewedListings').html(resultsHTML);

        if (actualResults.length &gt; 4) {
            $('.m_rslt_ftr').show();
        } else {
            $('.m_rslt_ftr').hide();
        }

        // Hide loading
        $('.m_load_icon').slideUp();
        searchInAction = false;
    }

    function ValidateSearch() {

        var errorMessage = '';       
        
        //Sale price range
        if ($('#TransactionTypeID').val() == 2) {
            if (parseInt($('#PriceMax').val()) &lt; parseInt($('#PriceMin').val()) &amp;&amp; parseInt($('#PriceMax').val()) != 0) {
                errorMessage = Message.SalePriceRange;
            }
        }

        //Rent price range
        if ($('#TransactionTypeID').val() == 3) { 
            if (parseInt($('#RentMax').val()) &lt; parseInt($('#RentMin').val()) &amp;&amp; parseInt($('#RentMax').val()) != 0) {
                errorMessage = Message.RentPriceRange;
            }
        }

        //Open house range 
        if ($('#PropertyTypeID').val() === 'ohs' &amp;&amp;  $('#OpenHouseEndDate').val() != '') {
            if ($('#OpenHouseStartDate').val() == '' &amp;&amp; $('#OpenHouseEndDate').val() != '') {
                //errorMessage = Message.InvalidDateRange;
            }
            else if ($('#OpenHouseStartDate').datepicker('getDate').getTime() &gt; $('#OpenHouseEndDate').datepicker('getDate').getTime()) {
                errorMessage = Message.InvalidDateRange;
            }
        }
        
        
     
        if (errorMessage != '') {        
            showMessage(errorMessage);
            return false;
        }

        return true;
    }

    function AddControlValToQuery(paramName, controlId) {
        var returnVal = '';

        if ($('#' + controlId).is(":visible") || $('#dk_label_' + controlId).is(":visible")) {
            switch ($('#' + controlId).prop('type')) {

                case "select-one":
                    if ($('#' + controlId).val() != "0") {
                        returnVal = "&amp;" + paramName + "=" + $('#' + controlId).val();
                    }
                    break;
                case "checkbox":
                    if ($('#' + controlId).prop('checked')) {
                        returnVal = "&amp;" + paramName + "=" + $('#' + controlId).val();
                    }
                    break;

                case "text":
                    if ($('#' + controlId).hasClass('hasDatepicker') &amp;&amp; !/[a-zA-Z]+/.test($('#' + controlId).val())) {
                        try {
                            var d = new Date();
                            returnVal = "&amp;" + paramName + "=" + Math.ceil((d.getTime() - $('#' + controlId).datepicker('getDate').getTime()) / ((1000 * 60 * 60 * 24)));
                        } catch (ex) { }
                    } else if (!$('#' + controlId).hasClass('hasDatepicker') &amp;&amp; $('#' + controlId).val() != '') {
                        returnVal = "&amp;" + paramName + "=" + $('#' + controlId).val();
                    }
                    break;

            }
        }

        return returnVal;
    }

    $("#Keywords").autocomplete({
        source: function (request, response) {
            var results = $.ui.autocomplete.filter(
              (autoCompleteWordArray || []), extractLast(request.term));
            response(results.slice(0, 4));
        },        
        search: function () {
            // custom minLength
            var term = extractLast(this.value);
            if (term.length &lt; 2) {
                return false;
            }
        },
        focus: function () {
            // prevent value inserted on focus
            return false;
        },
        select: function (event, ui) {
            var terms = split(this.value);
            // remove the current input
            terms.pop();
            // add the selected item
            terms.push(ui.item.value);
            // add placeholder to get the comma-and-space at the end
            terms.push("");
            this.value = terms.join(", ");
            return false;
        }
    });
