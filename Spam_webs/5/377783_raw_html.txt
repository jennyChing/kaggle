function setTVScreen(channel) {
        if (noLayers) {
                return
        }
//        var instrux = eval("rmain.document." + range + "telescreen" + styleObj)
//        instrux.visibility = "visible"
        if (channel &gt; maxPreCached) {
		rmain.document.tvScreen.src = imgRefArray[channel]
	  } else {
		rmain.document.tvScreen.src = preCachedArray[channel].src
	  }
// hide previous channel title layer
        var instrux = eval("rmain.document." + range + "titleText" + prevChannel + styleObj)
        instrux.visibility = "hidden"
        var instrux = eval("rmain.document." + range + "headingText" + styleObj)
        instrux.visibility = "hidden"
        if (channel == screen_dark) {
                tvOn = false
                window.clearStatus()
        } else {
                tvOn = true
                currChannel = channel
		prevChannel = channel
                window.clearStatus()
                if (titleArray[channel] != "") {
		        var instrux = eval("rmain.document." + range + "headingText" + styleObj)
		        instrux.visibility = "visible"
		        var instrux = eval("rmain.document." + range + "titleText" + channel + styleObj)
		        instrux.visibility = "visible"
                        window.showStatus("On the RetroTube: "+titleArray[channel])
                }
        }
}
function tvOnOff() {
        if (tvOn) {
                setTVScreen(screen_dark)
        } else {
                setTVScreen(currChannel)
        }
        return false
}
function channelChange() {
        currChannel = currChannel + 1
        if (currChannel &gt; maxChannelSel) {
                currChannel = 1
        }
        setTVScreen(currChannel)
        return false
}

function linkCheck() {
        if (linkArray[currChannel] != "") {

}

function createLayer(nme,x,y,content,z,v) {
        if (noLayers) {
                return
        }
	if (v) {
		vis = 'visible';
	} else {
		vis = 'hidden';
	}
        layerStr = '&lt;' + layerTag + ' id="' + nme + '"' + styleBegin + ' top' + delimBegin + x + delimEnd + ' left' + delimBegin + y + delimEnd;
	layerStr = layerStr + ' z-index' + delimBegin + z + delimEnd + ' visibility' + delimBegin + vis + delimEnd;
	layerStr = layerStr + styleEnd + '&gt;' + content + '&lt;/' + layerTag + '&gt;\n';
        rmain.document.write(layerStr);
}
