    function togglePriceGrid(selectedValue) {
        var s = '';

        for (var i = 0; i &lt; window.PriceGrid.length; i++) {
            var item = window.PriceGrid[i];

            if (item.Edp == selectedValue) {
                s += '&lt;div style="position: absolute; overflow: hidden; padding: 0px 0px 10px 0px; width: 170px; margin: 10px 0px 0px 0px;"&gt;';
                s += '  &lt;div&gt;';
                s += '    &lt;div class="difl tac b" style="width:80px;"&gt;Quantity&lt;/div&gt;';
                s += '    &lt;div class="b tac"&gt;Price&lt;/div&gt;';
                s += '  &lt;/div&gt;';

                var lastCount = 1;

                for (var y = 0; y &lt; item.Data.length - 1; y++) {  
                    var data = item.Data[y];

                    s += '  &lt;div&gt;';
                    s += '    &lt;div class="difl tac" style="width:80px;border-right:solid 1px #eeeeee;"&gt;';

                    if (lastCount == data.End)
                        s += data.End + '+';
                    //else if (data.End &lt; 99999)  
                    else if (data.End &lt; 99999 &amp;&amp; y + 1 &lt; item.Data.length - 1)
                        s += lastCount + ' to ' + data.End;
                    else
                        s += lastCount + '+';

                    lastCount = (data.End + 1);

                    s += '    &lt;/div&gt;';
                    s += '    &lt;div class="tac"&gt;' + data.Price + ' ea.&lt;/div&gt;';
                    s += '  &lt;/div&gt;';
                }

                s += '&lt;/div&gt;';

                return s;
            }
        }

        return s;
    }

    function moveBox() {
        resizeBackground();

        var win = $(window);
        var box = $('.sendemail');

        if (box.length == 0)
            return;

        var w = win.width();
        var h = win.height();
        var height = box.height();
        var width = box.width();
        var marginTop = (h - height) / 2;
        var marginLeft = (w - width) / 2;

        if (h &lt; height)
            marginTop = 0;

        if (w &lt; width)
            marginLeft = 0;

        box.css({ marginTop: marginTop, marginLeft: marginLeft });
    }

    function resizeBackground() {
        var win = $(window);
        var doc = $(document);
        var bg = $('.mask');
        var w = win.width();
        var h = win.height();

        if (doc.height() &gt; h)
            h = doc.height();

        if (doc.width() &gt; w)
            w = doc.width();

        bg.css({ height: h, width: w, opacity: .3 });
    }
