function validate_form(theForm)
{
	if(theForm.fullname.value == "" || theForm.fullname.value == "Full Name")
	{
		alert("Please enter your name.");
		theForm.fullname.focus();
		theForm.fullname.value="";
		return (false);
	}

	if(theForm.email.value == "" || theForm.email.value == "Email")
	{
		alert("Please enter a value for the \"Email\" field.");
		theForm.email.focus();
		theForm.email.value="";
		return (false);
	}

	if((theForm.email.value.indexOf("\@") == -1) || (theForm.email.value.indexOf(".")== -1) || isInvalidateEmail(theForm.email.value))
	{
		alert("Invalid Email Address");
		theForm.email.focus();
		theForm.email.value="";
		return (false);
	}

	if(theForm.phone.value == "" || theForm.phone.value == "Phone")
	{
		alert("Please enter a value for the \"Phone\" field.");
		theForm.phone.focus();
		theForm.phone.value="";
		return (false);
	}

    var regex = /(\d.*){10}/;
	if(!theForm.phone.value.match(regex))
	{
		alert("Please enter your complete 10 digit phone number.");
		theForm.phone.focus();
		return (false);
	}
	return true;
}
