    function SetSelectedTopNav() {

        var PageName = getParameterByName('category');

        $('#main-navigation ul li a').each(function () {
            if (PageName != null) {
                PageName = PageName.replace('%20', ' ')
                var NavName = $.trim($(this).text().toLowerCase());

                if (NavName == PageName.toLowerCase()) {
                    $(this).parent().addClass('active-link');
                    $('#main-navigation ul li:first').removeClass('active-link');
                }
            }
        });
    }

    function getParameterByName(name) {
        var match = RegExp('[?&amp;]' + name + '=([^&amp;]*)').exec(window.location.search);
        return match &amp;&amp; decodeURIComponent(match[1].replace(/\+/g, ' '));
    }
