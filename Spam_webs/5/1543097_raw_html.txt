		function bindFollow()
		{
			$.ajax({
		     url:"http://www.podbean.com/site/userCenter/getFollowStatus",
		     type:"GET",
		     data:"id=721800&amp;type=pbblog",
		     dataType:"jsonp",
		     context:$("#follow"),
		     success:function(data){  
		      
		      if(data.code=="200"&amp;&amp;data.followStatus=="2")
		      {
		      	 $(this).click(function(event) {
		            alert("You don't need to follow yourself.");
		            return false;
		         });
			  } 
		      else if(data.code=="200")
		      {
		       if(data.followStatus==1)
		       {
		        $(this).addClass("followed");
		        $(this).text("Following");
		       } 

		       $(this).click(function(){
		        var node = $(this);
		        if(node.hasClass("followed"))
		              { 
		                $.ajax({
		                    url: "http://www.podbean.com/site/userCenter/follow",
		                    type:"GET",
		                    data:"id=721800&amp;type=pbblog&amp;action=unfollow",
		                    context:node,
		                    dataType:"jsonp",
		                    success: function(data){
		                      if(data.code == 200)
		                      { 
		                       $(this).removeClass("followed");
		                          $(this).addClass("unfollow");
		                          $(this).text("Follow");
		                      }
		                    }
		                });
		            } 
		             else
		             {
		                $.ajax({
		                    url: "http://www.podbean.com/site/userCenter/follow",
		                    type:"GET",
		                    data:"id=721800&amp;type=pbblog&amp;action=follow",
		                    context:node,
		                    dataType:"jsonp",
		                    success: function(data){
		                      if(data.code == 200)

	var timer = null;
	function mobileLoadIframe(){
		if(timer !== null){
			clearTimeout(timer);
		}
		timer = setTimeout(function(){
			
			var window_height,scrollTop;
			window_height 	= $(window).height();
			scrollTop 		= $(window).scrollTop();
			$('.aplayer-panel').not('[data-uri]').each(function(){
				var offset_top = $(this).offset().top;
				var height = $(this).height();
				console.log(offset_top);
				if(offset_top&gt;=scrollTop-height&amp;&amp;offset_top&lt;=scrollTop+window_height){
					(showIframe.bind(this))(true);
				}
			});
		},300);
	}

	function blockPlay(){
		$('.aplayer-panel[rel="1"]').css('background-image','url("/pb-assets/assets/images/player_theme/aplayer-bg.png")')
							.find("iframe").remove()
							.end()
							.attr('rel','0')
							.children('.aplayer-bg')
							.css('display','block');
	}

	}
