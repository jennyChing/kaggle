				flyoutNewsForm.initForm(type, subType);
			} else {
				/* NO COOKIE  */

				flyoutNewsForm.setCookie("ManNewsForm", JSON.stringify(flyoutNewsForm.baseCookie), 1000);
				flyoutNewsForm.initForm(type, subType);
			}
		} else {
			flyoutNewsForm.initForm(type, subType);
		}
	},
	
	initForm: function(type, subType){
		/*  ONLY SHOW IF THEY HAVEN'T ALREADY SIGNED UP  */
		if((flyoutNewsForm.session.signed == false)){
			switch(type){
				case "timed":
					if(flyoutNewsForm.session.pageCount.article &gt;= 1){
						var modal = $('.flyout-newsletter-timed');
						// check if article referral is coming from taboola
						if(window.location.href.indexOf("utm_source=taboola&amp;utm_medium=referral")&gt; -1 ){
							// show newsletter in 5 secs

								if(document.cookie.indexOf("ManNewsModalCancel") &lt;= -1 &amp;&amp; flyoutNewsForm.session.signed == false){
									flyoutNewsForm.showForm(type, modal);
								}
							}, 3000);
						} else {
							setTimeout(function(){
								if(document.cookie.indexOf("ManNewsModalCancel") &lt;= -1 &amp;&amp; flyoutNewsForm.session.signed == false){

								}

						flyoutNewsForm.setCookie("ManNewsForm", JSON.stringify(flyoutNewsForm.session), 1000);

							flyoutNewsForm.initForm(type, subType);
						}
					}
					break;
				case "count":
					/*  ONLY IF NO CANCEL COOKIE  */
					if(flyoutNewsForm.session.pageCount.gallery &gt;= 1){

								isTaboola = true;
							}
							flyoutNewsForm.countInt = setInterval(function(){
								if(thePath != window.location.pathname){
									thePath = window.location.pathname;
									count++;
									if(isTaboola){
										if(count &gt;= 3){
											clearInterval(flyoutNewsForm.countInt);
											flyoutNewsForm.showForm(type, modal);
										}
									} else { 
										if(count &gt;= 10){
											clearInterval(flyoutNewsForm.countInt);
											flyoutNewsForm.showForm(type, modal);
										}
									}
								}

	setCookie: function(name,value,expires) {
		var date = new Date();
		date.setDate(date.getDate()+expires);
		var value = escape(value)+((expires == null) ? "" : "; expires="+date.toUTCString()+";path=/");
		
		document.cookie = name+"="+value;
	},
	
	getCookie: function(name) {
		var i,x,y,cookies = document.cookie.split(";");
		for (i = 0; i &lt; cookies.length; i++) {
			x = cookies[i].substr(0,cookies[i].indexOf("="));
			y = cookies[i].substr(cookies[i].indexOf("=")+1);
			x = x.replace(/^\s+|\s+$/g,"");
			if (x == name) return unescape(y);
		};
	},
