function auto_complete(tp)
{
	var val=document.getElementById('key').value;
	$('#key').removeClass('textbox');
	$('#key').addClass('new_text_key');
	$(".search_new_text").fadeIn(10);
	http.abort();
	http.open("REQUEST","https://handmadeartists.com/ajax/auto_complete.php?type="+tp+"&amp;value="+val, true);
	http.onreadystatechange=function() 
	{
		if(http.readyState == 4) 
		{
			//alert(http.responseText);
			document.getElementById('auto_holder').style.visibility= "visible";
			document.getElementById('auto').style.visibility= "visible";
			document.getElementById('auto').innerHTML=http.responseText;
			//setTimeout('hide_div()', 8000);
		}
	}
	http.send(null);
}

function select_val(val)
{

	$(".search_new_text").fadeOut(10);
	document.getElementById('key').value=val;
	document.getElementById('auto_holder').style.visibility= "hidden";
	document.getElementById('auto').style.visibility= "hidden";
	hide_div();
}
