    function invite_by_email(element){
    	
    	var parent_div = $(element).closest('#invite_by_email');
    	var wrapper_div = $(parent_div).parent();
    	var all_responses = $(wrapper_div).find('.response');
    	var progress = $(parent_div).find('.progress');
    	var response = $(parent_div).find('.response');
    	var txt_area = $(parent_div).find('textarea[name="friends_to_invite"]');
		var emails = $(txt_area).val().trim();

      	if ( emails !== '' ) {
			$(all_responses).hide();
			$(progress).show();
			$(element).attr('disabled','disabled');
			$.ajax({
				url: "/invite_email_send",
				data: {emails: emails, type: 'invite', combine: 'false'},
				type: "POST",
				dataType: "html",
				success: function(r) {
					$(txt_area).val('');
					$(progress).hide();
					$(response).show().html('Okay, invitation sent.  Anyone else?');
					$(element).removeAttr('disabled');
				},
				error: function(er){
					$(progress).hide();
					$(response).show().html('There is an error. Please try later.');
					$(element).removeAttr('disabled');
				}
			});
			_kmq.push(['record', 'Invite invite_email Sent', {'Count':emails.split(',').length}]);
		}
    }

    function fixPartialRows(){    
        var reposition = true;
        $last = $('.similar_products section .product').last();
        var lastPos = $last.position(), thisPos, quit = 10;
        while (true) {
          if (quit-- == 0) break; // traveled too far
          $last = $last.prev();
          if ($last.length == 0) break; // only 1 row?
          thisPos = $last.position();
          if (thisPos.left == lastPos.left) { break;}
          if (thisPos.left &gt; lastPos.left) {
            if (thisPos.top &lt; lastPos.top) {
              // hide everything after this
              while (($last = $last.next()).length &gt; 0) {
                $last.hide();
              }
            }
            break;
          }
        }
    }
