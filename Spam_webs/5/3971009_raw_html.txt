     var searchBox = function() {

	        var fade = function(d){

		        var a = alpha;
		        if((a != endalpha &amp;&amp; d == 1) || (a != 0 &amp;&amp; d == -1)){
			        var i = speed;
			        if(endalpha - a &lt; speed &amp;&amp; d == 1){
				        i = endalpha - a;
			        }else if(alpha &lt; speed &amp;&amp; d == -1){
				        i = a;
			        }
			        alpha = a + (i * d);

			        if (!ie) {
			            oToolTipDiv.style.filter = 'alpha(opacity=' + alpha + ')';

		        }else{
			        clearInterval(timer);

		        }
	        }

        	
	        return {

			        if(oToolTipDiv == null){

				        oToolTipDiv.innerHTML = sTooltipHtml;

    function GlobalNavigationChangeModel(selector){
        var oField=document.getElementById(selector);
        var sUrl=oField.options[oField.selectedIndex].value;
        var sUrlSiteSection=oField.options[oField.selectedIndex].text;
        if(sUrl!=""){
            top.location.href="http://automobiles.honda.com" + sUrl;
        }
        if (GlobalNavigationTrackEvent) {
            GlobalNavigationTrackEvent('selectahonda' , { vehicle : sUrlSiteSection , trackingName : oField.options[oField.selectedIndex].getAttribute('data-trackingname') } );
        }
    }
    function GlobalNavigationChangeShoppingTool(selector){
        var oField=document.getElementById(selector);
        var sUrl=oField.options[oField.selectedIndex].value;
        var sUrlSiteSection=oField.options[oField.selectedIndex].text;
        if(sUrl!=""){
            top.location.href=sUrl;
        }
        
        if (sUrl.indexOf('build') &gt; 0) {
            GlobalNavigationTrackEvent('build');
        } else if (sUrl.indexOf('estimate') &gt; 0) {
            GlobalNavigationTrackEvent('estimate');
        } else if (sUrl.indexOf('compare') &gt; 0) {
            GlobalNavigationTrackEvent('compare');
        } else if (sUrl.indexOf('quote') &gt; 0) {
            GlobalNavigationTrackEvent('quote');
        }

    }
    // This function will redirect to search results page.
    function GlobalNavigationSearchSite(selector){
    
        var searchValue = document.getElementById(selector).value;
        
        if (searchValue == 'Enter search terms to search site or ZIP code to find a dealer') searchValue = '';

        // If 5 digits, do a dealer locator search
        if (searchValue.match(/(\d{5})/g)) {
            top.location = 'http://automobiles.honda.com/tools/dealer-locator/results.aspx?address=&amp;city=&amp;state=&amp;zip=' + searchValue + '&amp;dealername=&amp;filters=&amp;fromsection=NEW';
            GlobalNavigationTrackEvent('dealer');
        } else {
            top.location = "http://automobiles.honda.com/search-results.aspx?question="+escape(searchValue);
            GlobalNavigationTrackEvent('search');
        }
        
    }

     function GlobalNavigationTrackEvent(sEvent , args) {
        
            switch (sEvent) {

                case 'selectahonda':
                    var trackName = (args['trackingName']) ? (args['trackingName']) : (args['vehicle'])
                    CallSDotTL(s, true, 'SELECT A HONDA LINK ON GLOBAL NAV', 'o', { prop1: trackName.toUpperCase(), linkTrackVars: 'prop1,prop25,prop37' });
                    break;
                case 'build':
                    CallSDotTL(s , true , 'BUILD AND PRICE LINK ON GLOBAL NAV' , 'o' , { linkTrackVars : 'prop25,prop37' } );
                    break;
                case 'offers':
                    CallSDotTL(s , true , 'CURRENT OFFERS LINK ON GLOBAL NAV' , 'o' , { linkTrackVars : 'prop25,prop37' } );
                    break;
                case 'estimate':
                    CallSDotTL(s , true , 'ESTIMATE PAYMENTS LINK ON GLOBAL NAV' , 'o' , { linkTrackVars : 'prop25,prop37' } );
                    break;
                case 'compare':
                    CallSDotTL(s , true , 'COMPARE VEHICLES LINK ON GLOBAL NAV' , 'o' , { linkTrackVars : 'prop25,prop37' } );
                    break;
                case 'quote':
                    CallSDotTL(s, true, 'REQUEST A QUOTE LINK ON GLOBAL NAV', 'o', { linkTrackVars: 'prop25,prop37' });
                    break;                  
                case 'dealer':
                    CallSDotTL(s , true , 'FIND A DEALER LINK ON GLOBAL NAV' , 'o' , { linkTrackVars : 'prop25,prop37' } );
                    break;
                case 'search':
                    CallSDotTL(s, true, 'SITE SEARCH LINK ON GLOBAL NAV', 'o', { linkTrackVars: 'prop25,prop37' });
                    break;
                case 'owners':
                    CallSDotTL(s, true, 'OWNERS LINK ON GLOBAL NAV', 'o', { linkTrackVars: 'prop25,prop37' });
                    break;
            }
        
        }
