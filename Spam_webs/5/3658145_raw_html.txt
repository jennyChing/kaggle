   function calculateNewValue() {
        var str = $('#counter').html();
        var count = parseInt(str.replace(/,/g, ''));
        if (count == 'NaN') {
            return;
        }
        count += Math.floor(Math.random() * incrementPerSecond);
        $('#counter').html(addCommas(count + ''));
        setTimeout(calculateNewValue, 5000);
    }

    var incrementPerSecond = 4;
    calculateNewValue();

    function addCommas(strv) {
        var len = strv.length;
        var result = '';
        while (len &gt; 3) {
            len = len - 3;
            result = ',' + strv.substr(len) + result;
            strv = strv.substr(0, len);
        }
        result = strv + result;
        return result;
    }

    }

        }
        return isSupport;
    }
