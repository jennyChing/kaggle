<!DOCTYPE html>
<html>
 <head>
  <meta content="1; ?js=false" http-equiv="refresh"/>
  <meta content="width=device-width, initial-scale=1, maximum-scale=1.0, user-scalable=no, minimal-ui" name="viewport"/>
 </head>
 <script>
  if(navigator.cookieEnabled){
   var width = (window.innerWidth &gt; 0) ? window.innerWidth : screen.width;
   var height = (window.innerHeight &gt; 0) ? window.innerHeight : screen.height;
   document.cookie = "viewport_width=" + Math.min(width, height);
    document.location.reload(true);
}
else{
   document.location.search = '?c=false';
}
 </script>
 <body>
  <!-- Browser feature check-->
 </body>
</html>
