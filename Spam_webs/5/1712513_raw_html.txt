	function insertParam(url, parameterName, parameterValue, atStart){

	    replaceDuplicates = true;

	    if(url.indexOf('#') &gt; 0){

	        var cl = url.indexOf('#');

	        urlhash = url.substring(url.indexOf('#'),url.length);

	    } else {

	        urlhash = '';

	        cl = url.length;

	    }

	    sourceUrl = url.substring(0,cl);



	    var urlParts = sourceUrl.split("?");

	    var newQueryString = "";



	    if (urlParts.length &gt; 1)

	    {

	        var parameters = urlParts[1].split("&amp;");

	        for (var i=0; (i &lt; parameters.length); i++)

	        {

	            var parameterParts = parameters[i].split("=");

	            if (!(replaceDuplicates &amp;&amp; parameterParts[0] == parameterName))

	            {

	                if (newQueryString == "") {

	                    newQueryString = "?";

	                }

	                else {

	                    newQueryString += "&amp;";

	                newQueryString += parameterParts[0] + "=" + (parameterParts[1]?parameterParts[1]:'');

	            	}

	            }

	        }

	    }

	    if (newQueryString == "")

	        newQueryString = "?";



	    if(atStart){

	        newQueryString = '?'+ parameterName + "=" + parameterValue + (newQueryString.length&gt;1?'&amp;'+newQueryString.substring(1):'');

	    } else {

	        if (newQueryString !== "" &amp;&amp; newQueryString != '?')

	            newQueryString += "&amp;";

	        newQueryString += parameterName + "=" + (parameterValue?parameterValue:'');

	    }

	    return urlParts[0] + newQueryString + urlhash;

	};

				function getFrameID(id){

				    var elem = document.getElementById(id);

				    if (elem) {

				        if(/^iframe$/i.test(elem.tagName)) return id; //Frame, OK

				        // else: Look for frame

				        var elems = elem.getElementsByTagName("iframe");

				        if (!elems.length) return null; //No iframe found, FAILURE

				        }

				        elem = elems[i]; //The only, or the best iFrame

				        if (elem.id) return elem.id; //Existing ID, return it

				        // else: Create a new ID

				        do { //Keep postfixing `-frame` until the ID is unique

				            id += "-frame";

				        } while (document.getElementById(id));

				        return id;

				    }

				    // If no element, return null.

				}

          <img alt="top 20 - 10a" class="aligncenter size-full wp-image-3921" height="690" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-10a.jpg" width="995"/>
         </a>
        </p>
        <h2>
         11.
         <a href="http://www.quicksprout.com/" target="_blank">
          Quick Sprout Website Analyser
         </a>
        </h2>
        <p>
         Quick Sprout has a tool that does a
         <a href="http://www.verveuk.eu/website-critique/" target="_blank">
          full analysis of your website
         </a>
         for free. Just enter your website address and it will display a series of results and recommendations. Whats good about the tool is that you can view an overview and then access more detailed information if need be. For example, the report details which elements of the site are taking the longest time to download.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-11b.jpg">
          <img alt="top 20 - 11b" class="aligncenter size-full wp-image-3969" height="768" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-11b.jpg" width="852"/>
         </a>
        </p>
        <h2>
         12.
         <a href="http://www.similarweb.com/" target="_blank">
          Similar Web
         </a>
        </h2>
        <p>
         Compare website traffic with
         <a href="http://www.similarweb.com/" target="_blank">
          SimilarWeb.com
         </a>
         s advanced traffic estimator tool. See any websites traffic sources &amp; uncover their online marketing strategies. SimilarWeb makes web analytics simple and accessible to everyone, showing you the story behind any website with data that will help you analyze your competitors traffic and identify growth opportunities online.
        </p>
        <p>
         <img alt="similar-web" class="aligncenter size-full wp-image-3874" height="598" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-12.jpg" width="1229"/>
        </p>
        <h2>
         13.
         <a href="http://snip.ly/" target="_blank">
          Sniply
         </a>
        </h2>
        <p>
         When sharing content, use
         <a href="http://snip.ly/" target="_blank">
          Sniply
         </a>
         to attach a message to the page and also include a link back to a page of your choice. This is particularly useful if you are directing people away from your site to a related article or through social media and want to encourage them to come back. It is a very simple tool that also includes solid analytical data.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-13a.jpg">
          <img alt="snip.ly" class="aligncenter size-full wp-image-3926" height="525" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-13a.jpg" width="488"/>
         </a>
        </p>
        <h2>
         14.
         <a href="http://moz.com/tools/seo-toolbar" target="_blank">
          Moz Tool Bar
         </a>
        </h2>
        <p>
         The SEO Toolbar from
         <a href="http://moz.com/tools/seo-toolbar" target="_blank">
          Moz
         </a>
         gives you quick access to many on-page SEO factors, Domain &amp; Page Authority plus a quick nofollow toggle. It provides a good general overview on a sites strength which is particularly useful for seeing sites who provide backlinks to your domain; It is also useful for checking how competitors are getting on. The MozBar is a web browser extension thats available for
         <a href="www.google.co.uk/chrome/" target="_blank">
          Google Chrome
         </a>
         and
         <a href="https://www.mozilla.org/en-US/firefox/new/" target="_blank">
          Mozilla Firefox
         </a>
         that provides valuable metrics about a website and its link profile for Search Engine Optimization (SEO) purposes.
        </p>
        <p>
         <img alt="moz-toolbar" class="aligncenter size-full wp-image-3876" height="47" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-14.jpg" width="982"/>
        </p>
        <h2>
         15.
         <a href="http://builtwith.com/" target="_blank">
          BuiltWith
         </a>
        </h2>
        <p>
         Web technology information profiler tool. Again this is very much what it says on the tin, see which technologies websites are utilising e.g. Google Analytics, WordPress etc The goal of
         <a href="http://builtwith.com/" target="_blank">
          BuiltWith
         </a>
         is to help web developers, researchers and designers find out which technologies webpages are using which may help them to decide the technologies that will need to be implemented on their own sites.
        </p>
        <p>
         <img alt="built-with" class="aligncenter size-full wp-image-3877" height="839" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-15.jpg" width="952"/>
        </p>
        <h2>
         16.
         <a href="https://ifttt.com/" target="_blank">
          IFTTT (If this, then that)
         </a>
        </h2>
        <p>
         IFTTT is a service that lets you create powerful connections with one simple statement, e.g.
         <strong>
          If
         </strong>
         you want something to trigger an action to happen.
         <strong>
          This
         </strong>
         could be a date
         <strong>
          that
         </strong>
         you specify to back up contacts or an email that is triggered if a certain weather condition is apparent.
        </p>
        <p>
         <img alt="IFTTT" class="aligncenter size-full wp-image-3878" height="746" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-16.jpg" width="992"/>
        </p>
        <h2>
         17.
         <a href="http://tools.seochat.com/tools/related-keywords-tool/" target="_blank">
          Related Key Words Tool  SEO Chat
         </a>
        </h2>
        <p>
         Discover valuable keyword opportunities by conducting deep research with their Related Keywords Tool. Export a spreadsheet of the related keywords which can used with Google AdWords or for on-page optimization.
        </p>
        <p>
         <img alt="suggestion-keyword-finder" class="aligncenter size-full wp-image-3879" height="558" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-17.jpg" width="992"/>
        </p>
        <h2>
         18.
         <a href="https://socialcrawlytics.com/" target="_blank">
          Social Crawlytics
         </a>
        </h2>
        <p>
         Identify influencers and your competitors most shared content and find out who shared their content. Social Crawlytics is a completely free
         <a href="http://www.verveuk.eu/services/digital-marketing/">
          social media
         </a>
         tool that can be scheduled to crawl periodically.
        </p>
        <p>
         <img alt="social-crawlytics" class="aligncenter size-full wp-image-3880" height="362" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-18.jpg" width="938"/>
        </p>
        <h2>
         19.
         <a href="http://www.semrush.com/" target="_blank">
          SEMrush
         </a>
        </h2>
        <p>
         SEMrush is created by SEO professionals for SEO professionals and is a very good tool for analysing competitors organic and paid performance. This knowledge can help you climb the search engine rankings by getting an edge over competitors.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-19b.jpg">
          <img alt="SEMrush" class="aligncenter size-full wp-image-3959" height="467" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-19b.jpg" width="1085"/>
         </a>
        </p>
        <h2>
         20.
         <a href="http://sitecheck.sucuri.net/" target="_blank">
          Sucuri SiteCheck
         </a>
        </h2>
        <p>
         Sucuri SiteCheck is a free &amp; remote scanner for your website.
         <a href="http://sitecheck.sucuri.net/" target="_blank">
          Sucuri SiteCheck
         </a>
         can identify potential threats to both the owner and user of the website by analysis of out of date web apps and plugins that could make your website vulnerable to attack.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-20a.jpg">
          <img alt="Sucuri Site" class="aligncenter size-full wp-image-3971" height="663" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-20a.jpg" width="801"/>
         </a>
        </p>
        <h2>
         21.
         <a href="http://www.hubspot.com/blog-topic-generator" target="_blank">
          Blog Topic Generator
         </a>
        </h2>
        <p>
         Enter the keyword you want to target and the
         <a href="http://www.verveuk.eu/marketing/71-terrific-templates-for-blog-post-title-ideas/" target="_blank">
          Blog Topic
         </a>
         Generator will give you a list of existing headlines using that keyword to choose from.
        </p>
        <p>
         <img alt="blog-topic-generator" class="aligncenter size-full wp-image-3883" height="860" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-21.jpg" width="779"/>
        </p>
        <h2>
         22.
         <a href="http://www.contentrow.com/tools/link-bait-title-generator" target="_blank">
          Link Bait Generator
         </a>
        </h2>
        <p>
         Generating great content ideas is not always the easiest task, luckily, this
         <a href="http://www.contentrow.com/tools/link-bait-title-generator" target="_blank">
          Link Bait Title Generator
         </a>
         can help you out. A fantastically simple tool giving effective results in producing provocative titles, funny headlines or simple blog lists. It can pose questions or make statements with a few simple keywords.
        </p>
        <p>
         <img alt="link-bait-generator" class="aligncenter size-full wp-image-3884" height="302" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-22.jpg" width="289"/>
        </p>
        <h2>
         23.
         <a href="http://www.woorank.com/" target="_blank">
          Woorank
         </a>
        </h2>
        <p>
         With their free
         <a href="http://www.verveuk.eu/services/digital-marketing/">
          on-page SEO
         </a>
         tools, you can instantly check to make sure your on-page SEO is correct and they will also give you a number of tips on what you can do to improve. The service is designed to let website publishers and marketers evaluate the SEO-friendliness and other aspects of their web sites on the fly, completely free of charge.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-23a.jpg">
          <img alt="WooRank" class="aligncenter size-full wp-image-3970" height="329" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-23a.jpg" width="1022"/>
         </a>
        </p>
        <h2>
         24.
         <a href="http://openlinkprofiler.org/" target="_blank">
          Open Link Profiler
         </a>
        </h2>
        <p>
         Open Link Profiler offers a great deal for a free to use tool.
         <a href="http://openlinkprofiler.org/" target="_blank">
          Open Link Profiler
         </a>
         also has a great filtering system for you to narrow down searches on your backlinks and another great feature is Link Influence Score, this shows the strength or benefit you get from backlinks to your site.
        </p>
        <p>
         <a href="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-25b.jpg">
          <img alt="Open Link Profiler" class="aligncenter size-full wp-image-3958" height="766" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-25b.jpg" width="1163"/>
         </a>
        </p>
        <h2>
         25.
         <a href="https://suite.searchmetrics.com" target="_blank">
          Search Metrics
         </a>
        </h2>
        <p>
         The Searchmetrics Suite is the international enterprise SEO platform for
         <a href="http://www.verveuk.eu/services/digital-marketing/">
          search engine optimization
         </a>
         tools from Searchmetrics. As an overview it shows both Organic and Paid Visibility and it also shows your social visibility, giving accurate numbers that can actively be used to see where you are in the market.
        </p>
        <p>
         <img alt="search-metrics" class="aligncenter size-full wp-image-3888" height="648" src="http://www.verveuk.eu/wp-content/uploads/2014/11/top-20-26.jpg" width="958"/>
        </p>
