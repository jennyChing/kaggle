   $(function() {
        var pull = $('#pull');
            menu = $('#site-navigation'), // This is the local nav for your curren tab
            secondary = $('#secondary');   // This is the actual pull down menu
            if (!secondary.length) {
              secondary = $('&lt;div&gt;');
            }
            menuHeight  = menu.height(),
            menuOpen = false,
            mheight = menu.height(),  // 50ish
            sheight = secondary.height(), // 500ish
            menuSpeed = 20,
            ratio = sheight / mheight,  // 10x
            isMobile = false,
            resizeTimer = 0;

    $(pull).on('click', function(e) {
        if (menuOpen === false){
            openMenu();
        }else if (menuOpen === true){
            closeMenu();
        }
    });

    // Close the pull down menu, then continue and close the
    // site nav menu (appears as one fluid closing menu this way).
    function closeMenu(){
        secondary.slideToggle(menuSpeed * ratio, "easeInCubic", function(){
            menu.slideToggle(menuSpeed, "linear");
            menuOpen = false;
            pull.removeClass('active');
        });
    }

    // Open the site nav first, then open the dropdown. Appears
    // as one fluid motion.
    function openMenu(){
        menu.slideToggle(menuSpeed, "linear", function(){
            secondary.slideToggle(menuSpeed * ratio, "easeOutCubic");
            menuOpen = true;
            pull.addClass('active');
        });
    }

    // Debounce a resize check to see if we're on mobile
    $(window).resize(function(){
        clearTimeout(resizeTimer);
        resizeTimer = setTimeout(checkIfMobile, 250);
    });

    // Using 'isMobile' to keep track of whether or not
    // the nav needs to be shown so that it doesn't get
    // repeatedly shown on every resize event.
    //
    // Also including the variable in the if so that
    // we don't set the variable to true or false on
    // every resize event.
    function checkIfMobile(){
        if (isMobile &amp;&amp; $(window).width() &gt; 760 ){
            isMobile = false;
            menu.show();
            secondary.show();
        }else if (!isMobile &amp;&amp; $(window).width() &lt; 760){
            isMobile = true;
        }
    }
});
  </script>
