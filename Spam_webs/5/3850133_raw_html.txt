// Main openlayers map
var wannaMap;

// World Geodetic System 1984 projection
var WGS84 = new OpenLayers.Projection("EPSG:4326");
 
// WGS84 Google Mercator projection
var WGS84_google_mercator = new OpenLayers.Projection("EPSG:900913");

var wannaControlPanel;
var wannaShowLabels = true;

function wanna_map_init(){
	
	//--- Base map
	wannaMap = new OpenLayers.Map({
		div: "mapBBOX",
		controls:[
	      	     new OpenLayers.Control.Navigation(),    //Allows user pan/zoom ability
		     new OpenLayers.Control.PanZoom(),	     //Displays the pan/zoom tools
		     new OpenLayers.Control.LayerSwitcher(), //Displays a layer switcher
		     new OpenLayers.Control.Attribution(),   //Display attributions
		],
    		projection: WGS84_google_mercator,
		displayProjection: WGS84
	});
	
	var gmap_st = new OpenLayers.Layer.Google("Google Streets",{numZoomLevels: 20});
	var gmap_ph = new OpenLayers.Layer.Google("Google Physical",{type: google.maps.MapTypeId.TERRAIN, numZoomLevels: 20});
	var gmap_hy = new OpenLayers.Layer.Google("Google Hybrid",{type: google.maps.MapTypeId.HYBRID, numZoomLevels: 22});
	var gmap_sa = new OpenLayers.Layer.Google("Google Satellite",{type: google.maps.MapTypeId.SATELLITE, numZoomLevels: 22});
	
	var mapnik = new OpenLayers.Layer.OSM("Mapnik OpenStreetMap");
	mapnik.attribution = "Data, imagery and map information provided by &lt;a href='http://www.openstreetmap.org/' target='_blank'&gt;Open Street Map&lt;/a&gt; and contributors, CC-BY-SA";
	var mapquest_osm = new OpenLayers.Layer.XYZ("MapQuest OpenStreetMap", [
                "http://otile1.mqcdn.com/tiles/1.0.0/map/${z}/${x}/${y}.png",
                "http://otile2.mqcdn.com/tiles/1.0.0/map/${z}/${x}/${y}.png",
                "http://otile3.mqcdn.com/tiles/1.0.0/map/${z}/${x}/${y}.png",
                "http://otile4.mqcdn.com/tiles/1.0.0/map/${z}/${x}/${y}.png"
            ], { attribution: "Data, imagery and map information provided by &lt;a href='http://www.mapquest.com/'  target='_blank'&gt;MapQuest&lt;/a&gt;, &lt;a href='http://www.openstreetmap.org/' target='_blank'&gt;Open Street Map&lt;/a&gt; and contributors, CC-BY-SA",
                transitionEffect: "resize", numZoomLevels: 19
            }
        );
	var esri = new OpenLayers.Layer.XYZ( "ESRI World Topo Map",
                    "http://server.arcgisonline.com/ArcGIS/rest/services/World_Topo_Map/MapServer/tile/${z}/${y}/${x}", {
		    sphericalMercator: true,
		    attribution: "Data, imagery and map information provided by &lt;a href='http://www.arcgis.com'&gt;ESRI ArcGis&lt;/a&gt; mapserver"
		    });
	var ovi_street = new OpenLayers.Layer.XYZ("OVI Street", ["http://a.maptile.maps.svc.ovi.com/maptiler/maptile/newest/normal.day/${z}/${x}/${y}/256/png8",
                                                "http://b.maptile.maps.svc.ovi.com/maptiler/maptile/newest/normal.day/${z}/${x}/${y}/256/png8"], {
						transitionEffect: 'resize',
						sphericalMercator: true,
						numZoomLevels: 21,
						attribution: "Data, imagery and map information provided by &lt;a href='http://api.maps.ovi.com'&gt;OVI&lt;/a&gt; mapserver"
						});
        var ovi_satellite = new OpenLayers.Layer.XYZ("OVI Satellite", ["http://e.maptile.maps.svc.ovi.com/maptiler/maptile/newest/hybrid.day/${z}/${x}/${y}/256/png8",
                                                   "http://f.maptile.maps.svc.ovi.com/maptiler/maptile/newest/hybrid.day/${z}/${x}/${y}/256/png8"], {
						   transitionEffect: 'resize', sphericalMercator: true, numZoomLevels: 21, attribution: "Data, imagery and map information provided by &lt;a href='http://api.maps.ovi.com'&gt;OVI&lt;/a&gt; mapserver"
						   });


	//--- BEGIN Wanna layer
	// Wanna layer styling
	var wannaLayerStyle = new OpenLayers.StyleMap(OpenLayers.Util.applyDefaults({
		 externalGraphic: "http://mw1.google.com/mw-earth-vectordb/dynamic/wannacorp/template/wannasurf/images/gps_exact.png",
		 graphicWidth: 32,
		 graphicHeight: 32,
		 fillOpacity: 1,
		 label: "${name}",
		 labelAlign: "cb",
		 labelYOffset:-20,
		 fontColor: "#666666",
		 fontFamily: "Arial",
		 fontSize: "12px",
		 fontWeight: "normal"
		 },
		OpenLayers.Feature.Vector.style["default"]
	));
	wannaVectorLayer = new OpenLayers.Layer.Vector("Wannasurf surf spots", {
	        attribution: "Wannasurf surf spots provided by Wannasurf.com Ltd",
		strategies: [new OpenLayers.Strategy.Refresh({force:true}), new OpenLayers.Strategy.BBOX({ratio:2, resFactor: 3})],
		projection: WGS84,
		styleMap: wannaLayerStyle,
		protocol: new OpenLayers.Protocol.HTTP({
			url: "/common/services/search-by-coordinates-gmap.php",
			params: {'37df00f4fc7b':'bc7c7576-fd73-4f00-b7cf-dd158729a728'},
			format: new OpenLayers.Format.KML({
				extractStyles: false, 
				extractAttributes: true,
				maxDepth: 2
			})
		})
	});
	wannaSelect = new OpenLayers.Control.SelectFeature(wannaVectorLayer);
	wannaVectorLayer.events.on({
		"featureselected": onFeatureSelect,
		"featureunselected": onFeatureUnselect
	});
	wannaMap.addControl(wannaSelect);
	wannaSelect.activate();
	//--- END Wanna layer
	
	//--- Add label button
	var wannaToggleLayerbutton = new OpenLayers.Control.Button({
	    autoActivate: true,
	    displayClass: "olControlWannaLabel",
	    trigger: wannaLayerVisibility,
	    title: 'Toggle labels'
	});
	wannaControlPanel = new OpenLayers.Control.Panel();
	wannaControlPanel.addControls(wannaToggleLayerbutton);
	wannaMap.addControl(wannaControlPanel);
	wannaLayerVisibility(); // Make sure to hide labels by default
	
	//--- Common Popup management				
	function onFeatureSelect(event) {
	  var feature = event.feature;
	  // Since KML is user-generated, do naive protection against Javascript.
	  var content = "&lt;h2&gt;"+feature.attributes.name + "&lt;/h2&gt;" + feature.attributes.description;
	  if (content.search("&lt;script") != -1) {
	     content = "Content contained Javascript! Escaped content below.&lt;br /&gt;" + content.replace(/&lt;/g, "&lt;");
	  }
	  popup = new OpenLayers.Popup.FramedCloud(
	                            "wannaSpotBubble", 
                                    feature.geometry.getBounds().getCenterLonLat(),
                                    new OpenLayers.Size(150,100),
                                    content,
                                    null,
				    true
				    );
	  feature.popup = popup;
                 wannaMap.addPopup(popup);
	}
	function onFeatureUnselect(event) {
	  var feature = event.feature;
	  if(feature.popup) {
	    wannaMap.removePopup(feature.popup);
                   feature.popup.destroy();
                   delete feature.popup;
	  }
	}

	//--- Add layers
	wannaMap.addLayers([ gmap_hy, gmap_st, gmap_ph, gmap_sa,
	               mapnik, mapquest_osm, esri,
		       ovi_street, ovi_satellite,
	                wannaVectorLayer,
		      ]);
	wannaMap.zoomToExtent(new OpenLayers.Bounds(-117.588686943050000,33.382791193367000,-117.588686943050000,33.382791193367000).transform(WGS84, wannaMap.getProjectionObject()));
	
	if (wannaMap.getZoom() &gt; 16) {
	  wannaMap.zoomTo(14);
	}
	
} // End wanna_map_init

//--- Toggle labels
function wannaLayerVisibility() {
  if (wannaShowLabels) {
    wannaVectorLayer.styleMap.styles.default.defaultStyle.label = '';
    wannaVectorLayer.redraw();
    wannaShowLabels = false;
  }
  else {
    wannaVectorLayer.styleMap.styles.default.defaultStyle.label = "${name}";
    wannaVectorLayer.redraw();
    wannaShowLabels = true;
  }
}
