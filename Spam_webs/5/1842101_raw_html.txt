    var ShopifyLogin = function () {

        this.setServiceUser = function (user, service) {
            FB.api('/me', function(response) {
                ssw.ajax({
                    type: 'POST',
                    url: '/apps/ssw/lite2/auth/setService',
                    data: {'user_id': user.authResponse.userID, 'access_token': user.authResponse.accessToken, 'user': response, 'service': service, '_sid': ssw.cookie("hesid")},
                    success: function (json) {

                            ssw(ssw('.socialconnect')[1]).prev().remove();

        this.login = function () {
            onOpenForm();
            loginLoader();
            var serviceCheckId = 0;

            if (service_id == 0 &amp;&amp; sl == 1)
                serviceCheckId = 1;
            else
                serviceCheckId = service_id;

            var data = ssw('#he_customer_login').serialize() + '&amp;_sid=' + ssw.cookie('hesid') + '&amp;hash_key=' + ssw.cookie('hash_key') + '&amp;service_id=' + sswSocialLogin;
            if(typeof sswCookie('cart') != 'undefined'){
                data = data + '&amp;cart_token=' + sswCookie('cart');
            }

            var url = getParameterByName('checkout_url');
            if (url != "") {
                var l = url.split('/');
                data = 'form_type=customer_login&amp;utf8=&amp;checkout_url=' + encodeURIComponent(url) + '&amp;cart=' + l[l.length - 1] + '&amp;utf8=%E2%9C%93&amp;customer%5Bemail%5D=' + ssw('#he_customer_login #customer_email').val()  + '&amp;customer%5Bpassword%5D=' + ssw('#he_customer_login #customer_password').val();
            }
            ssw.ajax({
                type: 'POST',
                url: '/account/login',
                data: data,
                beforeSend: function () {
                    onSubmitForm();
                    ssw('#hesy-login-submit').attr('disabled', 'disabled');
                    ssw('#login_modal .ssw-alert').sswAlert('close');
                },
                success: function (html) {
                    html = html.split('script').join('div');
                    var htmlElement = ssw('&lt;div&gt;').html(html);
                    var cid = htmlElement.find('#ssw_cid').val();
                    if (cid == 0) {
                        onOpenForm();
                        loginLoaderCancel();
                        ssw('#login_modal').sswModal('show');
                        if (sswSocialLogin) {
                            trackShopStats('invalid_social_login', 'all');
                        } else {
                            trackShopStats('invalid_email_login', 'all');
                        }

                        ssw('#hesy-login-submit').removeAttr('disabled').attr('enabled', 'enabled');
                        if (sl == 1) {
                            addMessage('#login_modal .login-failed', "Right now we&#39;re not able to log in you socially. Please type your email and password in the form below to log in.", 'warning');
                        } else {
                            addMessage('#login_modal .login-failed', "Invalid log in credentials." , 'warning');
                        }
                    } else {
                        ssw.post('/apps/ssw/lite2/auth/login', data, function (result) {
                            if (result.error) {
                                if (sswSocialLogin) {
                                    trackShopStats('invalid_social_login', 'all');
                                } else {
                                    trackShopStats('invalid_email_login', 'all');
                                }

                                if(typeof sswCookie('cart') == 'undefined' &amp;&amp; result.cart){
                                    ssw.cookie('cart', result.cart, {
                                        expires: 14,
                                        path: '/'
                                    });
