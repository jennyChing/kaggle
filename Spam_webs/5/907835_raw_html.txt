function rndCase(s){
    var o = ""
    for (var i=0; i&lt;s.length; i++){
        if(Math.random() &gt; .5){
            o += s[i].toUpperCase();
        }else{
            o += s[i].toLowerCase();
        }
    }
    return o;
}

function convert(){
    var re;
    var html = document.getElementById('htmlin').value;
    
    // doctype
    re = new RegExp('&lt;!DOCTYPE.*?&gt;');
    html = html.replace(re, '&lt;!DOCTYPE html&gt;')
    
    // closing tags
    re = new RegExp("&lt;/(td|tr|p|li|th)&gt;", "g")
    html = html.replace(re, function(m,a){
        if(Math.random() &gt; .5){
            return m;
        }else{
            return "";
        }
    });
    
    // no more tags
    re = new RegExp("&lt;/?(html|head|body).*?&gt;", "g")
    html = html.replace(re, "");
    
    // no more tags
    re = new RegExp("&lt;(script|style)(.*?)&gt;", "gi")
    html = html.replace(re, function(m,t,a){
        return '&lt;' + t + '&gt;';
    });
    
    // shorter meta charset
    re = new RegExp("&lt;meta.*?content=\"text/html; charset=UTF-8\".*?/&gt;", "gi")
    html = html.replace(re, "&lt;meta charset=\"UTF-8\"&gt;");
    
    // tag letters
    re = new RegExp("(&lt;/?)([^! &gt;]+)", "g")
    html = html.replace(re, function(m,t,a){
        return t + rndCase(a);
    });
    
    // tag letters
    re = new RegExp("&lt;(img|meta|input)(.*?)( ?/&gt;)", "gi")
    html = html.replace(re, function(m,t,a){
        return '&lt;' + t + a + '&gt;';
    });
    
    // tag letters
    re = new RegExp("(&lt;[a-z]+)\\s(.*?)(&gt;)", "gi")
    html = html.replace(re, function(m,b,a,c){
        var re = new RegExp("([a-z]+)(=\")(.*?)(\")", "gi");
        a = a.replace(
                re, function(m,a,e,v,q){
                        
                    var single = false
                    
                    var x = m.indexOf("disabled=\"");
                    if (x &gt; -1 &amp;&amp; x &lt; 1){
                        single = true;
                    }
                    
                    var x = m.indexOf("checked=\"");
                    if (x &gt; -1 &amp;&amp; x &lt; 1){
                        single = true;
                    }
                    
                    var x = m.indexOf("selected=\"");
                    if (x &gt; -1 &amp;&amp; x &lt; 1){
                        single = true;
                    }
                    
                    
                    if (single){
                        if(Math.random() &gt; .5){
                            v = "";
                            e = '';
                            q = "";
                        }
                        
                        if (v != ""){
                            if(Math.random() &gt; .5){
                                e = '=';
                                q = "";
                            }
                        }
                    }else{
                        if (v.indexOf(' ') == -1){
                            if(Math.random() &gt; .5){
                                e = '=';
                                q = "";
                            }
                        }
                    }
                    return rndCase(a) + e +  v + q;
                }
            )
        return b + ' ' + a + c;
    });
    
    document.getElementById('htmlout').value = html;
}
