           function doClick(buttonName, e)	{
		//the purpose of this function is to allow the enter key to 
		//point to the correct button to click.
		var key;
		if (window.event)	{key = window.event.keyCode}     //IE
		else	{key = e.which}     //firefox

		if (key == 13)	{
			//Get the button the user wants to have clicked
			var btn = document.getElementById(buttonName);
			if (btn != null)	{ //If we find the button click it
				btn.click();
				event.keyCode = 0
				}
			}
		}
		
	function getCheckedRadio(ctl) {
        var radioButtons = document.getElementsByName(ctl);
        for (var x = 0; x &lt; radioButtons.length; x++) {if (radioButtons[x].checked) { return radioButtons[x].value; }}
		}
      
    function getDropDown(ctl)	{
		var ele = document.getElementById(ctl);
		return ele[ele.selectedIndex].value;
		}
		
	function encodeText(s)	{
		return s.replace(/^\s+|\s+$/g, '').replace(/\s/, '%20').replace(/\//g, '%2f');
		}
		
	function isDate(sDate) {
		var scratch = new Date(sDate);
		if (scratch.toString() == "NaN" || scratch.toString() == "Invalid Date") {return false;} 
		else { return true; } 
		}

	function searchSite(queryCtl, imgCtl, artistCtl, frM, frD, frY, todM, todD, todY, scopeCtl)	{
		//searchstr = String.Format(searchstr, txtQuery.ClientID, rblImageType.ClientID, ddlArtists.ClientID, ddlFromMonth.ClientID, ddlFromDay.ClientID, ddlFromYear.ClientID, ddlToMonth.ClientID, ddlToDay.ClientID, ddlToYear.ClientID)
		var query = document.getElementById(queryCtl).value;
		var img = getCheckedRadio("ctrlSidebar:search:rblImageType");
		var scope = getCheckedRadio("ctrlSidebar:search:rblScope");
		var lang = getCheckedRadio("ctrlSidebar:search:rblLanguage");
		var artist = getDropDown(artistCtl);
		var fromD = getDropDown(frM) + '/' + getDropDown(frD) + '/' + getDropDown(frY);
		var toD = getDropDown(todM) + '/' + getDropDown(todD) + '/' + getDropDown(todY);
		
		var s = '/search.aspx?cmd=4&amp;mode=Advanced';
		s += '&amp;artist=' + artist + '&amp;zone=' + scope + '&amp;type=' + img + '&amp;lang=' + lang;
		if (query != '') {s += '&amp;query=' + encodeText(query);}
		if (isDate(fromD))	{s += '&amp;from=' + encodeText(fromD);}
		if (isDate(toD))	{s += '&amp;to=' + encodeText(toD);}
		
		document.location = s;

		return false;
		}
		
	function showAdv(dv, img)	{
		var ele = document.getElementById(dv);
		if (ele.style.display == 'block')	{ele.style.display = 'none'; img.src = '/lib/themes/default/arrdwn.gif';}
		else {ele.style.display = 'block'; img.src = '/lib/themes/default/arrup.gif';}
		}
