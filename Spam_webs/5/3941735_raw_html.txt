    function hideShowSectionIfItHasCheats(section) {
            section = $(section);
            section.show();

            var visibleItemCount = section.find('li:visible').size();

            if(visibleItemCount == 0) {
                section.hide();
            }
        }

        function getCurrentCheatSection() {
            var clicked_link = $("#cheats_filter li.active a").attr('href');

            // the section slug is stored in the hash of the clicked link!
            var section = clicked_link.split("#").pop();
            return section;
        }

        /*
            Display only cheats with the given section.
         */
        function filterByCheatSection(section) {
            var cheatSections = $('.cheat_section');
            
            if(section == "all") {
                cheatSections.each(function(i) {
                    $(this).show();
                    $('#walkthroughs-content').fadeIn('slow');
                });
            }
            else {
                cheatSections.each(function(i) {
                    if($(this).hasClass(section)) {
                        $(this).fadeIn('slow');
                    }
                    else {
                        $(this).hide();
                    }
                });
            }

            hideShowSectionIfItHasCheats(section);
        }

        function filterByPlatform(evt) {

            var platformId = $(evt.target).val();
            var cheat_section = getCurrentCheatSection();

            log('platformId [' + platformId + '], cheat_section [' + cheat_section + ']');


            if(platformId == 'all') {
                $('.list_page_item').fadeIn('slow');
            }
            else {
                $('.list_page_item').hide().filter('.'+platformId).show();
            }

            $('.cheat_section').each(function(i) {
                var section = $(this);
                hideShowSectionIfItHasCheats(section);
                section.find('li:visible').addClass('firstvisiblecheat');
            });
            
            // Finally filter on selected section
            if (cheat_section != 'all') {
                filterByCheatSection(cheat_section);
            }
        }


        /*
            Game Level Cheats filter navigation

            Filter by platform, or subsections: unlockables, hints, cheats, easter-eggs
            guides button links to guides template. 
            
            Hash ID allows history and linkback from guides
            TODO  8/13/13  -  ^ could use html5 push/pop state now maybe!
          
            Grab the hash if visitng from another URL. If no hash,
            set to all, otherwise, filter the content.
         */
        var cheatNavSelected = $("#cheats_filter li.active");

        $('#cheats_filter li a').click(function(e) {

            clicked_link = $(this).attr("href");

            if(clicked_link.endsWith("/grand-theft-auto-v/walkthroughs-guides/")) {
                return;  // follow the link, for SEO purposes
            }

            e.preventDefault();

            // first update the cheat filter links...
            $(cheatNavSelected).removeClass('active');
            $(this).parent().addClass('active');
            cheatNavSelected = $(this).parent();

            // the section slug is stored in the hash of the clicked link!
            var section = clicked_link.split("#").pop();
            filterByCheatSection(section);
            window.location.hash = '#' + section;
        });
