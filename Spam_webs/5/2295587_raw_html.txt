      function changeScreen(id) {
	document.getElementById('productScreen').innerHTML = unescape(document.getElementById(id).innerHTML);
	
	//resetup the tabs
	$('#infoTabs').tabs();
	magnify();
	setCart();
	updateOptions(id);
}

function changeSwatch(swatchID) {
	if (swatch = document.getElementById('swatch'+swatchID)) {
		document.getElementById('currentSwatch').innerHTML = swatch.innerHTML;
	}
}

function magnify() {
	//reset the image zoomer
	$(".jqzoom").jqzoom({
		zoomType: "standard",
		preloadText: "Loading Hi Res",
		preloadPosition: "center",
		zoomWidth: 600,
		zoomHeight: 600,
		xOffset: 6,
		yOffset: 0,
		position: "right"
	});
}

// loadProductImage
function loadProductImage(imageURL, huge_img) {
    if (gImageCapableBrowser) {
        $('#trashy_magnifying_glass').attr('href', huge_img);
        $('#mainImage').attr('src', imageURL);

        return false;
    } else {
        return true;
    }
}

function updateOptions(product_id) {
	$.ajax({
		url: '/AJAX/admin_size_color.php',
		data: { product_id: product_id, action: 'get' },
		dataType: 'json',
		success: function(data) {
			if (data) {
				//console.log(data);
				$("#attribute_1").html('');
				$("#attribute_3").html('');
				window.sizetocolor = data;
				
				$('#attribute_3').append($('&lt;option&gt;', { value: '' }).text('Select Size'));
				for (i in data) {
					$('#attribute_3').append($('&lt;option&gt;', { value: i }).text(attributes.Size[i]));
				}
				$("#attribute_3").change(function() { updateColor(this[this.selectedIndex].value); });;
			}
		}
	});
}

function updateColor(size) {
	$("#attribute_1").html('');
	$('#attribute_1').append($('&lt;option&gt;', { value: '' }).text('Select Color'));
	for (i in sizetocolor[size]) {
		$('#attribute_1').append($('&lt;option&gt;', { value: sizetocolor[size][i] }).text(attributes.Color[sizetocolor[size][i]]));
	}
} 

//on load get size to color options!
$(document).ready(function() {
	updateOptions($("#item_product_id").val());
});
