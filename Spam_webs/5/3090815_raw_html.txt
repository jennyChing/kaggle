    function signup() {

    function addGender(gender) {

            id: 'newsletter-bar-gender',
            name: 'gender',
            value: gender
        }).appendTo('#newsletter-bar-form');
    }


    /*
     * Get error message for backend validation
     */
    function getErrorMsg(key) {
        if (key === 'newsletter.bar.name.missing') {

            return "Please enter your email address";
        } else if (key === 'newsletter.bar.gender.missing') {
            return "Please select your gender";
        } else if (key === 'newsletter.bar.email.invalid') {
            return "Please enter a valid email address";
        } else if (key === 'newsletter.bar.tryagain') {
            return "Try again";
        }
