function BTC_Redir(url, target) { 
    if (url != '') { 
        if ((target == '_blank') || (target == '_new')) { 
            window.open(url); 
            return true; 
        }
        else if (target=='_self')
        {
            this.location.href = url;
            return true;
        }
        else if (target != '') {
            var frame;
            if(parent &amp;&amp; parent.frames) {
                frame = parent.frames[target];
            }
            try
            {
                if(!frame || (frame.CheckChanges &amp;&amp; !frame.CheckChanges()))
                {
                    return false;
                }
            }
            catch (ex) { 
                // When not a web page
            }
            frame.location.href = url; 
            if(typeof(frame.focus) == 'function'){
                frame.focus();
            }
            return true;
        } 
        else {
            parent.location.href = url; 
            return true; 
        } 
    }  
    
    return true; 
}
