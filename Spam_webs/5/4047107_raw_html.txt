   jQuery(document).ready(function($) {

  var winwidth = $(window).width();

  $(window).resize(function() {
      clearTimeout(this.id);
      this.id = setTimeout(doneResizing, 500);

  function doneResizing(){
    if ($(window).width() != winwidth) {
      window.location.reload();
    }
  };

  if ( !$('#text-10').length &amp;&amp; console) {
    console.log('"#text-10" element not found, please check Sticky Element plugin settings');
    return;
  }
  if ( !$('#footer').length &amp;&amp; console) {
    console.log('"#footer" element not found, please check Sticky Element plugin settings');
    return;
  }

  if (winwidth &gt;= 900) {

    var el = $("#text-10");
    var elwidthpx = el.css('width');
    var elleftmar = el.css('margin-left');
    var eltop;
    var elheight;
    var elleft = el.offset().left;

    $('&lt;div/&gt;').attr('id', 'clone').css('width',elwidthpx).insertAfter( $('#text-10') );

    var style = document.createElement('style');
    style.type = 'text/css';
    style.innerHTML = '.sticky-element-fixed { position: fixed; top: 0; width: ' + elwidthpx + ' !important; }';
    document.getElementsByTagName('head')[0].appendChild(style);

    $(window).scroll(function() {

      if (typeof eltop === "undefined" ) {
        eltop = el.offset().top;
      }
      elheight = el.outerHeight();

      var end = $("#footer");
      endtop = end.offset().top;
      var winscroll = $(window).scrollTop();

      if (winscroll &gt; eltop) {
        $("#text-10").addClass('sticky-element-fixed');
        $("#text-10").css("left", elleft);
        $("#text-10").css("margin-left", 0);
        $('#clone').css('height',elheight);
      } else {
        $("#text-10").removeClass('sticky-element-fixed');
        $("#text-10").css("left", "auto");
        $("#text-10").css("margin-left", elleftmar);
        $('#clone').css('height',0);
      }

      if (winscroll + elheight &gt; endtop) {
        var amount = endtop - (winscroll + elheight);
        $("#text-10").css("top", amount + "px");
      } else {
        var amount = endtop - (winscroll + elheight);
        $("#text-10").css("top", "");
      }

    });
  }

});
