                         Option Explicit
 
Sub Gantt_Chart()
    'Define the variables.
    Dim rge As String
    Dim ValueAxisMinValue As Date
    Dim shtname As String
    Dim Title As String, aChart As Chart
    'Store the location of the data as a string.
    rge = Selection.Address()
    'Store the start date for the chart.
    ValueAxisMinValue = Selection.Cells(2, 2).Value
    'Ask user for the Chart title.
    Title = InputBox("Please enter the title")
    'Store the sheet name.
    shtname = ActiveSheet.Name
    'Turn off screen updating.
    Application.ScreenUpdating = False
    'Create a chart located on a chart sheet.
    Set aChart = Charts.Add
    With aChart
        .ChartWizard Source:=Sheets(shtname).Range(rge), _
        Gallery:=xlBar, Format:=3, PlotBy:=xlColumns, _
        CategoryLabels:=1, SeriesLabels:=1, HasLegend:=1, _
        Title:=Title, CategoryTitle:="", ValueTitle:="", _
        ExtraTitle:=""
        'Remove the legend.
        .Legend.Delete
        'Create and format the series.
        With .SeriesCollection(1)
            With .Border
                .Weight = xlThin
                .LineStyle = xlNone
            End With
            .InvertIfNegative = False
            .Interior.ColorIndex = xlNone
        End With
        'Modify the category (x) axis.
        With .Axes(xlCategory)
            .ReversePlotOrder = True
            .TickLabelSpacing = 1
            .TickMarkSpacing = 1
            .AxisBetweenCategories = True
        End With
        'Modify the value (y) axis.
        With .Axes(xlValue)
            .MinimumScale = ValueAxisMinValue
            .MaximumScaleIsAuto = True
            .MinorUnitIsAuto = True
            .MajorUnitIsAuto = True
            .Crosses = xlAutomatic
            .ReversePlotOrder = False
            .ScaleType = False
            .HasMajorGridlines = True
            .HasMinorGridlines = False
        End With
     End With
    'Turn screen updating back on.
    Application.ScreenUpdating = True
 
 End Sub

            var getColumnHeight = function ()
            {
                var height = 0;
                var totalVisibleDivs = $thinColumn.children('div').not(':hidden');
                var divs = totalVisibleDivs.each(function (index, value)
                {
                    height += $(this).outerHeight(true);
                });

                topMargin = parseInt($(totalVisibleDivs[0]).css('margin-top'), 10);
                bottomMargin = parseInt($(totalVisibleDivs[$(totalVisibleDivs).length - 1]).css('margin-bottom'), 10);

                return height;
            };

            $window.on('scroll resize ', function ()
            {
                var columnHeight = getColumnHeight();
                var columnOffsetTop = scrollTop($primaryThinColumn);
                var left = scrollLeft($primaryThinColumn) + parseInt($primaryThinColumn.css('padding-left'), 10);

                if (columnOffsetTop - topSpace &lt; 0)
                {
                    $thinColumn.css({ 'position': 'fixed', 'top': (topSpace - topMargin) + 'px', 'left': left + 'px' });
                    $chatDiv.css({ 'left': left + 'px' });
                }

                var footerOffsetTop;
                if ($footer.length &gt; 0)
                {
                    footerOffsetTop = scrollTop($footer);
                }
                else
                {
                    footerOffsetTop = columnOffsetTop + $primaryThinColumn.height();
                }

                if (footerOffsetTop - columnHeight - bottomSpace - topSpace + bottomMargin &lt; 0)
                {
                    $primaryThinColumn.css({ 'position': 'relative' });

                }
                else if (columnOffsetTop &gt; topSpace)
                {
                    $thinColumn.css({ 'position': 'relative', 'top': 'auto', 'left': 'auto' });
                    $chatDiv.css({ 'top': 'auto' });
                }
            });

        {
            // If Url is passed open new window
            if (data.hasOwnProperty('URL') &amp;&amp; data.URL)
            {
                window.open(data.URL);
                return;
            }

            if (data.hasOwnProperty('PID') &amp;&amp; data.PID)

                // If it is more product redirect to all products page.
                if (data.PID === '00000')
                {
                    var allProductsUrl = '/AllProducts';
                    if (data.UseTarget)
                    {
                        window.open(allProductsUrl);
                    }
                    else
                    {
                        window.location.assign(allProductsUrl);
                    }
                }
                else
                {
                    var productHubUrlFormat = '/Product/{0}';
                    var formattedProduct = "";
                    if (data.ProductObj.Value.hasOwnProperty('Productlist') &amp;&amp; data.ProductObj.Value.Productlist)
                    {
                        var versionKey = data.ProductObj.Value.Productlist[0].Key.replace(' ', '-').replace('.', '-')

                    else
                    {
                        formattedProduct = data.PName.replace(' ', '-').replace('.', '-');

                    else
