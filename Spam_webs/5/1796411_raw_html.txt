		function getEpoParams() {
			var parts = document.referrer.replace(/^https?:\/\//, '').split('/');
			var defaultCatId = 1111;
			var defaultTypeId = 2;
			var epoParams = "module.epo";

			parts.shift();
			
			// From HOMEPAGE; show default typeId articles 
			if (parts.join('/') == "" &amp;&amp; document.referrer.indexOf(document.domain)) {
				
				
				epoParams += "&amp;typeId=" + defaultTypeId + "&amp;referrer=home";
				
			}
			// From ARTICLE: Show articles w referrer catId 
			else if (document.referrer != undefined &amp;&amp; document.referrer.indexOf('article') &gt;= 0) {
				var a = document.createElement('a');
				a.href = document.referrer;
				var uriParts = a.pathname.split('/');
				a = '';
				if (typeof uriParts[3] == 'undefined') {
					epoParams += "&amp;typeId=" + defaultTypeId + "&amp;referrer=home"; // default is 'home' behavior
				}
				else {
					var refCatSlug = uriParts[3];
					epoParams += "&amp;catSlug=" + refCatSlug + "&amp;referrer=article";
				}
			}
			// From SEARCH: Show article with catId same as current article
			else if (document.referrer.indexOf("google") &gt;= 0 || document.referrer.indexOf("yahoo") &gt;= 0 || document.referrer.indexOf("bing") &gt;= 0) {
				var categories = [2206];
				if (categories instanceof Array &amp;&amp; categories.length &gt; 0) {
					var primaryCatId = categories[0];
					epoParams += "&amp;catId=" + primaryCatId + "&amp;referrer=search";
				}
				else {
					epoParams += "&amp;typeId=" + defaultTypeId + "&amp;referrer=home"; // default is 'home' behavior
				}
			}
			// Default is to show like coming from homepage 
			else {
				
				
				epoParams += "&amp;typeId=" + defaultTypeId + "&amp;referrer=home";
				 // default is 'home' behavior
			}
			return epoParams;
		}
		
		/** 
		 * @param jqo Original jquery object target
		 * @param divString The div to be inserted.
		 * @return Difference in height between original placement target and final target.
		 * Checks first 6 elements for an allowable placement (600 pixel window).
		 * If none, check nearby for elements that are not right avoids.
		 * If none, place element before current target.
		 */
		function applyInsert(jqo, divString) { 
			if (debug) {
				console.log("applyInsert at top and jqo index is: " + jqo.index());
			}
			
			for (var i=0; i&lt;=6; i++) {
				$thisElement = jqo.nextAll().andSelf().slice(i, i+1);
				if (debug) {
					console.log("Checking first six and i is: " + i + " and this element index is " + $thisElement.index() );
				}
				if ($thisElement.index() &lt; 0) {
					break;
				}
				if (allowPlacement($thisElement)) {
					return addElement(jqo, $thisElement, divString);
				}  
			}
			// If got here, no good place to put it.. just put it before one that is not right avoid! 
			var $allowElement = null;
			if (($allowElement = findNearbyAllow(jqo)) != null) {
				return addElement(jqo, $allowElement, divString);
			}
			else {
				// nothing good so put in first spot that is not rightReject and not followed by reject.
				if (debug) {
					console.log("No nearby allows so just place in first spot that is not rightReject.");
				}
				var numElements = jqo.nextAll().length;
				var startIndex = jqo.index();
				for (var i=startIndex; i&lt;=numElements; i++) {
					var $element = $("#drr-container").children().eq(i);
					// This element is eligible when not null, not in placement index, and not a rightReject nor is next element 
					if ($element != null &amp;&amp; (placementIndex == null || placementIndex.indexOf(i) == -1)) {
						if (!isRightReject($element) &amp;&amp; !isRightReject($element.next())) {
							return addElement(jqo, $element, divString);
						}
					}
				}
				if (debug) {
					console.log("Not going to place element: return 0.");
				}
				return 0;
			} 
		}
		
		/**
		 * @param jqo Original jquery object
		 * @param allowElement Element that is good placement for module/ad
		 * @param divString The div to be inserted before the good element
		 * @return placementHeightDiff Diff in height between original placement target and current target.
		 *
		 * If element is not too close to the end the insert the div before allowable element. 
		 * Add element index to placementIndex to keep track of which elements already have placements
		 */
		function addElement(jqo, allowElement, divString) {
			var offset = allowElement.index() - jqo.index();
			if (debug) {
				console.log("addElement: jqo index is " + jqo.index() + " allowElement index is " + allowElement.index());
			}
			
			if (elementNotNearEnd(allowElement, rightPixelWindow)) {
				allowElement.before(divString);
				if (debug) {
					console.log("addElement: Adding " + allowElement.index() + " to placementIndex.");
				}
				placementIndex.push(allowElement.index());
				if (offset == 0) {
					return 0;
				}
				else {
					return getHeightDifference(jqo,allowElement);
				}

			}
			else {
				if (debug) {
					console.log("addElement: Near the end so do NOT add.");
				}
				return 0;
			} 
		}
		
		function getHeightDifference(jqo,allowElement) {
			var offset = allowElement.index() - jqo.index();
			var height = 0;
			var children = null;
			if (offset &gt; 0) {
				children = $("#drr-container").children().slice(jqo.index(), allowElement.index() );
			}
			else {
				children = $("#drr-container").children().slice(allowElement.index(), jqo.index());
			
			}
			if (children != null) {
				children.each(function(i) {
					if (debug) {
						console.log("About to add this element's height to heigh diff offset");
						console.log($(this));
					}
		   			height += $(this).height() + grafHeight;
				});
			}
			if (offset &lt; 0) {
				height *= -1;
			}
			if (debug) {
				console.log("getHeightDifference: offset was " + offset + " and height diff is : " + height);
			}
			
			return height;
		}
		
		/** 
		 * Return true if next 550 pixels do not include a right avoid; false otherwise.
		 */
		function allowPlacement(jqo) {
			$testElement = jqo;
			var height = 0;
			while (height &lt; 550) {
				if ($testElement != null &amp;&amp; !isRightAvoid($testElement)) {
					if (debug) {
						console.log("allowPlacement: this element height is " + $testElement.height() + " and index is: " + $testElement.index());
					}
					height += $testElement.height() + grafHeight;
				}
				else {
					return false;
				}
				$testElement = $testElement.next();
			}
			return true;
		}
		
		/** 
		 * Look ahead and back for element that is not rightAvoid
		 * Return the index of the closest allowable element or null if none found.
		 */
		function findNearbyAllow(jqo) {
			if (debug) {
				console.log("In nearby allow so could not find 600 px window.");
			}
			// Check current element first 
			if (!isRightAvoid(jqo) &amp;&amp; !isRightAvoid(jqo.next())) {
				return jqo;
			}
			for (var i=1; i&lt;=2; i++) {
				if (!isRightAvoid(jqo.nextAll().slice(i-1, i)) &amp;&amp; !isRightAvoid(jqo.nextAll().slice(i, i+1)) ) {
					return jqo.nextAll().slice(i-1, i);
				}  
				if (!isRightAvoid(jqo.prevAll().slice(i-1, i)) &amp;&amp; !isRightAvoid(jqo.prevAll().slice(i, i+1))  ) {
					return jqo.prevAll().slice(i, i+1);
				}  
			}
			return null;
		}
		 
		/**
		* Return true if this is avoid element or already has module/ad placed near it
		*/
		function isRightAvoid(jqo) {
			if (placementIndex == null || placementIndex.indexOf(parseInt(jqo.index())) == -1) {
				if (jqo.is("pre") &amp;&amp; jqo.height() &gt; 300) {
					if (debug) {
						console.log("isRightAvoid: found pre. return true");
					}
					return true;
				}
				if (jqo.is("figure") &amp;&amp; jqo.hasClass('large')) {
					if (debug) {
						console.log("isRightAvoid: found figure.large return true");
					}
					return true;
				}
				if (jqo.is("figure") &amp;&amp; jqo.hasClass('medium') &amp;&amp; jqo.hasClass('inline')) {
					if (debug) {

		function elementNotNearEnd(element, pixelWindow) {

			if (element == null) {
				return false;
			}
			var remainingHeight = 0;

			if (children == null) {

        <img src="http://images.techhive.com/images/idge/imported/article/nww/2008/09/l8math-100302147-orig.jpg"/>

       <p>
        It sounds like a math phobic's worst nightmare or perhaps
        <a href="http://www.imdb.com/title/tt0119217">
         Good Will Hunting
        </a>
        for the ages.
       </p>
       <p>
        Those wacky folks at he the
        <a href="/community/node/24814">
         Defense Advanced Research Projects Agency
        </a>
        have put out a research request it calls
        <a href="https://www.fbo.gov/download/9bc/9bce380aafb19f9ad3bda188bfc1ab20/DARPA-BAA-08-65.doc">
         Mathematical Challenges
        </a>
        , that has the mighty goal of "dramatically revolutionizing mathematics and thereby strengthening DoD's scientific and technological capabilities."
       </p>
       <p>
        The challenges are in fact 23 questions that if answered, would offer a high potential for major mathematical breakthroughs,
        <a href="/community/node/29252">
         DARPA
        </a>
        said. So if you have ever wanted to settle the
        <a href="http://en.wikipedia.org/wiki/Riemann_hypothesis">
         Riemann Hypothesis
        </a>
        , which I won't begin to describe but it is one of the great unanswered questions in math history, experts say. Or perhaps you've always had a theory about Dark Energy, which in a nutshell holds that the universe is ever-expanding, this may be your calling.
       </p>
       <p>
        DARPA perhaps obviously states research grants will be awarded individually but doesn't say how much they'd be worth. The agency does say you'd need to submit your research plan by Sept. 29, 2009.
       </p>
       <p>
        So if you're game, take your pick of the following questions and have at it.
       </p>
       <ul>
        <li>
         The Mathematics of the Brain: Develop a mathematical theory to build a functional model of the brain that is mathematically consistent and predictive rather than merely biologically inspired.
        </li>
        <li>
         The Dynamics of Networks: Develop the high-dimensional mathematics needed to accurately model and predict behavior in large-scale distributed networks that evolve over time occurring in communication, biology and the social sciences.
        </li>
        <li>
         Capture and Harness
         <a href="http://www.thefreedictionary.com/stochasticity">
          Stochasticity
         </a>
         in Nature: Address
         <a href="http://wmy2000.math.jussieu.fr/3_mumfo.html">
          Mumford's call for new mathematics for the 21st century
         </a>
         . Develop methods that capture persistence in stochastic environments.
        </li>
        <li>
         21st Century Fluids: Classical fluid dynamics and the
         <a href="http://en.wikipedia.org/wiki/Navier-Stokes_equations">
          Navier-Stokes Equation
         </a>
         were extraordinarily successful in obtaining quantitative understanding of shock waves, turbulence and
         <a href="http://en.wikipedia.org/wiki/Soliton">
          solitons
         </a>
         , but new methods are needed to tackle complex fluids such as foams, suspensions, gels and liquid crystals.
        </li>
        <li>
         Biological Quantum Field Theory: Quantum and statistical methods have had great success modeling virus evolution. Can such techniques be used to model more complex systems such as bacteria? Can these techniques be used to control pathogen evolution?
        </li>
        <li>
         Computational Duality: Duality in mathematics has been a profound tool for theoretical understanding. Can it be extended to develop principled computational techniques where duality and geometry are the basis for novel algorithms?
        </li>
        <li>
         <a href="http://www.kaushik.net/avinash/occams-razor-what">
          Occam's Razor
         </a>
         in Many Dimensions: As data collection increases can we "do more with less" by finding lower bounds for sensing complexity in systems? This is related to questions about entropy maximization algorithms.
        </li>
        <li>
         Beyond
         <a href="http://www.stanford.edu/~boyd/cvxbook">
          Convex Optimization
         </a>
         : Can linear algebra be replaced by algebraic geometry in a systematic way?
        </li>
        <li>
         What are the Physical Consequences of
         <a href="http://www.ams.org/notices/200608/comm-perelman.pdf">
          Perelman's Proof of Thurston's Geometrization Theorem
         </a>
         ?: Can profound theoretical advances in understanding three dimensions be applied to construct and manipulate structures across scales to fabricate novel materials?
        </li>
        <li>
         <a href="http://eands.caltech.edu/articles/LXVII1/origami.html">
          Algorithmic Origami
         </a>
         and Biology: Build a stronger mathematical theory for isometric and rigid embedding that can give insight into protein folding.
        </li>
        <li>
         Optimal Nanostructures: Develop new mathematics for constructing optimal globally symmetric structures by following simple local rules via the process of nanoscale self-assembly.
        </li>
        <li>
         The Mathematics of Quantum Computing, Algorithms, and
         <a href="/community/node/26422">
          Entanglement
         </a>
         : In the last century we learned how quantum phenomena shape our world. In the coming century we need to develop the mathematics required to control the quantum world.
        </li>
        <li>
         Creating a Game Theory that Scales: What new scalable mathematics is needed to replace the traditional
         <a href="http://math.ucsd.edu/~driver/231-02-03/lecture_notes.htm">
          Partial Differential Equations
         </a>
         (PDE) approach to differential games?
        </li>
        <li>
         An Information Theory for Virus Evolution: Can
         <a href="http://journal.media-culture.org.au/0603/03-sampson.php">
          Shannon's theory
         </a>
         shed light on this fundamental area of biology?
        </li>
        <li>
         The Geometry of Genome Space: What notion of distance is needed to incorporate biological utility?
        </li>
        <li>
         What are the Symmetries and Action Principles for Biology?: Extend our understanding of symmetries and action principles in biology along the lines of classical thermodynamics, to include important biological concepts such as robustness, modularity, evolvability and variability.
        </li>
        <li>
         <a href="http://www2.math.northwestern.edu/langlands">
          Geometric Langlands
         </a>
         and Quantum Physics: How does the Langlands program, which originated in number theory and representation theory, explain the fundamental symmetries of physics? And vice versa?
        </li>
        <li>
         Arithmetic Langlands, Topology, and Geometry: What is the role of homotopy theory in the classical, geometric, and quantum Langlands programs?
        </li>
        <li>
         Settle the Riemann Hypothesis:
         <a href="http://en.wikipedia.org/wiki/Monty_Python_and_the_Holy_Grail">
          The Holy Grail
         </a>
         of number theory.
        </li>
        <li>
         Computation at Scale: How can we develop asymptotics for a world with massively many degrees of freedom?
        </li>
        <li>
         Settle the
         <a href="http://en.wikipedia.org/wiki/Hodge_conjecture">
          Hodge Conjecture
         </a>
         : This conjecture in algebraic geometry is a metaphor for transforming transcendental computations into algebraic ones.
        </li>
        <li>
         Settle the
         <a href="http://www.claymath.org/millennium/Poincare_Conjecture/poincare.pdf">
          Smooth Poincare Conjecture in Dimension 4
         </a>
         : What are the implications for space-time and cosmology? And might the answer unlock the secret of "dark energy"?
        </li>
        <li>
         What are the Fundamental Laws of Biology?: This question will remain front and center for the next 100 years. DARPA places this challenge last as finding these laws will undoubtedly require the mathematics developed in answering several of the questions listed above.
        </li>
       </ul>
