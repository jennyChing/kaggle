                    data: $(this).serialize(),

                    success: function(data){
                        if(data.error){
                            $.pnotify({
                                text: data.error,
                                type: 'error'
                            });
                            if(data.validation){
                                $.each(data.validation, function(i){
                                    var input = $("[name='" + i + "']", t);
                                    var control = input.parents(".control-group");
                                    if(control.length){
                                        input.parent().append("&lt;span class='help-inline'&gt;" + this + "&lt;/span&gt;");
                                        input.parents(".control-group").addClass("error");
                                    }else{
                                        input.tooltip({
                                            title: this,
                                            placement: "top"
                                        });
                                        input.tooltip("show");
                                        setTimeout(function(){
                                            input.tooltip("destroy");
                                        }, 3000);
                                    }
                                });
                            }
                        }else if(data.success){
                            $t.replaceWith("&lt;div class='alert alert-success'&gt;" + data.success + "&lt;/div&gt;");
                        }
                    }
                });
            }
