}

//the right trim function
function RTrim(VALUE)
{
	var w_space = String.fromCharCode(32);
	var v_length = VALUE.length;
	var strTemp = "";
	if(v_length &lt; 0)
	{
		return"";
	}
	var iTemp = v_length -1;

	while(iTemp &gt; -1)
	{
		if(VALUE.charAt(iTemp) == w_space)
		{
		}
		else
		{
			strTemp = VALUE.substring(0,iTemp +1);
			break;
		}
		iTemp = iTemp-1;

	} //End While
	return strTemp;

} //End Function
//The left trim function
function LTrim(VALUE)
{
	var w_space = String.fromCharCode(32);
	if(v_length &lt; 1)
	{
		return"";
	}
	var v_length = VALUE.length;
	var strTemp = "";

	var iTemp = 0;

	while(iTemp &lt; v_length)
	{
		if(VALUE.charAt(iTemp) == w_space)
		{
		}
		else
		{
			strTemp = VALUE.substring(iTemp,v_length);
			break;
		}
		iTemp = iTemp + 1;
	} //End While
	return strTemp;
} //End Function

function numOnly()
{
    if(!((event.keyCode&gt;=48)&amp;&amp;(event.keyCode&lt;=57)))
    {
        event.keyCode=0;
    }
    else event.keyCode = event.keyCode;
}
function noSpace()
{
    if((event.keyCode==32))
    {
        event.keyCode=0;
    }
    else event.keyCode = event.keyCode;
}

function numCharsOnly()
{
    if(event.keyCode &gt;=97 &amp;&amp; event.keyCode&lt;=122)
		event.keyCode = event.keyCode;
		if((event.keyCode&gt;=48)&amp;&amp;(event.keyCode&lt;=57)||(event.keyCode&gt;=65)&amp;&amp;(event.keyCode&lt;=90)

}
