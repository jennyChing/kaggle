        /*
         Read Next Widget

        var readNextWidget = (function(){

            var settings = {
                articleImage: "",
                articleTitle: "",
                articleUrl: "#",
                articleId: "",
                currentArticleId: "",
                articleList: [],
                articleListToSkip: [],
                readNextCookieName: "tmg_read_next",
                openButton: "#readNextOpen a",
                closeButton: "#closeBtn a",
                articleContainer: "#readNextArticleContainer",
                widgetContainer: "#readNextWidget",
                displayThumbnail: false,
                panelOpen: false,
                autoOpen: true
            };

            var renderTemplate = function() {
                var template = '&lt;div id="readNextWidget" style="display:none"&gt;' +
                        '&lt;div id="readNextOpen"&gt;&lt;div class="topBar"&gt;&lt;/div&gt;&lt;div id="readNextOpenInner"&gt;&lt;a href="' + settings.articleUrl + '"&gt;&lt;span&gt;Next Story&lt;/span&gt;&lt;span class="icn"&gt;&lt;/span&gt;&lt;/a&gt;&lt;/div&gt;&lt;/div&gt;' +
                        '&lt;div id="readNextArticleContainer" style="display:none"&gt;' +
                        '&lt;div id="nextArticle"&gt;&lt;div class="topBar"&gt;&lt;/div&gt;' +
                        '&lt;div id="nextArticleWrapper"&gt;'+
                        '&lt;div id="nextArticleInnerWrapper"&gt;'+
                        '&lt;p id="readNextLabel"&gt;MOST VIEWED IN NEWS&lt;/p&gt;';
                        
				if(settings.articleImage &amp;&amp; settings.articleImage!="null" &amp;&amp; settings.displayThumbnail){
					template = template + '&lt;a id="nextPageThumbnail" href="' + settings.articleUrl + '"&gt;&lt;img src="'+ settings.articleImage +'"/&gt;&lt;/a&gt;';
				}
                        
				template = template + '&lt;a id="nextPageTitle" href="' + settings.articleUrl + '"&gt;&lt;span&gt;' + settings.articleTitle + '&lt;/span&gt;&lt;/a&gt;&lt;/div&gt;&lt;/div&gt;' +
                        '&lt;/div&gt;' +
                        '&lt;div id="closeBtn"&gt;&lt;div class="topBar"&gt;&lt;/div&gt;&lt;a href="#"&gt;&lt;span&gt;&lt;/span&gt;&lt;/a&gt;&lt;/div&gt;' +
                        '&lt;/div&gt;' +
                        '&lt;iframe style="border:none;position:absolute;top:0;left:0; height:100%;width:100%;z-index:-1;" id="iefix" src="about:blank"&gt;&lt;/iframe&gt;' + //fix for IE bug
                        '&lt;/div&gt;';
                return template;
            };


            var init = function(currentArticleId, articleList, displayThumbnail) {

                settings.articleList = articleList;
                settings.currentArticleId = currentArticleId;
                settings.displayThumbnail = displayThumbnail;

                settings.articleListToSkip = unescape($.cookie(settings.readNextCookieName)).split(',');

                if ($.cookie(settings.readNextCookieName) != null) {
                    chooseNextArticle(settings.currentArticleId, settings.articleList, settings.articleListToSkip);
                    updateCookieBasedOnArticleList(settings.currentArticleId, settings.articleList, settings.articleListToSkip);
                } else {
                    chooseNextArticle(settings.currentArticleId, settings.articleList, settings.articleListToSkip);
                    createCookieBasedOnArticleList(settings.currentArticleId, settings.articleList, settings.articleListToSkip);
                }


                if (settings.articleId != "") {

                    $(renderTemplate()).appendTo("body");

                    $("#nextArticleWrapper a, #readNextOpenInner a").click(function () {
                        _gaq.push(['_trackEvent', 'wayfarer_read_next', 'click_headline', 'version_1']);

                        getNavPos($(this).attr("href"), "wayfarerReadNext");
                    });

                    bindUIActions();

                    var displayed = false;

                    setTimeout(function () {
                        if (!displayed) {
                            displayed == true;
                            readNextWidget.display();
                        }

                    }, 10000);

                    $(window).scroll(function () {
                        if (isScrolledIntoView("#tmg-related-links")) {
                            if (!displayed) {
                                displayed == true;
                                readNextWidget.display();
                            }
                        }

                        if(isScrolledIntoView("#shareBottom") || isScrolledIntoView("#newGigyaShareBottom")) {
                            if(settings.autoOpen) {
                                openArticleContainer();
                            }
                        }

                        if($(window).scrollTop() + $(window).height() == $(document).height()) {
                            if(settings.autoOpen) {
                                openArticleContainer();
                            }
                        }
                    });
                }
            };

            var display = function () {
                $(settings.widgetContainer).fadeIn(2000, function () {
                    //animation complete
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'display_closed', 'version_1']);
                });
            };

            var bindUIActions = function () {
                $(settings.openButton).mouseenter(function (event) {
                    event.preventDefault();
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'hover_closed', 'version_1']);
                    openArticleContainer();
                });
                $(settings.widgetContainer).mouseleave(function (event) {
                    event.preventDefault();
                    closeArticleContainer();
                });
                $(settings.closeButton).click(function (event) {
                    event.preventDefault();
                    _gaq.push(['_trackEvent', 'wayfarer_read_next', 'click_close', 'version_1']);
                    updateCookie(settings.articleId, settings.articleListToSkip);
                    closeArticleContainer();
                    settings.autoOpen = false;
                });
            };

            var openArticleContainer = function () {
                if (!settings.panelOpen) {
                	$("#readNextArticleContainer").show();//show/hide because height is calculated incorrectly for elements with display:none
                    $("#readNextArticleContainer").hide();
                    $(settings.articleContainer).toggle('slide', {direction: 'left'}, function () {
                        //animation complete
                        _gaq.push(['_trackEvent', 'wayfarer_read_next', 'display_open', 'version_1']);
                    });
                    settings.panelOpen = !settings.panelOpen;
                    settings.autoOpen = false; //set auto open to false once user has viewed this once
                }
                return false;
            };

            var closeArticleContainer = function () {
                if (settings.panelOpen) {
                    $(settings.articleContainer).toggle('slide', {direction: 'left'}, function () {
                        //animation complete
                    });
                    settings.panelOpen = !settings.panelOpen;
                }
                return false;
            };


            var isScrolledIntoView = function(elem) {
                if($(elem).length) { //check if element exists
                    var docViewTop = $(window).scrollTop();
                    var docViewBottom = docViewTop + $(window).height();

                    var elemTop = $(elem).offset().top;
                    var elemBottom = elemTop + $(elem).height();

                    return ((elemBottom &lt;= docViewBottom) &amp;&amp; (elemTop &gt;= docViewTop));
                } else {
                    return false;
                }
            };

            var unique = function(array) {
                return array.filter(function (el, index, arr) {
                    return index == arr.indexOf(el);
                });
            }

            var createCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        articleListToSkip[0] = currentArticleId;
                        $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
                        return true;
                    }
                });
            }

            var updateCookieBasedOnArticleList = function(currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if (article.id == currentArticleId) {
                        updateCookie(currentArticleId, articleListToSkip);
                        return true;
                    }
                });
            }

            var updateCookie = function(articleIdToSkip, articleListToSkip) {
                articleListToSkip.push(articleIdToSkip);
                articleListToSkip = unique(articleListToSkip);
                $.cookie(settings.readNextCookieName, escape(articleListToSkip.join(',')), {path: '/', expires: 1, domain: 'telegraph.co.uk'});
            }

            var chooseNextArticle = function (currentArticleId, articleList, articleListToSkip) {
                articleList.some(function (article) {
                    if ((article.id != currentArticleId) &amp;&amp; ($.inArray(article.id, articleListToSkip) === -1)) {

                        //set article
                        settings.articleTitle = article.title;
                        settings.articleUrl = article.url;
                        settings.articleId = article.id;
						settings.articleImage = article.imgUrl;
                        return true;
                    }
                });
            }

            //public API
            return {
                settings: settings,
                init: init,
                display: display,
                openArticleContainer: openArticleContainer,
                closeArticleContainer: closeArticleContainer
            }
        })();

    };
