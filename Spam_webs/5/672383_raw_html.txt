 function hideMemeMenu () {
  if (meme_menu_shown) {
   resizeMemeMenu();
   for (var i = 0; i&lt;memes.length; i++) {
    var o = memes[i].o;
    if (shmm_iv == -1) {
     o.hfr = -(200/1)*(o.tx*.3 + o.ty)/memes.length;
    } else { // we were in the middle of showing! set hide frame to the inverse of regular frame
     o.hfr = Math.max(1, Math.min(19, 20-o.fr));
    }
   }
   if (shmm_iv != -1) {
    clearInterval(shmm_iv);
    shmm_iv = -1;
   }
   meme_menu_shown = false;
   meme_menu.tta = 0;
   meme_menu.tt.style.opacity = 0;
   $(searchholder).fadeOut(300);
   $("#redditInfo").fadeIn(300);
   if (mmtrg_iv != -1) {
    clearInterval(mmtrg_iv);
    mmtrg_iv = -1;
   }
   if (meme_menu_sel) {
    meme_menu_sel.o.div.style.top = meme_menu_sel.o.backdiv.style.top = meme_menu_sel.o.yy+'px';
    meme_menu_sel = false;
   }
   mmtrg.style.display = 'none';
   mmtrg2.style.display = 'none';
   if (himm_iv == -1) {
    lhimmtm = new Date().getTime();
    mm_hfr_r = .65 + (2.5 - Math.min(2.5, (new Date().getTime() - shmm_stm)/1000));
    if (mm_hiding_with_click) {
     mm_hfr_r = .65 + 2.5;
    }
    himm_iv = setInterval('hidingMemeMenu()', 33);
    hidingMemeMenu();
   }
  }
 }

 function showMemeMenu (force_repaint) {
  var was_already_shown = meme_menu_shown;
  if (!was_already_shown) {
   $(searchholder).fadeIn(300);
   $("#redditInfo").fadeOut(300);
   $(surtch).focus();
  }
  meme_menu_shown = true;
 
  var was_hiding = false;
  if (himm_iv != -1) {
   clearInterval(himm_iv);
   himm_iv = -1;
   was_hiding = true;
  }
  if (!force_repaint) {
   if (!was_hiding) {
    surtch.value = '';
    searchLoop();
   }
  }
 
  if (force_repaint) {
   mm_fr_r = 2.45;
  } else {
   mm_fr_r = 1.45;
  }
  resizeMemeMenu();
  if (!force_repaint) {
   for (var i = 0; i&lt;memes.length; i++) {
    var o = memes[i].o;
    if (!was_hiding) {
     o.fr = -(140/1)*Math.pow((o.tx + o.ty)/memes.length, .75);
    } else { // we were in the middle of hiding! set show frame to the inverse of hide frame
     o.fr = Math.max(1, Math.min(19, 20-o.hfr));
    }
   }
  }
  if (shmm_iv == -1) {
   lshmmtm = new Date().getTime();
   mm_show_num++;
   for (var i = 0; i&lt;memes.length; i++) {
    var o = memes[i].o;
    o.cur_alpha = 0;
    o.div_shown = false;
    o.div.style.display = o.backdiv.style.display = 'none';
    o.div.style.opacity = o.backdiv.style.opacity = 0;
    if (do3d) {
     o.div.style['-webkit-transform'] = o.div.style['-moz-transform'] = o.div.style.webkitTransform = o.div.style.MozTransform = o.div.style.transform = 'none';
     o.backdiv.style['-webkit-transform'] = o.backdiv.style['-moz-transform'] = o.backdiv.style.webkitTransform = o.backdiv.style.MozTransform = o.backdiv.style.transform = 'none';
    }
   }
   if (!was_hiding) {
    shmm_stm = new Date().getTime();
   }
   if (shmm_iv == -1) {
    shmm_iv = setInterval('showingMemeMenu()', 33);
    showingMemeMenu();
   }
  }
  if (do3d) {
   meme_menu.backdiv.style.display = 'inline';
  }
  meme_menu.div.style.display = 'inline';
  if (mmtrg_iv == -1) {
   mmtrg_fr = 0;
   mmtrg_lmotm = new Date().getTime();
   mmtrg_iv = setInterval('mmTrgAnim()', 33);
  }
 }
