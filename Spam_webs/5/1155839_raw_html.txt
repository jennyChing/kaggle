            this.reset();

        try {
            var parts = resp.msg.split(' - ',2);

                msg = resp.msg;
            } else {
                i = parseInt(parts[0]);
                if (i.toString() == parts[0]){
                    index = parts[0];
                    msg = parts[1];

                    index = -1;
                    msg = resp.msg;
                }
            }
        } catch(e){
            index = -1;

                    $(input_id).focus();

                    $('#mce-'+resp.result+'-response').show();
                    $('#mce-'+resp.result+'-response').html(msg);
                }
            }
        } catch(e){
            $('#mce-'+resp.result+'-response').show();
