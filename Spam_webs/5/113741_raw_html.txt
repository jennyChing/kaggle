          function check_search_parts_form() {
                        if($('#SelectedYearId').val()=="")
                        {
                            alert('Please select all car year, make, model and part type');
                            return false;
                        }
                        if($('#SelectedMakeId').val() =="")
                        {
                            alert('Please select all car make, model and part type');
                            return false;
                        }
                        if($('#SelectedModelId').val()=="")
                        {
                            alert('Please select all car model and part type');
                            return false;
                        }
                        if ($('#SelectedPartId').val() == "") 
                        {
                            alert('Please select all car part type');
                            return false;
                        }

                        $("input[id=iy]").val($('#SelectedYearId').val());
                        $("input[id=im]").val($('#SelectedMakeId').val());
                        $("input[id=imo]").val($('#SelectedModelId').val());
                        $("input[id=it]").val($('#SelectedPartId').val());

                        return true;
                    }

                    $(document).ready(function () {

                        var searchString = window.location.search.substring(1)
                            , params = searchString.split("&amp;")
                            , hash = {}
                            ;

                        for (var i = 0; i &lt; params.length; i++) {
                            var val = params[i].split("=");
                            hash[unescape(val[0])] = unescape(val[1]);
                        }

                        jQuery.ajaxSetup({ async: false });

                        if (hash["iy"] &amp;&amp; hash["iy"].length &gt; 0) {
                            $('#SelectedYearId').val(hash["iy"]);
                            $('#SelectedYearId').trigger('change');
                        }
                        else if(readCookie('iy')) {
                            $('#SelectedYearId').val(readCookie('iy'));
                            $('#SelectedYearId').trigger('change');
                        }

                                              
                        if (hash["im"] &amp;&amp; hash["im"].length &gt; 0) {
                            $('#SelectedMakeId').val(hash["im"]);
                            $('#SelectedMakeId').trigger('change');
                        }
                        else if (readCookie('im')) {
                            $('#SelectedMakeId').val(readCookie('im'));
                            $('#SelectedMakeId').trigger('change');
                        }


                        if (hash["imo"] &amp;&amp; hash["imo"].length &gt; 0) {
                            $('#SelectedModelId').val(hash["imo"]);
                            $('#SelectedModelId').trigger('change');
                        }
                        else if (readCookie('imo')) {
                            $('#SelectedModelId').val(readCookie('imo'));
                            $('#SelectedModelId').trigger('change');
                        }


                        if (hash["it"] &amp;&amp; hash["it"].length &gt; 0) {
                            $('#SelectedPartId').val(hash["it"]);
                            $('#SelectedPartId').trigger('change');
                        }
                        else if (readCookie('it')) {
                            $('#SelectedPartId').val(readCookie('it'));
                            $('#SelectedPartId').trigger('change');
                        }

                        jQuery.ajaxSetup({ async: true });
                    });

                    function readCookie(name) {
                        var nameEQ = name + "=";
                        var ca = document.cookie.split(';');
                        for (var i = 0; i &lt; ca.length; i++) {
                            var c = ca[i];
                            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
                            if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
                        }
                        return null;
                    }
