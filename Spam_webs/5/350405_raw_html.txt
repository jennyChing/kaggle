    function validateEmail(email){
      var re = /^(([^&lt;&gt;()[\]\\.,;:\s@\"]+(\.[^&lt;&gt;()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
      if(re.test(email)){
        return true;
      }else{
        return false;
      }
    };

    function validatePassword(password){
      var re = /^(?=.*[A-Z])[A-Za-z\d$@$!%*#?&amp;^()-_+=`~|]{8,}$/;
      if(re.test(password)){
        return true;
      }else{
        return false;
      }
    };

    $('input#registration_email').bind("change",function() {
	    if ( validateEmail($(this).val()) ) {
        	$(this).removeClass('error');
        	$('#registration_email_success').show();
        	$('#registration_email_error').hide();
        	$('#registration_email_help').slideUp(500);
    		email = $(this).val();
	    } else {
        	$(this).addClass('error');
        	$('#registration_email_success').hide();
        	$('#registration_email_error').show();
        	$('#registration_email_help').slideDown(500);
	    }
    });

    $('input#registration_password').bind("change",function() {
	    if ( validatePassword($(this).val()) ) {
        	$(this).removeClass('error');
        	$('#registration_password_success').show();
        	$('#registration_password_error').hide();
        	$('#registration_password_help').slideUp(500);
	    } else {
        	$(this).addClass('error');
        	$('#registration_password_success').hide();
        	$('#registration_password_error').show();
	    }
    });
})();

    var response = parseErrorResponse(jqxhr.responseText);
    if (response.errors) {
      _.each(response.errors, function(errors, key) {
        if (key == 'base') {
          errorMessages = errorMessages.concat(errors);
        }
        else {
          errors.forEach(function(error){ errorMessages.push(formatErrorMessage(key, error));});
          var inputSelector = 'input[name*="[' + key + ']"]';
          form.find(inputSelector).parents('.control-group').addClass('error');
          $('#registration_email_error').hide();
          $('#registration_password_error').hide();
        }
      });
    }
    if (errorMessages.length == 0) {
      errorMessages.push("Yikes, that didn't work. Please try again or &lt;a href=\"mailto:support@spiceworks.com\" onClick=\"_gaq.push([\'_trackEvent\', \'Form\', \'Links\', \'Contact Support\']);\"&gt;contact support&lt;/a&gt;.");
      $('#registration_email_error').hide();
      $('#registration_email_success').hide();
      $('#registration_password_error').hide();
      $('#registration_password_success').hide();
    }
    form.find('.generic-error').html(errorMessages.join("&lt;br /&gt;")).slideDown(250);
  };

    form.find('.control-group').removeClass('error');
