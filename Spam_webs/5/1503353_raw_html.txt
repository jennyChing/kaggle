function clearFilter() {
vFilterCategory.length=0;
vFilterType.length=0;
vFilterSoftware.length=0;
redrawFilter();

}

function removeProductFilter(vFilter) {
if (vFilterCategory.indexOf(vFilter.innerHTML)&gt;-1) {
	vFilterCategory.splice(vFilterCategory.indexOf(vFilter.innerHTML),1);
	}
if (vFilterType.indexOf(vFilter.innerHTML)&gt;-1) {
	vFilterType.splice(vFilterType.indexOf(vFilter.innerHTML),1);
	}
if (vFilterSoftware.indexOf(vFilter.innerHTML)&gt;-1) {
	vFilterSoftware.splice(vFilterSoftware.indexOf(vFilter.innerHTML),1);
	}
redrawFilter();
}

function addProductFilter(vType,vFilter) {
vStartRecord=1;
if (vType=='Category') {
	if (vFilterCategory.indexOf(vFilter)&lt;0) {
		vFilterCategory.push(vFilter);
		}
	}
if (vType=='Type') {
	if (vFilterType.indexOf(vFilter)&lt;0) {
		vFilterType.push(vFilter);
		}
	}
if (vType=='Software') {
	if (vFilterSoftware.indexOf(vFilter)&lt;0) {
		vFilterSoftware.push(vFilter);
		}
	}
redrawFilter();
}
