function getRadVal(myRad)
{
	for (j=0;j&lt;myRad.length;j++)
	{
		if (myRad[j].checked)
		{
			return parseInt(myRad[j].value);
		}
	} 
	return 0;
}
function moveq(i)
{
	var qNumPrev = qNum;
	qDiv = document.getElementById('q' + (qNum+i).toString());
	if(qNum == 8 &amp;&amp; i == 1) // they have hit finish
	{
		document.getElementById("nextTop").style.display = 'none';
		document.getElementById("nextBottom").style.display = 'none';
		document.getElementById('q8').style.display = 'none';
		document.getElementById('result').style.display = 'block';
		var score = 0;
		for(k=1;k&lt;=8;k++)
		{
			var rad = document.forms["quiz"].elements["q" + k.toString() + "rad"];
			score += getRadVal(rad);
		}
		document.getElementById("scoreText").innerHTML = score.toString();
		if(score &lt;= 13){document.getElementById("result1").style.display = 'block'}
		if(score &gt;= 14 &amp;&amp; score &lt;= 19){document.getElementById("result2").style.display = 'block'}
		if(score &gt;= 20 &amp;&amp; score &lt;= 24){document.getElementById("result3").style.display = 'block'}
		return;
	}
	if(qDiv) // there is an element to move to - should never hit this if the button visibility behaves
	{
		qNum += i;
		qDiv.style.display = 'inline';
		qDivOld = document.getElementById('q' + (qNumPrev).toString());
		if(qDivOld)
		{
		  qDivOld.style.display = 'none';
		}
		if(qNum == 7 &amp;&amp; i == -1)
		{
			document.getElementById("nbut1").value = "Next";
			document.getElementById("nbut2").value = "Next";	  
		}
		if(qNum == 8)
		{
			document.getElementById("nbut1").value = "Finish";
			document.getElementById("nbut2").value = "Finish";	
		}
		if(qNum != 1)
		{
			document.getElementById("nprev1").style.visibility = "visible";
			document.getElementById("nprev2").style.visibility = "visible";
		}
		else
		{
			document.getElementById("nprev1").style.visibility = "hidden";
			document.getElementById("nprev2").style.visibility = "hidden";
		}
		// reveal next if they have already got an answer here
		var newRad = document.forms["quiz"].elements["q" + qNum + "rad"];
		if(getRadVal(newRad) != 0)
		{
			document.getElementById("nbut1").style.visibility = "visible";
			document.getElementById("nbut2").style.visibility = "visible";
		}
		else
		{
			document.getElementById("nbut1").style.visibility = "hidden";
			document.getElementById("nbut2").style.visibility = "hidden";
		}
	}
}

function getWinWidth()
{
	var w = 0;

	if(!window.innerWidth)
	{
		if(!(document.documentElement.clientWidth == 0))
		{
			w = document.documentElement.clientWidth;
		}
		else
		{
			w = document.body.clientWidth;
		}
	}
	else
	{
		w = window.innerWidth;
	}
	return w;
}

function cellEntries(json)
{
	var loadIm = document.getElementById('initIm');
	if(loadIm){loadIm.parentNode.removeChild(loadIm)}
	theComments = new Array();
	horizMultiplier = 0;
	imDiv = document.getElementById('imgDiv');
        grRow = document.getElementById('graphRow');

	var start = 0;
	var chartCount = 0;
	var ls = document.getElementById('leftScale');
	var rs = document.getElementById('rightScale');
	ls.src = 'http://chart.apis.google.com/chart?chs=26x' + (chartHeight + 10) + '&amp;chxt=y&amp;chxr=0,10,13,0.5&amp;cht=lc&amp;chd=t:';
	rs.src = 'http://chart.apis.google.com/chart?chs=18x' + (chartHeight + 10) + '&amp;chxt=r&amp;chxr=0,5,25,1&amp;cht=lc&amp;chd=t:';
	while(start &lt; json.feed.entry.length)
	{
		var endVal;
		if(start + 180 &gt; json.feed.entry.length)
		{
			endVal = json.feed.entry.length;
		}
		else
		{
			endVal = start + 180;
		}
		var imgURL = 'http://chart.apis.google.com/chart?chs=' + ((endVal - start) * chartFactor) + 'x' + chartHeight + '&amp;chco=0000FF,00FF00&amp;chls=4|4&amp;cht=lc&amp;chd=t:[data]|[data2]';
		var theData = ''; // total
		var theData2 = ''; // fat
		var thisDate = new Date();
		var previousMonth = 13;
		var previousYear = 2100;
		var prevVal = 0;
		var prevFatVal = 0;
		for (var i=start; i &lt; endVal; i++)
		{
			var entry = json.feed.entry[i];
			var val = parseFloat(entry.gsx$weight.$t);
			var fat = parseFloat(entry.gsx$fat.$t);
			if(isNaN(val))
			{
				val = prevVal;
			}
			if(isNaN(fat))
			{
				fat = prevFatVal;
			}
			if(val&gt;10 &amp;&amp; val &lt;13 &amp;&amp; fat &gt; 5 &amp;&amp; fat &lt;25)
			{
				theData += ((val-10)*100/3).toFixed(1) + ',';
				theData2 += ((fat-5)*100/20).toFixed(0) + ',';
			}
			prevVal = val;
			prevFatVal = fat;
			if(i==0)
			{
				startDate = new Date(entry.gsx$date.$t);
			}
			if(i==json.feed.entry.length - 1)
			{
				endDate = new Date(entry.gsx$date.$t);
			}
			if(entry.gsx$comments.$t != '')
			{

				theComments.push(new Array(entry.gsx$comments.$t, getHorizChartPos(i), ((chartHeight)*(13-val)/3)-16, entry.gsx$commentoffset.$t));
			}
			var thisDate = new Date(entry.gsx$date.$t);

			if(thisDate.getMonth() &gt; previousMonth || thisDate.getFullYear() &gt; previousYear)
			{
				mnthLab = document.createElement('DIV');
				mnthLab.className = 'scaleLbl mnthLab';
				mnthLab.style.left = (i*chartFactor - 20) + 'px';
				mnthLab.innerHTML = month[thisDate.getMonth()];
				imDiv.appendChild(mnthLab);
			}
			if(thisDate.getFullYear() &gt; previousYear)
			{
				yrLab = document.createElement('DIV');
				yrLab.className = 'scaleLbl yrLab';
				yrLab.style.left = (i*chartFactor - 25) + 'px';
				yrLab.innerHTML = thisDate.getFullYear();
				imDiv.appendChild(yrLab);
			}
			previousMonth = thisDate.getMonth();
			previousYear = thisDate.getFullYear();
		}
		var fullURL = imgURL.replace('[data]',theData.replace(/,$/,'')).replace('[data2]',theData2.replace(/,$/,''));
		var chartCell = document.createElement('TD');
		var chartImg = document.createElement('IMG');
		chartImg.src = fullURL;
		chartImg.className = 'chartImg'
                chartImg.style.border = 'none';
                chartImg.style.padding = '0px';
                chartImg.style.margin = '0px';
		chartCell.appendChild(chartImg);
		grRow.appendChild(chartCell);
		drawComments();
		start += 180;
		chartCount ++;
	}
        imDiv.scrollLeft = imDiv.scrollWidth;
}

function findAbsolutePos(axis,obj)
{
	var offPar = obj.offsetParent;
	var total = 0;
	while(offPar)
	{
		if(!isNaN((axis=='left')?offPar.offsetLeft:offPar.offsetTop))
		{
			total += parseInt((axis=='left')?(offPar.offsetLeft+offPar.clientLeft):(offPar.offsetTop+offPar.clientTop));
		}
		offPar = offPar.offsetParent;
	}
	return ((axis=='left')?(obj.offsetLeft+obj.clientLeft):(obj.offsetTop+obj.clientTop)) + total;
}

          <h3 class="post-title entry-title">
           <a href="http://paynowlivelater.blogspot.com/2009/05/first-2-weeks-of-transitioning-to-paleo.html">
            First 2 Weeks of Transitioning to  Paleo Diet: A Case Study
           </a>
          </h3>
          <div class="post-header-line-1">
          </div>
          <div class="post-body entry-content">
           <strong style="font-weight: normal;">
            This is a guest post by
            <a href="http://www.straighttothebar.com/" target="_blank">
             Straight to the Bar
            </a>
            's Scott Bird.
           </strong>
           <br/>
           <br/>
           <div class="articlepic" style="width: 340px; float: left; margin-right: 10px; margin-bottom: 10px;">
            <strong>
             <img alt="Scott Bird. Artwork by Vince Palko." height="240" src="http://www.straighttothebar.com/images/scott_med.jpg" title="Scott Bird. Artwork by Vince Palko." width="320"/>
            </strong>
            <div class="articlepiccaption" style="width: 320px; margin-top: 7px; color: gray; text-align: center;">
             Yes, I love the deadlift.
            </div>
           </div>
           <span style="font-size:85%;">
            <strong>
             Although my diet's been gradually getting cleaner for about 5 years now
            </strong>
            ,
           </span>
           it wasn't until
           <span style="font-size:85%;">
            <strong>
             Methuselah
            </strong>
           </span>
           wrote the superb
           <em>
            <a href="http://www.straighttothebar.com/2009/05/transitioning_to_a_paleo_diet.html" target="_blank">
             Transitioning to a Paleo Diet
            </a>
           </em>
           that I decided to investigate the Paleo approach. And following a weekend of intensive research, I decided to make the switch two weeks ago.
           <br/>
           <br/>
           Here's a look at the result of my first two weeks of Paleo eating, and the many benefits I'm already experiencing
           <span class="fullpost">
            . Fantastic.
            <h5 class="my_header">
             Making the Decision
            </h5>
            When it comes to making major, life-changing decisions like this, I typically fall into the '
            <em>
             obsessive and single-minded
            </em>
            ' group that
            <span style="font-size:85%;">
             <strong>
              Methuselah
             </strong>
            </span>
            mentioned. Aside from anything else, it's much easier to go '
            <em>
             cold turkey
            </em>
            ' than to make small changes over a long period of time. In my case, anyway.
            <br/>
            <br/>
            Following the abovementioned weekend of intensive research, on Monday morning I decided to banish three major elements from my diet. These were :
            <br/>
            <ul>
             <li>
              grains (all bread, cereals &amp; pasta),
             </li>
             <li>
              dairy products, and
             </li>
             <li>
              legumes (including peanuts, which are - strictly speaking - in this category)
             </li>
            </ul>
            The first two of these were fairly obvious and simple to avoid; the third took a little thought on a a few occasions.
            <h5 class="my_header">
             What do I Have in Place Of ...?
            </h5>
            The first few days were filled with '
            <em>
             I usually eat this, what should I have instead?
            </em>
            ' moments. The most notable of these came with breakfast, as I've been eating cereal+milk &amp; toast for years now. What could I eat instead?
            <br/>
            <br/>
            This bothered me for a while, until I remembered two occasions when the '
            <em>
             cereal+milk
            </em>
            ' thing was not an option. These were :
            <br/>
            <ul>
             <li>
              breakfast in an Edinburgh cafe, which consisted of scrambled eggs and slices of salmon
             </li>
             <li>
              breakfast in an Amsterdam hotel, which consisted of herrings, fruits, juice and coffee
             </li>
            </ul>
            These meals were both elegantly simple, and both rank high on my '
            <em>
             great breakfasts
            </em>
            ' list. Problem solved.
            <h5 class="my_header">
             Grocery Shopping
            </h5>
            I found myself shopping for groceries almost a week later, and wondering what to get. In the end, the changes were fairly minor.
            <br/>
            <br/>
            The first difference was the fact that I instinctively selected a much great variety of foods. In addition to staples such as chicken and beef were various organ and game meats, my usual purchase of 5-6 pieces of fruit became 5-6 varieties, and the cans of sardines became bags of fresh mussels.
            <br/>
            <br/>
            <strong>
             <span style="font-size:85%;">
              Best of all
             </span>
             :
            </strong>
            although I'd been bracing myself for a massive increase in cost, the difference was actually quite reasonable.
            <h5 class="my_header">
             Take-Away Food &amp; Restaurants
            </h5>
            The day after shopping was
            <em>
             Mothers' Day
            </em>
            , and I ended up going on a 600km journey with my parents. Not only was this an ideal opportunity to explore a small rural town, it also forced me to rapidly adapt Paleo eating to both take-away and restaurant menus.
            <br/>
            <br/>
            First up was a franchise of the ubiquitous McDonalds. Although I rarely eat there (perhaps one meal every 2-3 years), it was early Sunday morning in a country town. There was nothing else open.
            <br/>
            <br/>
            After staring at the menu for several minutes and wondering why the
            <em>
             Big Breakfast
            </em>
            is no longer on it (in that store, at least), I elected to start with an
            <em>
             Egg &amp; Bacon Roll
            </em>
            , and simply remove the roll. Not bad, considering the options.
            <br/>
            <br/>
            In fact, I'd probably stick with this '
            <em>
             remove the roll
            </em>
            ' mentality for a number of things on the menu. That applies to any fast-food restaurant of this type.
            <br/>
            <br/>
            Lunch was a little easier. Take-away food in Australian country towns focuses on a mix of hamburgers, pizzas and seafood. I opted for grilled fish and fish cakes, which avoided the '
            <em>
             remove the roll
            </em>
            ' thing entirely.
            <h5 class="my_header">
             Supplements
            </h5>
            For the first month or so of the transition, I'm putting my usual collection of supplements to one side. This is simply as my nutritional needs are likely to change considerably with this style of eating (particularly my intake of Vitamin C and Omega 3), and I'll find out exactly what my requirements are at the end of the first four weeks.
            <br/>
            <br/>
            One thing to note though, my intake of Vitamin B2 (
            <a href="http://en.wikipedia.org/wiki/Riboflavin" target="_blank">
             Riboflavin
            </a>
            ) has obviously increased dramatically. As the excess comes out in the urine (turning it a brilliant, pale yellow in the process), it's extremely easy to spot.
            <h5 class="my_header">
             Unexpected Benefits
            </h5>
            <div class="articlepic" style="float: right; margin-left: 10px; margin-bottom: 10px; width: 330px;">
             <img alt="Sleep, Glorious Sleep" height="240" src="http://www.straighttothebar.com/images/posts/061010_nap.jpg" title="Sleep, Glorious Sleep" width="320"/>
             <div class="articlepiccaption" style="width: 320px; color: gray; text-align: center; margin-top: 7px;">
              Sleep, Glorious Sleep.
             </div>
            </div>
            <span style="font-size:85%;">
             <strong>
              Better Sleep
             </strong>
            </span>
            : since
            <a href="http://www.straighttothebar.com/2006/10/biphasic_sleep_30_day_summary.html" target="_blank">
             I changed over to Biphasic Sleep in 2006
            </a>
            , the quality of my slumber has been fantastic. In fact, before switching to Paleo I was doubtful whether I could improve it much.
            <br/>
            <br/>
            However, after only a few days of Paleo eating, I noted that I was getting markedly better sleep. Factors such as a constant loud noise or a particularly hectic day were having less of an impact, and I awoke feeling refreshed regardless of the starting point.
            <br/>
            <br/>
            This change alone has major carry-over benefits to other activities throughout the day.
            <br/>
            <br/>
            <strong>
             <span style="font-size:85%;">
              The Disappearing Cough
             </span>
             :
            </strong>
            for the past five years I've had a regular, yet minor, cough. After this length of time I'd forgotten all about it, as I'd long ago put it in the '
            <em>
             mildly annoying, yet easily ignored
            </em>
            ' basket.
            <br/>
            <br/>
            It was only after a week of eating Paleo that I noticed it was changing - becoming far less frequent. After only two weeks it has almost disappeared entirely.
            <br/>
            <br/>
            Perhaps the cough was due to a mild intolerance to
            <em>
             <a href="http://en.wikipedia.org/wiki/Gluten" target="_blank">
              gluten
             </a>
            </em>
            , or possibly
            <em>
             <a href="http://en.wikipedia.org/wiki/Casein" target="_blank">
              casein
             </a>
            </em>
            . Whatever the case, it's fantastic to be finally getting rid of it.
            <br/>
            <br/>
            <span style="font-size:85%;">
             <strong>
              Craving for Fruit &amp; Veg
             </strong>
            </span>
            : perhaps the most unexpected - and wonderful - change was in my desire to eat fruits and vegetables. Previously I ate fruit only occasionally, and a small amount of mixed veg with meals; now I have an undeniable craving to eat every piece of fruit in sight.
            <br/>
            <br/>
            I suspect that a large part of this is simply my body looking for carbs &amp; sugars (now that other sources have been taken away); whatever the cause, I welcome the change.
            <h5 class="my_header">
             Final Thoughts on the First Two Weeks of Paleo
            </h5>
            <div>
             <div class="articlepic" style="width: 330px; float: left; margin-right: 10px; margin-bottom: 10px;">
              <img alt="Summer Fruits. Photo by The Wandering Angel." height="240" src="http://www.straighttothebar.com/images/posts/090514_summerfruits.jpg" title="Summer Fruits. Photo by The Wandering Angel." width="320"/>
              <div class="articlepiccaption" style="width: 320px; margin-top: 7px; color: gray; text-align: center;">
               Love 'em.
               <a href="http://www.flickr.com/photos/wandering_angel/848730858/" target="_blank">
                Photo by The Wandering Angel
               </a>
               .
              </div>
             </div>
             Overall, I cannot recommended it highly enough. I certainly have no plans to change anytime soon.
             <br/>
             <br/>
             Given that my starting point was a reasonably simple, healthy diet; I'm amazed at just how big a difference the Paleo approach makes.
             <span style="font-size:85%;">
              <strong>
               If better sleep, greater nutrient intake and faster recovery (from training) are things you'd love to have, give it a go. Superb.
               <br/>
               <br/>
               <br/>
               <br/>
               <br/>
               <br/>
               <br/>
               <br/>
              </strong>
             </span>
