   &lt;!--
/* JavaScript Functions */

	var varSetOver, varSetOut, varGetHikesEnabled; /* Global Vars */

	/* Keyboard handler */
	function Enter_Handler(e){
		if(navigator.appName == "Netscape")	{
			if(e.which == 13){
				disable_GetHikes(2);process_SetPref();return(false);
			}
		}
		if(navigator.appName == "Microsoft Internet Explorer"){	
			if(event.keyCode == 13){
				disable_GetHikes(2);process_SetPref();event.returnValue=false;
			}
		}
	}


	function disable_GetHikes(type)
	{
		document.GetHikes.src='images/GetHikes_Disabled.gif';
		varSetOver='images/GetHikes_Disabled.gif';
		varSetOut='images/GetHikes_Disabled.gif';
		varGetHikesEnabled=false;
		if(type == 2) //Only reset FIPS if Location Changed
			document.MainMenu.FIPS.value='_____';
	}


	function process_ChangeMiles()  /* Miles changed for Radius, only call if hikes enabled*/
	{
		document.MainMenu.GetHikesIn[4].checked = true;
		/* if disabled button is clicked, return */
		if(varGetHikesEnabled == true){
			document.MainMenu.action = 'search.asp?ChangeMiles=1'; /*Use ChangeMiles to prevent SetPrefStateID from incrementing */
			document.MainMenu.submit();
		}
	}

	function process_SetPref() /* SetPreferences Button Pressed */
	{
		/* Check for empty preference settings */
		if((document.MainMenu.Fair.checked==false &amp; document.MainMenu.Good.checked==false &amp; document.MainMenu.Great.checked==false)
			|(document.MainMenu.Easy.checked==false &amp; document.MainMenu.Moderate.checked==false &amp; document.MainMenu.Challenging.checked==false)
			|(document.MainMenu.CoupleHours.checked==false &amp; document.MainMenu.HalfDay.checked==false &amp; document.MainMenu.AllDay.checked==false))
		{
			window.alert('At least on check box must be selected in Scenery, Difficulty, and Time Available.');
			return;
		}

		if(document.MainMenu.Location.value==''){
			window.alert('You must specify a location.');
			return;
		}
		
		if(document.MainMenu.Location.value.indexOf("\'") != -1){
			window.alert('You cannot use apostrophes in a City Name.\nUse a space instead.');
			return;
		}
		
		if(document.MainMenu.Location.value.indexOf(",") != -1){
			window.alert('Enter just the city name, without the state.');
			return;
		}
		/* Check for valid Zip */
		if(parseInt(document.MainMenu.Location.value) &gt; 0 &amp;&amp;  document.MainMenu.Location.value.length != 5){
			window.alert('Zip Code must have 5 digits.');
			return;
		}

		/* Call asp code using Submit method to process Set Preferences */
		document.MainMenu.action = 'search.asp';
		document.MainMenu.submit();
	}

//--&gt;
  </script>
  <script language="JavaScript" type="text/javascript">
   &lt;!--
function process_GetHikes()  /* GetHikes Button Pressed */
{
var nhikes = new Array();
var x;
/* if disabled button is clicked, return */
if(varGetHikesEnabled == false){
window.alert('Before getting hikes, you must first click the Set Preferences button');
return;}
// Verify Hikes &gt; 0
nhikes = document.MainMenu.NumHikes.value.split(":")
for(x=0;x&lt;5;++x){
if(document.MainMenu.GetHikesIn[x].checked == true &amp;&amp; nhikes[x] == "0"){
window.alert("There are no hikes available in the selected area.\nTry selecting a larger area, such as Metro Area or State.");
return;
}
}
/* Call asp code to process Get Hikes */
document.MainMenu.action = 'GetHikes.asp?ActiveHike=SetPref&amp;SetPrefStateID=';
document.MainMenu.submit();
}
//--&gt;
  </script>
